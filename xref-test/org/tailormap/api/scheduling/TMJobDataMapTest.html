6   package org.tailormap.api.scheduling;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertNotNull;
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  
12  import java.util.Map;
13  import org.junit.jupiter.api.Test;
14  
15  class TMJobDataMapTest {
16  
17    /** Test the creation using a map with missing required parameters. */
18    @Test
19    void testInvalidMap() {
20      assertThrows(
21          IllegalArgumentException.class,
22          () -> new TMJobDataMap(Map.of(Task.TYPE_KEY, "test")),
23          "JobDataMap should not be created for missing parameters");
24    }
25  
26    /** Test the creation using a map with all required parameters. */
27    @Test
28    void testMap() {
29      TMJobDataMap jobDataMap = new TMJobDataMap(Map.of(Task.TYPE_KEY, "test", Task.DESCRIPTION_KEY, "test"));
30      assertNotNull(jobDataMap, "JobDataMap should not be null");
31      assertEquals("NONE", jobDataMap.getState().name(), "State should be NONE");
32      assertEquals(5, jobDataMap.getPriority(), "Priority should be 5");
33    }
34  
35    /** Test the creation using a map with required parameters and negative priority. */
36    @Test
37    void testMapWithPriority() {
38      TMJobDataMap jobDataMap =
39          new TMJobDataMap(Map.of(Task.TYPE_KEY, "test", Task.DESCRIPTION_KEY, "test", Task.PRIORITY_KEY, -1));
40      assertNotNull(jobDataMap, "JobDataMap should not be null");
41      assertEquals("NONE", jobDataMap.getState().name(), "State should be NONE");
42      assertEquals(0, jobDataMap.getPriority(), "Priority should be 0");
43    }
44  }
