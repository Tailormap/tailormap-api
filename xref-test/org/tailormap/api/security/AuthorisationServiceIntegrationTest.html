27  @PostgresIntegrationTest
28  class AuthorisationServiceIntegrationTest {
29    @Autowired
30    private ApplicationRepository applicationRepository;
31  
32    @Autowired
33    private GeoServiceRepository geoServiceRepository;
34  
35    @Autowired
36    AuthorisationService authorisationService;
37  
38    @ParameterizedTest
39    @ValueSource(strings = {"secured", "secured-auth"})
40    @WithMockUser(
41        username = "user",
42        authorities = {"test-foo"},
43        password = "user")
44    void testUserAllowedToViewApplication(String applicationId) {
45      Application app = applicationRepository.findByName(applicationId);
46      assertNotNull(app, () -> "Application " + applicationId + " should exist");
47      assertThat(
48          "Application " + applicationId + " should have authorization rules",
49          app.getAuthorizationRules().size(),
50          greaterThan(0));
51      assertTrue(authorisationService.userAllowedToViewApplication(app));
52    }
53  
54    @ParameterizedTest
55    @ValueSource(strings = {"secured", "secured-auth"})
56    @WithMockUser(
57        username = "tm-admin",
58        authorities = {ADMIN},
59        password = "tm-admin")
60    void testAdminUserAllowedToViewApplication(String applicationId) {

98    @Test
99    @WithMockUser(
100       username = "foo",
101       authorities = {"test-foo"},
102       password = "foo")
103   void testFooUserNotAllowedToViewBGT() {
104     final String applicationId = "secured-auth";
105     Application app = applicationRepository.findByName(applicationId);
106     assertNotNull(app, () -> "Application " + applicationId + " should exist");
107     assertThat(
108         "Application " + applicationId + " should have authorization rules",
109         app.getAuthorizationRules().size(),
110         greaterThan(0));
111     assertTrue(authorisationService.userAllowedToViewApplication(app));
112     // this id matches title "Test GeoServer (with authorization rules)"
113     // this service is allowed for user with authority test-foo
114     GeoService geoService =
115         geoServiceRepository.findById("filtered-snapshot-geoserver").orElse(null);
116     assertNotNull(geoService, () -> "GeoService should exist");
117     assertTrue(authorisationService.userAllowedToViewGeoService(geoService));
118 
119     GeoServiceLayer bgtLayer = geoService.getLayers().stream()
120         .filter(l -> "BGT".equals(l.getName()))
121         .findFirst()
122         .orElse(null);
123     assertNotNull(bgtLayer, () -> "BGT layer should exist");
124     // this layer is not allowed for user with authority test-foo
125     assertFalse(authorisationService.userAllowedToViewGeoServiceLayer(geoService, bgtLayer));
126 
127     // this layer is allowed for user with authority test-foo
128     GeoServiceLayer terreindeelLayer = geoService.getLayers().stream()
129         .filter(l -> "postgis:begroeidterreindeel".equals(l.getName()))
130         .findFirst()
131         .orElse(null);
132     assertNotNull(terreindeelLayer, () -> "begroeidterreindeel layer should exist");
133     assertTrue(authorisationService.userAllowedToViewGeoServiceLayer(geoService, terreindeelLayer));
134   }
