79    @ParameterizedTest(name = "#{index}: should return all unique values from database: {0}")
80    @MethodSource("databaseArgumentsProvider")
81    void bronhouder_unique_values_test(String url, String... expected) throws Exception {
82      url = apiBasePath + url;
83      MvcResult result = mockMvc.perform(
84              get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
85          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
86          .andExpect(status().isOk())
87          .andExpect(jsonPath("$.filterApplied").value(false))
88          .andExpect(jsonPath("$.values").isArray())
89          .andExpect(jsonPath("$.values").isNotEmpty())
90          .andReturn();
91  
92      final String body = result.getResponse().getContentAsString();
93      List<String> values = JsonPath.read(body, "$.values");
94      final Set<String> uniqueValues = new HashSet<>(values);
95  
96      assertEquals(values.size(), uniqueValues.size(), "Unique values should be unique");
97      assertTrue(uniqueValues.containsAll(Set.of(expected)), "not all values are present");
98    }

130   @ParameterizedTest(
131       name = "#{index}: should return unique bronhouder from database when filtered on bronhouder: {0}")
132   @MethodSource("databaseArgumentsProvider")
133   void bronhouder_with_filter_on_bronhouder_unique_values_test(String url, String... expected) throws Exception {
134     url = apiBasePath + url;
135     String cqlFilter = "bronhouder='G0344'";
136     if (url.contains("BRONHOUDER")) {
137       // uppercase oracle cql filter
138       cqlFilter = cqlFilter.toUpperCase(Locale.ROOT);
139     }
140 
141     MvcResult result = mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON)
142             .with(setServletPath(url))
143             .param("filter", cqlFilter))
144         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
145         .andExpect(status().isOk())
146         .andExpect(jsonPath("$.filterApplied").value(true))
147         .andExpect(jsonPath("$.values").isArray())
148         .andExpect(jsonPath("$.values").isNotEmpty())
149         .andReturn();
150 
151     final String body = result.getResponse().getContentAsString();
152     List<String> values = JsonPath.read(body, "$.values");
153 
154     assertEquals(1, values.size(), "there should only be 1 value");
155     assertTrue(List.of(expected).contains(values.get(0)), "not all values are present");
156   }

178   @RetryingTest(2)
179   // https://b3partners.atlassian.net/browse/HTM-758
180   void unique_values_from_wfs() throws Exception {
181     final String url = apiBasePath + provinciesWfsUrl;
182     MvcResult result = mockMvc.perform(
183             get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
184         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
185         .andExpect(status().isOk())
186         .andExpect(jsonPath("$.filterApplied").value(false))
187         .andExpect(jsonPath("$.values").isArray())
188         .andExpect(jsonPath("$.values").isNotEmpty())
189         .andReturn();
190 
191     final String body = result.getResponse().getContentAsString(/* for Fryslân */ StandardCharsets.UTF_8);
192     final List<String> values = JsonPath.read(body, "$.values");
193     assertEquals(12, values.size(), "there should be 12 provinces");
194 
195     final Set<String> uniqueValues = new HashSet<>(values);
196     assertEquals(values.size(), uniqueValues.size(), "Unique values should be unique");
197     assertTrue(
198         uniqueValues.containsAll(Arrays.asList(
199             "Noord-Brabant",
200             "Zuid-Holland",
201             "Utrecht",
202             "Groningen",
203             "Drenthe",
204             "Fryslân",
205             "Zeeland",
206             "Limburg",
207             "Noord-Holland",
208             "Gelderland",
209             "Flevoland",
210             "Overijssel")),
211         "not all values are present");
212 
213     final List<String> sortedValues = values.stream().sorted().collect(Collectors.toList());
214     assertEquals(sortedValues, values, "Unique values should be sorted");
215   }
