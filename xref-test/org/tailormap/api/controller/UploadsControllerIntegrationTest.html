7   package org.tailormap.api.controller;
8   
9   import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
10  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.header;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
13  
14  import org.junit.jupiter.api.MethodOrderer;
15  import org.junit.jupiter.api.Order;
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.TestMethodOrder;
18  import org.springframework.beans.factory.annotation.Autowired;
19  import org.springframework.beans.factory.annotation.Value;
20  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
21  import org.springframework.core.io.ClassPathResource;
22  import org.springframework.test.web.servlet.MockMvc;
23  import org.springframework.test.web.servlet.MvcResult;
24  import org.springframework.transaction.annotation.Transactional;
25  import org.tailormap.api.annotation.PostgresIntegrationTest;
26  import org.tailormap.api.persistence.Upload;
27  import org.tailormap.api.repository.UploadRepository;
28  
29  @PostgresIntegrationTest
30  @AutoConfigureMockMvc
31  @TestMethodOrder(MethodOrderer.OrderAnnotation.class)
32  class UploadsControllerIntegrationTest {
33    @Autowired
34    private MockMvc mockMvc;
35  
36    @Autowired
37    private UploadRepository uploadRepository;
38  
39    @Value("${tailormap-api.base-path}")
40    private String apiBasePath;
41  
42    private static String logoUrl;
43  
44    private static MvcResult logoResult;
45  
46    @Test
47    void test404() throws Exception {
48      mockMvc.perform(get(apiBasePath + "/uploads/something/a10457df-9643-4240-b70b-bf6038ec88f5/file.txt"))
49          .andExpect(status().isNotFound());
50    }
51  
52    @Test
53    void testBadRequest() throws Exception {
54      mockMvc.perform(get(apiBasePath + "/uploads/something/not-a-uuid/file.txt"))
55          .andExpect(status().isBadRequest());
56    }
57  
58    @Test
59    @Order(1)
60    @Transactional
61    void testExists() throws Exception {
62      Upload logo = uploadRepository.findByCategory(Upload.CATEGORY_APP_LOGO).get(0);
63  
64      logoUrl = apiBasePath + "/uploads/%s/%s/file.txt".formatted(logo.getCategory(), logo.getId());
65      logoResult = mockMvc.perform(get(logoUrl))
66          .andExpect(status().isOk())
67          .andExpect(content().contentType("image/svg+xml"))
68          .andExpect(header().longValue("content-length", logo.getContentLength()))
69          .andReturn();
70    }
71  
72    @Test
73    @Order(2)
74    void testConditionalNotModified() throws Exception {
75      mockMvc.perform(get(logoUrl)
76              .header("If-Modified-Since", logoResult.getResponse().getHeader("Last-Modified")))
77          .andExpect(status().isNotModified());
78    }
79  
80    @Test
81    @Order(3)
82    void testConditionalModified() throws Exception {
83      mockMvc.perform(get(logoUrl).header("If-Modified-Since", "Wed, 12 Jun 2001 09:48:38 GMT"))
84          .andExpect(status().isOk())
85          .andExpect(content().bytes(new ClassPathResource("test/gradient-logo.svg").getContentAsByteArray()));
86    }
87  }
