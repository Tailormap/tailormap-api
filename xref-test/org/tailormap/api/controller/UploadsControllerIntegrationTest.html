7   package org.tailormap.api.controller;
8   
9   import static org.hamcrest.MatcherAssert.assertThat;
10  import static org.hamcrest.Matchers.endsWith;
11  import static org.hamcrest.Matchers.startsWith;
12  import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
13  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
14  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.header;
15  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
16  
17  import com.jayway.jsonpath.JsonPath;
18  import org.junit.jupiter.api.MethodOrderer;
19  import org.junit.jupiter.api.Order;
20  import org.junit.jupiter.api.Test;
21  import org.junit.jupiter.api.TestMethodOrder;
22  import org.springframework.beans.factory.annotation.Autowired;
23  import org.springframework.beans.factory.annotation.Value;
24  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
25  import org.springframework.core.io.ClassPathResource;
26  import org.springframework.test.web.servlet.MockMvc;
27  import org.springframework.test.web.servlet.MvcResult;
28  import org.springframework.transaction.annotation.Transactional;
29  import org.tailormap.api.annotation.PostgresIntegrationTest;
30  import org.tailormap.api.persistence.Upload;
31  import org.tailormap.api.repository.UploadRepository;
32  
33  @PostgresIntegrationTest
34  @AutoConfigureMockMvc
35  @TestMethodOrder(MethodOrderer.OrderAnnotation.class)
36  class UploadsControllerIntegrationTest {
37    @Autowired
38    private MockMvc mockMvc;
39  
40    @Autowired
41    private UploadRepository uploadRepository;
42  
43    @Value("${tailormap-api.base-path}")
44    private String apiBasePath;
45  
46    private static String logoUrl;
47  
48    private static MvcResult logoResult;
49  
50    @Test
51    void test404() throws Exception {
52      mockMvc.perform(get(apiBasePath + "/uploads/something/a10457df-9643-4240-b70b-bf6038ec88f5/file.txt"))
53          .andExpect(status().isNotFound());
54    }
55  
56    @Test
57    void testBadRequest() throws Exception {
58      mockMvc.perform(get(apiBasePath + "/uploads/something/not-a-uuid/file.txt"))
59          .andExpect(status().isBadRequest());
60    }
61  
62    @Test
63    @Order(1)
64    @Transactional
65    void testExists() throws Exception {
66      Upload logo = uploadRepository.findByCategory(Upload.CATEGORY_APP_LOGO).get(0);
67  
68      logoUrl = apiBasePath + "/uploads/%s/%s/file.txt".formatted(logo.getCategory(), logo.getId());
69      logoResult = mockMvc.perform(get(logoUrl))
70          .andExpect(status().isOk())
71          .andExpect(content().contentType("image/svg+xml"))
72          .andExpect(header().longValue("content-length", logo.getContentLength()))
73          .andReturn();
74    }
75  
76    @Test
77    @Order(2)
78    void testConditionalNotModified() throws Exception {
79      mockMvc.perform(get(logoUrl)
80              .header("If-Modified-Since", logoResult.getResponse().getHeader("Last-Modified")))
81          .andExpect(status().isNotModified());
82    }
83  
84    @Test
85    @Order(3)
86    void testConditionalModified() throws Exception {
87      mockMvc.perform(get(logoUrl).header("If-Modified-Since", "Wed, 12 Jun 2001 09:48:38 GMT"))
88          .andExpect(status().isOk())
89          .andExpect(content().bytes(new ClassPathResource("test/gradient-logo.svg").getContentAsByteArray()));
90    }
91  
92    @Test
93    @Transactional
94    void testDrawingStyle() throws Exception {
95      final Upload theOnlyStyle =
96          uploadRepository.findByCategory(Upload.CATEGORY_DRAWING_STYLE).get(0);
97  
98      MvcResult result = mockMvc.perform(get(apiBasePath
99              + "/uploads/%s/%s/style.json".formatted(theOnlyStyle.getCategory(), theOnlyStyle.getId())))
100         .andExpect(status().isOk())
101         .andExpect(content().contentType("application/json"))
102         .andExpect(header().longValue("content-length", theOnlyStyle.getContentLength()))
103         .andReturn();
104 
105     final String body = result.getResponse().getContentAsString();
106     final String drinkwaterUrl = JsonPath.parse(body).read("$.styles[0].style.markerImage", String.class);
107     assertThat(drinkwaterUrl, startsWith("/uploads/drawing-style-image/"));
108     assertThat(drinkwaterUrl, endsWith("/drinkwater.svg"));
109 
110     mockMvc.perform(get(apiBasePath + drinkwaterUrl))
111         .andExpect(status().isOk())
112         .andExpect(content().contentType("image/svg+xml"))
113         .andExpect(content().bytes(new ClassPathResource("test/drinkwater.svg").getContentAsByteArray()));
114 
115     final String firstAidUrl = JsonPath.parse(body).read("$.styles[1].style.markerImage", String.class);
116     mockMvc.perform(get(apiBasePath + firstAidUrl))
117         .andExpect(status().isOk())
118         .andExpect(content().contentType("image/svg+xml"))
119         .andExpect(content().bytes(new ClassPathResource("test/first-aid.svg").getContentAsByteArray()));
120   }
121 
122   @Test
123   void testLatestUpload() throws Exception {
124     mockMvc.perform(get(apiBasePath + "/uploads/%s/latest".formatted(Upload.CATEGORY_DRAWING_STYLE_IMAGE)))
125         .andExpect(status().isOk())
126         .andExpect(content().contentType("image/svg+xml"))
127         .andExpect(content().bytes(new ClassPathResource("test/first-aid.svg").getContentAsByteArray()));
128   }
129 
130   @Test
131   void testNonExistentLatestUpload() throws Exception {
132     mockMvc.perform(get(apiBasePath + "/uploads/%s/latest".formatted("non-existent-category")))
133         .andExpect(status().isNotFound());
134   }
135 }
