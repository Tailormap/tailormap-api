9   import static org.junit.jupiter.api.Assumptions.assumeTrue;
10  import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.header;
13  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
14  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
15  
16  import java.math.BigDecimal;
17  import java.util.NoSuchElementException;
18  import org.junit.jupiter.api.BeforeEach;
19  import org.junit.jupiter.api.Test;
20  import org.junit.jupiter.api.parallel.Execution;
21  import org.junit.jupiter.api.parallel.ExecutionMode;
22  import org.junit.jupiter.api.util.DefaultTimeZone;
23  import org.junitpioneer.jupiter.Stopwatch;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.beans.factory.annotation.Value;
26  import org.springframework.boot.webmvc.test.autoconfigure.AutoConfigureMockMvc;
27  import org.springframework.http.MediaType;
28  import org.springframework.security.test.context.support.WithMockUser;
29  import org.springframework.test.web.servlet.MockMvc;
30  import org.tailormap.api.annotation.PostgresIntegrationTest;
31  import org.tailormap.api.persistence.Group;
32  import org.tailormap.api.repository.FeatureSourceRepository;
33  import org.tailormap.api.repository.FeatureTypeRepository;
34  
35  @PostgresIntegrationTest
36  @AutoConfigureMockMvc
37  @Execution(ExecutionMode.SAME_THREAD)
38  @Stopwatch
39  class AttributeStatisticsAdminControllerIntegrationTest {
40    @Value("${tailormap-api.admin.base-path}/statistics/")
41    private String adminStatsPath;
42  
43    @Autowired
44    private MockMvc mockMvc;
45  
46    @Autowired
47    private FeatureTypeRepository featureTypeRepository;
48  
49    @Autowired
50    private FeatureSourceRepository featureSourceRepository;
51  
52    private long kad_perceelFeatureTypeId;
53    private long bakFeatureTypeId;
54  
55    @BeforeEach
56    void setup() {
57      kad_perceelFeatureTypeId = featureTypeRepository
58          .getTMFeatureTypeByNameAndFeatureSource(
59              "kadastraal_perceel",
60              featureSourceRepository
61                  .getByTitle("PostGIS")
62                  .orElseThrow(() -> new NoSuchElementException("PostGIS feature source not found")))
63          .orElseThrow(() -> new NoSuchElementException("Feature type kadastraal_perceel not found"))
64          .getId();
65      assumeTrue(kad_perceelFeatureTypeId > 0, "Feature type id should be greater than 0");
66  
67      bakFeatureTypeId = featureTypeRepository
68          .getTMFeatureTypeByNameAndFeatureSource(
69              "bak",
70              featureSourceRepository
71                  .getByTitle("PostGIS")
72                  .orElseThrow(() -> new NoSuchElementException("PostGIS feature source not found")))
73          .orElseThrow(() -> new NoSuchElementException("Feature type bak not found"))
74          .getId();
75      assumeTrue(bakFeatureTypeId > 0, "Feature type id should be greater than 0");
76    }
77  
78    @Test
79    void unauthenticated_get_kadastraal_perceel_deltax_statistics() throws Exception {
80      String url = adminStatsPath + kad_perceelFeatureTypeId + "/deltax";
81      mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON))
82          .andExpect(status().isFound())
83          .andExpect(header().string("Location", "/api/unauthorized"));
84    }
85  
86    @Test
87    @WithMockUser(
88        username = "tm-admin",
89        authorities = {Group.ADMIN})
90    void get_kadastraal_perceel_deltax_statistics() throws Exception {
91      mockMvc.perform(get(adminStatsPath + kad_perceelFeatureTypeId + "/deltax")
92              .accept(MediaType.APPLICATION_JSON))
93          .andExpect(status().isOk())
94          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
95          .andExpect(jsonPath("$.count").value(4518))
96          .andExpect(jsonPath("$.min").value(-37.862))
97          .andExpect(jsonPath("$.max").value(24.861))
98          .andExpect(jsonPath("$.sum").value(closeTo(-280.1140, 0.001)))
99          .andExpect(jsonPath("$.avg").value(closeTo(BigDecimal.valueOf(-0.0619995), BigDecimal.valueOf(0.01))));
100   }
101 
102   @Test
103   @WithMockUser(
104       username = "tm-admin",
105       authorities = {Group.ADMIN})
106   @DefaultTimeZone("GMT")
107   void get_bak_creationdate_statistics() throws Exception {
108     mockMvc.perform(get(adminStatsPath + bakFeatureTypeId + "/creationdate").accept(MediaType.APPLICATION_JSON))
109         .andExpect(status().isOk())
110         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
111         .andExpect(jsonPath("$.count").value(201))
112         .andExpect(jsonPath("$.min").value("2017-01-26T00:00:00.000Z"))
113         .andExpect(jsonPath("$.max").value("2022-02-23T00:00:00.000Z"));
114   }
115 
116   @Test
117   @WithMockUser(
118       username = "tm-admin",
119       authorities = {Group.ADMIN})
120   @DefaultTimeZone("GMT")
121   void get_bak_tijdstipregistratie_statistics() throws Exception {
122     mockMvc.perform(get(adminStatsPath + bakFeatureTypeId + "/tijdstipregistratie")
123             .accept(MediaType.APPLICATION_JSON))
124         .andExpect(status().isOk())
125         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
126         .andExpect(jsonPath("$.count").value(201))
127         .andExpect(jsonPath("$.min").value("2017-03-08T12:02:57.000Z"))
128         .andExpect(jsonPath("$.max").value("2022-03-01T15:56:05.000Z"));
129   }
130 
131   @Test
132   @WithMockUser(
133       username = "tm-admin",
134       authorities = {Group.ADMIN})
135   @DefaultTimeZone("GMT")
136   void get_bak_inonderzoek_statistics() throws Exception {
137     mockMvc.perform(get(adminStatsPath + bakFeatureTypeId + "/inonderzoek").accept(MediaType.APPLICATION_JSON))
138         .andExpect(status().isBadRequest())
139         .andExpect(content().contentType(MediaType.APPLICATION_JSON));
140   }
141 }
