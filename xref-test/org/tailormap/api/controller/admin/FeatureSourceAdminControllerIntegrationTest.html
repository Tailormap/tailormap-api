30  @PostgresIntegrationTest
31  class FeatureSourceAdminControllerIntegrationTest {
32    @Autowired
33    private WebApplicationContext context;
34  
35    @Value("${tailormap-api.admin.base-path}")
36    private String adminBasePath;
37  
38    @Test
39    @WithMockUser(
40        username = "admin",
41        authorities = {Group.ADMIN})
42    void refreshJdbcFeatureSourceCapabilities() throws Exception {
43      MockMvc mockMvc = MockMvcBuilders.webAppContextSetup(context).build(); // Required for Spring Data Rest APIs
44  
45      String host = "localhost";
46      int port = 54322;
47      String database = "geodata";
48      String user = "geodata";
49      String password = "980f1c8A-25933b2";
50  
51      DriverManagerDataSource dataSource = new DriverManagerDataSource();
52      dataSource.setDriverClassName("org.postgresql.Driver");
53      dataSource.setUrl(String.format("jdbc:postgresql://%s:%s/%s", host, port, database));
54      dataSource.setUsername(user);
55      dataSource.setPassword(password);
56      JdbcTemplate jdbcTemplate = new JdbcTemplate(dataSource);
57      jdbcTemplate.execute("drop table if exists test");
58  
59      String featureSourcePOSTBody = String.format(
60          """
61  {
62  "title": "My Test Source",
63  "protocol": "JDBC",
64  "url": "",
65  "refreshCapabilities": true,
66  "jdbcConnection": {
67  "dbtype": "postgis",
68  "port": %s,
69  "host": "%s",
70  "database": "%s",
71  "schema": "public"
72  },
73  "authentication": {
74  "method": "password",
75  "username": "%s",
76  "password": "%s"
77  }
78  }""",
79          port, host, database, user, password);
80  
81      MvcResult result = mockMvc.perform(post(adminBasePath + "/feature-sources")
82              .contentType(MediaType.APPLICATION_JSON)
83              .content(featureSourcePOSTBody))
84          .andExpect(status().isCreated())
85          .andExpect(jsonPath("$.id").isNotEmpty())
86          .andExpect(jsonPath("$.allFeatureTypes").isArray())
87          .andExpect(jsonPath("$.allFeatureTypes.length()").value(31))
88          .andReturn();
89      Integer featureSourceId = JsonPath.read(result.getResponse().getContentAsString(), "$.id");
90      String selfLink = JsonPath.read(result.getResponse().getContentAsString(), "$._links.self.href");
91  
92      try {
93        jdbcTemplate.execute("create table test(id serial primary key)");
94  
95        mockMvc.perform(post(
96                adminBasePath + String.format("/feature-sources/%s/refresh-capabilities", featureSourceId)))
97            .andExpect(status().isFound())
98            .andExpect(header().string("Location", equalTo(selfLink)));
99  
100       mockMvc.perform(get(selfLink).accept(MediaType.APPLICATION_JSON))
101           .andExpect(status().isOk())
102           .andExpect(jsonPath("$.id").isNotEmpty())
103           .andExpect(jsonPath("$.allFeatureTypes").isArray())
104           .andExpect(jsonPath("$.allFeatureTypes.length()").value(32))
105           .andExpect(jsonPath("$.allFeatureTypes[?(@.name=='test')]").isNotEmpty());
106     } finally {
107       try {
108         new JdbcTemplate(dataSource).execute("drop table test");
109       } catch (Exception ignored) {
110         // tell the compiler we want to ignore any exceptions
111       }
112     }
113 
114     mockMvc.perform(post(
115             adminBasePath + String.format("/feature-sources/%s/refresh-capabilities", featureSourceId)))
116         .andExpect(status().isFound())
117         .andExpect(header().string("Location", equalTo(selfLink)));
118 
119     mockMvc.perform(get(selfLink).accept(MediaType.APPLICATION_JSON))
120         .andExpect(status().isOk())
121         .andExpect(jsonPath("$.id").isNotEmpty())
122         .andExpect(jsonPath("$.allFeatureTypes").isArray())
123         .andExpect(jsonPath("$.allFeatureTypes.length()").value(31))
124         .andExpect(jsonPath("$.allFeatureTypes[?(@.name=='test')]").isEmpty());
125   }
126 }
