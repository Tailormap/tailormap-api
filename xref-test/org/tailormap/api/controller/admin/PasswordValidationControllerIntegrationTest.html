28  @AutoConfigureMockMvc
29  @PostgresIntegrationTest
30  @TestInstance(TestInstance.Lifecycle.PER_CLASS)
31  class PasswordValidationControllerIntegrationTest {
32  
33    @Autowired
34    private WebApplicationContext context;
35  
36    private MockMvc mockMvc;
37  
38    @BeforeAll
39    void initialize() {
40      mockMvc = MockMvcBuilders.webAppContextSetup(context).build();
41    }
42  
43    @Value("${tailormap-api.admin.base-path}")
44    private String adminBasePath;
45  
46    @ParameterizedTest
47    @CsvSource({
48      "short,false",
49      "jI@#%09,false",
50      "jI@#%09y,true",
51      "very long password without special characters but unguessable,true"
52    })
53    @WithMockUser(
54        username = "tm-admin",
55        authorities = {Group.ADMIN})
56    void test(String password, String expected) throws Exception {
57      mockMvc.perform(post(adminBasePath + "/validate-password")
58              //                    .servletPath(adminBasePath + "/validate-password")
59              .contentType(MediaType.APPLICATION_FORM_URLENCODED_VALUE)
60              .param("password", password)
61              .accept(MediaType.APPLICATION_JSON))
62          .andExpect(status().isOk())
63          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
64          .andExpect(jsonPath("$.result").value(expected));
65    }
66  }
