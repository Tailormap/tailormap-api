33  
34  @PostgresIntegrationTest
35  @AutoConfigureMockMvc
36  @Execution(ExecutionMode.CONCURRENT)
37  @Stopwatch
38  @Order(1)
39  class UniqueValuesAdminControllerIntegrationTest {
40    @Value("${tailormap-api.admin.base-path}/unique-values")
41    private String adminUniqueValuesPath;
42  
43    @Autowired
44    private MockMvc mockMvc;
45  
46    @Test
47    @WithMockUser(
48        username = "tm-admin",
49        authorities = {Group.ADMIN})
50    void bronhouder_unique_values_test() throws Exception {
51      String url = adminUniqueValuesPath + "/39/bronhouder";
52      MvcResult result = mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON))
53          .andExpect(status().isOk())
54          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
55          .andExpect(jsonPath("$.filterApplied").value(false))
56          .andExpect(jsonPath("$.values").isArray())
57          .andExpect(jsonPath("$.values").isNotEmpty())
58          .andReturn();
59  
60      final String body = result.getResponse().getContentAsString();
61      List<String> values = JsonPath.read(body, "$.values");
62      final Set<String> uniqueValues = new HashSet<>(values);
63  
64      assertEquals(values.size(), uniqueValues.size(), "Unique values should be unique");
65      assertTrue(uniqueValues.containsAll(Set.of("G0344", "P0026", "G1904")), "not all values are present");
66    }
67  }
