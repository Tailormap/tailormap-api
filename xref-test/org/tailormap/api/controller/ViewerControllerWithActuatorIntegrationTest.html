6   package org.tailormap.api.controller;
7   
8   import static org.hamcrest.Matchers.containsStringIgnoringCase;
9   import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
10  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
12  import static org.tailormap.api.TestRequestProcessor.setServletPath;
13  import static org.tailormap.api.prometheus.TagNames.METRICS_APP_REQUEST_COUNTER_NAME;
14  
15  import org.junit.jupiter.api.Test;
16  import org.springframework.beans.factory.annotation.Autowired;
17  import org.springframework.beans.factory.annotation.Value;
18  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
19  import org.springframework.http.MediaType;
20  import org.springframework.security.test.context.support.WithMockUser;
21  import org.springframework.test.context.TestPropertySource;
22  import org.springframework.test.web.servlet.MockMvc;
23  import org.tailormap.api.annotation.PostgresIntegrationTest;
24  import org.tailormap.api.persistence.Group;
25  
26  @PostgresIntegrationTest
27  @AutoConfigureMockMvc
28  @TestPropertySource(
29      properties = {
30        "management.endpoints.web.exposure.include=prometheus",
31        "management.endpoint.prometheus.access=read_only",
32        "management.prometheus.metrics.export.enabled=true",
33        "management.prometheus.metrics.export.descriptions=true",
34      })
35  class ViewerControllerWithActuatorIntegrationTest {
36    @Autowired
37    private MockMvc mockMvc;
38  
39    @Value("${tailormap-api.base-path}")
40    private String apiBasePath;
41  
42    @Value("${management.endpoints.web.base-path}")
43    private String managementBasePath;
44    /**
45     * Test for the Prometheus metrics endpoint. This test checks if the endpoint is accessible and returns the expected
46     * content type and pattern. The assumption is that there have been several requests made to ViewerController
47     */
48    @Test
49    @WithMockUser(
50        username = "tm-admin",
51        authorities = {Group.ADMIN})
52    void check_prometheus_metrics_endpoint_after_app_request() throws Exception {
53      // generate some application-use metrics by accessing the ViewerController
54      final String path = apiBasePath + "/app/default";
55      mockMvc.perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
56          .andExpect(status().isOk())
57          .andExpect(content().contentType(MediaType.APPLICATION_JSON));
58  
59      // check that the metrics endpoint is accessible and returns the expected content type
60      mockMvc.perform(get(managementBasePath + "/prometheus")
61              .accept("text/plain")
62              .param("includedNames", METRICS_APP_REQUEST_COUNTER_NAME))
63          .andExpect(status().isOk())
64          .andExpect(content().contentType("text/plain;version=0.0.4;charset=utf-8"))
65          .andExpect(content()
66              .string(containsStringIgnoringCase(
67                  "tailormap_app_request_total{appId=\"1\",appName=\"default\",appType=\"app\"} 1.0")));
68    }
69  }
