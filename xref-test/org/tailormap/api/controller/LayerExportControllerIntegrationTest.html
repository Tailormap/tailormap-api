102   @Test

138         .andExpect(jsonPath("$.features.length()").value(1))

143   @Test
144   void shouldExportGeoJSONWithFilterAndSort() throws Exception {
145     final String url = apiBasePath + layerWaterdeel + downloadPath;
146     mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON)
147             .with(setServletPath(url))
148             .param("outputFormat", MediaType.APPLICATION_JSON_VALUE)
149             .param("filter", "(BRONHOUDER IN ('G1904','L0002','L0004'))")
150             .param("sortBy", "CLASS")
151             .param("sortOrder", "asc"))
152         .andExpect(status().isOk())
153         // GeoServer returns application/json;charset=UTF-8; but this is deprecated
154         // .andExpect(content().contentType(MediaType.APPLICATION_JSON_UTF8_VALUE))
155         .andExpect(jsonPath("$.type").value("FeatureCollection"))
156         .andExpect(jsonPath("$.features.length()").value(19))
157         .andExpect(jsonPath("$.features[0].geometry.type").value("Polygon"))
158         .andExpect(jsonPath("$.features[0].properties.CLASS").value("greppel, droge sloot"));
159 
160     mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON)
161             .with(setServletPath(url))
162             .param("outputFormat", MediaType.APPLICATION_JSON_VALUE)
163             .param("filter", "(BRONHOUDER IN ('G1904','L0002','L0004'))")
164             .param("sortBy", "CLASS")
165             .param("sortOrder", "desc"))
166         .andExpect(status().isOk())
167         // GeoServer returns application/json;charset=UTF-8; but this is deprecated
168         // .andExpect(content().contentType(MediaType.APPLICATION_JSON_UTF8_VALUE))
169         .andExpect(jsonPath("$.type").value("FeatureCollection"))
170         .andExpect(jsonPath("$.features.length()").value(19))
171         .andExpect(jsonPath("$.features[0].geometry.type").value("Polygon"))
172         .andExpect(jsonPath("$.features[0].properties.CLASS").value("watervlakte"));

175   @Test
176   void shouldNotExportHiddenAttributesInGeoJSONWhenRequested() throws Exception {
177     final String url = apiBasePath + layerBegroeidTerreindeelPostgis + downloadPath;
178     mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON)
179             .with(setServletPath(url))
180             .param("outputFormat", MediaType.APPLICATION_JSON_VALUE)
181             // terminationdate,geom_kruinlijn are hidden attributes
182             .param("attributes", "identificatie,bronhouder,class,terminationdate,geom_kruinlijn"))
183         .andExpect(status().is4xxClientError())
184         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
185         .andExpect(jsonPath("$.message")
186             .value("One or more requested attributes are not available on the feature type"));
187   }

189   @Test
190   @Issue("https://b3partners.atlassian.net/browse/SUPPORT-14840")
191   @DisabledUntil(
192       date = "2025-01-23",
193       reason = "This should be fixed in GeoServer 2.26.2, which is planned for 2025-01-18")
194   void shouldNotExportHiddenAttributesInGeoJSON() throws Exception {
195     final String url = apiBasePath + layerBakPostgis + downloadPath;
196     mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON)
197             .with(setServletPath(url))
198             .param("outputFormat", MediaType.APPLICATION_JSON_VALUE)
199             .param("filter", "(identificatie = 'P0026.8abeacd54c5b7500047b2112796cab56')"))
200         .andDo(MockMvcResultHandlers.print())
201         .andExpect(status().isOk())
202         .andExpect(jsonPath("$.type").value("FeatureCollection"))
203         .andExpect(jsonPath("$.features.length()").value(1))
204         // all attributes are hidden except bronhouder and identificatie
205         .andExpect(jsonPath("$.features[0].properties.identificatie")
206             .value("P0026.8abeacd54c5b7500047b2112796cab56"))
207         .andExpect(
208             jsonPath("$.features[0].properties.lv_publicatiedatum").doesNotHaveJsonPath())
209         .andExpect(jsonPath("$.features[0].properties.creationdate").doesNotHaveJsonPath())
210         .andExpect(
211             jsonPath("$.features[0].properties.tijdstipregistratie").doesNotHaveJsonPath())
212         .andExpect(jsonPath("$.features[0].properties.bronhouder").value("P0026"))
213         .andExpect(jsonPath("$.features[0].properties.inonderzoek").doesNotHaveJsonPath())
214         .andExpect(jsonPath("$.features[0].properties.relatievehoogteligging")
215             .doesNotHaveJsonPath())
216         .andExpect(jsonPath("$.features[0].properties.bgt_status").doesNotHaveJsonPath())
217         .andExpect(jsonPath("$.features[0].properties.function_").doesNotHaveJsonPath())
218         .andExpect(jsonPath("$.features[0].properties.plus_type").doesNotHaveJsonPath());
219   }
