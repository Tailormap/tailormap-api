7   package org.tailormap.api.controller;
8   
9   import static org.hamcrest.Matchers.everyItem;
10  import static org.hamcrest.Matchers.startsWith;
11  import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
13  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
14  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
15  import static org.tailormap.api.TestRequestProcessor.setServletPath;
16  
17  import org.hamcrest.Matchers;
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.api.parallel.Execution;
20  import org.junit.jupiter.api.parallel.ExecutionMode;
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.beans.factory.annotation.Value;
23  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
24  import org.springframework.http.MediaType;
25  import org.springframework.security.test.context.support.WithMockUser;
26  import org.springframework.test.web.servlet.MockMvc;
27  import org.tailormap.api.annotation.PostgresIntegrationTest;
28  
29  @PostgresIntegrationTest
30  @AutoConfigureMockMvc
31  @Execution(ExecutionMode.CONCURRENT)
32  class PageControllerIntegrationTest {
33    @Autowired
34    private MockMvc mockMvc;
35  
36    @Value("${tailormap-api.base-path}")
37    private String apiBasePath;
38  
39    @Test
40    void testPageNotFound() throws Exception {
41      final String url = apiBasePath + "/page/does-not-exist";
42      mockMvc.perform(get(url).with(setServletPath(url))).andExpect(status().isNotFound());
43    }
44  
45    @Test
46    void testHomePageWithFilteredTile() throws Exception {
47      final String url = apiBasePath + "/page";
48      mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
49          .andExpect(status().isOk())
50          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
51          .andExpect(jsonPath("$.name").value("home"))
52          .andExpect(jsonPath("$.tiles.length()").value(Matchers.greaterThan(3)))
53          .andExpect(jsonPath("$.tiles[?(@.applicationUrl == '/app/default')]")
54              .exists())
55          .andExpect(jsonPath(
56              "$.tiles[?(@.title == 'Default app')].image",
57              everyItem(startsWith("http://localhost/api/uploads/portal-image/"))))
58          .andExpect(jsonPath("$.tiles[?(@.title == 'About')].pageUrl").value("/page/about"))
59          .andExpect(jsonPath(
60                  "$.tiles[?(@.applicationUrl == '/app/secured' && @.title == 'Secured app (unfiltered)')]")
61              .exists())
62          .andExpect(jsonPath("$.tiles[?(@.applicationUrl == '/app/secured' && @.title == 'Secured app')]")
63              .doesNotHaveJsonPath());
64    }
65  
66    @Test
67    @WithMockUser(
68        username = "tm-admin",
69        authorities = {"admin"})
70    void testFilteredTileShownWhenAuthorized() throws Exception {
71      final String url = apiBasePath + "/page";
72      mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
73          .andExpect(status().isOk())
74          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
75          .andExpect(jsonPath("$.name").value("home"))
76          .andExpect(jsonPath("$.tiles[?(@.applicationUrl == '/app/secured' && @.title == 'Secured app')]")
77              .exists());
78    }
79  
80    @Test
81    void testMenuItems() throws Exception {
82      final String url = apiBasePath + "/page/home";
83      String aboutItemJsonPath = "$.menu[?(@.label == 'About')].pageUrl";
84      String b3pWebsiteItemJsonPath = "$.menu[?(@.label == 'B3Partners website')]";
85      mockMvc.perform(get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
86          .andExpect(status().isOk())
87          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
88          .andExpect(jsonPath(aboutItemJsonPath).value("/page/about"))
89          .andExpect(jsonPath(b3pWebsiteItemJsonPath).doesNotHaveJsonPath());
90  
91      mockMvc.perform(get(apiBasePath + "/page/about")
92              .accept(MediaType.APPLICATION_JSON)
93              .with(setServletPath(url)))
94          .andExpect(status().isOk())
95          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
96          .andExpect(jsonPath(aboutItemJsonPath).exists())
97          .andExpect(jsonPath(b3pWebsiteItemJsonPath).exists());
98    }
99  }
