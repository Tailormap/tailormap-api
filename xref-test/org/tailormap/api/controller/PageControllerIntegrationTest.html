7   package org.tailormap.api.controller;
8   
9   import static org.hamcrest.Matchers.everyItem;
10  import static org.hamcrest.Matchers.startsWith;
11  import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
13  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
14  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
15  import static org.tailormap.api.TestRequestProcessor.setServletPath;
16  
17  import org.hamcrest.Matchers;
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.api.parallel.Execution;
20  import org.junit.jupiter.api.parallel.ExecutionMode;
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.beans.factory.annotation.Value;
23  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
24  import org.springframework.http.MediaType;
25  import org.springframework.security.test.context.support.WithMockUser;
26  import org.springframework.test.web.servlet.MockMvc;
27  import org.tailormap.api.annotation.PostgresIntegrationTest;
28  
29  @PostgresIntegrationTest
30  @AutoConfigureMockMvc
31  @Execution(ExecutionMode.CONCURRENT)
32  class PageControllerIntegrationTest {
33    @Autowired private MockMvc mockMvc;
34  
35    @Value("${tailormap-api.base-path}")
36    private String apiBasePath;
37  
38    @Test
39    void testPageNotFound() throws Exception {
40      final String url = apiBasePath + "/page/does-not-exist";
41      mockMvc.perform(get(url).with(setServletPath(url))).andExpect(status().isNotFound());
42    }
43  
44    @Test
45    void testHomePageWithFilteredTile() throws Exception {
46      final String url = apiBasePath + "/page";
47      mockMvc
48          .perform(get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
49          .andExpect(status().isOk())
50          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
51          .andExpect(jsonPath("$.name").value("home"))
52          .andExpect(jsonPath("$.tiles.length()").value(Matchers.greaterThan(3)))
53          .andExpect(jsonPath("$.tiles[?(@.applicationUrl == '/app/default')]").exists())
54          .andExpect(
55              jsonPath(
56                  "$.tiles[?(@.title == 'Default app')].image",
57                  everyItem(startsWith("http://localhost/api/uploads/portal-image/"))))
58          .andExpect(jsonPath("$.tiles[?(@.title == 'About')].pageUrl").value("/page/about"))
59          .andExpect(
60              jsonPath(
61                      "$.tiles[?(@.applicationUrl == '/app/secured' && @.title == 'Secured app (unfiltered)')]")
62                  .exists())
63          .andExpect(
64              jsonPath("$.tiles[?(@.applicationUrl == '/app/secured' && @.title == 'Secured app')]")
65                  .doesNotExist());
66    }
67  
68    @Test
69    @WithMockUser(
70        username = "tm-admin",
71        authorities = {"admin"})
72    void testFilteredTileShownWhenAuthorized() throws Exception {
73      final String url = apiBasePath + "/page";
74      mockMvc
75          .perform(get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
76          .andExpect(status().isOk())
77          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
78          .andExpect(jsonPath("$.name").value("home"))
79          .andExpect(
80              jsonPath("$.tiles[?(@.applicationUrl == '/app/secured' && @.title == 'Secured app')]")
81                  .exists());
82    }
83  
84    @Test
85    void testMenuItems() throws Exception {
86      final String url = apiBasePath + "/page/home";
87      String aboutItemJsonPath = "$.menu[?(@.label == 'About')].pageUrl";
88      String b3pWebsiteItemJsonPath = "$.menu[?(@.label == 'B3Partners website')]";
89      mockMvc
90          .perform(get(url).accept(MediaType.APPLICATION_JSON).with(setServletPath(url)))
91          .andExpect(status().isOk())
92          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
93          .andExpect(jsonPath(aboutItemJsonPath).value("/page/about"))
94          .andExpect(jsonPath(b3pWebsiteItemJsonPath).doesNotExist());
95  
96      mockMvc
97          .perform(
98              get(apiBasePath + "/page/about")
99                  .accept(MediaType.APPLICATION_JSON)
100                 .with(setServletPath(url)))
101         .andExpect(status().isOk())
102         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
103         .andExpect(jsonPath(aboutItemJsonPath).exists())
104         .andExpect(jsonPath(b3pWebsiteItemJsonPath).exists());
105   }
106 }
