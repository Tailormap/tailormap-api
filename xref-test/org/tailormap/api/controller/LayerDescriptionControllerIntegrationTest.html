27  @PostgresIntegrationTest

31    @Autowired
32    private MockMvc mockMvc;
33  
34    @Value("${tailormap-api.base-path}")
35    private String apiBasePath;
36  
37    @Test
38    void app_not_found_404() throws Exception {
39      final String path = apiBasePath + "/app/1234/layer/76/describe";
40      mockMvc.perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
41          .andExpect(status().isNotFound())
42          .andExpect(content().contentTypeCompatibleWith(MediaType.APPLICATION_JSON))
43          .andExpect(jsonPath("$.message").value("Not Found"));
44    }
45  
46    @Test
47    void public_app() throws Exception {
48      final String path = apiBasePath + layerBegroeidTerreindeelPostgis + "/describe";
49      mockMvc.perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
50          .andExpect(status().isOk())
51          .andExpect(content().contentTypeCompatibleWith(MediaType.APPLICATION_JSON))
52          .andExpect(jsonPath("$.featureTypeName").value("begroeidterreindeel"))
53          .andExpect(jsonPath("$.geometryAttribute").value("geom"))
54          .andExpect(jsonPath("$.id").value("lyr:snapshot-geoserver:postgis:begroeidterreindeel"))
55          .andExpect(jsonPath("$.serviceId").value("snapshot-geoserver"))
56          .andExpect(jsonPath("$.attributes").isArray())
57          .andExpect(jsonPath("$.attributes[?(@.name == 'relatievehoogteligging')].type")
58              .value("integer"))
59          .andExpectAll(
60              jsonPath("$.attributes[?(@.name == 'terminationdate')]").doesNotHaveJsonPath(),
61              jsonPath("$.attributes[?(@.name == 'geom_kruinlijn')]").doesNotHaveJsonPath())
62          .andExpect(
63              jsonPath("$.attributes[?(@.name == 'gmlid')].nullable").value(false))
64          .andExpect(
65              jsonPath("$.attributes[?(@.name == 'gmlid')].editable").value(false))
66          .andExpect(jsonPath("$.editable").value(true));
67    }
68  
69    @Test
70    void appLayer_with_feature_type_but_no_editable_setting() throws Exception {
71      final String path =
72          apiBasePath + "/app/default/layer/lyr:snapshot-geoserver-proxied:postgis:begroeidterreindeel/describe";
73      mockMvc.perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
74          .andExpect(status().isOk())
75          .andExpect(content().contentTypeCompatibleWith(MediaType.APPLICATION_JSON))
76          .andExpect(jsonPath("$.featureTypeName").value("begroeidterreindeel"))
77          .andExpect(jsonPath("$.geometryAttribute").value("geom"))
78          .andExpect(jsonPath("$.id").value("lyr:snapshot-geoserver-proxied:postgis:begroeidterreindeel"))
79          .andExpect(jsonPath("$.serviceId").value("snapshot-geoserver-proxied"))
80          .andExpect(jsonPath("$.attributes").isArray())
81          .andExpect(jsonPath("$.attributes[?(@.name == 'relatievehoogteligging')].type")
82              .value("integer"))
83          .andExpect(
84              jsonPath("$.attributes[?(@.name == 'gmlid')].nullable").value(false))
85          .andExpect(
86              jsonPath("$.attributes[?(@.name == 'gmlid')].editable").value(false))
87          .andExpect(jsonPath("$.editable").value(false));
88    }

90    @Test
91    void test_wms_secured_app_denied() throws Exception {
92      final String path =
93          apiBasePath + "/app/secured/layer/lyr:snapshot-geoserver-proxied:postgis:begroeidterreindeel/describe";
94      mockMvc.perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
95          .andExpect(status().isUnauthorized())
96          .andExpect(content().contentTypeCompatibleWith(MediaType.APPLICATION_JSON))
97          .andExpect(jsonPath("$.code").value(401))
98          .andExpect(jsonPath("$.url").value("/login"));
99    }
