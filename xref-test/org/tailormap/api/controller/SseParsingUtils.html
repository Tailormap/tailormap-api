8   import org.tailormap.api.viewer.model.ServerSentEventResponse;
9   import tools.jackson.core.JacksonException;
10  import tools.jackson.databind.ObjectMapper;
11  
12  abstract class SseParsingUtils {
13  
14    /**
15     * Parse the last non-empty line from the SSE stream that looks something like:
16     * {@code data:{"details":{"message":"Extract task
17     * completed","progress":100,"downloadId":"begroeidterreindeel15061479295163305053.csv"},"eventType":"extract-completed","id":"019d6838-7f48-7053-9256-dd4b57c14264"}
18     * } as JSON and extract the file from the details.
19     */
20    String getLastCompletedEventJson(String sseMessages) {
21      return java.util.Arrays.stream(sseMessages.split("\\R"))
22          .map(String::trim)
23          .filter(line -> !line.isEmpty())
24          .filter(line -> line.startsWith("data:"))
25          .filter(line -> line.contains("\"eventType\":\"extract-completed\""))
26          .reduce((first, second) -> second)
27          .orElseThrow()
28          .substring("data:".length());
29    }
30  
31    String getDownloadId(String eventJson) throws JacksonException {
32      return new ObjectMapper()
33          .readTree(eventJson)
34          .path("details")
35          .path("downloadId")
36          .asString();
37    }
38  
39    int count_completed_messages(String s) {
40      int count = 0;
41      int index = 0;
42      final String marker = "\"eventType\":\"" + ServerSentEventResponse.EventTypeEnum.EXTRACT_COMPLETED + "\"";
43      while ((index = s.indexOf(marker, index)) != -1) {
44        count++;
45        index += marker.length();
46      }
47      return count;
48    }
