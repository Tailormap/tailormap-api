125   @SuppressModernizer
126   private static void assertImagesEqual(String expectedImageFileName, MvcResult result, TestInfo testInfo)
127       throws Exception {
128     BufferedImage expectedImage = ImageComparisonUtil.readImageFromResources(expectedImageFileName);
129 
130     BufferedImage actualImage =
131         ImageIO.read(new ByteArrayInputStream(result.getResponse().getContentAsByteArray()));
132     ImageIO.write(actualImage, "png", new File("./target/" + testInfo.getDisplayName() + "_actual.png"));
133 
134     ImageComparisonResult imageComparisonResult = new ImageComparison(expectedImage, actualImage)
135         .setAllowingPercentOfDifferentPixels(5)
136         .setDestination(new File("./target/" + testInfo.getDisplayName() + "_comparison.png"))
137         .compareImages();
138 
139     assertEquals(ImageComparisonState.MATCH, imageComparisonResult.getImageComparisonState());
140   }

340   private static class StringIsNotZeroMatcher extends TypeSafeMatcher<String> {
341     @Override
342     protected boolean matchesSafely(String item) {
343       return Long.parseLong(item) > 0;
344     }
345 
346     @Override
347     public void describeTo(Description description) {
348       description.appendText("is not zero");
349     }
