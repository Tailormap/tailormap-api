138   @SuppressModernizer
139   private static void assertImagesEqual(String expectedImageFileName, MvcResult result, TestInfo testInfo)
140       throws Exception {
141     BufferedImage expectedImage = ImageComparisonUtil.readImageFromResources(expectedImageFileName);
142 
143     BufferedImage actualImage =
144         ImageIO.read(new ByteArrayInputStream(result.getResponse().getContentAsByteArray()));
145     ImageIO.write(actualImage, "png", new File("./target/" + testInfo.getDisplayName() + "_actual.png"));
146 
147     ImageComparisonResult imageComparisonResult = new ImageComparison(expectedImage, actualImage)
148         .setAllowingPercentOfDifferentPixels(5)
149         .setDestination(new File("./target/" + testInfo.getDisplayName() + "_comparison.png"))
150         .compareImages();
151 
152     assertEquals(ImageComparisonState.MATCH, imageComparisonResult.getImageComparisonState());
153   }

422   private static class StringIsNotZeroMatcher extends TypeSafeMatcher<String> {
423     @Override
424     protected boolean matchesSafely(String item) {
425       return Long.parseLong(item) > 0;
426     }
427 
428     @Override
429     public void describeTo(Description description) {
430       description.appendText("is not zero");
