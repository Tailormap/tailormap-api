8   import static org.hamcrest.MatcherAssert.assertThat;
9   import static org.hamcrest.Matchers.containsStringIgnoringCase;
10  import static org.hamcrest.Matchers.greaterThan;
11  import static org.hamcrest.Matchers.is;
12  import static org.junit.jupiter.api.Assertions.assertEquals;
13  import static org.junit.jupiter.api.Assertions.assertTrue;

16  import com.fasterxml.jackson.databind.JsonNode;
17  import java.io.IOException;
18  import org.junit.jupiter.api.MethodOrderer;
19  import org.junit.jupiter.api.Order;
20  import org.junit.jupiter.api.Test;
21  import org.junit.jupiter.api.TestMethodOrder;
22  import org.junitpioneer.jupiter.DisableIfTestFails;
23  import org.junitpioneer.jupiter.Stopwatch;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.tailormap.api.annotation.PostgresIntegrationTest;
26  
27  @PostgresIntegrationTest
28  @Stopwatch
29  @TestMethodOrder(MethodOrderer.OrderAnnotation.class)
30  @DisableIfTestFails
31  @Order(Integer.MAX_VALUE - 1)
32  class PrometheusServiceIntegrationTest {
33    @Autowired
34    private PrometheusService prometheusService;
35  
36    @Test
37    @Order(1)
38    void isPrometheusAvailable() {
39      boolean isAvailable = prometheusService.isPrometheusAvailable();
40      assertTrue(isAvailable, "Prometheus server should be available");
41    }
42  
43    @Test
44    @Order(2)
45    void testQueryExecution() {
46      final String query = "scrape_duration_seconds";
47      try {
48        final JsonNode root = prometheusService.executeQuery(query);
49        assertEquals("success", root.path("status").asText());
50        assertTrue(root.path("data").path("result").isArray(), "Results should be an array");
51        assertThat(
52            root.path("data").path("result").get(0).path("value").get(1).asDouble(), is(greaterThan(0.0)));
53      } catch (Exception e) {
54        fail("Query execution failed: " + e.getMessage());
55      }
56    }
57  
58    @Test
59    @Order(2)
60    void testQueryExecutionLargeQuery() {
61      final String query =
62          """
63  label_replace(floor(increase(tailormap_app_request_total[90d])), "type", "total", "__name__", ".*")\s
64  or\s
65  label_replace(floor(time()-max_over_time(timestamp(changes(tailormap_app_request_total[5m])>0)[90d:1m])), "type", "last_updated", "__name__", ".*")
66  """;
67      try {
68        final JsonNode root = prometheusService.executeQuery(query);
69        assertEquals("success", root.path("status").asText());
70        assertTrue(root.path("data").path("result").isArray(), "Results should be an array");
71        assertThat(
72            root.path("data").path("result").get(0).path("value").get(1).asDouble(), is(greaterThan(0.0)));
73      } catch (Exception e) {
74        fail("Query execution failed: " + e.getMessage());
75      }
76    }
77  
78    @Test
79    @Order(2)
80    void testQueryExecutionWithUnknownMetric() {
81      final String query = "unknown_metric_name";
82      try {
83        JsonNode result = prometheusService.executeQuery(query);
84        // Even if the metric is unknown, prometheus will still return a valid response
85        assertEquals("success", result.path("status").asText());
86        assertTrue(result.path("data").path("result").isArray(), "Results should be an array");
87        assertEquals(0, result.path("data").path("result").size(), "Result should be empty for unknown metric");
88      } catch (Exception e) {
89        fail("Query execution failed: " + e.getMessage());
90      }
91    }
92  
93    @Test
94    @Order(2)
95    void testQueryExecutionWithEmptyQuery() {
96      try {
97        prometheusService.executeQuery("");
98        fail("Expected an exception for empty query");
99      } catch (IOException e) {
100       assertThat(e.getMessage(), containsStringIgnoringCase("invalid parameter"));
101     }
102   }
