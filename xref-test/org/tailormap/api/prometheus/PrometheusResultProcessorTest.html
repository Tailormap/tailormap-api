6   package org.tailormap.api.prometheus;
7   
8   import static org.junit.jupiter.api.Assertions.assertAll;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  import static org.junit.jupiter.api.Assertions.fail;
11  import static org.junit.jupiter.api.Assumptions.assumingThat;
12  
13  import com.fasterxml.jackson.core.JsonProcessingException;
14  import java.lang.invoke.MethodHandles;
15  import java.util.Map;
16  import org.junit.jupiter.api.Test;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  
20  class PrometheusResultProcessorTest {
21    private static final Logger logger =
22        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
23    private final PrometheusResultProcessor processor = new PrometheusResultProcessor();
24  
25    @Test
26    void testCombinedQueryResults() {
27      String jsonResponse =
28          """
29  {
30  "status": "success",
31  "data": {
32  "resultType": "vector",
33  "result": [
34  {
35  "metric": {
36  "appId": "1",
37  "appName": "default",
38  "appType": "app",
39  "application": "tailormap-api",
40  "hostname": "localhost",
41  "instance": "host.docker.internal:8080",
42  "job": "tailormap-api-snapshot",
43  "type": "total"
44  }, "value": [1755163922.424, "35" ]},
45  {
46  "metric": {
47  "appId": "2",
48  "appName": "snapshot-geoserver",
49  "appType": "service",
50  "application": "tailormap-api",
51  "hostname": "localhost",
52  "instance": "host.docker.internal:8080",
53  "job": "tailormap-api-snapshot",
54  "type": "total"
55  }, "value": [1755163922.424,"5"] },
56  {
57  "metric": {
58  "appId": "5",
59  "appName": "austria",
60  "appType": "app",
61  "application": "tailormap-api",
62  "hostname": "localhost",
63  "instance": "host.docker.internal:8080",
64  "job": "tailormap-api-snapshot",
65  "type": "total"
66  }, "value": [1755163922.424,"17"] },
67  {
68  "metric": {
69  "appId": "1",
70  "appName": "default",
71  "appType": "app",
72  "application": "tailormap-api",
73  "hostname": "localhost",
74  "instance": "host.docker.internal:8080",
75  "job": "tailormap-api-snapshot",
76  "type": "last_updated"
77  }, "value": [1755163922.424,"6777" ] },
78  {
79  "metric": {
80  "appId": "5",
81  "appName": "austria",
82  "appType": "app",
83  "application": "tailormap-api",
84  "hostname": "localhost",
85  "instance": "host.docker.internal:8080",
86  "job": "tailormap-api-snapshot",
87  "type": "last_updated"
88  }, "value": [1755163922.424,"11577" ]},
89  {
90  "metric": {
91  "appId": "2",
92  "appName": "snapshot-geoserver",
93  "appType": "service",
94  "application": "tailormap-api",
95  "hostname": "localhost",
96  "instance": "host.docker.internal:8080",
97  "job": "tailormap-api-snapshot",
98  "type": "last_updated"
99  }, "value": [1755163922.424,"11577"]}
100 ]}}
101 """;
102     try {
103       Map<String, Map<String, String>> processedResults = processor.processPrometheusResults(jsonResponse);
104       assertEquals(3, processedResults.size());
105       processedResults.forEach((appId, metrics) -> {
106         logger.info("appId: {}\t{}", appId, metrics);
107 
108         assumingThat(
109             "1".equals(appId),
110             () -> assertAll(
111                 "Default App Metrics",
112                 () -> assertEquals(
113                     "default", metrics.get(PrometheusResultProcessor.METRICS_APP_NAME_TAG)),
114                 () -> assertEquals("35", metrics.get("total")),
115                 () -> assertEquals("6777", metrics.get("last_updated"))));
116         assumingThat(
117             "2".equals(appId),
118             () -> assertAll(
119                 "Snapshot Geoserver Metrics",
120                 () -> assertEquals(
121                     "snapshot-geoserver",
122                     metrics.get(PrometheusResultProcessor.METRICS_APP_NAME_TAG)),
123                 () -> assertEquals("5", metrics.get("total")),
124                 () -> assertEquals("11577", metrics.get("last_updated"))));
125         assumingThat(
126             "5".equals(appId),
127             () -> assertAll(
128                 "Austria App Metrics",
129                 () -> assertEquals(
130                     "austria", metrics.get(PrometheusResultProcessor.METRICS_APP_NAME_TAG)),
131                 () -> assertEquals("17", metrics.get("total")),
132                 () -> assertEquals("11577", metrics.get("last_updated"))));
133       });
134     } catch (JsonProcessingException e) {
135       fail("Failed to process JSON response: " + e.getMessage());
136     }
137   }
138 }
