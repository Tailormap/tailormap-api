6   package org.tailormap.api.repository.validation;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertFalse;
10  import static org.junit.jupiter.api.Assertions.assertTrue;
11  import static org.junit.jupiter.api.Assertions.fail;
12  
13  import org.junit.jupiter.api.BeforeEach;
14  import org.junit.jupiter.api.Test;
15  import org.springframework.beans.factory.annotation.Autowired;
16  import org.springframework.validation.BeanPropertyBindingResult;
17  import org.springframework.validation.Errors;
18  import org.tailormap.api.annotation.PostgresIntegrationTest;
19  import org.tailormap.api.persistence.SearchIndex;
20  import org.tailormap.api.repository.FeatureSourceRepository;
21  import org.tailormap.api.repository.FeatureTypeRepository;
22  
23  @PostgresIntegrationTest
24  class SearchIndexValidatorIntegrationTest {
25    private SearchIndexValidator searchIndexValidator;
26  
27    @Autowired private FeatureTypeRepository featureTypeRepository;
28    @Autowired private FeatureSourceRepository featureSourceRepository;
29  
30    @BeforeEach
31    void setUp() {
32      searchIndexValidator = new SearchIndexValidator(featureTypeRepository);
33    }
34  
35    @Test
36    void supports() {
37      assertTrue(searchIndexValidator.supports(SearchIndex.class));
38      assertFalse(searchIndexValidator.supports(Object.class));
39    }
40  
41    @Test
42    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
43    void testWFSFeatureType() {
44      featureSourceRepository
45          .getByTitle("WFS for Test GeoServer")
46          .ifPresentOrElse(
47              featureSource -> {
48                featureTypeRepository
49                    .getTMFeatureTypeByNameAndFeatureSource(
50                        "postgis:begroeidterreindeel", featureSource)
51                    .ifPresentOrElse(
52                        featureType -> {
53                          SearchIndex wfsIndex =
54                              new SearchIndex().setFeatureTypeId(featureType.getId());
55                          Errors errors = new BeanPropertyBindingResult(wfsIndex, "");
56                          searchIndexValidator.validate(wfsIndex, errors);
57                          assertEquals(1, errors.getErrorCount(), "Expected 1 error");
58                          assertEquals(
59                              "This feature type is not available for indexing.",
60                              errors.getAllErrors().get(0).getDefaultMessage(),
61                              "Did not get expected error message");
62                        },
63                        () -> fail("Feature type not found"));
64              },
65              () -> fail("Feature source not found"));
66    }
67  
68    @Test
69    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
70    void testJDBCFeatureType() {
71      featureSourceRepository
72          .getByTitle("PostGIS")
73          .ifPresentOrElse(
74              featureSource -> {
75                featureTypeRepository
76                    .getTMFeatureTypeByNameAndFeatureSource("begroeidterreindeel", featureSource)
77                    .ifPresentOrElse(
78                        featureType -> {
79                          SearchIndex jdbcIndex =
80                              new SearchIndex().setFeatureTypeId(featureType.getId());
81                          Errors errors = new BeanPropertyBindingResult(jdbcIndex, "");
82                          searchIndexValidator.validate(jdbcIndex, errors);
83                          assertEquals(0, errors.getErrorCount(), "Expected no errors");
84                        },
85                        () -> fail("Feature type not found"));
86              },
87              () -> fail("Feature source not found"));
88    }
89  }
