6   package org.tailormap.api.repository.validation;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertFalse;
10  import static org.junit.jupiter.api.Assertions.assertTrue;
11  import static org.junit.jupiter.api.Assertions.fail;
12  
13  import org.junit.jupiter.api.BeforeEach;
14  import org.junit.jupiter.api.Test;
15  import org.springframework.beans.factory.annotation.Autowired;
16  import org.springframework.validation.BeanPropertyBindingResult;
17  import org.springframework.validation.Errors;
18  import org.tailormap.api.annotation.PostgresIntegrationTest;
19  import org.tailormap.api.persistence.SearchIndex;
20  import org.tailormap.api.repository.FeatureSourceRepository;
21  import org.tailormap.api.repository.FeatureTypeRepository;
22  
23  @PostgresIntegrationTest
24  class SearchIndexValidatorIntegrationTest {
25    private SearchIndexValidator searchIndexValidator;
26  
27    @Autowired
28    private FeatureTypeRepository featureTypeRepository;
29  
30    @Autowired
31    private FeatureSourceRepository featureSourceRepository;
32  
33    @BeforeEach
34    void setUp() {
35      searchIndexValidator = new SearchIndexValidator(featureTypeRepository);
36    }
37  
38    @Test
39    void supports() {
40      assertTrue(searchIndexValidator.supports(SearchIndex.class));
41      assertFalse(searchIndexValidator.supports(Object.class));
42    }
43  
44    @Test
45    @SuppressWarnings("PMD.UnitTestShouldIncludeAssert")
46    void testWFSFeatureType() {
47      featureSourceRepository
48          .getByTitle("WFS for Test GeoServer")
49          .ifPresentOrElse(
50              featureSource -> featureTypeRepository
51                  .getTMFeatureTypeByNameAndFeatureSource("postgis:begroeidterreindeel", featureSource)
52                  .ifPresentOrElse(
53                      featureType -> {
54                        SearchIndex wfsIndex =
55                            new SearchIndex().setFeatureTypeId(featureType.getId());
56                        Errors errors = new BeanPropertyBindingResult(wfsIndex, "");
57                        searchIndexValidator.validate(wfsIndex, errors);
58                        assertEquals(1, errors.getErrorCount(), "Expected 1 error");
59                        assertEquals(
60                            "This feature type is not available for indexing.",
61                            errors.getAllErrors().get(0).getDefaultMessage(),
62                            "Did not get expected error message");
63                      },
64                      () -> fail("Feature type not found")),
65              () -> fail("Feature source not found"));
66    }
67  
68    @Test
69    @SuppressWarnings("PMD.UnitTestShouldIncludeAssert")
70    void testJDBCFeatureType() {
71      featureSourceRepository
72          .getByTitle("PostGIS")
73          .ifPresentOrElse(
74              featureSource -> featureTypeRepository
75                  .getTMFeatureTypeByNameAndFeatureSource("begroeidterreindeel", featureSource)
76                  .ifPresentOrElse(
77                      featureType -> {
78                        SearchIndex jdbcIndex =
79                            new SearchIndex().setFeatureTypeId(featureType.getId());
80                        Errors errors = new BeanPropertyBindingResult(jdbcIndex, "");
81                        searchIndexValidator.validate(jdbcIndex, errors);
82                        assertEquals(0, errors.getErrorCount(), "Expected no errors");
83                      },
84                      () -> fail("Feature type not found")),
85              () -> fail("Feature source not found"));
86    }
87  }
