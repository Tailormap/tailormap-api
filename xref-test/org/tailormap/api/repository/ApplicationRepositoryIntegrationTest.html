8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertNotNull;
10  import static org.junit.jupiter.api.Assertions.assertNull;
11  
12  import org.junit.jupiter.api.Test;
13  import org.springframework.beans.factory.annotation.Autowired;
14  import org.tailormap.api.annotation.PostgresIntegrationTest;
15  import org.tailormap.api.persistence.Application;
16  
17  @PostgresIntegrationTest
18  class ApplicationRepositoryIntegrationTest {
19    @Autowired
20    private ApplicationRepository applicationRepository;
21  
22    @Test
23    void should_find_application_by_name() {
24      Application a = applicationRepository.findByName("default");
25      assertNotNull(a);
26      assertEquals("default", a.getName(), "application name is incorrect");
27    }
28  
29    @Test
30    void should_not_find_application_by_nonexistent_name() {
31      Application a = applicationRepository.findByName("does-not-exist");
32      assertNull(a);
33    }
34  }
