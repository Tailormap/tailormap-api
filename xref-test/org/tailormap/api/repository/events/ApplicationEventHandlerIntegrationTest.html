37  @Order(Integer.MAX_VALUE)
38  @AutoConfigureMockMvc
39  @PostgresIntegrationTest
40  @TestInstance(TestInstance.Lifecycle.PER_CLASS)
41  class ApplicationEventHandlerIntegrationTest {
42  
43    @Autowired
44    private ApplicationEventHandler applicationEventHandler;
45  
46    @Autowired
47    private ApplicationRepository applicationRepository;
48  
49    @Autowired
50    private PrometheusService prometheusService;
51  
52    @Autowired
53    private WebApplicationContext context;
54  
55    private MockMvc mockMvc;
56  
57    @Value("${tailormap-api.admin.base-path}")
58    private String adminBasePath;
59  
60    @BeforeAll
61    void initialize() {
62      mockMvc = MockMvcBuilders.webAppContextSetup(context).build();
63    }
64  
65    @Test
66    @WithMockUser(
67        username = "tm-admin",
68        authorities = {Group.ADMIN})
69    @Order(2)
70    void testApplicationDeleted() throws Exception {
71      Application application = applicationRepository.findByName("default");
72      assertEquals(1, application.getId(), "default application should have id 1");
73      assumeTrue(prometheusService.isPrometheusAvailable(), "Prometheus must be available for this test");
74  
75      // trigger the delete event handler
76      applicationEventHandler.afterDeleteApplicationEventHandler(application);
77  
78      mockMvc.perform(get(adminBasePath + "/graph/applications"))
79          .andExpect(status().isOk())
80          .andDo(print())
81          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
82          .andExpect(jsonPath("$.applications").isArray())
83          .andExpect(jsonPath("$.applications[0].appId").value("5"));
84    }
85  
86    @Test
87    @WithMockUser(
88        username = "tm-admin",
89        authorities = {Group.ADMIN})
90    @Order(1)
91    void testAllAppLayersRemoved() throws Exception {
92      Application application = applicationRepository.findById(1L).orElseThrow();
93      assumeTrue("default".equals(application.getName()), "Application with id 1 should be 'default'");
94      assumeTrue(prometheusService.isPrometheusAvailable(), "Prometheus must be available for this test");
95      assertTrue(
96          application.getAllOldAppTreeLayerNode().findAny().isEmpty(), "No old node IDs should exist initially");
97  
98      // create a new app content (only the part we need for this test)
99      final AppContent appContentCopy = new AppContent();
100     // Remove all app layers and update the application
101     appContentCopy.setLayerNodes(Collections.emptyList());
102     appContentCopy.setBaseLayerNodes(Collections.emptyList());
103     application.setContentRoot(appContentCopy);
104 
105     assertTrue(application.getAllOldAppTreeLayerNode().findAny().isPresent(), "Old node IDs should now exist");
106 
107     // trigger the delete event handler
108     applicationEventHandler.beforeSaveApplicationEventHandler(application);
109 
110     mockMvc.perform(get(adminBasePath + "/graph/applayers/1"))
111         .andExpect(status().isOk())
112         .andDo(print())
113         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
114         .andExpect(jsonPath("$.applicationLayers").isArray())
115         .andExpect(jsonPath("$.applicationLayers").isEmpty());
116   }
117 }
