35  @Transactional
36  @Order(Integer.MAX_VALUE)
37  @AutoConfigureMockMvc
38  @PostgresIntegrationTest
39  @TestInstance(TestInstance.Lifecycle.PER_CLASS)
40  class ApplicationEventHandlerIntegrationTest {
41    @Autowired
42    private ApplicationEventHandler applicationEventHandler;
43  
44    @Autowired
45    private ApplicationRepository applicationRepository;
46  
47    @Autowired
48    private PrometheusService prometheusService;
49  
50    @Autowired
51    private WebApplicationContext context;
52  
53    private MockMvc mockMvc;
54  
55    @Value("${tailormap-api.admin.base-path}")
56    private String adminBasePath;
57  
58    @BeforeAll
59    void initialize() {
60      mockMvc = MockMvcBuilders.webAppContextSetup(context).build();
61    }
62  
63    @Test
64    @WithMockUser(
65        username = "tm-admin",
66        authorities = {Group.ADMIN})
67    void test() throws Exception {
68      Application application = applicationRepository.findByName("default");
69      assertEquals(1, application.getId(), "default application should have id 1");
70      assertTrue(prometheusService.isPrometheusAvailable());
71  
72      // trigger the delete event handler
73      applicationEventHandler.afterDeleteApplicationEventHandler(application);
74  
75      mockMvc.perform(get(adminBasePath + "/graph/applications"))
76          .andExpect(status().isOk())
77          .andDo(print())
78          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
79          .andExpect(jsonPath("$.applications").isArray())
80          .andExpect(jsonPath("$.applications[0].appId").value("5"));
81    }
82  }
