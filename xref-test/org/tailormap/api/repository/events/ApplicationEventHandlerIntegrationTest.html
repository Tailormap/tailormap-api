36  @Order(Integer.MAX_VALUE)
37  @AutoConfigureMockMvc
38  @PostgresIntegrationTest
39  @TestInstance(TestInstance.Lifecycle.PER_CLASS)
40  class ApplicationEventHandlerIntegrationTest {
41  
42    @Autowired
43    private ApplicationEventHandler applicationEventHandler;
44  
45    @Autowired
46    private ApplicationRepository applicationRepository;
47  
48    @Autowired
49    private PrometheusService prometheusService;
50  
51    @Autowired
52    private WebApplicationContext context;
53  
54    private MockMvc mockMvc;
55  
56    @Value("${tailormap-api.admin.base-path}")
57    private String adminBasePath;
58  
59    @BeforeAll
60    void initialize() {
61      mockMvc = MockMvcBuilders.webAppContextSetup(context).build();
62    }
63  
64    @Test
65    @WithMockUser(
66        username = "tm-admin",
67        authorities = {Group.ADMIN})
68    @Order(2)
69    void testApplicationDeleted() throws Exception {
70      Application application = applicationRepository.findByName("default");
71      assertEquals(1, application.getId(), "default application should have id 1");
72      assumeTrue(prometheusService.isPrometheusAvailable(), "Prometheus must be available for this test");
73  
74      // trigger the delete event handler
75      applicationEventHandler.afterDeleteApplicationEventHandler(application);
76  
77      mockMvc.perform(get(adminBasePath + "/graph/applications"))
78          .andExpect(status().isOk())
79          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
80          .andExpect(jsonPath("$.applications").isArray())
81          .andExpect(jsonPath("$.applications[0].appId").value("5"));
82    }
83  
84    @Test
85    @WithMockUser(
86        username = "tm-admin",
87        authorities = {Group.ADMIN})
88    @Order(1)
89    void testAllAppLayersRemoved() throws Exception {
90      Application application = applicationRepository.findById(1L).orElseThrow();
91      assumeTrue("default".equals(application.getName()), "Application with id 1 should be 'default'");
92      assumeTrue(prometheusService.isPrometheusAvailable(), "Prometheus must be available for this test");
93      assertTrue(
94          application.getAllOldAppTreeLayerNode().findAny().isEmpty(), "No old node IDs should exist initially");
95  
96      // create a new app content (only the part we need for this test)
97      final AppContent appContentCopy = new AppContent();
98      // Remove all app layers and update the application
99      appContentCopy.setLayerNodes(Collections.emptyList());
100     appContentCopy.setBaseLayerNodes(Collections.emptyList());
101     application.setContentRoot(appContentCopy);
102 
103     assertTrue(application.getAllOldAppTreeLayerNode().findAny().isPresent(), "Old node IDs should now exist");
104 
105     // trigger the delete event handler
106     applicationEventHandler.beforeSaveApplicationEventHandler(application);
107 
108     mockMvc.perform(get(adminBasePath + "/graph/applayers/1"))
109         .andExpect(status().isOk())
110         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
111         .andExpect(jsonPath("$.applicationLayers").isArray())
112         .andExpect(jsonPath("$.applicationLayers").isEmpty());
113   }
114 }
