6   package org.tailormap.api.repository.events;
7   
8   import static org.hamcrest.MatcherAssert.assertThat;
9   import static org.hamcrest.Matchers.containsString;
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  import static org.junit.jupiter.api.Assumptions.assumingThat;
12  
13  import org.junit.jupiter.api.Test;
14  import org.junitpioneer.jupiter.Issue;
15  import org.quartz.SchedulerException;
16  import org.springframework.beans.factory.annotation.Autowired;
17  import org.tailormap.api.annotation.PostgresIntegrationTest;
18  import org.tailormap.api.persistence.SearchIndex;
19  import org.tailormap.api.repository.SearchIndexRepository;
20  
21  @PostgresIntegrationTest
22  class SearchIndexEventHandlerIntegrationTest {
23    @Autowired SearchIndexEventHandler searchIndexEventHandler;
24    @Autowired SearchIndexRepository searchIndexRepository;
25  
26    /**
27     * Test that a {@code SearchIndex} with a scheduled task that already exists cannot be saved with
28     * a new task through Spring Data REST.
29     */
30    @Test
31    @Issue("HTM-1258")
32    @SuppressWarnings("PMD.UnitTestShouldIncludeAssert")
33    void testBeforeSaveSearchIndexEventHandler() {
34      SearchIndex existingSearchIndexWithSchedule = searchIndexRepository.findById(1L).orElse(null);
35  
36      assumingThat(
37          null != existingSearchIndexWithSchedule
38              && null != existingSearchIndexWithSchedule.getSchedule()
39              && null != existingSearchIndexWithSchedule.getSchedule().getUuid(),
40          () -> {
41            final Exception actual =
42                assertThrows(
43                    SchedulerException.class,
44                    () -> {
45                      existingSearchIndexWithSchedule.getSchedule().setUuid(null);
46                      searchIndexEventHandler.beforeSaveSearchIndexEventHandler(
47                          existingSearchIndexWithSchedule);
48                    },
49                    "Test did not invoke expected exception");
50  
51            assertThat(actual.getMessage(), containsString("scheduled task already exists"));
52          });
53    }
54  }
