6   package org.tailormap.api.repository.events;
7   
8   import static org.hamcrest.MatcherAssert.assertThat;
9   import static org.hamcrest.Matchers.containsString;
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  import static org.junit.jupiter.api.Assumptions.assumingThat;
12  
13  import org.junit.jupiter.api.Test;
14  import org.junitpioneer.jupiter.Issue;
15  import org.quartz.SchedulerException;
16  import org.springframework.beans.factory.annotation.Autowired;
17  import org.tailormap.api.annotation.PostgresIntegrationTest;
18  import org.tailormap.api.persistence.SearchIndex;
19  import org.tailormap.api.repository.SearchIndexRepository;
20  
21  @PostgresIntegrationTest
22  class SearchIndexEventHandlerIntegrationTest {
23    @Autowired
24    SearchIndexEventHandler searchIndexEventHandler;
25  
26    @Autowired
27    SearchIndexRepository searchIndexRepository;
28  
29    /**
30     * Test that a {@code SearchIndex} with a scheduled task that already exists cannot be saved with a new task through
31     * Spring Data REST.
32     */
33    @Test
34    @Issue("HTM-1258")
35    @SuppressWarnings("PMD.UnitTestShouldIncludeAssert")
36    void testBeforeSaveSearchIndexEventHandler() {
37      SearchIndex existingSearchIndexWithSchedule =
38          searchIndexRepository.findById(1L).orElse(null);
39  
40      assumingThat(
41          null != existingSearchIndexWithSchedule
42              && null != existingSearchIndexWithSchedule.getSchedule()
43              && null != existingSearchIndexWithSchedule.getSchedule().getUuid(),
44          () -> {
45            final Exception actual = assertThrows(
46                SchedulerException.class,
47                () -> {
48                  existingSearchIndexWithSchedule.getSchedule().setUuid(null);
49                  searchIndexEventHandler.beforeSaveSearchIndexEventHandler(
50                      existingSearchIndexWithSchedule);
51                },
52                "Test did not invoke expected exception");
53  
54            assertThat(actual.getMessage(), containsString("scheduled task already exists"));
55          });
56    }
57  }
