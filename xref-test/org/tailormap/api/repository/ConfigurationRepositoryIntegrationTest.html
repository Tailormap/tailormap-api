6   package org.tailormap.api.repository;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertNotNull;
10  import static org.junit.jupiter.api.Assertions.assertNull;
11  import static org.tailormap.api.persistence.Configuration.DEFAULT_APP;
12  
13  import org.junit.jupiter.api.MethodOrderer.OrderAnnotation;
14  import org.junit.jupiter.api.Order;
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.TestMethodOrder;
17  import org.springframework.beans.factory.annotation.Autowired;
18  import org.springframework.transaction.annotation.Transactional;
19  import org.tailormap.api.annotation.PostgresIntegrationTest;
20  import org.tailormap.api.persistence.Configuration;
21  import org.tailormap.api.security.InternalAdminAuthentication;
22  
23  @PostgresIntegrationTest
24  @TestMethodOrder(OrderAnnotation.class)
25  class ConfigurationRepositoryIntegrationTest {
26  
27    @Autowired
28    private ConfigurationRepository configurationRepository;
29  
30    @Test
31    @Order(1)
32    void it_should_findByConfigKeyDefaultApplication() {
33      final Configuration c = configurationRepository.findByKey(DEFAULT_APP).orElse(null);
34      assertNotNull(c);
35      assertEquals("default", c.getValue());
36    }
37  
38    @Test
39    @Transactional
40    @Order(2)
41    void it_should_not_find_value_after_deleting_key() {
42      InternalAdminAuthentication.setInSecurityContext();
43      try {
44        configurationRepository.deleteConfigurationByKey(DEFAULT_APP);
45        final Configuration c =
46            configurationRepository.findByKey(DEFAULT_APP).orElse(null);
47        assertNull(c);
48      } finally {
49        InternalAdminAuthentication.clearSecurityContextAuthentication();
50      }
51    }
52  }
