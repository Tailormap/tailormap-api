6   package org.tailormap.api.repository;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertTrue;
10  
11  import java.time.Instant;
12  import java.time.ZoneOffset;
13  import java.util.Optional;
14  import org.junit.jupiter.api.Test;
15  import org.springframework.beans.factory.annotation.Autowired;
16  import org.springframework.transaction.annotation.Transactional;
17  import org.tailormap.api.annotation.PostgresIntegrationTest;
18  import org.tailormap.api.persistence.Upload;
19  
20  @PostgresIntegrationTest
21  class UploadRepositoryIntegrationTest {
22  
23    @Autowired
24    private UploadRepository uploadRepository;
25  
26    @Test
27    @Transactional(label = "inserts and removes an upload")
28    void should_find_latest_upload_by_category() {
29      Upload testUpload =
30          uploadRepository.findByCategory(Upload.CATEGORY_DRAWING_STYLE).get(0);
31      // create a copy of the upload with a different filename and younger last-modified time
32      Upload copiedUpload = new Upload()
33          .setCategory(testUpload.getCategory())
34          .setFilename(testUpload.getFilename())
35          .setContent(testUpload.getContent())
36          .setMimeType(testUpload.getMimeType())
37          .setLastModified(Instant.now().atOffset(ZoneOffset.UTC))
38          .setFilename("latest-drawing-style.json");
39      uploadRepository.save(copiedUpload);
40  
41      Optional<Upload> upload =
42          uploadRepository.findFirstWithContentByCategoryOrderByLastModifiedDesc(Upload.CATEGORY_DRAWING_STYLE);
43      assertTrue(upload.isPresent(), "A latest upload should be present for the drawing style category");
44      assertEquals(
45          2,
46          uploadRepository.findByCategory(Upload.CATEGORY_DRAWING_STYLE).size(),
47          "There should be two uploads for the drawing style category after adding the latest upload");
48  
49      final Upload actual = upload.orElseThrow();
50      assertEquals(Upload.CATEGORY_DRAWING_STYLE, actual.getCategory());
51      assertEquals(
52          "application/json",
53          actual.getMimeType(),
54          "Mime type should be application/json for drawing style uploads");
55      assertEquals(
56          "latest-drawing-style.json",
57          actual.getFilename(),
58          "Name should be latest-drawing-style.json for the latest drawing style upload");
59    }
60  }
