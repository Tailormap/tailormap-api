7   package org.tailormap.api.persistence.helper;
8   
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  import static org.tailormap.api.persistence.helper.TMFeatureTypeHelper.getConfiguredAttributes;
11  
12  import java.util.ArrayList;
13  import java.util.List;
14  import java.util.Map;
15  import java.util.function.Function;
16  import org.apache.commons.lang3.tuple.Pair;
17  import org.junit.jupiter.api.Test;
18  import org.tailormap.api.persistence.TMFeatureType;
19  import org.tailormap.api.persistence.json.AppLayerSettings;
20  import org.tailormap.api.persistence.json.AttributeSettings;
21  import org.tailormap.api.persistence.json.FeatureTypeSettings;
22  import org.tailormap.api.persistence.json.TMAttributeDescriptor;
23  
24  class TMFeatureTypeHelperTest {
25  
26    @Test
27    void testGetConfiguredAttributes() {
28      TMFeatureType ft = new TMFeatureType();
29      final Function<String, TMAttributeDescriptor> att =
30          (name) -> new TMAttributeDescriptor().name(name).comment("comment for " + name);
31      ft.setAttributes(List.of(
32          att.apply("a"),
33          att.apply("b"),
34          att.apply("c"),
35          att.apply("d"),
36          att.apply("e"),
37          att.apply("f"),
38          att.apply("m"),
39          att.apply("n"),
40          att.apply("o"),
41          att.apply("q")));
42  
43      ft.setSettings(new FeatureTypeSettings()
44          .putAttributeSettingsItem("d", new AttributeSettings().title("d title"))
45          .addHideAttributesItem("b")
46          .addHideAttributesItem("q")
47          .addHideAttributesItem("o")
48          .addAttributeOrderItem("c")
49          .addAttributeOrderItem("d")
50          .addAttributeOrderItem("e")
51          .addAttributeOrderItem("o")
52          .addAttributeOrderItem("a"));
53  
54      AppLayerSettings appLayerSettings = new AppLayerSettings();
55      appLayerSettings.setHideAttributes(List.of("e", "f"));
56  
57      Map<String, Pair<TMAttributeDescriptor, AttributeSettings>> configuredAttributes =
58          getConfiguredAttributes(ft, appLayerSettings);
59      // Compare using Lists to explicitly check the ordering
60      assertEquals(
61          List.of(
62              "c",
63              "d",
64              // "e" and "f" are hidden in app layer settings
65              // "o" is hidden in feature type settings
66              "a",
67              // after ordered attributes, remaining attributes in original feature type order
68              "m",
69              "n"),
70          new ArrayList<>(configuredAttributes.keySet()));
71  
72      assertEquals("d title", configuredAttributes.get("d").getRight().getTitle());
73  
74      assertEquals("comment for m", configuredAttributes.get("m").getLeft().getComment());
75    }
76  }
