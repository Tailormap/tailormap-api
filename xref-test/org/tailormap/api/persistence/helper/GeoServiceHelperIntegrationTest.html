10  import org.junit.jupiter.api.Test;
11  import org.springframework.beans.factory.annotation.Autowired;
12  import org.tailormap.api.annotation.PostgresIntegrationTest;
13  import org.tailormap.api.persistence.GeoService;
14  import org.tailormap.api.persistence.json.GeoServiceLayer;
15  import org.tailormap.api.repository.GeoServiceRepository;

17  @PostgresIntegrationTest
18  class GeoServiceHelperIntegrationTest {
19  
20    @Autowired private GeoServiceRepository geoServiceRepository;
21  
22    @Test
23    void testGetLayerLegendUrlFromStyles() {
24      GeoService service = geoServiceRepository.findById("demo").stream().findFirst().orElseThrow();
25      GeoServiceLayer layer = service.findLayer("geomorfologie");
26      String expected =
27          "https://demo.tailormap.com/geoserver/geodata/ows?service=WMS&version=1.3.0&request=GetLegendGraphic&format=image%2Fpng&width=20&height=20&layer=geomorfologie";
28      String actual = GeoServiceHelper.getLayerLegendUrlFromStyles(service, layer).toString();
29      assertEquals(expected, actual, "Expected and actual legend url are not equal");
30    }
