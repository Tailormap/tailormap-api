8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   
10  import org.junit.jupiter.api.Test;
11  import org.springframework.beans.factory.annotation.Autowired;
12  import org.tailormap.api.annotation.PostgresIntegrationTest;
13  import org.tailormap.api.persistence.GeoService;
14  import org.tailormap.api.persistence.json.GeoServiceLayer;
15  import org.tailormap.api.repository.GeoServiceRepository;
16  
17  @PostgresIntegrationTest
18  class GeoServiceHelperIntegrationTest {
19  
20    @Autowired
21    private GeoServiceRepository geoServiceRepository;
22  
23    @Test
24    void testGetLayerLegendUrlFromStyles() {
25      GeoService service =
26          geoServiceRepository.findById("demo").stream().findFirst().orElseThrow();
27      GeoServiceLayer layer = service.findLayer("geomorfologie");
28      String expected =
29          "https://demo.tailormap.com/geoserver/geodata/ows?service=WMS&version=1.3.0&request=GetLegendGraphic&format=image%2Fpng&width=20&height=20&layer=geomorfologie";
30      String actual =
31          GeoServiceHelper.getLayerLegendUrlFromStyles(service, layer).toString();
32      assertEquals(expected, actual, "Expected and actual legend url are not equal");
33    }
34  }
