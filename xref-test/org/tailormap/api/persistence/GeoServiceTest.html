9   import static org.junit.jupiter.params.provider.Arguments.arguments;

11  import java.util.stream.Stream;

17  class GeoServiceTest {
18  
19    private GeoService geoService;
20  
21    static Stream<Arguments> urlProvider() {
22      return Stream.of(
23          arguments("https://geoserver.nl/wms", "https://geoserver.nl/wms"),
24          arguments(
25              "https://geoserver.nl/wms?encoded=aapjes%20%3D%20!%263%3D1",
26              "https://geoserver.nl/wms?encoded=aapjes%20%3D%20!%263%3D1"),
27          arguments("https://geoserver.nl/wms?", "https://geoserver.nl/wms"),
28          arguments("https://geoserver.nl/wms?request=getcapabilities", "https://geoserver.nl/wms"),
29          arguments(
30              "https://geoserver.nl/wms?request=getcapabilities&REQUEST=getMap", "https://geoserver.nl/wms"),
31          arguments("https://geoserver.nl/wmts?REQUEST=getcapabilities", "https://geoserver.nl/wmts"),
32          arguments(
33              "https://geoserver.nl/wms?request=getcapabilities&version=1",
34              "https://geoserver.nl/wms?version=1"),
35          arguments("https://geoserver.nl/wms?&version=1", "https://geoserver.nl/wms?version=1"),
36          arguments("https://geoserver.nl/wms?&version=1,2", "https://geoserver.nl/wms?version=1,2"),
37          arguments(
38              "https://service.pdok.nl/hwh/luchtfotorgb/wmts/v1_0",
39              "https://service.pdok.nl/hwh/luchtfotorgb/wmts/v1_0"),
40          arguments(
41              "https://basemap.at/wmts/1.0.0/WMTSCapabilities.xml",
42              "https://basemap.at/wmts/1.0.0/WMTSCapabilities.xml"),
43          arguments(
44              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS",
45              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS"),

48              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS"));
49    }
50  
51    @BeforeEach
52    void setGeoService() {
53      geoService = new GeoService();
54    }
55  
56    @ParameterizedTest(name = "#{index}: should sanitise url: {0}")
57    @MethodSource("urlProvider")
58    void testSetUrl(final String input, final String expected) {
59      geoService.setUrl(input);
60      assertEquals(expected, geoService.getUrl(), () -> input + "not sanitised properly, expected " + expected);
61    }
62  }
