37  /** Basic test to verify session serialization/deserialization works with configured ConversionService. */
38  @JsonTest
39  class JdbcSessionConfigurationTest {
40  
41    private ConversionService conversionService;
42  
43    @Autowired
44    private ObjectMapper objectMapper;
45  
46    @BeforeEach
47    void setUp() {
48      JdbcSessionConfiguration cfg = new JdbcSessionConfiguration();
49      cfg.setBeanClassLoader(this.getClass().getClassLoader());
50      conversionService = cfg.springSessionConversionService(objectMapper);
51    }
52  
53    @Test
54    void should_serialize_and_deserialize_security_context_with_tm_userdetails_round_trip() {
55      Set<Group> groups = Set.of(new Group()
56          .setName("admin")
57          .setAdditionalProperties(List.of(new AdminAdditionalProperty("grouptest", true, "group"))));
58      User user = new User()
59          .setUsername("tm-admin")
60          .setAdditionalProperties(List.of(new AdminAdditionalProperty("usertest", true, "user")))
61          .setOrganisation("Tailormap")
62          .setGroups(groups);
63      TailormapUserDetailsImpl userDetails = new TailormapUserDetailsImpl(user, null);
64  
65      Authentication auth = new UsernamePasswordAuthenticationToken(
66          userDetails, null, List.of(new SimpleGrantedAuthority("admin")));
67      SecurityContextImpl ctx = new SecurityContextImpl(auth);
68  
69      // serialize
70      byte[] bytes = conversionService.convert(ctx, byte[].class);
71      assertNotNull(bytes);
72  
73      // deserialize
74      Object back = conversionService.convert(bytes, Object.class);
75      assertNotNull(back);
76      assertEquals(SecurityContextImpl.class, back.getClass());
77      SecurityContextImpl ctxBack = (SecurityContextImpl) back;
78      assertNotNull(ctxBack.getAuthentication());
79      assertEquals(
80          UsernamePasswordAuthenticationToken.class,
81          ctxBack.getAuthentication().getClass());
82  
83      TailormapUserDetailsImpl userDetailsBack =
84          (TailormapUserDetailsImpl) ctxBack.getAuthentication().getPrincipal();
85      assertEquals(userDetails.getAuthorities(), userDetailsBack.getAuthorities());
86      assertEquals(userDetails.getUsername(), userDetailsBack.getUsername());
87      assertEquals(userDetails.getOrganisation(), userDetailsBack.getOrganisation());
88      Collection<String> in = userDetails.getAdditionalProperties().stream()
89          .map(Objects::toString)
90          .collect(Collectors.toList());
91      Collection<String> out = userDetailsBack.getAdditionalProperties().stream()
92          .map(Objects::toString)
93          .collect(Collectors.toList());
94      assertEquals(in, out);
95      Set<String> ag = userDetails.getAdditionalGroupProperties().stream()
96          .map(Object::toString)
97          .collect(Collectors.toSet());
98      Set<String> bg = userDetailsBack.getAdditionalGroupProperties().stream()
99          .map(Object::toString)
100         .collect(Collectors.toSet());
101     assertEquals(ag, bg);
102     assertEquals(userDetails.isAccountNonExpired(), userDetailsBack.isAccountNonExpired());
103     assertEquals(userDetails.isEnabled(), userDetailsBack.isEnabled());
104     assertEquals(userDetails.isCredentialsNonExpired(), userDetailsBack.isCredentialsNonExpired());
105   }

107   @Test
108   void should_serialize_and_deserialize_security_context_with_oidc_userdetails_round_trip() {
109     TailormapOidcUser userDetails = new TailormapOidcUser(
110         List.of(new SimpleGrantedAuthority("admin")),
111         OidcIdToken.withTokenValue("test")
112             .claim(StandardClaimNames.NAME, "tm-admin")
113             .build(),
114         OidcUserInfo.builder().name("tm-admin").build(),
115         "name",
116         "oidc-registration",
117         List.of(new TailormapAdditionalProperty("grouptest", true, "group")));
118 
119     assertEquals("tm-admin", userDetails.getUsername());
120 
121     Authentication auth =
122         new OAuth2AuthenticationToken(userDetails, List.of(new SimpleGrantedAuthority("admin")), "client-id");
123     SecurityContextImpl ctx = new SecurityContextImpl(auth);
124 
125     // serialize
126     byte[] bytes = conversionService.convert(ctx, byte[].class);
127     assertNotNull(bytes);
128 
129     // deserialize
130     Object back = conversionService.convert(bytes, Object.class);
131     assertNotNull(back);
132     assertEquals(SecurityContextImpl.class, back.getClass());
133     SecurityContextImpl ctxBack = (SecurityContextImpl) back;
134     assertNotNull(ctxBack.getAuthentication());
135     assertEquals(auth.getClass(), ctxBack.getAuthentication().getClass());
136 
137     TailormapOidcUser userDetailsBack =
138         (TailormapOidcUser) ctxBack.getAuthentication().getPrincipal();
139     assertEquals(userDetails.getAuthorities(), userDetailsBack.getAuthorities());
140     assertEquals(userDetails.getUsername(), userDetailsBack.getUsername());
141     assertEquals(userDetails.getOrganisation(), userDetailsBack.getOrganisation());
142     assertEquals(userDetails.getAdditionalProperties(), userDetailsBack.getAdditionalProperties());
143     assertEquals(
144         userDetails.getAdditionalGroupProperties().stream()
145             .map(Object::toString)
146             .collect(Collectors.toSet()),
147         userDetailsBack.getAdditionalGroupProperties().stream()
148             .map(Object::toString)
149             .collect(Collectors.toSet()));
150     assertEquals(userDetails.isAccountNonExpired(), userDetailsBack.isAccountNonExpired());
151     assertEquals(userDetails.isEnabled(), userDetailsBack.isEnabled());
152     assertEquals(userDetails.isCredentialsNonExpired(), userDetailsBack.isCredentialsNonExpired());
153   }
