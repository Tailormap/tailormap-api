45    private static final String SELECT_ATTRIBUTE_BYTES_SQL =
46        """
47  select jsonb_pretty(attribute_bytes) as attribute from spring_session_attributes
48  where attribute_name = 'SPRING_SECURITY_CONTEXT'
49  and session_primary_id = (select primary_id from spring_session where session_id = ?)
50  """;
51  
52    /** this is in fact a org.springframework.session.jdbc.JdbcIndexedSessionRepository. */
53    @Autowired
54    @SuppressWarnings("rawtypes")
55    private FindByIndexNameSessionRepository sessionRepository;
56  
57    @Autowired
58    private GroupRepository groupRepository;
59  
60    @Autowired
61    private DataSource dataSource;
62  
63    /** this is in fact a org.springframework.session.jdbc.JdbcIndexedSessionRepository.JdbcSession. */
64    private Session session;
65  
66    private String sessionId;
67    private JdbcTemplate jdbcTemplate;
68  
69    private String getAttribute(String sessionId) {
70      return jdbcTemplate.queryForObject(
71          SELECT_ATTRIBUTE_BYTES_SQL, (rs, rowNum) -> rs.getString("attribute"), sessionId);
72    }
73  
74    @BeforeEach
75    void setUp() {
76      session = sessionRepository.createSession();
77      sessionId = session.getId();
78      jdbcTemplate = new JdbcTemplate(dataSource);
79    }
80  
81    @AfterEach
82    void tearDown() {
83      jdbcTemplate.update("DELETE FROM spring_session WHERE session_id=?", sessionId);
84    }

86    @Test
87    void should_create_session_attribute_using_postgresql_convert_from_and_jsonb() {
88      User user = new User()
89          .setUsername("test-user")
90          .setAdditionalProperties(List.of(new AdminAdditionalProperty("userkey", true, "uservalue")))
91          .setGroups(Set.of(new Group().setName("test-bar")));
92      // Note: group additional properties are normally loaded from the database via GroupRepository
93      TailormapUserDetailsImpl userDetails = new TailormapUserDetailsImpl(user, groupRepository);
94  
95      Authentication auth =
96          new UsernamePasswordAuthenticationToken(userDetails, null, List.of(new SimpleGrantedAuthority("USER")));
97      SecurityContextImpl ctx = new SecurityContextImpl(auth);
98  
99      session.setAttribute("SPRING_SECURITY_CONTEXT", ctx);
100     sessionRepository.save(session);
101 
102     String jsonbContent = getAttribute(session.getId());
103     assertNotNull(jsonbContent, "JSONB content should not be null");
104     assertTrue(jsonbContent.contains("test-user"), "JSON should contain username");
105     assertTrue(
106         jsonbContent.contains("UsernamePasswordAuthenticationToken"),
107         "JSON should contain authentication type");
108 
109     Session retrievedSession = sessionRepository.findById(sessionId);
110     assertNotNull(retrievedSession, "Session should be retrievable");
111 
112     SecurityContextImpl retrievedCtx = retrievedSession.getAttribute("SPRING_SECURITY_CONTEXT");
113     assertNotNull(retrievedCtx, "Security context should be retrievable");
114     assertNotNull(retrievedCtx.getAuthentication(), "Authentication should be present");
115 
116     TailormapUserDetailsImpl retrievedUserDetails =
117         (TailormapUserDetailsImpl) retrievedCtx.getAuthentication().getPrincipal();
118     assertEquals("test-user", retrievedUserDetails.getUsername(), "Username should match");
119     assertEquals(1, retrievedUserDetails.getAdditionalProperties().size(), "Should have one additional property");
120     assertEquals(2, retrievedUserDetails.getAdditionalGroupProperties().size(), "Should have two group properties");
121   }

123   @Test
124   void should_update_session_attribute_using_postgresql_encode_and_jsonb() {
125     User user = new User().setUsername("initial-user");
126     TailormapUserDetailsImpl userDetails = new TailormapUserDetailsImpl(user, null);
127     Authentication auth =
128         new UsernamePasswordAuthenticationToken(userDetails, null, List.of(new SimpleGrantedAuthority("USER")));
129     SecurityContextImpl ctx = new SecurityContextImpl(auth);
130 
131     session.setAttribute("SPRING_SECURITY_CONTEXT", ctx);
132     sessionRepository.save(session);
133 
134     Session retrievedSession = sessionRepository.findById(sessionId);
135     assertNotNull(retrievedSession, "Session should exist");
136 
137     User updatedUser = new User()
138         .setUsername("updated-user")
139         .setAdditionalProperties(List.of(new AdminAdditionalProperty("newkey", false, "newvalue")));
140     TailormapUserDetailsImpl updatedUserDetails = new TailormapUserDetailsImpl(updatedUser, null);
141     Authentication updatedAuth = new UsernamePasswordAuthenticationToken(
142         updatedUserDetails, null, List.of(new SimpleGrantedAuthority("ADMIN")));
143     SecurityContextImpl updatedCtx = new SecurityContextImpl(updatedAuth);
144 
145     retrievedSession.setAttribute("SPRING_SECURITY_CONTEXT", updatedCtx);
146     sessionRepository.save(retrievedSession);
147     String jsonbContent = getAttribute(sessionId);
148 
149     assertNotNull(jsonbContent, "Updated JSONB content should not be null");
150     assertTrue(jsonbContent.contains("updated-user"), "JSON should contain updated username");
151     assertTrue(jsonbContent.contains("ADMIN"), "JSON should contain updated authority");
152 
153     Session finalSession = sessionRepository.findById(sessionId);
154     assertNotNull(finalSession, "Updated session should be retrievable");
155 
156     SecurityContextImpl finalCtx = finalSession.getAttribute("SPRING_SECURITY_CONTEXT");
157     assertNotNull(finalCtx, "Updated security context should be retrievable");
158 
159     TailormapUserDetailsImpl finalUserDetails =
160         (TailormapUserDetailsImpl) finalCtx.getAuthentication().getPrincipal();
161     assertEquals("updated-user", finalUserDetails.getUsername(), "Username should be updated");
162     assertEquals(1, finalUserDetails.getAdditionalProperties().size(), "Should have updated properties");
163     assertEquals(
164         "newkey",
165         finalUserDetails.getAdditionalProperties().stream()
166             .findFirst()
167             .orElseThrow()
168             .key(),
169         "Property key should match");
170   }

172   @Test
173   void should_handle_multiple_attributes_in_same_session() {
174     User user = new User().setUsername("multi-attr-user");
175     TailormapUserDetailsImpl userDetails = new TailormapUserDetailsImpl(user, null);
176     Authentication auth =
177         new UsernamePasswordAuthenticationToken(userDetails, null, List.of(new SimpleGrantedAuthority("USER")));
178     SecurityContextImpl ctx = new SecurityContextImpl(auth);
179     session.setAttribute("SPRING_SECURITY_CONTEXT", ctx);
180 
181     session.setAttribute("CUSTOM_DATA", "test-value");
182     session.setAttribute("LOGIN_TIME", Instant.now().toEpochMilli());
183     sessionRepository.save(session);
184 
185     Integer attributeCount = jdbcTemplate.queryForObject(
186         """
187 select count(*) from spring_session_attributes
188 where session_primary_id = (select primary_id from spring_session where session_id = ?)
189 """,
190         Integer.class,
191         sessionId);
192 
193     assertEquals(3, attributeCount, "Should have three attributes");
194 
195     Session retrievedSession = sessionRepository.findById(sessionId);
196     assertNotNull(retrievedSession, "Session should be retrievable");
197 
198     SecurityContextImpl retrievedCtx = retrievedSession.getAttribute("SPRING_SECURITY_CONTEXT");
199     assertNotNull(retrievedCtx, "Security context should be retrievable");
200 
201     String customData = retrievedSession.getAttribute("CUSTOM_DATA");
202     assertEquals("test-value", customData, "Custom data should match");
203 
204     Long loginTime = retrievedSession.getAttribute("LOGIN_TIME");
205     assertNotNull(loginTime, "Login time should be retrievable");
206   }

208   @Test
209   void should_handle_jsonb_queries_with_special_characters() {
210 
211     User user = new User()
212         .setUsername("user-with-\"quotes\"")
213         .setAdditionalProperties(List.of(new AdminAdditionalProperty(
214             "key with spaces", true, "value with 'quotes' and \"double quotes\"")));
215     TailormapUserDetailsImpl userDetails = new TailormapUserDetailsImpl(user, null);
216     Authentication auth =
217         new UsernamePasswordAuthenticationToken(userDetails, null, List.of(new SimpleGrantedAuthority("USER")));
218     SecurityContextImpl ctx = new SecurityContextImpl(auth);
219 
220     session.setAttribute("SPRING_SECURITY_CONTEXT", ctx);
221     sessionRepository.save(session);
222 
223     Session retrievedSession = sessionRepository.findById(sessionId);
224     assertNotNull(retrievedSession, "Session should be retrievable");
225 
226     SecurityContextImpl retrievedCtx = retrievedSession.getAttribute("SPRING_SECURITY_CONTEXT");
227     assertNotNull(retrievedCtx, "Security context should be retrievable");
228 
229     TailormapUserDetailsImpl retrievedUserDetails =
230         (TailormapUserDetailsImpl) retrievedCtx.getAuthentication().getPrincipal();
231     assertEquals(
232         "user-with-\"quotes\"",
233         retrievedUserDetails.getUsername(),
234         "Username with special characters should match");
235 
236     TailormapAdditionalProperty prop = retrievedUserDetails.getAdditionalProperties().stream()
237         .findFirst()
238         .orElseThrow();
239     assertEquals("key with spaces", prop.key(), "Key with spaces should match");
240     assertTrue(((String) prop.value()).contains("'quotes'"), "Value should contain single quotes");
241     assertTrue(((String) prop.value()).contains("\"double quotes\""), "Value should contain double quotes");
242   }

244   @Test
245   void should_verify_jsonb_storage_type_in_database() {
246     String dataTypeSql =
247         """
248 SELECT data_type
249 FROM information_schema.columns
250 WHERE table_name = 'spring_session_attributes'
251 AND column_name = 'attribute_bytes'
252 """;
253 
254     String dataType = jdbcTemplate.queryForObject(dataTypeSql, String.class);
255     assertNotNull(dataType, "Data type should not be null");
256     assertEquals("jsonb", dataType.toLowerCase(java.util.Locale.ROOT), "ATTRIBUTE_BYTES should be JSONB type");
257   }
