27  /**
28   * setup testdata from property file.
29   *
30   * @author mprins
31   */
32  public class StaticTestData {
33    public static final Properties testData = new Properties();
34  
35    static {
36      try {
37        testData.load(StaticTestData.class.getResourceAsStream("/StaticTestData.properties"));
38      } catch (IOException e) {
39        throw new RuntimeException(e);
40      }
41    }
42  
43    public static String get(String key) {
44      return testData.getProperty(key);
45    }
46  
47    public static String getResourceString(Resource resource) throws IOException {
48      return StreamUtils.copyToString(resource.getInputStream(), StandardCharsets.UTF_8);
49    }

51    /**
52     * Create a random feature source with the given number of features, within the given (EPSG:28992) bounds and date
53     * range.
54     *
55     * @param randomFeatureCount the number of random features to create
56     * @param minX the minimum X coordinate, the maximum X coordinate will be 10000 greater
57     * @param minY the minimum Y coordinate, the maximum Y coordinate will be 10000 greater
58     * @param minEpochMillis the minimum epoch milliseconds for the date range
59     * @param maxEpochMillis the maximum epoch milliseconds for the date range
60     * @return a SimpleFeatureSource containing the generated features
61     * @throws IOException if an I/O error occurs
62     * @throws SchemaException if there is an error creating the feature type
63     */
64    public static SimpleFeatureSource createRandomFeatureSource(
65        int randomFeatureCount, int minX, int minY, long minEpochMillis, long maxEpochMillis)
66        throws IOException, SchemaException {
67      final Random random = new Random();
68      SimpleFeatureType inputType = DataUtilities.createType(
69          "test",
70          "id:Integer,randomNumber:Long,date:Date,localdate:java.time.LocalDate,*location:Point:srid=28992");
71      MemoryDataStore dataStore = new MemoryDataStore(inputType);
72  
73      int[] xCoords = random.ints(randomFeatureCount, minX, minX + 10000).toArray();
74      int[] yCoords = random.ints(randomFeatureCount, minY, minY + 10000).toArray();
75      final SimpleFeatureBuilder fb = new SimpleFeatureBuilder(inputType);
76      final GeometryFactory gf = new GeometryFactory();
77      IntStream.range(0, randomFeatureCount).forEach(id -> {
78        Instant randomDate = Instant.ofEpochMilli(random.nextLong(minEpochMillis, maxEpochMillis));
79  
80        fb.set("id", id);
81        fb.set("randomNumber", random.nextLong(randomFeatureCount - 1));
82        fb.set("date", Date.from(randomDate));
83        fb.set("localdate", randomDate.atZone(ZoneOffset.UTC).toLocalDate());
84        fb.set("location", gf.createPoint(new Coordinate(xCoords[id], yCoords[id])));
85  
86        dataStore.addFeature(fb.buildFeature(String.valueOf(id)));
87      });
88  
89      return dataStore.getFeatureSource(inputType.getName());
90    }
