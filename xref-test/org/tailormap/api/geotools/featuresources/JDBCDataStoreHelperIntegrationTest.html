6   package org.tailormap.api.geotools.featuresources;

8   import static org.hamcrest.MatcherAssert.assertThat;
9   import static org.hamcrest.Matchers.startsWithIgnoringCase;
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  import static org.junit.jupiter.api.Assertions.assertNotNull;
12  import static org.junit.jupiter.api.Assertions.fail;
13  
14  import java.util.Locale;
15  import java.util.stream.Stream;
16  import org.junit.jupiter.api.BeforeEach;
17  import org.junit.jupiter.params.ParameterizedTest;
18  import org.junit.jupiter.params.provider.Arguments;
19  import org.junit.jupiter.params.provider.MethodSource;
20  import org.springframework.beans.factory.annotation.Autowired;
21  import org.tailormap.api.annotation.PostgresIntegrationTest;
22  import org.tailormap.api.persistence.TMFeatureType;
23  import org.tailormap.api.repository.FeatureSourceRepository;
24  import org.tailormap.api.repository.FeatureTypeRepository;
25  
26  @PostgresIntegrationTest
27  class JDBCDataStoreHelperIntegrationTest {
28    private JDBCDataStoreHelper helper;
29    private TMFeatureType featureType;
30  
31    @Autowired
32    private FeatureSourceRepository featureSourceRepository;
33  
34    @Autowired
35    private FeatureTypeRepository featureTypeRepository;
36  
37    @BeforeEach
38    void setUp() {
39      helper = new JDBCDataStoreHelper();
40    }
41  
42    private static Stream<Arguments> titlesAndNamesForFeatureSourcesAndFeatureTypes() {
43      return Stream.of(
44          Arguments.of("PostGIS", "bord"), Arguments.of("MS SQL Server", "bord"), Arguments.of("Oracle", "BORD"));
45    }
46  
47    @ParameterizedTest
48    @MethodSource("titlesAndNamesForFeatureSourcesAndFeatureTypes")
49    void testGetCreateAttachmentsForFeatureTypeStatements(String fsTitle, String ftName) {
50      featureType = featureTypeRepository
51          .getTMFeatureTypeByNameAndFeatureSource(
52              ftName, featureSourceRepository.getByTitle(fsTitle).orElseThrow())
53          .orElseThrow();
54  
55      try {
56        String sql = helper.getCreateAttachmentsForFeatureTypeStatements(featureType);
57        assertNotNull(sql);
58        assertThat(sql, startsWithIgnoringCase("CREATE TABLE " + ftName + "_attachments (\n" + ftName + "_pk "));
59      } catch (Exception e) {
60        fail(e.getMessage());
61      }
62    }
63  
64    @ParameterizedTest
65    @MethodSource("titlesAndNamesForFeatureSourcesAndFeatureTypes")
66    void testGetCreateAttachmentsIndexForFeatureTypeStatements(String fsTitle, String ftName) {
67      featureType = featureTypeRepository
68          .getTMFeatureTypeByNameAndFeatureSource(
69              ftName, featureSourceRepository.getByTitle(fsTitle).orElseThrow())
70          .orElseThrow();
71  
72      try {
73        String actual = helper.getCreateAttachmentsIndexForFeatureTypeStatements(featureType);
74        assertNotNull(actual);
75        String expected = "CREATE INDEX " + ftName + "_attachments_" + ftName + "_pk ON " + ftName
76            + "_attachments (" + ftName + "_pk);";
77        if (fsTitle.equals("Oracle")) expected = expected.toUpperCase(Locale.ROOT);
78        assertEquals(expected, actual);
79      } catch (Exception e) {
80        fail(e.getMessage());
81      }
82    }
83  }
