121   @BeforeEach

123     featureType = featureTypeRepository
124         .getTMFeatureTypeByNameAndFeatureSource(
125             featureTypeName,

128                 .orElseThrow(() -> new IllegalArgumentException(
129                     "Feature source not found: " + featureSourceTitle)))

133     AttachmentAttributeType attachmentAttributeType = new AttachmentAttributeType()

135         .maxAttachmentSize(4_000_000L)
136         .mimeType("image/jpeg, image/svg+xml, image/png");

138     attachmentData = new ClassPathResource("test/lichtpunt.svg").getContentAsByteArray();

183   @Order(2)
184   @Test
185   @DisplayName("Add an attachment to the first feature of a type.")
186   @WithMockUser(username = "unittestuser")
187   void addAttachmentToFirstFeatureOfFeatureType() {
188     try {
189       SimpleFeatureSource fs = ds.getFeatureSource(featureTypeName);
190       Query query = new Query(featureTypeName);
191       query.setMaxFeatures(1);
192       query.setPropertyNames(featureType.getPrimaryKeyAttribute());
193 
194       try (FeatureIterator<SimpleFeature> featureIterator =
195           fs.getFeatures(query).features()) {
196         if (featureIterator.hasNext()) {
197           featurePrimaryKey = featureIterator.next().getAttribute(featureType.getPrimaryKeyAttribute());
198         } else {
199           fail("No feature found in feature type table to get a valid UUID primary key value.");
200         }
201       }
202     } catch (IOException e) {
203       fail("Failed to retrieve UUID primary key value: " + e.getMessage());
204     }
205 
206     try {
207       AttachmentMetadata inserted = AttachmentsHelper.insertAttachment(
208           featureType, attachmentMetadata, featurePrimaryKey, attachmentData);
209 
210       assertNotNull(inserted);
211       assertNotNull(inserted.getAttachmentId());
212       attachmentAttributePKvalue = inserted.getAttachmentId();
213 
214       try (Connection conn = ds.getDataSource().getConnection();
215           Statement stmt = conn.createStatement();
216           ResultSet rs = stmt.executeQuery(
217               "select count(*) from " + schemaPrefix + featureTypeName + "_attachments")) {
218         if (rs.next()) {
219           int count = rs.getInt(1);
220           assertEquals(1, count, "Attachment was not inserted correctly.");
221         } else {
222           fail("Attachments table '%s_attachments' does not exist.".formatted(featureTypeName));
223         }
224       }
225     } catch (SQLException | IOException e) {
226       fail(e.getMessage());
227     }

276   @Order(4)
277   @Test
278   @DisplayName("Get attachments for a list of features of a type.")
279   void testListAttachmentsForFeaturesByFeatureId() {
280     try {
281       Object validFeaturePrimaryKey = null;
282       if (featurePrimaryKey instanceof byte[] bytes) {
283         // byte[] does not implement equals/hashCode properly, but GeoTools uses byte[] for PKs in Oracle when
284         // UUID is used
285         validFeaturePrimaryKey = ByteBuffer.wrap(bytes);
286       } else if (featurePrimaryKey instanceof Comparable<?> key) {
287         validFeaturePrimaryKey = key;
288       }
289       assertNotNull(validFeaturePrimaryKey);
290       Map<@NotNull Object, List<AttachmentMetadata>> listAttachments =
291           AttachmentsHelper.listAttachmentsForFeaturesByFeatureId(
292               featureType, List.of(validFeaturePrimaryKey));
293 
294       assertNotNull(listAttachments);
295       assertEquals(1, listAttachments.size(), "Expected exactly one feature.");
296       if (featurePrimaryKey instanceof byte[] bytes) {
297         // byte[] does not implement equals/hashCode properly, but GeoTools uses byte[] for PKs in Oracle when
298         // UUID is used
299         assertNotNull(listAttachments.get(ByteBuffer.wrap(bytes)));
300         assertEquals(1, listAttachments.get(ByteBuffer.wrap(bytes)).size(), "Expected exactly one attachment.");
301       } else {
302         assertNotNull(listAttachments.get(validFeaturePrimaryKey));
303         assertEquals(1, listAttachments.get(validFeaturePrimaryKey).size(), "Expected exactly one attachment.");
304       }
305     } catch (RuntimeException | IOException e) {
306       fail(e);
307     }
308   }
