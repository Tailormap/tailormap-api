6   package org.tailormap.api.geotools.featuresources;
7   
8   import static java.nio.ByteBuffer.wrap;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  
12  import java.util.UUID;
13  import org.junit.jupiter.api.Test;
14  import org.tailormap.api.persistence.TMFeatureType;
15  
16  class AttachmentsHelperTest {
17    static final String ftName = "testFeatureType";
18    static final TMFeatureType featureType = org.mockito.Mockito.mock(TMFeatureType.class, invocation -> {
19      if ("getName".equals(invocation.getMethod().getName())) {
20        return ftName;
21      }
22      return org.mockito.Answers.RETURNS_DEFAULTS.answer(invocation);
23    });
24  
25    @Test
26    void check_array_of_byte_object() {
27      UUID uuid = UUID.randomUUID();
28      Object testData = new byte[16];
29      wrap((byte[]) testData).putLong(uuid.getMostSignificantBits()).putLong(uuid.getLeastSignificantBits());
30      String result = AttachmentsHelper.fidFromPK(featureType, testData);
31      assertEquals(ftName + "." + uuid, result);
32    }
33  
34    @Test
35    void check_integer_object() {
36      Object testData = 12345;
37      String result = AttachmentsHelper.fidFromPK(featureType, testData);
38      assertEquals(ftName + "." + testData, result);
39    }
40  
41    @Test
42    void check_null() {
43      assertThrows(IllegalArgumentException.class, () -> AttachmentsHelper.fidFromPK(featureType, null));
44    }
45  }
