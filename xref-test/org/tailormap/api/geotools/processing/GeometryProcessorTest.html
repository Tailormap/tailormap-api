45    @Stopwatch
46    @Test
47    void reprojectPoint() throws ParseException, FactoryException {
48      final Geometry p = new WKTReader2().read(testData.getProperty("RDpointWkt"));
49      MathTransform transform = CRS.findMathTransform(CRS.decode("EPSG:28992"), CRS.decode("EPSG:4326"), true);
50  
51      final Geometry reprojected = new WKTReader2().read(GeometryProcessor.processGeometry(p, true, true, transform));
52      final Geometry expected = new WKTReader2().read(testData.getProperty("WGS84pointWkt"));
53      assertEquals(
54          expected.getCoordinate().getX(),
55          reprojected.getCoordinate().getX(),
56          .1,
57          "X-coord of simplified, reprojected geometry should match");
58      assertEquals(
59          expected.getCoordinate().getY(),
60          reprojected.getCoordinate().getY(),
61          .1,
62          "Y-coord of simplified, reprojected geometry should match");
63    }

65    @Stopwatch
66    @Test
67    void transformPoint() throws ParseException, FactoryException {
68      final Geometry p = new WKTReader2().read(testData.getProperty("RDpointWkt"));
69      MathTransform transform = CRS.findMathTransform(CRS.decode("EPSG:28992"), CRS.decode("EPSG:4326"), true);
70  
71      final Geometry reprojected = GeometryProcessor.transformGeometry(p, transform);
72      final Geometry expected = new WKTReader2().read(testData.getProperty("WGS84pointWkt"));
73      assertEquals(
74          expected.getCoordinate().getX(),
75          reprojected.getCoordinate().getX(),
76          .1,
77          "X-coord of simplified, reprojected geometry should match");
78      assertEquals(
79          expected.getCoordinate().getY(),
80          reprojected.getCoordinate().getY(),
81          .1,
82          "Y-coord of simplified, reprojected geometry should match");
83    }

85    @Stopwatch
86    @Test
87    void simplifyPolygon() throws ParseException {
88      final Geometry p = new WKTReader2().read(testData.getProperty("RDpolygonWkt"));
89      final String simplified = GeometryProcessor.processGeometry(p, true, true, null);
90      assertNotEquals(testData.getProperty("RDpolygonWkt"), simplified, "simplified geometry should not match");
91  
92      final Geometry p2 = new WKTReader2().read(simplified);
93      assertTrue(p.overlaps(p2), "source polygon should overlap simplified");
94      //    /* PMD.JUnitAssertionsShouldIncludeMessage */
95      //    assertThat(100 * p2.intersection(p).getArea() / p.getArea())
96      //        .as("intersection area should be about 99% of original")
97      //        .isCloseTo(99, within(1d));
98      assertThat(
99          "intersection area should be about 99% of original",
100         100 * p2.intersection(p).getArea() / p.getArea(), closeTo(99, 1d));
101 
102     //        .isCloseTo(99, within(1d));

141   @Test
142   void testJsonOutput() throws ParseException {
143     Geometry p = new WKTReader2().read(testData.getProperty("RDpointWkt"));
144     assertEquals(
145         "{\"type\":\"Point\",\"coordinates\":[141247,458118]}",
146         GeometryProcessor.geometryToJson(p),
147         "json output should match");
148 
149     // this is ignored by the json output
150     p.setSRID(28992);
151     assertEquals(
152         "{\"type\":\"Point\",\"coordinates\":[141247,458118]}",
153         GeometryProcessor.geometryToJson(p),
154         "json output should match");
155 
156     p = null;
157     //noinspection ConstantValue
158     assertEquals("null", GeometryProcessor.geometryToJson(p), "json output should match");
159   }
