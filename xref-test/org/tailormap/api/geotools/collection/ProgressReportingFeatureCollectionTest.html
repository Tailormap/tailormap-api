32  class ProgressReportingFeatureCollectionTest {
33    private static final int randomFeatureCount = 104;
34    private SimpleFeatureSource randomFeatureSource = null;
35  
36    @BeforeEach
37    void createRandomFeatures() throws IOException, SchemaException {
38      SimpleFeatureType inputType =
39          DataUtilities.createType("test", "id:Integer,label:String,date:Date,location:Point:28992");
40      MemoryDataStore dataStore = new MemoryDataStore(inputType);
41  
42      int[] xCoords = new Random().ints(randomFeatureCount, 155000, 165000).toArray();
43      int[] yCoords = new Random().ints(randomFeatureCount, 463000, 473000).toArray();
44      long minEpoch = LocalDate.of(2000, 1, 1).toEpochDay() * 86400L * 1000L;
45      long maxEpoch = LocalDate.of(2025, 12, 31).toEpochDay() * 86400L * 1000L;
46      final Random random = new Random();
47      final SimpleFeatureBuilder fb = new SimpleFeatureBuilder(inputType);
48      final GeometryFactory gf = new GeometryFactory();
49      IntStream.range(0, randomFeatureCount).forEach(id -> {
50        fb.set("id", id);
51        fb.set("label", "Feature number " + id);
52        @SuppressWarnings("JavaUtilDate")
53        Date randomDate = new Date(minEpoch + (long) (random.nextDouble() * (maxEpoch - minEpoch)));
54        fb.set("date", randomDate);
55        fb.set("location", gf.createPoint(new Coordinate(xCoords[id], yCoords[id])));
56        dataStore.addFeature(fb.buildFeature(String.valueOf(id)));
57      });
58  
59      randomFeatureSource = dataStore.getFeatureSource(inputType.getName());
60      assumeTrue(randomFeatureSource != null, "Failed to create random feature source");
61    }
62  
63    @AfterEach
64    void cleanup() {
65      if (randomFeatureSource != null) {
66        randomFeatureSource.getDataStore().dispose();
67      }
68    }
69  
70    @Test
71    void test_progress_works() throws IOException, SchemaException {
72      SimpleFeatureCollection source = randomFeatureSource.getFeatures();
73      AtomicInteger progressCount = new AtomicInteger(0);
74      ProgressReportingFeatureCollection collection =
75          new ProgressReportingFeatureCollection(source, 10, progressCount::set);
76      assertEquals(randomFeatureCount, collection.size());
77      try (SimpleFeatureIterator iterator = collection.features()) {
78        while (iterator.hasNext()) {
79          iterator.next();
80        }
81        assertEquals(10 * (randomFeatureCount / 10), progressCount.get());
82      }
83    }
84  
85    @Test
86    void allows_null_progress() throws IOException {
87      SimpleFeatureCollection source = randomFeatureSource.getFeatures();
88      ProgressReportingFeatureCollection collection = new ProgressReportingFeatureCollection(source, 10, null);
89      assertEquals(randomFeatureCount, collection.size());
90      try (SimpleFeatureIterator iterator = collection.features()) {
91        while (iterator.hasNext()) {
92          iterator.next();
93        }
94      }
95    }
96  
97    @Test
98    void disallows_negative_progress_interval() throws IOException {
99      SimpleFeatureCollection source = randomFeatureSource.getFeatures();
100     assertThrows(
101         IllegalArgumentException.class, () -> new ProgressReportingFeatureCollection(source, -1, count -> {}));
102   }
103 }
