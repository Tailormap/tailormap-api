6   package org.tailormap.api;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertFalse;
10  import static org.junit.jupiter.api.Assertions.fail;
11  
12  import java.io.IOException;
13  import java.lang.invoke.MethodHandles;
14  import java.util.ArrayList;
15  import java.util.Enumeration;
16  import java.util.List;
17  import java.util.zip.ZipEntry;
18  import java.util.zip.ZipFile;
19  import org.junit.jupiter.api.Test;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  class ValidateGeoToolsInArtifactIntegrationTest {
24  
25    private static final Logger logger =
26        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
27  
28    /** this name is set in the pom file */
29    private static final String RUNNABLE_JAR = "target/tailormap-api-exec.jar";
30  
31    /**
32     * check if there is only one gt-main and only one gt-epsg jar in the runnable jar, so we can be sure there are no
33     * or EPSG factory version conflicts. This test will fail if there is than one gt-main (versions) or gt-epsg
34     * (factory providers) in the runnable jar.
35     */
36    @Test
37    void checkArtifact() {
38      logger.debug("Checking Tailormap API artifact: {} for GeoTools artifacts.", RUNNABLE_JAR);
39      try (ZipFile zipFile = new ZipFile(RUNNABLE_JAR)) {
40        Enumeration<? extends ZipEntry> e = zipFile.entries();
41        List<ZipEntry> gtMainJars = new ArrayList<>();
42        List<ZipEntry> gtEPSGJars = new ArrayList<>();
43  
44        while (e.hasMoreElements()) {
45          ZipEntry entry = e.nextElement();
46          if (!entry.isDirectory()) {
47            if (entry.getName().contains("gt-main")) {
48              gtMainJars.add(entry);
49            }
50            if (entry.getName().contains("gt-epsg")) {
51              gtEPSGJars.add(entry);
52            }
53          }
54        }
55  
56        assertFalse(gtMainJars.isEmpty(), "No gt-main artifact in the runnable jar file");
57        assertEquals(1, gtMainJars.size(), "There are more than 1 gt-main artifacts in the runnable jar file");
58  
59        assertFalse(gtEPSGJars.isEmpty(), "No gt-epsg artifact in the runnable jar file");
60        assertEquals(1, gtEPSGJars.size(), "There are more than 1 gt-epsg artifacts in the runnable jar file");
61  
62      } catch (IOException e) {
63        logger.error("Failed", e);
64        fail(e.getLocalizedMessage());
65      }
66    }
67  }
