6   package org.tailormap.api.util;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertNotNull;
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  import static org.junit.jupiter.api.Assertions.assertTrue;
12  
13  import com.fasterxml.jackson.core.JsonParser;
14  import com.fasterxml.jackson.databind.DeserializationContext;
15  import com.fasterxml.jackson.databind.ObjectMapper;
16  import java.io.IOException;
17  import org.junit.jupiter.api.Test;
18  import org.tailormap.api.security.InvalidPasswordException;
19  
20  class TMPasswordDeserializerTest {
21    private final String testJsonTemplate = "{\"password\":\"%s\"}";
22  
23    @Test
24    void testDeserializeNullPassword() {
25      final String testJson = "{\"password\":null}";
26      assertThrows(
27          InvalidPasswordException.class,
28          () -> this.deserializeJson(testJson),
29          "null password should throw InvalidPasswordException");
30    }
31  
32    @Test
33    void testDeserializeEmptyInput() {
34      assertThrows(
35          InvalidPasswordException.class,
36          () -> this.deserializeJson("{}"),
37          "empty input should throw InvalidPasswordException");
38    }
39  
40    @Test
41    void testEmptyPassword() {
42      final String testJson = String.format(testJsonTemplate, "");
43      assertThrows(
44          InvalidPasswordException.class,
45          () -> this.deserializeJson(testJson),
46          "empty password should throw InvalidPasswordException");
47    }
48  
49    @Test
50    void testWeakPassword() {
51      final String testJson = String.format(testJsonTemplate, "flamingo");
52      assertThrows(
53          InvalidPasswordException.class,
54          () -> this.deserializeJson(testJson),
55          "empty password should throw InvalidPasswordException");
56    }
57  
58    @Test
59    void testValidPassword() throws IOException {
60      final String testJson = String.format(testJsonTemplate, "myValidSecret$@12");
61      String actual = this.deserializeJson(testJson);
62  
63      assertNotNull(actual, "encrypted password should not be null");
64      assertTrue(actual.startsWith("{bcrypt}$2a$"), "bcrypted password should start with {bcrypt}$2a$");
65      assertEquals(68, actual.length(), "bcrypted password should be 8+60 characters");
66    }
67  
68    private String deserializeJson(String json) throws IOException {
69  
70      ObjectMapper objectMapper = new ObjectMapper();
71      DeserializationContext deserializationContext = objectMapper.getDeserializationContext();
72      try (JsonParser parser = new ObjectMapper().getFactory().createParser(json)) {
73        // step though the templated json
74        // skip START_OBJECT
75        parser.nextToken();
76        // skip FIELD_NAME
77        parser.nextToken();
78        // use FIELD_VALUE
79        parser.nextToken();
80  
81        return new TMPasswordDeserializer().deserialize(parser, deserializationContext);
82      }
83    }
84  }
