6   package org.tailormap.api.util;

8   import static org.hamcrest.MatcherAssert.assertThat;
9   import static org.hamcrest.Matchers.closeTo;
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  import static org.junit.jupiter.api.Assertions.assertNotNull;
12  import static org.junit.jupiter.api.Assertions.assertTrue;
13  
14  import java.time.Instant;
15  import java.util.ArrayList;
16  import java.util.UUID;
17  import org.junit.jupiter.api.Test;
18  
19  class UUIDv7Test {
20    @Test
21    void testExtractUuid() throws InterruptedException {
22      ArrayList<UUID> uuids = new ArrayList<>();
23      for (int i = 0; i < 10; i++) {
24        Thread.sleep(10); // Ensure different timestamps for each UUID
25        UUID uuid = UUIDv7.randomV7();
26        assertNotNull(uuid);
27        uuids.add(uuid);
28        assertEquals(7, uuid.version(), () -> "Expected version 7, got " + uuid.version());
29        assertEquals(2, uuid.variant(), () -> "Expected RFC 4122 variant, got " + uuid.variant());
30        assertThat((double) (Instant.now().toEpochMilli() - UUIDv7.timestamp(uuid)), closeTo(0d, 100d));
31      }
32  
33      UUID[] cloned = uuids.toArray(new UUID[0]).clone();
34      uuids.sort(UUID::compareTo);
35      for (int i = 0; i < uuids.size(); i++) {
36        assertEquals(cloned[i], uuids.get(i), "Expected UUIDs to be in the same order after sorting " + i);
37        if (i > 0) {
38          assertTrue(
39              UUIDv7.timestampAsInstant(uuids.get(i)).isAfter(UUIDv7.timestampAsInstant(uuids.get(i - 1))),
40              "Expected timestamps to be in ascending order " + i);
41        }
42      }
43    }
44  
45    @Test
46    void testRoundTrip() {
47      UUID uuid = UUIDv7.randomV7();
48      Instant timestamp = UUIDv7.timestampAsInstant(uuid);
49      assertNotNull(timestamp);
50      // uuid v4 check
51      assertNotNull(UUID.fromString(uuid.toString()));
52  
53      UUID parsed = UUIDv7.fromString(uuid.toString());
54      assertNotNull(parsed);
55      assertEquals(7, parsed.version(), () -> "Expected version 7, got " + parsed.version());
56      assertEquals(timestamp.toEpochMilli(), UUIDv7.timestamp(parsed));
57      assertEquals(uuid, parsed);
58    }
59  }
