257   @Test
258   @WithMockUser(username = "user")
259   @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
260   void test_obk_wmts_GetTile_Conditional() throws Exception {
261     final String path = apiBasePath + obkUrl;
262 
263     final DateTimeFormatter httpDateHeaderFormatter =
264         DateTimeFormatter.ofPattern("EEE, dd MMM yyyy HH:mm:ss z", Locale.ENGLISH)
265             .withZone(ZoneId.of("GMT"));
266     mockMvc
267         .perform(
268             get(path)
269                 .param("SERVICE", "WMTS")
270                 .param("VERSION", "1.0.0")
271                 .param("REQUEST", "GetTile")
272                 .param("LAYER", "osm")
273                 .param("STYLE", "default")
274                 .param("FORMAT", "image/png")
275                 .param("TILEMATRIXSET", "rd")
276                 .param("TILEMATRIX", "4")
277                 .param("TILEROW", "7")
278                 .param("TILECOL", "8")
279                 .with(setServletPath(path))
280                 .header("If-Modified-Since", httpDateHeaderFormatter.format(Instant.now())))
281         .andExpect(status().isNotModified());
282   }

284   private static class StringIsNotZeroMatcher extends TypeSafeMatcher<String> {
285     @Override
286     protected boolean matchesSafely(String item) {
287       return Long.parseLong(item) > 0;
288     }
289 
290     @Override
291     public void describeTo(Description description) {
292       description.appendText("is not zero");
293     }
294   }
