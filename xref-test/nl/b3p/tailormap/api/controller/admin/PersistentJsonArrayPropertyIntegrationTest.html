35  import org.springframework.web.context.WebApplicationContext;

37  @PostgresIntegrationTest
38  class PersistentJsonArrayPropertyIntegrationTest {
39    private static final Logger logger =
40        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
41  
42    @Autowired private WebApplicationContext context;
43  
44    @Value("${tailormap-api.admin.base-path}")
45    private String adminBasePath;
46  
47    @Test
48    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
49    @WithMockUser(
50        username = "admin",
51        authorities = {Group.ADMIN})
52    void testUpdateApplicationComponents() throws Exception {
53      ObjectMapper objectMapper = new ObjectMapper();
54  
55      MockMvc mockMvc =
56          MockMvcBuilders.webAppContextSetup(context).build(); // Required for Spring Data Rest APIs
57  
58      Application app =
59          new Application()
60              .setName("test-json-array-property")
61              .setTitle("Testcase")
62              .setCrs("EPSG:3857");
63  
64      MvcResult result =
65          mockMvc
66              .perform(
67                  post(adminBasePath + "/applications")
68                      .contentType(MediaType.APPLICATION_JSON)
69                      .content(objectMapper.writeValueAsString(app)))
70              .andExpect(status().isCreated())
71              .andExpect(jsonPath("$.id").isNotEmpty())
72              .andExpect(jsonPath("$.components").isArray())
73              .andExpect(jsonPath("$.components.length()").value(0))
74              .andExpect(jsonPath("$.name").value("test-json-array-property"))
75              .andReturn();
76  
77      Integer appId = JsonPath.read(result.getResponse().getContentAsString(), "$.id");
78  
79      app.setComponents(
80          List.of(
81              new Component().type("test 1"),
82              new Component().type("test 2"),
83              new Component().type("test 3")));
84  
85      mockMvc
86          .perform(
87              patch(adminBasePath + "/applications/" + appId)
88                  .contentType(MediaType.APPLICATION_JSON)
89                  .content(getApplicationComponentsPatchBody(objectMapper, app)))
90          .andExpect(status().isOk())
91          .andExpect(jsonPath("$.id").isNotEmpty())
92          .andExpect(jsonPath("$.components").isArray())
93          .andExpect(jsonPath("$.components.length()").value(3))
94          .andExpect(jsonPath("$.components[0].type").value("test 1"))
95          .andExpect(jsonPath("$.components[1].type").value("test 2"))
96          .andExpect(jsonPath("$.components[2].type").value("test 3"));
97  
98      app.setComponents(
99          List.of(
100             new Component().type("test 1"),
101             new Component().type("test 2 [modified without changing array size]"),
102             new Component().type("test 3")));
103 
104     mockMvc
105         .perform(
106             patch(adminBasePath + "/applications/" + appId)
107                 .contentType(MediaType.APPLICATION_JSON)
108                 .content(getApplicationComponentsPatchBody(objectMapper, app)))
109         .andExpect(status().isOk())
110         .andExpect(jsonPath("$.id").isNotEmpty())
111         .andExpect(jsonPath("$.components").isArray())
112         .andExpect(jsonPath("$.components.length()").value(3))
113         .andExpect(
114             jsonPath("$.components[1].type")
115                 .value("test 2 [modified without changing array size]"));
116 
117     app.setComponents(
118         List.of(new Component().type("test 2 [shrink array]"), new Component().type("test 3")));
119 
120     mockMvc
121         .perform(
122             patch(adminBasePath + "/applications/" + appId)
123                 .contentType(MediaType.APPLICATION_JSON)
124                 .content(getApplicationComponentsPatchBody(objectMapper, app)))
125         .andExpect(status().isOk())
126         .andExpect(jsonPath("$.id").isNotEmpty())
127         .andExpect(jsonPath("$.components").isArray())
128         .andExpect(jsonPath("$.components.length()").value(2))
129         .andExpect(jsonPath("$.components[0].type").value("test 2 [shrink array]"));
130 
131     app.setComponents(
132         List.of(
133             new Component().type("test 1"),
134             new Component().type("test 2"),
135             new Component().type("test 3"),
136             new Component().type("test 4 [test growing array]")));
137 
138     mockMvc
139         .perform(
140             patch(adminBasePath + "/applications/" + appId)
141                 .contentType(MediaType.APPLICATION_JSON)
142                 .content(getApplicationComponentsPatchBody(objectMapper, app)))
143         .andExpect(status().isOk())
144         .andExpect(jsonPath("$.id").isNotEmpty())
145         .andExpect(jsonPath("$.components").isArray())
146         .andExpect(jsonPath("$.components.length()").value(4))
147         .andExpect(jsonPath("$.components[3].type").value("test 4 [test growing array]"));
148   }
149 
150   private static String getApplicationComponentsPatchBody(
151       ObjectMapper objectMapper, Application app) throws JsonProcessingException {
152     ObjectNode node = objectMapper.createObjectNode();
153     node.set("components", objectMapper.convertValue(app, JsonNode.class).get("components"));
154     String patchBody = objectMapper.writeValueAsString(node);
155     logger.info("PATCH body: " + patchBody);
156     return patchBody;
157   }
158 }
