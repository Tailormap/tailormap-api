8   
9   import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.post;
10  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
12  
13  import com.fasterxml.jackson.databind.ObjectMapper;
14  import nl.b3p.tailormap.api.annotation.PostgresIntegrationTest;
15  import nl.b3p.tailormap.api.persistence.Group;
16  import org.junit.jupiter.api.Test;
17  import org.springframework.beans.factory.annotation.Autowired;
18  import org.springframework.beans.factory.annotation.Value;
19  import org.springframework.http.MediaType;
20  import org.springframework.security.test.context.support.WithMockUser;
21  import org.springframework.test.web.servlet.MockMvc;
22  import org.springframework.test.web.servlet.setup.MockMvcBuilders;
23  import org.springframework.web.context.WebApplicationContext;
24  
25  @PostgresIntegrationTest
26  class JSR303ValidationIntegrationTest {
27    @Autowired private WebApplicationContext context;
28  
29    @Value("${tailormap-api.admin.base-path}")
30    private String adminBasePath;
31  
32    @Test
33    @WithMockUser(
34        username = "admin",
35        authorities = {Group.ADMIN})
36    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
37    void testUrlRequired() throws Exception {
38      MockMvc mockMvc =
39          MockMvcBuilders.webAppContextSetup(context).build(); // Required for Spring Data Rest APIs
40  
41      String geoServicePOSTBody =
42          new ObjectMapper()
43              .createObjectNode()
44              .put("protocol", "wms")
45              .put("title", "test")
46              .put("refreshCapabilities", true)
47              .put("url", (String) null)
48              .toPrettyString();
49  
50      mockMvc
51          .perform(
52              post(adminBasePath + "/geo-services")
53                  .contentType(MediaType.APPLICATION_JSON)
54                  .content(geoServicePOSTBody))
55          .andExpect(status().isBadRequest())
56          .andExpect(
57              content()
58                  .json(
59                      "{\"errors\":[{\"entity\":\"GeoService\",\"property\":\"url\",\"invalidValue\":null,\"message\":\"must not be null\"}]}"));
60    }
61  }
