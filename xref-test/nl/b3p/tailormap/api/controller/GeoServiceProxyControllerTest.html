6   package nl.b3p.tailormap.api.controller;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   
10  import org.junit.jupiter.api.Test;
11  import org.springframework.util.MultiValueMap;
12  import org.springframework.web.util.UriComponents;
13  import org.springframework.web.util.UriComponentsBuilder;
14  
15  class GeoServiceProxyControllerTest {
16  
17    @Test
18    void buildOgcProxyRequestParams() {
19      String originalUrl = "https://service/path1?param=abc";
20      String requestUrl = "https://api/path2?request=GetMap&other=value";
21      assertEquals(
22          UriComponentsBuilder.fromHttpUrl(
23                  "https://service/path1?request=GetMap&param=abc&other=value")
24              .build(),
25          getProxyUrl(originalUrl, requestUrl),
26          "bad proxy params");
27  
28      originalUrl = "https://service/path1?param=abc&REQUEST=GetCapabilities&original=value";
29      requestUrl = "https://api/path2?request=GetMap&param=xyz";
30      assertEquals(
31          UriComponentsBuilder.fromHttpUrl(
32                  "https://service/path1?request=GetMap&param=xyz&original=value")
33              .build(),
34          getProxyUrl(originalUrl, requestUrl),
35          "bad proxy params");
36    }
37  
38    private static UriComponents getProxyUrl(String originalUrl, String requestUrl) {
39      UriComponentsBuilder uriComponentsBuilder = UriComponentsBuilder.fromHttpUrl(originalUrl);
40      uriComponentsBuilder.replaceQueryParams(
41          GeoServiceProxyController.buildOgcProxyRequestParams(
42              getUrlParams(originalUrl), getUrlParams(requestUrl)));
43      return uriComponentsBuilder.build();
44    }
45  
46    private static MultiValueMap<String, String> getUrlParams(String url) {
47      return UriComponentsBuilder.fromHttpUrl(url).build().getQueryParams();
48    }
49  }
