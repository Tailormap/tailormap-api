50    @Test
51    void services_should_be_unique() throws Exception {
52      // GET https://snapshot.tailormap.nl/api/app/default/map
53      final String path = apiBasePath + "/app/default/map";
54      MvcResult result =
55          mockMvc
56              .perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
57              .andExpect(status().isOk())
58              .andExpect(content().contentType(MediaType.APPLICATION_JSON))
59              .andExpect(jsonPath("$.initialExtent").isMap())
60              .andExpect(jsonPath("$.maxExtent").isMap())
61              .andExpect(jsonPath("$.services").isArray())
62              .andExpect(jsonPath("$.crs.code").value("EPSG:28992"))
63              .andReturn();
64  
65      String body = result.getResponse().getContentAsString();
66      List<Service> allSvc = JsonPath.read(body, "$.services");
67      Set<Service> uniqueSvc = new HashSet<>(allSvc);
68      assertTrue(1 < allSvc.size(), "there must be more than one service, otherwise change the url");
69      assertEquals(
70          allSvc.size(),
71          uniqueSvc.size(),
72          () -> ("services array contains non-unique items: " + allSvc));
73    }

104         .andExpect(
105             jsonPath("$.services[?(@.id == 'snapshot-geoserver-proxied')].url")

107         .andExpect(
108             jsonPath(
109                     "$.appLayers[?(@.id === 'lyr:snapshot-geoserver-proxied:postgis:begroeidterreindeel')].url")
110                 // Need to use contains() because jsonPath() returns an array even
111                 // when the expression resolves to a single scalar property
112                 .value(
113                     contains(
114                         endsWith(
115                             "/app/default/layer/lyr%3Asnapshot-geoserver-proxied%3Apostgis%3Abegroeidterreindeel/proxy/wms"))))
116         .andExpect(
117             jsonPath("$.services[?(@.id == 'pdok-kadaster-bestuurlijkegebieden')].url")
118                 .value(contains(nullValue())))
119         .andExpect(
120             jsonPath(
121                     "$.appLayers[?(@.id === 'lyr:pdok-kadaster-bestuurlijkegebieden:Provinciegebied')].url")

123                     contains(
124                         endsWith(

250       username = "foo",
251       authorities = {"test-foo"})
252   void should_filter_layers() throws Exception {
253     final String path = apiBasePath + "/app/secured-auth/map";
254     // the group "test-foo" has READ permissions to the Application and the GeoService, and is
255     // denied access to the filtered "BGT" layer.
256     // (postgis:begroeidterreindeel has no explicit rule, so it is allowed.)
257     mockMvc
258         .perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
259         .andExpect(status().isOk())
260         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
261         .andExpect(jsonPath("$.services.length()").value(2))
262         .andExpect(jsonPath("$.appLayers.length()").value(2))
263         .andExpect(
264             jsonPath("$.appLayers[0].id")
265                 .value("lyr:filtered-snapshot-geoserver:postgis:begroeidterreindeel"))
266         .andExpect(jsonPath("$.appLayers[1].id").value("lyr:snapshot-geoserver:BGT"));
267   }
