6   package nl.b3p.tailormap.api.repository;
7   
8   import static nl.b3p.tailormap.api.persistence.Configuration.DEFAULT_APP;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  import static org.junit.jupiter.api.Assertions.assertNotNull;
11  import static org.junit.jupiter.api.Assertions.assertNull;
12  
13  import nl.b3p.tailormap.api.annotation.PostgresIntegrationTest;
14  import nl.b3p.tailormap.api.persistence.Configuration;
15  import nl.b3p.tailormap.api.security.InternalAdminAuthentication;
16  import org.junit.jupiter.api.MethodOrderer.OrderAnnotation;
17  import org.junit.jupiter.api.Order;
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.api.TestMethodOrder;
20  import org.springframework.beans.factory.annotation.Autowired;
21  import org.springframework.transaction.annotation.Transactional;
22  
23  @PostgresIntegrationTest
24  @TestMethodOrder(OrderAnnotation.class)
25  class ConfigurationRepositoryIntegrationTest {
26  
27    @Autowired private ConfigurationRepository configurationRepository;
28  
29    @Test
30    @Order(1)
31    void it_should_findByConfigKeyDefaultApplication() {
32      final Configuration c = configurationRepository.findByKey(DEFAULT_APP).orElse(null);
33      assertNotNull(c);
34      assertEquals("default", c.getValue());
35    }
36  
37    @Test
38    @Transactional
39    @Order(2)
40    void it_should_not_find_value_after_deleting_key() {
41      InternalAdminAuthentication.setInSecurityContext();
42      try {
43        configurationRepository.deleteConfigurationByKey(DEFAULT_APP);
44        final Configuration c = configurationRepository.findByKey(DEFAULT_APP).orElse(null);
45        assertNull(c);
46      } finally {
47        InternalAdminAuthentication.clearSecurityContextAuthentication();
48      }
49    }
50  }
