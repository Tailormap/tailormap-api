8   import static nl.b3p.tailormap.api.TestRequestProcessor.setServletPath;
9   import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
10  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
13  
14  import nl.b3p.tailormap.api.annotation.PostgresIntegrationTest;
15  import nl.b3p.tailormap.api.persistence.Group;
16  import org.junit.jupiter.api.Test;
17  import org.springframework.beans.factory.annotation.Autowired;
18  import org.springframework.beans.factory.annotation.Value;
19  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
20  import org.springframework.http.MediaType;
21  import org.springframework.security.test.context.support.WithMockUser;
22  import org.springframework.test.web.servlet.MockMvc;
23  
24  /**
25   * Integration test for the API security configuration. The test indirectly verifies that the API
26   * {@code app} endpoint is secured as expected.
27   */
28  @PostgresIntegrationTest
29  @AutoConfigureMockMvc
30  class ApiSecurityConfigurationIntegrationTest {
31  
32    @Autowired private MockMvc mockMvc;
33  
34    @Value("${tailormap-api.base-path}")
35    private String apiBasePath;
36  
37    @Test
38    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
39    void anonymousSecuredAppTest() throws Exception {
40      String path = apiBasePath + "/app/secured";
41      mockMvc
42          .perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
43          .andExpect(status().isUnauthorized())
44          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
45          .andExpect(jsonPath("$.code").value(401))
46          .andExpect(jsonPath("$.url").value("/login"));
47    }
48  
49    @Test
50    @WithMockUser(
51        username = "user",
52        authorities = {Group.AUTHENTICATED})
53    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
54    void authenticatedSecuredAppTest() throws Exception {
55      String path = apiBasePath + "/app/secured";
56      mockMvc
57          .perform(get(path).accept(MediaType.APPLICATION_JSON).with(setServletPath(path)))
58          .andExpect(status().isOk())
59          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
60          .andExpect(jsonPath("$.kind").value("app"));
61    }
62  }
