8   import static org.hamcrest.Matchers.containsString;
9   import static org.hamcrest.Matchers.startsWith;
10  import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
13  
14  import nl.b3p.tailormap.api.annotation.PostgresIntegrationTest;
15  import nl.b3p.tailormap.api.persistence.Group;
16  import org.junit.jupiter.api.Test;
17  import org.springframework.beans.factory.annotation.Autowired;
18  import org.springframework.beans.factory.annotation.Value;
19  import org.springframework.boot.test.autoconfigure.actuate.observability.AutoConfigureObservability;
20  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
21  import org.springframework.http.MediaType;
22  import org.springframework.security.test.context.support.WithMockUser;
23  import org.springframework.test.context.TestPropertySource;
24  import org.springframework.test.web.servlet.MockMvc;
25  
26  /**
27   * Integration test for the actuator security configuration. The test verifies that the actuator
28   * endpoints are secured as expected; {@code health} is publicly accessible, {@code prometheus}
29   * requires authentication.
30   */
31  @PostgresIntegrationTest
32  @AutoConfigureMockMvc
33  @AutoConfigureObservability
34  @TestPropertySource(
35      properties = {
36        "management.endpoints.web.exposure.include=health,prometheus",
37        "management.endpoint.health.enabled=true",
38        "management.endpoint.prometheus.enabled=true",
39        "management.prometheus.metrics.export.enabled=true",
40        "management.prometheus.metrics.export.descriptions=true",
41      })
42  class ActuatorSecurityConfigurationIntegrationTest {
43    @Autowired private MockMvc mockMvc;
44  
45    @Value("${management.endpoints.web.base-path}")
46    private String basePath;
47  
48    @Test
49    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
50    void testUnAuthenticatedHealth() throws Exception {
51      mockMvc
52          .perform(get(basePath + "/health").accept(MediaType.APPLICATION_JSON))
53          .andExpect(status().isOk())
54          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
55          .andExpect(content().json("{\"status\":\"UP\"}"));
56    }
57  
58    @Test
59    @WithMockUser(
60        username = "tm-admin",
61        authorities = {Group.ADMIN})
62    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
63    void testAuthenticatedHealth() throws Exception {
64      mockMvc
65          .perform(get(basePath + "/health").accept(MediaType.APPLICATION_JSON))
66          .andExpect(status().isOk())
67          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
68          .andExpect(content().json("{\"status\":\"UP\"}"));
69    }
70  
71    @Test
72    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
73    void testUnAuthenticatedPrometheus() throws Exception {
74      mockMvc
75          .perform(get(basePath + "/prometheus").accept(MediaType.TEXT_PLAIN))
76          .andExpect(status().isUnauthorized());
77    }
78  
79    @Test
80    @WithMockUser(
81        username = "tm-admin",
82        authorities = {Group.ADMIN})
83    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
84    void testAuthenticatedPrometheus() throws Exception {
85      mockMvc
86          .perform(get(basePath + "/prometheus").accept(MediaType.TEXT_PLAIN))
87          .andExpect(status().isOk())
88          .andExpect(content().contentTypeCompatibleWith(MediaType.TEXT_PLAIN))
89          .andExpect(content().string(startsWith("# HELP")))
90          .andExpect(
91              content()
92                  .string(containsString("# HELP jvm_memory_used_bytes The amount of used memory")));
93    }
94  }
