8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.params.provider.Arguments.arguments;
10  
11  import java.util.stream.Stream;
12  import org.junit.jupiter.api.BeforeEach;
13  import org.junit.jupiter.params.ParameterizedTest;
14  import org.junit.jupiter.params.provider.Arguments;
15  import org.junit.jupiter.params.provider.MethodSource;
16  
17  class GeoServiceTest {
18  
19    private GeoService geoService;
20  
21    static Stream<Arguments> urlProvider() {
22      return Stream.of(
23          arguments("https://geoserver.nl/wms", "https://geoserver.nl/wms"),
24          arguments(
25              "https://geoserver.nl/wms?encoded=aapjes%20%3D%20!%263%3D1",
26              "https://geoserver.nl/wms?encoded=aapjes%20%3D%20!%263%3D1"),
27          arguments("https://geoserver.nl/wms?", "https://geoserver.nl/wms"),
28          arguments("https://geoserver.nl/wms?request=getcapabilities", "https://geoserver.nl/wms"),
29          arguments(
30              "https://geoserver.nl/wms?request=getcapabilities&REQUEST=getMap",
31              "https://geoserver.nl/wms"),
32          arguments("https://geoserver.nl/wmts?REQUEST=getcapabilities", "https://geoserver.nl/wmts"),
33          arguments(
34              "https://geoserver.nl/wms?request=getcapabilities&version=1",
35              "https://geoserver.nl/wms?version=1"),
36          arguments("https://geoserver.nl/wms?&version=1", "https://geoserver.nl/wms?version=1"),
37          arguments("https://geoserver.nl/wms?&version=1,2", "https://geoserver.nl/wms?version=1,2"),
38          arguments(
39              "https://service.pdok.nl/hwh/luchtfotorgb/wmts/v1_0",
40              "https://service.pdok.nl/hwh/luchtfotorgb/wmts/v1_0"),
41          arguments(
42              "https://basemap.at/wmts/1.0.0/WMTSCapabilities.xml",
43              "https://basemap.at/wmts/1.0.0/WMTSCapabilities.xml"),
44          arguments(
45              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS",
46              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS"),
47          arguments(
48              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS&request=getMap",
49              "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wms/v1_0?service=WMS"));
50    }
51  
52    @BeforeEach
53    void setGeoService() {
54      geoService = new GeoService();
55    }
56  
57    @ParameterizedTest(name = "#{index}: should sanitise url: {0}")
58    @MethodSource("urlProvider")
59    void testSetUrl(final String input, final String expected) {
60      geoService.setUrl(input);
61      assertEquals(
62          expected,
63          geoService.getUrl(),
64          () -> input + "not sanitised properly, expected " + expected);
65    }
66  }
