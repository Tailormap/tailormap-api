22  import org.junit.jupiter.api.Test;
23  
24  class TMFeatureTypeHelperTest {
25  
26    @Test
27    void testGetConfiguredAttributes() {
28      TMFeatureType ft = new TMFeatureType();
29      final Function<String, TMAttributeDescriptor> att =
30          (name) -> new TMAttributeDescriptor().name(name).comment("comment for " + name);
31      ft.setAttributes(
32          List.of(
33              att.apply("a"),
34              att.apply("b"),
35              att.apply("c"),
36              att.apply("d"),
37              att.apply("e"),
38              att.apply("f"),
39              att.apply("m"),
40              att.apply("n"),
41              att.apply("o"),
42              att.apply("q")));
43  
44      ft.setSettings(
45          new FeatureTypeSettings()
46              .putAttributeSettingsItem("d", new AttributeSettings().title("d title"))
47              .addHideAttributesItem("b")
48              .addHideAttributesItem("q")
49              .addHideAttributesItem("o")
50              .addAttributeOrderItem("c")
51              .addAttributeOrderItem("d")
52              .addAttributeOrderItem("e")
53              .addAttributeOrderItem("o")
54              .addAttributeOrderItem("a"));
55  
56      AppLayerSettings appLayerSettings = new AppLayerSettings();
57      appLayerSettings.setHideAttributes(List.of("e", "f"));
58  
59      Map<String, Pair<TMAttributeDescriptor, AttributeSettings>> configuredAttributes =
60          getConfiguredAttributes(ft, appLayerSettings);
61      // Compare using Lists to explicitly check the ordering
62      assertEquals(
63          List.of(
64              "c",
65              "d",
66              // "e" and "f" are hidden in app layer settings
67              // "o" is hidden in feature type settings
68              "a",
69              // after ordered attributes, remaining attributes in original feature type order
70              "m",
71              "n"),
72          new ArrayList<>(configuredAttributes.keySet()));
73  
74      assertEquals("d title", configuredAttributes.get("d").getRight().getTitle());
75  
76      assertEquals("comment for m", configuredAttributes.get("m").getLeft().getComment());
77    }
78  }
