39  @AppRestController
40  @Validated
41  @RequestMapping(
42      path = "${tailormap-api.base-path}/{viewerKind}/{viewerName}/layer/{appLayerId}/describe",
43      produces = MediaType.APPLICATION_JSON_VALUE)
44  public class LayerDescriptionController {
45  
46    private final FeatureSourceRepository featureSourceRepository;
47  
48    public LayerDescriptionController(FeatureSourceRepository featureSourceRepository) {
49      this.featureSourceRepository = featureSourceRepository;
50    }
51  
52    @Transactional
53    @GetMapping
54    public ResponseEntity<Serializable> getAppLayerDescription(
55        @ModelAttribute Application application,
56        @ModelAttribute AppTreeLayerNode appTreeLayerNode,
57        @ModelAttribute GeoService service,
58        @ModelAttribute GeoServiceLayer layer) {
59  
60      if (layer == null) {
61        throw new ResponseStatusException(
62            HttpStatus.NOT_FOUND, "Can't find layer " + appTreeLayerNode);
63      }
64  
65      TMFeatureType tmft = service.findFeatureTypeForLayer(layer, featureSourceRepository);
66      if (tmft == null) {
67        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "Layer does not have feature type");
68      }
69  
70      LayerDetails r =
71          new LayerDetails()
72              .id(appTreeLayerNode.getId())
73              .serviceId(appTreeLayerNode.getServiceId())
74              .featureTypeName(tmft.getName())
75              .geometryAttribute(tmft.getDefaultGeometryAttribute())
76              // TODO deduce from tmft.getDefaultGeometryAttribute()
77              .geometryAttributeIndex(null /* TODO */)
78              .geometryType(
79                  tmft.getDefaultGeometryDescriptor()
80                      .map(TMAttributeDescriptor::getType)
81                      .map(TMAttributeType::getValue)
82                      .map(TMGeometryType::fromValue)
83                      .orElse(null))
84              .editable(TMFeatureTypeHelper.isEditable(application, appTreeLayerNode, tmft));
85  
86      AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
87      Set<String> readOnlyAttributes =
88          TMFeatureTypeHelper.getReadOnlyAttributes(tmft, appLayerSettings);
89  
90      getConfiguredAttributes(tmft, appLayerSettings).values().stream()
91          .map(
92              pair -> {
93                TMAttributeDescriptor a = pair.getLeft();
94                AttributeSettings settings = pair.getRight();
95                return new Attribute()
96                    .featureType(tmft.getId())
97                    .key(a.getName())
98                    // Only return generic 'geometry' type for now, frontend doesn't
99                    // handle different geometry types. For the default geometry
100                   // attribute there is a specific geometry type set
101                   .type(isGeometry(a.getType()) ? TMAttributeType.GEOMETRY : a.getType())
102                   // primary key can never be edited
103                   .editable(
104                       !a.getName().equals(tmft.getPrimaryKeyAttribute())
105                           && !readOnlyAttributes.contains(a.getName()))
106                   .editAlias(Optional.ofNullable(settings.getTitle()).orElse(a.getName()))
107                   .defaultValue(a.getDefaultValue())
108                   .nullable(a.getNullable())
109                   .valueList(/*String[]*/ null /* TODO */)
110                   .allowValueListOnly(false /* TODO */);
111             })
112         .forEach(r::addAttributesItem);
113 
114     return ResponseEntity.ok(r);
115   }
116 }
