9   import static ch.rasc.sse.eventbus.SseEvent.DEFAULT_EVENT;
10  import static nl.b3p.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.KEEP_ALIVE;
11  
12  import ch.rasc.sse.eventbus.SseEvent;
13  import ch.rasc.sse.eventbus.SseEventBus;
14  import com.fasterxml.jackson.core.JsonProcessingException;
15  import com.fasterxml.jackson.databind.ObjectMapper;
16  import java.lang.invoke.MethodHandles;
17  import nl.b3p.tailormap.api.admin.model.ServerSentEvent;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  import org.springframework.scheduling.annotation.Scheduled;
21  import org.springframework.web.bind.annotation.GetMapping;
22  import org.springframework.web.bind.annotation.PathVariable;
23  import org.springframework.web.bind.annotation.RestController;
24  import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;
25  
26  @RestController
27  public class ServerSentEventsAdminController {
28    private static final Logger logger =
29        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
30  
31    private final SseEventBus eventBus;
32  
33    private final ObjectMapper objectMapper;
34  
35    public ServerSentEventsAdminController(SseEventBus eventBus, ObjectMapper objectMapper) {
36      this.eventBus = eventBus;
37      this.objectMapper = objectMapper;
38    }
39  
40    /**
41     * Endpoint for the single-page admin frontend to receive updates to entities from other windows.
42     * The frontend does not load all data on each navigation as a traditional frontend and loads and
43     * stores entities in memory. This requires updating stale entities when these get updated in
44     * other windows (from other sessions or the same session in a different browser tab), otherwise
45     * data might get lost or overwritten.
46     *
47     * <p>This endpoint uses <a
48     * href="https://html.spec.whatwg.org/multipage/server-sent-events.html">Server-sent events</a>,
49     * which technically is a long-poll HTTP request which allows unidirectional server-initiated
50     * communication. Not bidirectional like websockets, but simpler because of the long-poll HTTP
51     * request. The webserver should ideally use HTTP/2 or higher because HTTP/1.1 limits the amount
52     * of simultaneous connections to a server per browser to 6.
53     *
54     * @return the server-sent events emitter
55     */
56    @GetMapping(path = "${tailormap-api.admin.base-path}/events/{clientId}")
57    public SseEmitter entityEvents(
58        /*@RequestParam(required = false) String[] events,*/ @PathVariable("clientId")
59            String clientId) {
60      logger.debug("New SSE client: {}, all clients: {}", clientId, this.eventBus.getAllClientIds());
61      return this.eventBus.createSseEmitter(clientId, 3600_000L, DEFAULT_EVENT);
62    }
63  
64    @Scheduled(fixedRate = 60_000)
65    public void keepAlive() throws JsonProcessingException {
66      this.eventBus.handleEvent(
67          SseEvent.ofData(
68              objectMapper.writeValueAsString(new ServerSentEvent().eventType(KEEP_ALIVE))));
69    }
70  }
