6   package nl.b3p.tailormap.api.controller;
7   
8   import java.io.Serializable;
9   import nl.b3p.tailormap.api.annotation.AppRestController;
10  import nl.b3p.tailormap.api.persistence.Configuration;
11  import nl.b3p.tailormap.api.repository.ConfigurationRepository;
12  import nl.b3p.tailormap.api.viewer.model.ConfigResponse;
13  import org.springframework.http.HttpStatus;
14  import org.springframework.http.MediaType;
15  import org.springframework.http.ResponseEntity;
16  import org.springframework.web.bind.annotation.GetMapping;
17  import org.springframework.web.bind.annotation.PathVariable;
18  import org.springframework.web.server.ResponseStatusException;
19  
20  /** Provides global configuration values which are available to viewers */
21  @AppRestController
22  public class ConfigurationController {
23  
24    private final ConfigurationRepository configurationRepository;
25  
26    public ConfigurationController(ConfigurationRepository configurationRepository) {
27      this.configurationRepository = configurationRepository;
28    }
29  
30    @GetMapping(
31        path = "${tailormap-api.base-path}/config/{key}",
32        produces = MediaType.APPLICATION_JSON_VALUE)
33    public ResponseEntity<Serializable> getConfig(@PathVariable String key) {
34      Configuration config =
35          configurationRepository
36              .getAvailableForViewer(key)
37              .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND));
38  
39      ConfigResponse response = new ConfigResponse();
40      response.setKey(key);
41      response.setValue(config.getValue());
42      response.setObject(config.getJsonValue());
43      return ResponseEntity.status(HttpStatus.OK).body(response);
44    }
45  }
