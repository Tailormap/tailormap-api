91    private static void checkFeatureHasOnlyValidAttributes(
92        Feature feature, TMFeatureType tmFeatureType, AppLayerSettings appLayerSettings) {
93  
94      if (!getNonHiddenAttributeNames(tmFeatureType, appLayerSettings)
95          .containsAll(feature.getAttributes().keySet())) {
96        throw new ResponseStatusException(
97            HttpStatus.BAD_REQUEST,
98            "Feature cannot be edited, one or more requested attributes are not available on the feature type");
99      }
100     if (!Collections.disjoint(
101         getReadOnlyAttributes(tmFeatureType, appLayerSettings), feature.getAttributes().keySet())) {
102       throw new ResponseStatusException(
103           HttpStatus.BAD_REQUEST,
104           "Feature cannot be edited, one or more requested attributes are not editable on the feature type");
105     }
106   }

108   @Transactional
109   @PostMapping(
110       consumes = MediaType.APPLICATION_JSON_VALUE,
111       produces = MediaType.APPLICATION_JSON_VALUE)
112   @Timed(value = "create_feature", description = "time spent to process create feature call")
113   @Counted(value = "create_feature", description = "number of create feature calls")
114   public ResponseEntity<Serializable> createFeature(
115       @ModelAttribute AppTreeLayerNode appTreeLayerNode,
116       @ModelAttribute GeoService service,
117       @ModelAttribute GeoServiceLayer layer,
118       @ModelAttribute Application application,
119       @RequestBody Feature completeFeature) {
120 
121     checkAuthentication();
122 
123     TMFeatureType tmFeatureType =
124         getEditableFeatureType(application, appTreeLayerNode, service, layer);
125     Map<String, Object> attributesMap = completeFeature.getAttributes();
126 
127     AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
128 
129     checkFeatureHasOnlyValidAttributes(completeFeature, tmFeatureType, appLayerSettings);
130 
131     Feature newFeature;
132     SimpleFeatureSource fs = null;
133 
134     try (Transaction transaction = new DefaultTransaction("create")) {
135       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
136 
137       SimpleFeature simpleFeature;
138       SimpleFeatureBuilder simpleFeatureBuilder = new SimpleFeatureBuilder(fs.getSchema());
139       if (null != completeFeature.getFid() && !completeFeature.getFid().isEmpty()) {
140         simpleFeature = simpleFeatureBuilder.buildFeature(completeFeature.getFid());
141         simpleFeature.getUserData().put(Hints.USE_PROVIDED_FID, Boolean.TRUE);
142       } else {
143         simpleFeature = simpleFeatureBuilder.buildFeature(null);
144       }
145 
146       handleGeometryAttributesInput(
147           tmFeatureType, appLayerSettings, completeFeature, attributesMap, application, fs);
148       for (Map.Entry<String, Object> entry : attributesMap.entrySet()) {
149         simpleFeature.setAttribute(entry.getKey(), entry.getValue());
150       }
151 
152       if (fs instanceof SimpleFeatureStore) {
153         ((SimpleFeatureStore) fs).setTransaction(transaction);
154         List<FeatureId> newFids =
155             ((SimpleFeatureStore) fs).addFeatures(DataUtilities.collection(simpleFeature));
156 
157         transaction.commit();
158         // find the created feature to return
159         newFeature = getFeature(fs, ff.id(newFids.get(0)), application);
160       } else {
161         throw new ResponseStatusException(
162             HttpStatus.BAD_REQUEST, "Feature cannot be added, datasource is not editable");
163       }
164     } catch (RuntimeException | IOException | FactoryException e) {
165       // either opening datastore, modify or transaction failed
166       logger.error("Error creating new feature {}", completeFeature, e);
167       String message = e.getMessage();
168       if (null != e.getCause() && null != e.getCause().getMessage()) {
169         message = e.getCause().getMessage();
170       }
171       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, message, e);
172     } finally {
173       if (fs != null) {
174         fs.getDataStore().dispose();
175       }
176     }
177     return new ResponseEntity<>(newFeature, HttpStatus.OK);
178   }

180   @Transactional
181   @PatchMapping(
182       consumes = MediaType.APPLICATION_JSON_VALUE,
183       produces = MediaType.APPLICATION_JSON_VALUE,
184       path = "/{fid}")
185   @Timed(value = "update_feature", description = "time spent to process patch feature call")
186   @Counted(value = "update_feature", description = "number of patch feature calls")
187   public ResponseEntity<Serializable> patchFeature(
188       @ModelAttribute AppTreeLayerNode appTreeLayerNode,
189       @ModelAttribute GeoService service,
190       @ModelAttribute GeoServiceLayer layer,
191       @ModelAttribute Application application,
192       @PathVariable String fid,
193       @RequestBody Feature partialFeature) {
194 
195     checkAuthentication();
196 
197     TMFeatureType tmFeatureType =
198         getEditableFeatureType(application, appTreeLayerNode, service, layer);
199     AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
200 
201     Map<String, Object> attributesMap = partialFeature.getAttributes();
202 
203     checkFeatureHasOnlyValidAttributes(partialFeature, tmFeatureType, appLayerSettings);
204 
205     Feature patchedFeature;
206     SimpleFeatureSource fs = null;
207     try (Transaction transaction = new DefaultTransaction("edit")) {
208       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
209       // find feature to update
210       final Filter filter = ff.id(ff.featureId(fid));
211       if (!fs.getFeatures(filter).isEmpty() && fs instanceof SimpleFeatureStore) {
212         handleGeometryAttributesInput(
213             tmFeatureType, appLayerSettings, partialFeature, attributesMap, application, fs);
214         // NOTE geotools does not report back that the feature was updated, no error === success
215         ((SimpleFeatureStore) fs).setTransaction(transaction);
216         ((SimpleFeatureStore) fs)
217             .modifyFeatures(
218                 attributesMap.keySet().toArray(new String[] {}),
219                 attributesMap.values().toArray(),
220                 filter);
221         transaction.commit();
222         // find the updated feature to return
223         patchedFeature = getFeature(fs, filter, application);
224       } else {
225         throw new ResponseStatusException(
226             HttpStatus.BAD_REQUEST,
227             "Feature cannot be edited, it does not exist or is not editable");
228       }
229     } catch (RuntimeException | IOException | FactoryException e) {
230       // either opening datastore, modify or transaction failed
231       logger.error("Error patching feature {}", partialFeature, e);
232       String message = e.getMessage();
233       if (null != e.getCause() && null != e.getCause().getMessage()) {
234         message = e.getCause().getMessage();
235       }
236       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, message, e);
237     } finally {
238       if (fs != null) {
239         fs.getDataStore().dispose();
240       }
241     }
242     return new ResponseEntity<>(patchedFeature, HttpStatus.OK);
243   }

245   @Transactional
246   @DeleteMapping(path = "/{fid}")
247   @Timed(value = "delete_feature", description = "time spent to process delete feature call")
248   @Counted(value = "delete_feature", description = "number of delete feature calls")
249   public ResponseEntity<Void> deleteFeature(
250       @ModelAttribute AppTreeLayerNode appTreeLayerNode,
251       @ModelAttribute GeoService service,
252       @ModelAttribute GeoServiceLayer layer,
253       @ModelAttribute Application application,
254       @PathVariable String fid) {
255 
256     checkAuthentication();
257 
258     TMFeatureType tmFeatureType =
259         getEditableFeatureType(application, appTreeLayerNode, service, layer);
260 
261     SimpleFeatureSource fs = null;
262     try (Transaction transaction = new DefaultTransaction("delete")) {
263       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
264       Filter filter = ff.id(ff.featureId(fid));
265       if (fs instanceof FeatureStore) {
266         // NOTE geotools does not report back that the feature does not exist, nor the number of
267         // deleted features, no error === success
268         ((FeatureStore<?, ?>) fs).setTransaction(transaction);
269         ((FeatureStore<?, ?>) fs).removeFeatures(filter);
270         transaction.commit();
271       } else {
272         throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Layer cannot be edited");
273       }
274     } catch (IOException e) {
275       // either opening datastore or commit failed
276       logger.error("Error deleting feature {}", fid, e);
277       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, e.getMessage(), e);
278     } finally {
279       if (fs != null) {
280         fs.getDataStore().dispose();
281       }
282     }
283 
284     return new ResponseEntity<>(HttpStatus.NO_CONTENT);
285   }

287   private void checkAuthentication() {
288     Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
289     boolean isAuthenticated =
290         authentication != null && !(authentication instanceof AnonymousAuthenticationToken);
291     if (!isAuthenticated) {
292       throw new ResponseStatusException(HttpStatus.UNAUTHORIZED, "Not authenticated");
293     }
294   }

296   /**
297    * Get editable feature type, throws exception if not found or not editable. Will throw a {@link
298    * ResponseStatusException} if the layer does not have an editable featuretype.
299    *
300    * @param application the application that has the editable layer
301    * @param appTreeLayerNode the layer to edit
302    * @param service the service that has the layer
303    * @param layer the layer to edit
304    * @return the editable feature type
305    */
306   private TMFeatureType getEditableFeatureType(
307       Application application,
308       AppTreeLayerNode appTreeLayerNode,
309       GeoService service,
310       GeoServiceLayer layer) {
311 
312     if (null == layer) {
313       throw new ResponseStatusException(
314           HttpStatus.NOT_FOUND, "Cannot find layer " + appTreeLayerNode);
315     }
316 
317     AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
318     if (!Boolean.TRUE.equals(appLayerSettings.getEditable())) {
319       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Layer is not editable");
320     }
321 
322     TMFeatureType tmft = service.findFeatureTypeForLayer(layer, featureSourceRepository);
323     if (null == tmft) {
324       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Layer does not have feature type");
325     }
326     if (!tmft.isWriteable()) {
327       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Feature type is not writeable");
328     }
329 
330     return tmft;
331   }

333   private static Feature getFeature(SimpleFeatureSource fs, Filter filter, Application application)
334       throws IOException, FactoryException {
335     Feature modelFeature = null;
336     try (SimpleFeatureIterator feats = fs.getFeatures(filter).features()) {
337       if (feats.hasNext()) {
338         SimpleFeature simpleFeature = feats.next();
339         modelFeature =
340             new Feature()
341                 .geometry(
342                     GeometryProcessor.processGeometry(
343                         simpleFeature.getDefaultGeometry(),
344                         false,
345                         true,
346                         TransformationUtil.getTransformationToApplication(application, fs)))
347                 .fid(simpleFeature.getID());
348         for (AttributeDescriptor att : simpleFeature.getFeatureType().getAttributeDescriptors()) {
349           Object value = simpleFeature.getAttribute(att.getName());
350           if (value instanceof Geometry) {
351             value =
352                 GeometryProcessor.transformGeometry(
353                     (Geometry) value,
354                     TransformationUtil.getTransformationToApplication(application, fs));
355             value = GeometryProcessor.geometryToWKT((Geometry) value);
356           }
357           modelFeature.putAttributesItem(att.getLocalName(), value);
358         }
359       }
360     }
361     return modelFeature;
362   }

364   /** Handle geometry attributes, essentially this is a WKT to Geometry conversion. */
365   private static void handleGeometryAttributesInput(
366       TMFeatureType tmFeatureType,
367       AppLayerSettings appLayerSettings,
368       Feature modelFeature,
369       Map<String, Object> attributesMap,
370       Application application,
371       SimpleFeatureSource fs)
372       throws FactoryException {
373 
374     final MathTransform transform =
375         TransformationUtil.getTransformationToDataSource(application, fs);
376     getNonHiddenAttributes(tmFeatureType, appLayerSettings).stream()
377         .filter(attr -> TMAttributeTypeHelper.isGeometry(attr.getType()))
378         .filter(attr -> modelFeature.getAttributes().containsKey(attr.getName()))
379         .forEach(
380             attr -> {
381               Geometry geometry =
382                   GeometryProcessor.wktToGeometry(
383                       (String) modelFeature.getAttributes().get(attr.getName()));
384               if (transform != null && geometry != null) {
385                 geometry.setSRID(
386                     Integer.parseInt(application.getCrs().substring("EPSG:".length())));
387                 if (logger.isTraceEnabled()) {
388                   logger.trace(
389                       "Transforming geometry {} from {} to {}",
390                       geometry.toText(),
391                       geometry.getSRID(),
392                       fs.getSchema().getCoordinateReferenceSystem().getIdentifiers());
393                 }
394                 geometry = GeometryProcessor.transformGeometry(geometry, transform);
395               }
396               attributesMap.put(attr.getName(), geometry);
397             });
398   }
