53  @AppRestController
54  @Validated
55  @RequestMapping(
56      path =
57          "${tailormap-api.base-path}/{viewerKind}/{viewerName}/layer/{appLayerId}/unique/{attributeName}",
58      produces = MediaType.APPLICATION_JSON_VALUE)
59  public class UniqueValuesController {
60    private static final Logger logger =
61        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
62  
63    private final FeatureSourceFactoryHelper featureSourceFactoryHelper;
64  
65    private final FeatureSourceRepository featureSourceRepository;
66  
67    @Value("${tailormap-api.unique.use_geotools_unique_function:true}")
68    private boolean useGeotoolsUniqueFunction;
69  
70    private final FilterFactory ff = CommonFactoryFinder.getFilterFactory(GeoTools.getDefaultHints());
71  
72    public UniqueValuesController(
73        FeatureSourceFactoryHelper featureSourceFactoryHelper,
74        FeatureSourceRepository featureSourceRepository) {
75      this.featureSourceFactoryHelper = featureSourceFactoryHelper;
76      this.featureSourceRepository = featureSourceRepository;
77    }
78  
79    @Transactional
80    @RequestMapping(method = {GET, POST})
81    @Timed(
82        value = "get_unique_attributes",
83        description = "time spent to process get unique attributes call")
84    public ResponseEntity<Serializable> getUniqueAttributes(
85        @ModelAttribute GeoService service,
86        @ModelAttribute GeoServiceLayer layer,
87        @ModelAttribute Application app,
88        @ModelAttribute AppTreeLayerNode appTreeLayerNode,
89        @PathVariable("attributeName") String attributeName,
90        @RequestParam(required = false) String filter) {
91      if (StringUtils.isBlank(attributeName)) {
92        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Attribute name is required");
93      }
94  
95      TMFeatureType tmft = service.findFeatureTypeForLayer(layer, featureSourceRepository);
96      AppLayerSettings appLayerSettings = app.getAppLayerSettings(appTreeLayerNode);
97      if (tmft == null) {
98        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "Layer does not have feature type");
99      }
100     if (!getNonHiddenAttributeNames(tmft, appLayerSettings).contains(attributeName)) {
101       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Attribute does not exist");
102     }
103     UniqueValuesResponse uniqueValuesResponse = getUniqueValues(tmft, attributeName, filter);
104     return ResponseEntity.status(HttpStatus.OK).body(uniqueValuesResponse);
105   }
106 
107   private UniqueValuesResponse getUniqueValues(
108       TMFeatureType tmft, String attributeName, String filter) {
109     final UniqueValuesResponse uniqueValuesResponse =
110         new UniqueValuesResponse().filterApplied(false);
111     SimpleFeatureSource fs = null;
112     try {
113       Filter existingFilter = null;
114       if (null != filter) {
115         existingFilter = ECQL.toFilter(filter);
116       }
117       logger.trace("existingFilter: {}", existingFilter);
118 
119       Filter notNull = ff.not(ff.isNull(ff.property(attributeName)));
120       Filter f = notNull;
121       if (null != existingFilter) {
122         f = ff.and(notNull, existingFilter);
123         uniqueValuesResponse.filterApplied(true);
124       }
125 
126       Query q = new Query(tmft.getName(), f);
127       q.setPropertyNames(attributeName);
128       q.setSortBy(ff.sort(attributeName, SortOrder.ASCENDING));
129       logger.trace("Unique values query: {}", q);
130 
131       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmft);
132       // and then there are 2 scenarios:
133       // there might be a performance benefit for one or the other
134       if (!useGeotoolsUniqueFunction) {
135         // #1 use a feature visitor to get the unique values
136         // not recommended, as it may not be performant
137         logger.trace("Using feature visitor to get unique values");
138         fs.getFeatures(q)
139             .accepts(
140                 feature ->
141                     uniqueValuesResponse.addValuesItem(
142                         feature.getProperty(attributeName).getValue()),
143                 null);
144       } else {
145         // #2 or use a Function to get the unique values
146         // this is the recommended way, uses SQL "distinct"
147         logger.trace("Using geotools unique collection function to get unique values");
148         Function unique = ff.function("Collection_Unique", ff.property(attributeName));
149         Object o = unique.evaluate(fs.getFeatures(q));
150         if (o instanceof Set) {
151           @SuppressWarnings("unchecked")
152           Set<Object> uniqueValues = (Set<Object>) o;
153           uniqueValuesResponse.setValues(new TreeSet<>(uniqueValues));
154         }
155       }
156     } catch (CQLException e) {
157       logger.error("Could not parse requested filter", e);
158       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Could not parse requested filter");
159     } catch (IOException e) {
160       logger.error("Could not retrieve attribute data", e);
161     } finally {
162       if (fs != null) {
163         fs.getDataStore().dispose();
164       }
165     }
166 
167     return uniqueValuesResponse;
168   }
