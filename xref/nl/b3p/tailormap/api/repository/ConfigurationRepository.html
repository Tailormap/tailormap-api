6   package nl.b3p.tailormap.api.repository;
7   
8   import java.util.Optional;
9   import nl.b3p.tailormap.api.persistence.Configuration;
10  import nl.b3p.tailormap.api.security.annotation.PreAuthorizeAdmin;
11  import org.springframework.data.jpa.repository.JpaRepository;
12  import org.springframework.data.rest.core.annotation.RepositoryRestResource;
13  import org.springframework.security.access.prepost.PreAuthorize;
14  
15  @PreAuthorizeAdmin
16  @RepositoryRestResource(
17      path = "configs",
18      collectionResourceRel = "configs",
19      itemResourceRel = "config")
20  public interface ConfigurationRepository extends JpaRepository<Configuration, String> {
21    @PreAuthorize("permitAll()")
22    Optional<Configuration> findByKey(String key);
23  
24    @PreAuthorize("permitAll()")
25    default String get(String key) {
26      return get(key, null);
27    }
28  
29    @PreAuthorize("permitAll()")
30    default String get(String key, String defaultValue) {
31      return findByKey(key).map(Configuration::getValue).orElse(defaultValue);
32    }
33  
34    void deleteConfigurationByKey(String configKey);
35  }
