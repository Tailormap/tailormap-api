6   package nl.b3p.tailormap.api.repository.events;
7   
8   import nl.b3p.tailormap.api.persistence.OIDCConfiguration;
9   import nl.b3p.tailormap.api.security.OIDCRepository;
10  import org.springframework.data.rest.core.annotation.HandleAfterCreate;
11  import org.springframework.data.rest.core.annotation.HandleAfterDelete;
12  import org.springframework.data.rest.core.annotation.HandleAfterSave;
13  import org.springframework.data.rest.core.annotation.HandleBeforeCreate;
14  import org.springframework.data.rest.core.annotation.HandleBeforeSave;
15  import org.springframework.data.rest.core.annotation.RepositoryEventHandler;
16  import org.springframework.stereotype.Component;
17  
18  @RepositoryEventHandler
19  @Component
20  public class OIDCConfigurationEventHandler {
21    private final OIDCRepository oidcRepository;
22  
23    public OIDCConfigurationEventHandler(OIDCRepository oidcRepository) {
24      this.oidcRepository = oidcRepository;
25    }
26  
27    @HandleBeforeCreate
28    @HandleBeforeSave
29    public void handleBeforeCreateOrSave(OIDCConfiguration configuration) {
30      // If the user provided a "full" OIDC discovery URL, strip it.
31      if (configuration.getIssuerUrl().endsWith("/.well-known/openid-configuration")) {
32        String issuerUrl = configuration.getIssuerUrl();
33        issuerUrl =
34            issuerUrl.substring(0, issuerUrl.lastIndexOf("/.well-known/openid-configuration"));
35        configuration.setIssuerUrl(issuerUrl);
36      }
37    }
38  
39    @HandleAfterCreate
40    @HandleAfterSave
41    @HandleAfterDelete
42    public void handleAfterCreateOrSaveOrDelete(OIDCConfiguration configuration) {
43      oidcRepository.synchronize();
44    }
45  }
