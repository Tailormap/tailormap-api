6   package nl.b3p.tailormap.api;
7   
8   import io.micrometer.core.instrument.Tag;
9   import io.micrometer.core.instrument.Tags;
10  
11  import nl.tailormap.viewer.config.app.Application;
12  import nl.tailormap.viewer.config.app.ApplicationLayer;
13  import nl.tailormap.viewer.config.services.GeoService;
14  import nl.tailormap.viewer.config.services.Layer;
15  import nl.tailormap.viewer.config.services.SimpleFeatureType;
16  
17  import java.util.ArrayList;
18  import java.util.List;
19  import java.util.stream.Collectors;
20  
21  public class MicrometerHelper {
22      public static Tags getTags(Object... objects) {
23          List<Tag> tags = new ArrayList<>();
24          for (Object o : objects) {
25              if (o != null) {
26                  if (o instanceof Application) {
27                      tags.add(Tag.of("appId", ((Application) o).getId() + ""));
28                  } else if (o instanceof ApplicationLayer) {
29                      tags.add(Tag.of("appLayerId", ((ApplicationLayer) o).getId() + ""));
30                  } else if (o instanceof GeoService) {
31                      tags.add(Tag.of("serviceId", ((GeoService) o).getId() + ""));
32                  } else if (o instanceof Layer) {
33                      tags.add(Tag.of("serviceLayerName", ((Layer) o).getName()));
34                  } else if (o instanceof SimpleFeatureType) {
35                      SimpleFeatureType featureType = (SimpleFeatureType) o;
36                      tags.add(Tag.of("featureTypeName", featureType.getTypeName()));
37                      tags.add(Tag.of("featureSourceUrl", featureType.getFeatureSource().getUrl()));
38                  }
39              }
40          }
41          return Tags.of(tags);
42      }
43  
44      public static String tagsToString(Tags tags) {
45          return tags.stream()
46                  .map(tag -> tag.getKey() + '=' + tag.getValue())
47                  .collect(Collectors.joining(","));
48      }
49  }
