8   import javax.validation.constraints.NotNull;
9   import nl.b3p.tailormap.api.persistence.Application;
10  import org.geotools.api.data.SimpleFeatureSource;
11  import org.geotools.api.referencing.FactoryException;
12  import org.geotools.api.referencing.crs.CoordinateReferenceSystem;
13  import org.geotools.api.referencing.operation.MathTransform;
14  import org.geotools.referencing.CRS;
15  
16  public class TransformationUtil {
17    private TransformationUtil() {
18      // utility class
19    }
20  
21    /**
22     * Determine whether we need to transform geometries to the application CRS. Note that this uses
23     * the "default geometry" attribute of the feature source, in cases where a feature source has
24     * multiple geometry attributes (with possibly different CRSs) or heterogenous CRSs across a
25     * single geometry attribute this may not be accurate.
26     *
27     * @param application the referenced application
28     * @param simpleFeatureSource the feature source used in the application
29     * @return {@code null} when no transform is required and a valid transform otherwise
30     * @throws FactoryException when the CRS cannot be decoded
31     */
32    public static MathTransform getTransformationToApplication(
33        @NotNull Application application, @NotNull SimpleFeatureSource simpleFeatureSource)
34        throws FactoryException {
35      MathTransform transform = null;
36      // this is the CRS of the "default geometry" attribute
37      final CoordinateReferenceSystem dataSourceCRS =
38          simpleFeatureSource.getSchema().getCoordinateReferenceSystem();
39      final CoordinateReferenceSystem appCRS = CRS.decode(application.getCrs());
40      if (!CRS.equalsIgnoreMetadata(dataSourceCRS, appCRS)) {
41        transform = CRS.findMathTransform(dataSourceCRS, appCRS);
42      }
43      return transform;
44    }
