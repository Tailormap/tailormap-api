7   package nl.b3p.tailormap.api.geotools;
8   
9   import java.io.ByteArrayInputStream;
10  import javax.xml.XMLConstants;
11  import javax.xml.parsers.DocumentBuilder;
12  import javax.xml.parsers.DocumentBuilderFactory;
13  import org.apache.commons.lang3.StringUtils;
14  import org.w3c.dom.Document;
15  import org.w3c.dom.Element;
16  import org.w3c.dom.Node;
17  import org.w3c.dom.NodeList;
18  
19  public class WMSServiceExceptionUtil {
20  
21    /**
22     * Tries to parse a WMS service exception XML response of any WMS version and extract an error
23     * message.
24     *
25     * @param body The response body from a text/xml WMS response
26     * @return A service exception message or null if it can't be extracted
27     */
28    public static String tryGetServiceExceptionMessage(byte[] body) {
29      try {
30        DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
31        documentBuilderFactory.setNamespaceAware(true);
32        documentBuilderFactory.setExpandEntityReferences(false);
33        documentBuilderFactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
34        documentBuilderFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
35        DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
36        Document doc = builder.parse(new ByteArrayInputStream(body));
37  
38        String code = null;
39        String locator = null;
40        String text = null;
41  
42        if ("ServiceExceptionReport".equals(doc.getDocumentElement().getNodeName())) {
43          // WMS 1.3.0
44  
45          NodeList exceptions = doc.getDocumentElement().getElementsByTagName("ServiceException");
46          if (exceptions.getLength() > 0) {
47            Element exception = (Element) exceptions.item(0);
48            code = exception.getAttribute("code");
49            locator = exception.getAttribute("locator");
50            text = exception.getTextContent().trim();
51          }
52        } else if (doc.getDocumentElement().getNodeName().contains(":ExceptionReport")) {
53          // WMS 1.0.0 and 1.1.0
54  
55          NodeList children = doc.getDocumentElement().getChildNodes();
56          Element exception = null;
57          for (int i = 0; i < children.getLength(); i++) {
58            if (children.item(i).getNodeType() == Node.ELEMENT_NODE) {
59              exception = (Element) children.item(i);
60              break;
61            }
62          }
63          if (exception != null) {
64            code = exception.getAttribute("exceptionCode");
65            locator = exception.getAttribute("locator");
66            children = exception.getChildNodes();
67            for (int i = 0; i < children.getLength(); i++) {
68              if (children.item(i).getNodeType() == Node.ELEMENT_NODE) {
69                text = children.item(i).getTextContent().trim();
70                break;
71              }
72            }
73          }
74        }
75  
76        if (text != null) {
77          if (StringUtils.isNotBlank(locator)) {
78            text = "locator: " + locator + ": " + text;
79          }
80          if (StringUtils.isNotBlank(code)) {
81            text = "code: " + code + ": " + text;
82          }
83          return text;
84        } else {
85          return null;
86        }
87      } catch (Exception e) {
88        return null;
89      }
90    }
91  }
