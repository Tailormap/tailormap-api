6   package nl.b3p.tailormap.api.geotools.wfs;
7   
8   import static nl.b3p.tailormap.api.util.HttpProxyUtil.addForwardedForRequestHeaders;
9   import static nl.b3p.tailormap.api.util.HttpProxyUtil.passthroughRequestHeaders;
10  import static nl.b3p.tailormap.api.util.HttpProxyUtil.setHttpBasicAuthenticationHeader;
11  
12  import java.io.InputStream;
13  import java.net.URI;
14  import java.net.http.HttpClient;
15  import java.net.http.HttpRequest;
16  import java.net.http.HttpResponse;
17  import java.util.Set;
18  import javax.servlet.http.HttpServletRequest;
19  
20  public class WFSProxy {
21    public static HttpResponse<InputStream> proxyWfsRequest(
22        URI wfsRequest, String username, String password, HttpServletRequest request)
23        throws Exception {
24      final HttpClient httpClient = HttpClient.newBuilder().build();
25  
26      HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(wfsRequest);
27  
28      addForwardedForRequestHeaders(requestBuilder, request);
29  
30      // Just a few headers for logging, conditional or range requests not likely to be supported by a
31      // WFS
32      passthroughRequestHeaders(requestBuilder, request, Set.of("Referer", "User-Agent"));
33  
34      setHttpBasicAuthenticationHeader(requestBuilder, username, password);
35  
36      return httpClient.send(requestBuilder.build(), HttpResponse.BodyHandlers.ofInputStream());
37    }
38  }
