79    public static URI getOGCRequestURL(
80        String url,
81        String service,
82        String version,
83        String request,
84        MultiValueMap<String, String> parameters) {
85  
86      final MultiValueMap<String, String> params = new LinkedMultiValueMap<>();
87      params.add("SERVICE", service);
88      params.add("VERSION", version);
89      params.add("REQUEST", request);
90      if (parameters != null) {
91        // We need to encode the parameters manually because UriComponentsBuilder annoyingly does not
92        // encode '+' as used in mime types for output formats, see
93        // https://stackoverflow.com/questions/18138011
94        parameters.replaceAll(
95            (key, values) ->
96                values.stream()
97                    .map(s -> URLEncoder.encode(s, StandardCharsets.UTF_8))
98                    .collect(Collectors.toList()));
99        params.addAll(parameters);
100     }
101     return UriComponentsBuilder.fromHttpUrl(url).replaceQueryParams(params).build(true).toUri();
102   }

114   public static List<String> getOutputFormats(
115       String wfsUrl, String typeName, String username, String password, HttpClient httpClient)
116       throws Exception {
117 
118     URI wfsGetCapabilities = getWFSRequestURL(wfsUrl, "GetCapabilities");
119 
120     HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(wfsGetCapabilities);
121 
122     setHttpBasicAuthenticationHeader(requestBuilder, username, password);
123 
124     HttpResponse<InputStream> response =
125         httpClient.send(requestBuilder.build(), HttpResponse.BodyHandlers.ofInputStream());
126 
127     // Parse capabilities in DOM
128 
129     DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
130     documentBuilderFactory.setNamespaceAware(true);
131     documentBuilderFactory.setExpandEntityReferences(false);
132     documentBuilderFactory.setValidating(false);
133     documentBuilderFactory.setXIncludeAware(false);
134     documentBuilderFactory.setCoalescing(true);
135     documentBuilderFactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
136     DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
137     Document doc = documentBuilder.parse(response.body());
138 
139     // WFS 1.1.0 and WFS 2.0.0 use different namespaces, but the same local element names
140     boolean wfs2 = "2.0.0".equals(doc.getDocumentElement().getAttribute("version"));
141     XPath xPath = XPathFactory.newInstance().newXPath();
142     SimpleNamespaceContext namespaceContext = new SimpleNamespaceContext();
143     namespaceContext.bindNamespaceUri("ows", "http://www.opengis.net/ows" + (wfs2 ? "/1.1" : ""));
144     namespaceContext.bindNamespaceUri("wfs", "http://www.opengis.net/wfs" + (wfs2 ? "/2.0" : ""));
145     xPath.setNamespaceContext(namespaceContext);
146 
147     List<String> outputFormats = null;
148 
149     // Find all feature types and loop through to find the typeName. We can't use XPath with the
150     // typeName parameter in it to find the exact FeatureType because escaping characters like '
151     // in the typeName is not possible.
152     NodeList featureTypes =
153         (NodeList)
154             xPath
155                 .compile("/wfs:WFS_Capabilities" + "/wfs:FeatureTypeList" + "/wfs:FeatureType")
156                 .evaluate(doc, XPathConstants.NODESET);
157 
158     for (int i = 0; i < featureTypes.getLength(); i++) {
159       Element n = (Element) featureTypes.item(i);
160       Element name = DomUtils.getChildElementByTagName(n, "Name");
161       if (name != null && typeName.equals(DomUtils.getTextValue(name))) {
162         Element formatsNode = DomUtils.getChildElementByTagName(n, "OutputFormats");
163         if (formatsNode != null) {
164           outputFormats =
165               DomUtils.getChildElementsByTagName(formatsNode, "Format").stream()
166                   .map(DomUtils::getTextValue)
167                   .collect(Collectors.toList());
168         }
169         break;
170       }
171     }
172 
173     // No output formats found (or maybe not even the featureType...), return output formats from
174     // OperationsMetadata
175     if (outputFormats == null) {
176       String xpathExpr =
177           "/wfs:WFS_Capabilities"
178               + "/ows:OperationsMetadata"
179               + "/ows:Operation[@name='GetFeature']"
180               + "/ows:Parameter[@name='outputFormat']"
181               + "//ows:Value";
182       NodeList nodes = (NodeList) xPath.compile(xpathExpr).evaluate(doc, XPathConstants.NODESET);
183 
184       outputFormats = new ArrayList<>();
185       for (int i = 0; i < nodes.getLength(); i++) {
186         outputFormats.add(DomUtils.getTextValue((Element) nodes.item(i)));
187       }
188     }
189 
190     return outputFormats;
191   }

193   public static WebMapServer getWebMapServer(String url, String username, String password)
194       throws IOException, ServiceException {
195     HTTPClient client = new SimpleHttpClient();
196     client.setUser(username);
197     client.setPassword(password);
198     client.setConnectTimeout(TIMEOUT);
199     client.setReadTimeout(TIMEOUT);
200     return new WebMapServer(new URL(url), client);
201   }

208   public static Map<String, SimpleWFSLayerDescription> describeWMSLayers(
209       String url, String username, String password, List<String> layers) {
210     try {
211       WebMapServer wms = getWebMapServer(url, username, password);
212       // Directly create WMS 1.1.1 request. Creating it from WebMapServer errors with GeoServer
213       // about unsupported request in capabilities unless we override WebMapServer to set up
214       // specifications.
215       DescribeLayerRequest describeLayerRequest =
216           new WMS1_1_1().createDescribeLayerRequest(new URL(url));
217       // XXX Otherwise GeoTools will send VERSION=1.1.0...
218       describeLayerRequest.setProperty("VERSION", "1.1.1");
219       describeLayerRequest.setLayers(String.join(",", layers));
220       // GeoTools will throw a ClassCastException when a WMS ServiceException is returned
221       DescribeLayerResponse describeLayerResponse = wms.issueRequest(describeLayerRequest);
222 
223       Map<String, SimpleWFSLayerDescription> descriptions = new HashMap<>();
224       for (LayerDescription ld : describeLayerResponse.getLayerDescs()) {
225         String wfsUrl = getWfsUrl(ld, wms);
226 
227         if (wfsUrl != null && ld.getQueries() != null && ld.getQueries().length != 0) {
228           descriptions.put(ld.getName(), new SimpleWFSLayerDescription(wfsUrl, ld.getQueries()));
229         }
230       }
231       return Collections.unmodifiableMap(descriptions);
232     } catch (Exception e) {
233       String msg =
234           String.format(
235               "Error in DescribeLayer request to WMS \"%s\": %s: %s",
236               url, e.getClass(), e.getMessage());
237       if (logger.isTraceEnabled()) {
238         logger.trace(msg, e);
239       } else {
240         logger.debug(msg + ". Set log level to TRACE for stacktrace.");
241       }
242     }
243 
244     return Collections.emptyMap();
245   }

247   private static String getWfsUrl(LayerDescription ld, WebMapServer wms) {
248     String wfsUrl = (ld.getWfs() != null) ? ld.getWfs().toString() : null;
249     if (wfsUrl == null && "WFS".equalsIgnoreCase(ld.getOwsType())) {
250       wfsUrl = ld.getOwsURL().toString();
251     }
252     // OGC 02-070 Annex B says the wfs/owsURL attributed are not required but implied. Some
253     // Deegree instance encountered has all attributes empty, and apparently the meaning is that
254     // the WFS URL is the same as the WMS URL (not explicitly defined in the spec).
255     if (wfsUrl == null) {
256       wfsUrl = wms.getInfo().getSource().toString();
257     }
258     return wfsUrl;
259   }
