30  public class JDBCFeatureSourceHelper extends FeatureSourceHelper {
31    private static final Map<JDBCConnectionProperties.DbtypeEnum, Integer> defaultPorts =
32        Map.of(
33            JDBCConnectionProperties.DbtypeEnum.POSTGIS, 5432,
34            JDBCConnectionProperties.DbtypeEnum.ORACLE, 1521,
35            JDBCConnectionProperties.DbtypeEnum.SQLSERVER, 1433);
36  
37    @Override
38    public DataStore createDataStore(TMFeatureSource tmfs, Integer timeout) throws IOException {
39      if (tmfs.getProtocol() != TMFeatureSource.Protocol.JDBC) {
40        throw new IllegalArgumentException(tmfs.getProtocol().getValue());
41      }
42      Objects.requireNonNull(tmfs.getJdbcConnection());
43      Objects.requireNonNull(tmfs.getAuthentication());
44      if (tmfs.getAuthentication().getMethod() != ServiceAuthentication.MethodEnum.PASSWORD) {
45        throw new IllegalArgumentException(tmfs.getAuthentication().getMethod().getValue());
46      }
47  
48      JDBCConnectionProperties c = tmfs.getJdbcConnection();
49      Objects.requireNonNull(c.getDbtype());
50      String connectionOpts =
51          Optional.ofNullable(c.getAdditionalProperties().get("connectionOptions")).orElse("");
52      if (c.getDbtype() == JDBCConnectionProperties.DbtypeEnum.POSTGIS
53          && !connectionOpts.contains("ApplicationName")) {
54        connectionOpts =
55            connectionOpts
56                + (connectionOpts.contains("?") ? "&amp;" : "?")
57                + "ApplicationName=tailormap-api";
58      }
59  
60      Map<String, Object> params = new HashMap<>();
61      params.put(DBTYPE.key, c.getDbtype().getValue());
62      params.put(HOST.key, c.getHost());
63      params.put(PORT.key, c.getPort() != null ? c.getPort() : defaultPorts.get(c.getDbtype()));
64      params.put(DATABASE.key, c.getDatabase() + connectionOpts);
65      params.put(SCHEMA.key, c.getSchema());
66      params.put(USER.key, tmfs.getAuthentication().getUsername());
67      params.put(PASSWD.key, tmfs.getAuthentication().getPassword());
68      params.put(FETCHSIZE.key, c.getFetchSize());
69      params.put(EXPOSE_PK.key, true);
70      params.put(PK_METADATA_TABLE.key, c.getPrimaryKeyMetadataTable());
71      if (c.getDbtype() != JDBCConnectionProperties.DbtypeEnum.ORACLE) {
72        // this key is available in ao. Oracle and MS SQL datastore factories, but not in the common
73        // parent...
74        // we need this for mssql to determine a feature type on an empty table
75        params.put(GEOMETRY_METADATA_TABLE.key, "geometry_columns");
76      }
77      return openDatastore(params, PASSWD.key);
78    }
79  }
