6   package nl.b3p.tailormap.api.configuration;
7   
8   import jakarta.annotation.PostConstruct;
9   import java.io.File;
10  import java.io.IOException;
11  import java.io.PrintWriter;
12  import java.lang.invoke.MethodHandles;
13  import java.nio.charset.StandardCharsets;
14  import org.geotools.api.referencing.FactoryException;
15  import org.geotools.metadata.iso.citation.Citations;
16  import org.geotools.referencing.CRS;
17  import org.geotools.referencing.ReferencingFactoryFinder;
18  import org.geotools.referencing.factory.PropertyAuthorityFactory;
19  import org.geotools.referencing.factory.ReferencingFactoryContainer;
20  import org.geotools.referencing.wkt.Formattable;
21  import org.geotools.util.factory.Hints;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.context.annotation.Configuration;
25  
26  @Configuration
27  public class GeoToolsEPSGAliasesConfiguration {
28    private static final Logger logger =
29        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
30  
31    // Registering this unofficial EPSG code for web mercator is required to properly load a CRS like
32    // urn:ogc:def:crs:EPSG:6.3:900913 used by MapCache, for example here:
33    // https://www.openbasiskaart.nl/mapcache/wmts?REQUEST=GetCapabilities&SERVICE=WMTS
34    private static final int[][] EPSG_ALIASES = {{900913, 3857}};
35  
36    @PostConstruct
37    public void addEPSGAliases() throws IOException, FactoryException {
38      // The PropertyAuthorityFactory only takes a URL parameter, create a temporary file
39      File f = File.createTempFile("epsg", "properties");
40      try (PrintWriter writer = new PrintWriter(f, StandardCharsets.US_ASCII)) {
41        for (int[] alias : EPSG_ALIASES) {
42          writer.printf("%d=%s\n", alias[0], ((Formattable) CRS.decode("EPSG:" + alias[1])).toWKT(0));
43        }
44      }
45  
46      Hints hints = new Hints(Hints.CRS_AUTHORITY_FACTORY, PropertyAuthorityFactory.class);
47      ReferencingFactoryContainer referencingFactoryContainer =
48          ReferencingFactoryContainer.instance(hints);
49  
50      PropertyAuthorityFactory factory =
51          new PropertyAuthorityFactory(
52              referencingFactoryContainer, Citations.fromName("EPSG"), f.toURI().toURL());
53  
54      ReferencingFactoryFinder.addAuthorityFactory(factory);
55      ReferencingFactoryFinder.scanForPlugins();
56  
57      for (int[] alias : EPSG_ALIASES) {
58        logger.info("Added CRS alias to GeoTools: EPSG:{} -> EPSG:{}", alias[0], alias[1]);
59      }
60    }
61  }
