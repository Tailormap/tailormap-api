8   import jakarta.servlet.http.HttpServletRequest;
9   import java.nio.charset.StandardCharsets;
10  import java.util.List;
11  import java.util.Locale;
12  import nl.b3p.tailormap.api.persistence.Application;
13  import nl.b3p.tailormap.api.persistence.Configuration;
14  import nl.b3p.tailormap.api.repository.ApplicationRepository;
15  import nl.b3p.tailormap.api.repository.ConfigurationRepository;
16  import org.springframework.beans.factory.annotation.Value;
17  import org.springframework.context.annotation.Lazy;
18  import org.springframework.stereotype.Controller;
19  import org.springframework.web.bind.annotation.GetMapping;
20  import org.springframework.web.bind.annotation.PathVariable;
21  import org.springframework.web.servlet.i18n.AcceptHeaderLocaleResolver;
22  import org.springframework.web.util.UriUtils;
23  
24  /**
25   * Front controller which forwards requests to URLs created by the Angular routing module to the
26   * proper viewer or admin index.html of the Angular bundle, for example when the user refreshes the
27   * page. Unfortunately, a proper front controller would first check if the file which the requested
28   * URL maps to exists and only return the index.html file if it doesn't. That is quite complicated
29   * to do in Spring Boot so as a poor-mans alternative we explicitly match the specific router URLs
30   * navigated to by the Angular frontends which don't map to files from the bundle.
31   *
32   * <p>It could happen that a frontend developer adds a route and forgets to update this front
33   * controller, and it is only noticed later when someone presses F5 and gets a 404.
34   */
35  @Controller
36  public class FrontController {
37    // Hardcoded list for now. In the future scan the spring.web.resources.static-locations directory
38    // for subdirectories of locale-specific frontend bundles.
39    private static final AcceptHeaderLocaleResolver localeResolver = new AcceptHeaderLocaleResolver();
40  
41    static {
42      localeResolver.setSupportedLocales(
43          List.of(new Locale("en"), new Locale("nl"), new Locale("de")));
44      localeResolver.setDefaultLocale(localeResolver.getSupportedLocales().get(0));
45    }
46  
47    private final ConfigurationRepository configurationRepository;
48    private final ApplicationRepository applicationRepository;
49  
50    @Value("${spring.profiles.active:}")
51    private String activeProfile;
52  
53    public FrontController(
54        @Lazy ConfigurationRepository configurationRepository,
55        @Lazy ApplicationRepository applicationRepository) {
56      this.configurationRepository = configurationRepository;
57      this.applicationRepository = applicationRepository;
58    }
59  
60    @GetMapping(
61        value = {"/", "/login", "/app", "/app/", "/app/**", "/service/**", "/admin/**", "/ext/**"})
62    public String appIndex(HttpServletRequest request) {
63      String path = request.getRequestURI().substring(request.getContextPath().length());
64      Application app = null;
65  
66      if (!activeProfile.contains("static-only")) {
67        // The language setting of the (default) app takes precedence over the Accept-Language header
68        if ("/".equals(path) || "/app".equals(path) || "/app/".equals(path)) {
69          String defaultAppName = configurationRepository.get(Configuration.DEFAULT_APP);
70          app = applicationRepository.findByName(defaultAppName);
71        } else if (path.startsWith("/app/")) {
72          String[] parts = path.split("/");
73          if (parts.length > 2) {
74            String appName = UriUtils.decode(parts[2], StandardCharsets.UTF_8);
75            app = applicationRepository.findByName(appName);
76          }
77        }
78      }
79  
80      if (app != null && app.getSettings().getI18nSettings() != null) {
81        String appLanguage = app.getSettings().getI18nSettings().getDefaultLanguage();
82        if (appLanguage != null
83            && localeResolver.getSupportedLocales().stream()
84                .anyMatch(l -> l.toLanguageTag().equals(appLanguage))) {
85          return "forward:/" + appLanguage + "/index.html";
86        }
87      }
88  
89      // Resolve using Accept-Language header
90      Locale locale = localeResolver.resolveLocale(request);
91      return "forward:/" + locale.toLanguageTag() + "/index.html";
92    }
93  
94    @GetMapping(
95        value = {
96          "/{locale}/",
97          "/{locale}/login",
98          // Need to avoid matching /api/app/ etc
99          "/{locale:^(?!api)[a-zA-Z-]+}/app/**",
100         "/{locale:^(?!api)[a-zA-Z-]+}/service/**",
101         "/{locale:^(?!api)[a-zA-Z-]+}/admin/**",
102         "/{locale:^(?!api)[a-zA-Z-]+}/ext/**"
103       })
104   public String localePrefixedAppIndex(
105       @PathVariable("locale") String locale, HttpServletRequest request) {
106     if (localeResolver.getSupportedLocales().stream()
107         .anyMatch(l -> l.toLanguageTag().equals(locale))) {
108       return "forward:/" + locale + "/index.html";
109     }
110     return appIndex(request);
111   }
112 
113   @SuppressWarnings("SameReturnValue")
114   @GetMapping(value = {"/swagger-ui", "/swagger-ui/"})
115   public String swaggerUiWelcomePage() {
116     return "redirect:/swagger-ui/index.html";
117   }
118 }
