6   package nl.b3p.tailormap.api.configuration.dev;
7   
8   import org.springframework.boot.autoconfigure.flyway.FlywayMigrationStrategy;
9   import org.springframework.context.annotation.Bean;
10  import org.springframework.context.annotation.Configuration;
11  
12  @Configuration
13  public class CleanSchema {
14    @Bean
15    public FlywayMigrationStrategy flywayCleanMigrationStrategy() {
16      return flyway -> {
17        flyway.clean();
18        flyway.migrate();
19      };
20    }
21  }
