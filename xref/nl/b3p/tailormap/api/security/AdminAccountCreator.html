6   package nl.b3p.tailormap.api.security;
7   
8   import static org.apache.commons.lang3.StringUtils.isNotBlank;
9   
10  import java.io.IOException;
11  import java.lang.invoke.MethodHandles;
12  import java.nio.charset.StandardCharsets;
13  import java.util.List;
14  import java.util.Map;
15  import java.util.UUID;
16  import nl.b3p.tailormap.api.persistence.Group;
17  import nl.b3p.tailormap.api.persistence.User;
18  import nl.b3p.tailormap.api.repository.GroupRepository;
19  import nl.b3p.tailormap.api.repository.UserRepository;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.springframework.beans.factory.annotation.Value;
23  import org.springframework.boot.ansi.AnsiPropertySource;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
25  import org.springframework.boot.context.event.ApplicationReadyEvent;
26  import org.springframework.context.annotation.Configuration;
27  import org.springframework.context.event.EventListener;
28  import org.springframework.core.env.MapPropertySource;
29  import org.springframework.core.env.MutablePropertySources;
30  import org.springframework.core.env.PropertySourcesPropertyResolver;
31  import org.springframework.core.io.ClassPathResource;
32  import org.springframework.security.crypto.factory.PasswordEncoderFactories;
33  import org.springframework.security.crypto.password.PasswordEncoder;
34  import org.springframework.transaction.annotation.Transactional;
35  import org.springframework.util.StreamUtils;
36  
37  @Configuration
38  @ConditionalOnProperty(
39      name = "tailormap-api.security.admin.create-if-not-exists",
40      havingValue = "true")
41  public class AdminAccountCreator {
42    private static final Logger logger =
43        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
44  
45    @Value("${tailormap-api.security.admin.username}")
46    private String newAdminUsername;
47  
48    @Value("${tailormap-api.security.admin.hashed-password}")
49    private String newAdminHashedPassword;
50  
51    private final UserRepository userRepository;
52    private final GroupRepository groupRepository;
53  
54    private final PasswordEncoder passwordEncoder =
55        PasswordEncoderFactories.createDelegatingPasswordEncoder();
56  
57    public AdminAccountCreator(UserRepository userRepository, GroupRepository groupRepository) {
58      this.userRepository = userRepository;
59      this.groupRepository = groupRepository;
60    }
61  
62    @EventListener(ApplicationReadyEvent.class)
63    @Transactional
64    public void postConstruct() throws IOException {
65      logger.trace("Checking whether an admin account exists...");
66  
67      InternalAdminAuthentication.setInSecurityContext();
68      try {
69        if (!userRepository.existsByGroupsNameIn(List.of(Group.ADMIN))) {
70          if (isNotBlank(newAdminHashedPassword) && newAdminHashedPassword.startsWith("{bcrypt}")) {
71            createAdmin(newAdminUsername, newAdminHashedPassword);
72            logger.info(
73                "New admin account \"{}\" created with hashed password from environment",
74                newAdminUsername);
75          } else {
76            // Create a new admin account with a random generated password
77            String password = UUID.randomUUID().toString();
78            createAdmin(newAdminUsername, passwordEncoder.encode(password));
79            // Log generated password
80            logger.info(getAccountBanner(newAdminUsername, password));
81          }
82        }
83      } finally {
84        InternalAdminAuthentication.clearSecurityContextAuthentication();
85      }
86    }
87  
88    private void createAdmin(String username, String hashedPassword) {
89      User u = new User().setUsername(username).setPassword(hashedPassword);
90      u.getGroups().add(groupRepository.getReferenceById(Group.ADMIN));
91      userRepository.saveAndFlush(u);
92    }
93  
94    private static String getAccountBanner(String username, String password) throws IOException {
95      String accountBanner =
96          StreamUtils.copyToString(
97              new ClassPathResource("account-banner.txt").getInputStream(),
98              StandardCharsets.US_ASCII);
99      MutablePropertySources sources = new MutablePropertySources();
100     sources.addFirst(new AnsiPropertySource("ansi", true));
101     sources.addFirst(
102         new MapPropertySource(
103             "account",
104             Map.of(
105                 "username", username,
106                 "password", password)));
107     return new PropertySourcesPropertyResolver(sources).resolvePlaceholders(accountBanner);
108   }
109 }
