41  @Configuration
42  @EnableWebSecurity
43  @EnableMethodSecurity
44  public class ApiSecurityConfiguration {
45    @Value("${tailormap-api.base-path}")
46    private String apiBasePath;
47  
48    @Value("${tailormap-api.admin.base-path}")
49    private String adminApiBasePath;
50  
51    @Value("${tailormap-api.security.disable-csrf:false}")
52    private boolean disableCsrf;
53  
54    @Bean
55    public CookieCsrfTokenRepository csrfTokenRepository() {
56      // Spring CSRF protection requires an X-XSRF-TOKEN header read from the XSRF-TOKEN cookie by
57      // JavaScript so set HttpOnly to false. Angular has automatic XSRF protection support:
58      // https://angular.io/guide/http#security-xsrf-protection
59      CookieCsrfTokenRepository csrfTokenRepository = CookieCsrfTokenRepository.withHttpOnlyFalse();
60      // Allow cross-domain non-GET (unsafe) requests for embedding with an iframe
61      csrfTokenRepository.setCookieCustomizer(
62          cookieCustomizer -> {
63            // Do not set SameSite=None when testing with HTTP instead of HTTPS
64            // Ideally use HttpServletRequest.isSecure(), but look at built cookie for now...
65            if (cookieCustomizer.build().isSecure()) {
66              cookieCustomizer.sameSite(Cookie.SameSite.NONE.attributeValue());
67            }
68          });
69      return csrfTokenRepository;
70    }
71  
72    @Bean
73    public SecurityFilterChain apiFilterChain(
74        HttpSecurity http, CookieCsrfTokenRepository csrfTokenRepository) throws Exception {
75  
76      // Disable CSRF protection for development with HAL explorer
77      // https://github.com/spring-projects/spring-data-rest/issues/1347
78      if (disableCsrf) {
79        http.csrf(AbstractHttpConfigurer::disable);
80      } else {
81        // https://docs.spring.io/spring-security/reference/servlet/exploits/csrf.html#csrf-integration-javascript-spa
82        http.csrf(
83            csrf ->
84                csrf.csrfTokenRequestHandler(new CsrfTokenRequestAttributeHandler())
85                    .csrfTokenRepository(csrfTokenRepository)
86                    // This uses POST for large filter in body, but is safe (read-only)
87                    .ignoringRequestMatchers(
88                        apiBasePath + "/{viewerKind}/{viewerName}/layer/{appLayerId}/features"));
89        http.addFilterAfter(new CsrfCookieFilter(), BasicAuthenticationFilter.class);
90      }
91  
92      // Before redirecting the user to the OAuth2 authorization endpoint, store the requested
93      // redirect URL.
94      RedirectStrategy redirectStrategy =
95          new DefaultRedirectStrategy() {
96            @Override
97            public void sendRedirect(
98                HttpServletRequest request, HttpServletResponse response, String url)
99                throws IOException {
100             String redirectUrl = request.getParameter("redirectUrl");
101             if (redirectUrl != null && redirectUrl.startsWith("/")) {
102               request.getSession().setAttribute("redirectUrl", redirectUrl);
103             }
104             super.sendRedirect(request, response, url);
105           }
106         };
107 
108     // When OAuth2 authentication succeeds, use the redirect URL stored in the session to send them
109     // back.
110     AuthenticationSuccessHandler authenticationSuccessHandler =
111         (request, response, authentication) -> {
112           HttpSession session = request.getSession(false);
113           if (session != null) {
114             String redirectUrl = (String) session.getAttribute("redirectUrl");
115             if (redirectUrl != null) {
116               response.sendRedirect(redirectUrl);
117               return;
118             }
119           }
120           response.sendRedirect("/");
121         };
122 
123     http.securityMatchers(matchers -> matchers.requestMatchers(apiBasePath + "/**"))
124         .addFilterAfter(
125             /* (debug) log user making the request */ new AuditInterceptor(),
126             AnonymousAuthenticationFilter.class)
127         .authorizeHttpRequests(
128             authorize -> {
129               authorize.requestMatchers(adminApiBasePath + "/**").hasAuthority(Group.ADMIN);
130               authorize.requestMatchers(apiBasePath + "/**").permitAll();
131             })
132         .formLogin(
133             formLogin ->
134                 formLogin
135                     .loginPage(apiBasePath + "/unauthorized")
136                     .loginProcessingUrl(apiBasePath + "/login"))
137         .oauth2Login(
138             login ->
139                 login
140                     .authorizationEndpoint(
141                         endpoint ->
142                             endpoint
143                                 .baseUri(apiBasePath + "/oauth2/authorization")
144                                 .authorizationRedirectStrategy(redirectStrategy))
145                     .redirectionEndpoint(
146                         endpoint -> endpoint.baseUri(apiBasePath + "/oauth2/callback"))
147                     .successHandler(authenticationSuccessHandler))
148         .anonymous(anonymous -> anonymous.authorities(Group.ANONYMOUS))
149         .logout(
150             logout ->
151                 logout
152                     .logoutUrl(apiBasePath + "/logout")
153                     .logoutSuccessHandler(
154                         (request, response, authentication) ->
155                             response.sendError(HttpStatus.OK.value(), "OK")));
156     return http.build();
157   }
158 
159   @Bean
160   public OIDCRepository clientRegistrationRepository(OIDCConfigurationRepository repository) {
161     return new OIDCRepository(repository);
162   }
163 
164   @Bean
165   public GrantedAuthoritiesMapper userAuthoritiesMapper(GroupRepository repository) {
166     return (authorities) -> {
167       Set<GrantedAuthority> mappedAuthorities = new HashSet<>();
168       Set<String> wantedGroups = new HashSet<>();
169 
170       try {
171         authorities.forEach(
172             authority -> {
173               mappedAuthorities.add(authority);
174               if (authority instanceof OidcUserAuthority oidcUserAuthority) {
175                 OidcIdToken idToken = oidcUserAuthority.getIdToken();
176 
177                 List<String> roles = idToken.getClaimAsStringList("roles");
178                 if (roles != null) {
179                   wantedGroups.addAll(roles);
180                 }
181               }
182             });
183 
184         for (String groupName : wantedGroups) {
185           if (repository.findById(groupName).isEmpty()) {
186             Group group = new Group();
187             group.setName(groupName);
188             group.setDescription("<imported from SSO>");
189             repository.save(group);
190           }
191 
192           mappedAuthorities.add(new SimpleGrantedAuthority(groupName));
193         }
194       } catch (Exception e) {
195         // Ignore
196       }
197 
198       return mappedAuthorities;
199     };
200   }
