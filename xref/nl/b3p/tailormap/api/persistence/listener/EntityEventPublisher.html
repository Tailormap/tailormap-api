7   package nl.b3p.tailormap.api.persistence.listener;
8   
9   import static ch.rasc.sse.eventbus.SseEvent.DEFAULT_EVENT;
10  import static nl.b3p.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.ENTITY_CREATED;
11  import static nl.b3p.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.ENTITY_DELETED;
12  import static nl.b3p.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.ENTITY_UPDATED;
13  
14  import ch.rasc.sse.eventbus.SseEvent;
15  import ch.rasc.sse.eventbus.SseEventBus;
16  import com.fasterxml.jackson.databind.ObjectMapper;
17  import java.lang.invoke.MethodHandles;
18  import javax.persistence.EntityManagerFactory;
19  import javax.persistence.PostPersist;
20  import javax.persistence.PostRemove;
21  import javax.persistence.PostUpdate;
22  import nl.b3p.tailormap.api.admin.model.EntityEvent;
23  import nl.b3p.tailormap.api.admin.model.ServerSentEvent;
24  import nl.b3p.tailormap.api.persistence.TMFeatureType;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.beans.factory.annotation.Autowired;
28  import org.springframework.context.annotation.Lazy;
29  import org.springframework.data.rest.webmvc.config.RepositoryRestMvcConfiguration;
30  import org.springframework.stereotype.Component;
31  import org.springframework.web.context.request.RequestContextHolder;
32  
33  @Component
34  public class EntityEventPublisher {
35    private static final Logger logger =
36        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
37  
38    @Autowired @Lazy private EntityManagerFactory entityManagerFactory;
39  
40    @Autowired @Lazy private ObjectMapper objectMapper;
41  
42    @Autowired @Lazy private SseEventBus eventBus;
43  
44    @Autowired @Lazy private RepositoryRestMvcConfiguration repositoryRestMvcConfiguration;
45  
46    public EntityEventPublisher() {}
47  
48    private void sendEvent(
49        ServerSentEvent.EventTypeEnum eventTypeEnum, Object entity, boolean serializeEntity) {
50      Object id = null;
51  
52      if (RequestContextHolder.getRequestAttributes() == null) {
53        // No current request -- do not send events / serialize entities during app startup
54        return;
55      }
56  
57      try {
58        id = entityManagerFactory.getPersistenceUnitUtil().getIdentifier(entity);
59        EntityEvent entityEvent =
60            new EntityEvent().entityName(entity.getClass().getSimpleName()).id(String.valueOf(id));
61        if (serializeEntity) {
62          entityEvent.setObject(repositoryRestMvcConfiguration.objectMapper().valueToTree(entity));
63        }
64        ServerSentEvent event = new ServerSentEvent().eventType(eventTypeEnum).details(entityEvent);
65        this.eventBus.handleEvent(SseEvent.of(DEFAULT_EVENT, objectMapper.writeValueAsString(event)));
66      } catch (Exception e) {
67        logger.error(
68            "Error sending SSE for event type {}, entity {}, id {}",
69            eventTypeEnum,
70            entity != null ? entity.getClass().getSimpleName() : null,
71            id,
72            e);
73      }
74    }
75  
76    @PostPersist
77    public void postPersist(Object entity) {
78      // Feature types are created when TMFeatureSource is created, so do not send event
79      // Note that when TMFeatureSource created event is sent, the oneToMany allFeatureTypes
80      // have no ID's yet. So for a created feature source the frontend must retrieve the
81      // TMFeatureSource separately or use the return value of the POST request.
82      if (!(entity instanceof TMFeatureType)) {
83        sendEvent(ENTITY_CREATED, entity, true);
84      }
85    }
86  
87    @PostRemove
88    public void postRemove(Object entity) {
89      // Feature types are only deleted when refreshing the feature source, so the updated/deleted
90      // event for TMFeatureSource suffices.
91      if (!(entity instanceof TMFeatureType)) {
92        sendEvent(ENTITY_DELETED, entity, false);
93      }
94    }
95  
96    @PostUpdate
97    public void postUpdate(Object entity) {
98      // Note that for an updated TMFeatureSource, new TMFeatureTypes do appear to have ID's set.
99      sendEvent(ENTITY_UPDATED, entity, true);
100   }
101 }
