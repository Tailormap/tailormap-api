84    public Application getServiceApplication(
85        String baseAppName, String projection, GeoService service) {
86      if (baseAppName == null) {
87        baseAppName =
88            Optional.ofNullable(service.getSettings().getPublishing())
89                .map(ServicePublishingSettings::getBaseApp)
90                .orElseGet(() -> configurationRepository.get(Configuration.DEFAULT_BASE_APP));
91      }
92  
93      Application baseApp = null;
94      if (baseAppName != null) {
95        baseApp = applicationRepository.findByName(baseAppName);
96        if (baseApp != null) {
97          // Caller may be changing the app content to add layers from this service, detach so those
98          // aren't saved
99          entityManager.detach(baseApp);
100       }
101     }
102 
103     Application app =
104         baseApp != null ? baseApp : new Application().setContentRoot(new AppContent());
105 
106     if (projection != null) {
107       // TODO: filter layers by projection parameter (layer.crs must inherit crs from parent layers)
108       throw new UnsupportedOperationException("Projection filtering not yet supported");
109     } else {
110       if (baseApp != null) {
111         projection = baseApp.getCrs();
112       } else {
113         projection = DEFAULT_WEB_MERCATOR_CRS;
114       }
115     }
116 
117     app.setName(service.getId()).setTitle(service.getTitle()).setCrs(projection);
118 
119     return app;
120   }

122   @Transactional
123   public MapResponse toMapResponse(Application app) {
124     MapResponse mapResponse = new MapResponse();
125     setCrsAndBounds(app, mapResponse);
126     setLayers(app, mapResponse);
127     return mapResponse;
128   }

130   public void setCrsAndBounds(Application a, MapResponse mapResponse) {
131     CoordinateReferenceSystem gtCrs = a.getGeoToolsCoordinateReferenceSystem();
132     if (gtCrs == null) {
133       throw new IllegalArgumentException("Invalid CRS: " + a.getCrs());
134     }
135 
136     TMCoordinateReferenceSystem crs =
137         new TMCoordinateReferenceSystem()
138             .code(a.getCrs())
139             .definition(((Formattable) gtCrs).toWKT(0))
140             .bounds(GeoToolsHelper.fromCRS(gtCrs))
141             .unit(
142                 Optional.ofNullable(CRSUtilities.getUnit(gtCrs.getCoordinateSystem()))
143                     .map(Objects::toString)
144                     .orElse(null));
145 
146     Bounds maxExtent = a.getMaxExtent() != null ? a.getMaxExtent() : crs.getBounds();
147     Bounds initialExtent = a.getInitialExtent() != null ? a.getInitialExtent() : maxExtent;
148 
149     mapResponse.crs(crs).maxExtent(maxExtent).initialExtent(initialExtent);
150   }

156   private String getProxyUrl(
157       GeoService geoService, Application application, AppTreeLayerNode appTreeLayerNode) {
158     return linkTo(
159             GeoServiceProxyController.class,
160             Map.of(
161                 "viewerKind", "app", // XXX
162                 "viewerName", application.getName(),
163                 "appLayerId", appTreeLayerNode.getId(),
164                 "protocol", geoService.getProtocol().getValue()))
165         .toString();
166   }

179     public void buildLayers() {
180       if (app.getContentRoot() != null) {
181         buildBackgroundLayers();
182         buildOverlayLayers();
183       }
184     }

186     private void buildBackgroundLayers() {
187       if (app.getContentRoot().getBaseLayerNodes() != null) {
188         for (AppTreeNode node : app.getContentRoot().getBaseLayerNodes()) {
189           addAppTreeNodeItem(node, mr.getBaseLayerTreeNodes());
190         }
191       }
192     }
193 
194     private void buildOverlayLayers() {
195       if (app.getContentRoot().getLayerNodes() != null) {
196         for (AppTreeNode node : app.getContentRoot().getLayerNodes()) {
197           addAppTreeNodeItem(node, mr.getLayerTreeNodes());
198         }
199       }
200     }

202     private void addAppTreeNodeItem(AppTreeNode node, List<LayerTreeNode> layerTreeNodeList) {
203       LayerTreeNode layerTreeNode = new LayerTreeNode();
204       if ("AppTreeLayerNode".equals(node.getObjectType())) {
205         AppTreeLayerNode appTreeLayerNode = (AppTreeLayerNode) node;
206         layerTreeNode.setId(appTreeLayerNode.getId());
207         layerTreeNode.setAppLayerId(appTreeLayerNode.getId());
208         if (!addAppLayerItem(appTreeLayerNode)) {
209           return;
210         }
211         // This name is not displayed in the frontend, the title from the appLayer node is used
212         layerTreeNode.setName(appTreeLayerNode.getLayerName());
213         layerTreeNode.setDescription(appTreeLayerNode.getDescription());
214       } else if ("AppTreeLevelNode".equals(node.getObjectType())) {
215         AppTreeLevelNode appTreeLevelNode = (AppTreeLevelNode) node;
216         layerTreeNode.setId(appTreeLevelNode.getId());
217         layerTreeNode.setChildrenIds(appTreeLevelNode.getChildrenIds());
218         layerTreeNode.setRoot(Boolean.TRUE.equals(appTreeLevelNode.getRoot()));
219         // The name for a level node does show in the frontend
220         layerTreeNode.setName(appTreeLevelNode.getTitle());
221         layerTreeNode.setDescription(appTreeLevelNode.getDescription());
222       }
223       layerTreeNodeList.add(layerTreeNode);
224     }

226     private boolean addAppLayerItem(AppTreeLayerNode layerRef) {
227       Triple<GeoService, GeoServiceLayer, GeoServiceLayerSettings> serviceWithLayer =
228           findServiceLayer(layerRef);
229       GeoService service = serviceWithLayer.getLeft();
230       GeoServiceLayer serviceLayer = serviceWithLayer.getMiddle();
231 
232       if (service == null || serviceLayer == null) {
233         return false;
234       }
235 
236       // Some settings can be set on the app layer level, layer level or service level (default
237       // layer settings). These settings should be used in-order: from app layer if set, otherwise
238       // from layer level, from default layer setting or the default.
239 
240       // An empty (blank) string means it is not set. To explicitly clear a layer level string
241       // setting for an app, an admin should set the app layer setting to a spaces.
242 
243       // The JSON wrapper classes have "null" values as defaults which means not-set. The defaults
244       // (such as tilingDisabled being false) are applied below although the frontend would also
245       // treat null as non-truthy.
246 
247       // When default layer settings or settings for a specific layer are missing, construct new
248       // settings objects so no null check is needed. All properties are initialized to null
249       // (not-set) by default.
250       GeoServiceDefaultLayerSettings defaultLayerSettings =
251           Optional.ofNullable(service.getSettings().getDefaultLayerSettings())
252               .orElseGet(GeoServiceDefaultLayerSettings::new);
253       GeoServiceLayerSettings serviceLayerSettings =
254           Optional.ofNullable(serviceWithLayer.getRight()).orElseGet(GeoServiceLayerSettings::new);
255 
256       AppLayerSettings appLayerSettings = app.getAppLayerSettings(layerRef);
257 
258       String title =
259           Objects.requireNonNullElse(
260               nullIfEmpty(appLayerSettings.getTitle()),
261               // Get title from layer settings, title from capabilities or the layer name -- never
262               // null
263               service.getTitleWithSettingsOverrides(layerRef.getLayerName()));
264 
265       // These settings can be overridden per appLayer
266 
267       String description =
268           ObjectUtils.firstNonNull(
269               nullIfEmpty(appLayerSettings.getDescription()),
270               nullIfEmpty(serviceLayerSettings.getDescription()),
271               nullIfEmpty(defaultLayerSettings.getDescription()));
272 
273       String attribution =
274           ObjectUtils.firstNonNull(
275               nullIfEmpty(appLayerSettings.getAttribution()),
276               nullIfEmpty(serviceLayerSettings.getAttribution()),
277               nullIfEmpty(defaultLayerSettings.getAttribution()));
278 
279       // These settings can't be overridden per appLayer but can be set on a per-layer and
280       // service-level default basis
281 
282       boolean tilingDisabled =
283           ObjectUtils.firstNonNull(
284               serviceLayerSettings.getTilingDisabled(),
285               defaultLayerSettings.getTilingDisabled(),
286               false);
287       Integer tilingGutter =
288           ObjectUtils.firstNonNull(
289               serviceLayerSettings.getTilingGutter(), defaultLayerSettings.getTilingGutter(), 0);
290       boolean hiDpiDisabled =
291           ObjectUtils.firstNonNull(
292               serviceLayerSettings.getHiDpiDisabled(),
293               defaultLayerSettings.getHiDpiDisabled(),
294               false);
295       TileLayerHiDpiMode hiDpiMode =
296           ObjectUtils.firstNonNull(
297               serviceLayerSettings.getHiDpiMode(), defaultLayerSettings.getHiDpiMode(), null);
298       // Do not get from defaultLayerSettings because a default wouldn't make sense
299       String hiDpiSubstituteLayer = serviceLayerSettings.getHiDpiSubstituteLayer();
300 
301       TMFeatureType tmft = service.findFeatureTypeForLayer(serviceLayer, featureSourceRepository);
302 
303       boolean proxied = service.getSettings().getUseProxy();
304 
305       mr.addAppLayersItem(
306           new AppLayer()
307               .id(layerRef.getId())
308               .serviceId(serviceLayerServiceIds.get(serviceLayer))
309               .layerName(layerRef.getLayerName())
310               .hasAttributes(tmft != null)
311               .editable(TMFeatureTypeHelper.isEditable(app, layerRef, tmft))
312               .url(proxied ? getProxyUrl(service, app, layerRef) : null)
313               // Can't set whether layer is opaque, not mapped from WMS capabilities by GeoTools
314               // gt-wms Layer class?
315               .maxScale(serviceLayer.getMaxScale())
316               .minScale(serviceLayer.getMinScale())
317               .title(title)
318               .tilingDisabled(tilingDisabled)
319               .tilingGutter(tilingGutter)
320               .hiDpiDisabled(hiDpiDisabled)
321               .hiDpiMode(hiDpiMode)
322               .hiDpiSubstituteLayer(hiDpiSubstituteLayer)
323               .minZoom(serviceLayerSettings.getMinZoom())
324               .maxZoom(serviceLayerSettings.getMaxZoom())
325               .tileSize(serviceLayerSettings.getTileSize())
326               .tileGridExtent(serviceLayerSettings.getTileGridExtent())
327               .opacity(appLayerSettings.getOpacity())
328               .visible(layerRef.getVisible())
329               .attribution(attribution)
330               .description(description));
331       return true;
332     }

334     private Triple<GeoService, GeoServiceLayer, GeoServiceLayerSettings> findServiceLayer(
335         AppTreeLayerNode layerRef) {
336       GeoService service = geoServiceRepository.findById(layerRef.getServiceId()).orElse(null);
337       if (service == null) {
338         logger.warn(
339             "App {} references layer \"{}\" of missing service {}",
340             app.getId(),
341             layerRef.getLayerName(),
342             layerRef.getServiceId());
343         return Triple.of(null, null, null);
344       }
345 
346       if (!authorizationService.mayUserRead(service)) {
347         return Triple.of(null, null, null);
348       }
349 
350       GeoServiceLayer serviceLayer = service.findLayer(layerRef.getLayerName());
351 
352       if (serviceLayer == null) {
353         logger.warn(
354             "App {} references layer \"{}\" not found in capabilities of service {}",
355             app.getId(),
356             layerRef.getLayerName(),
357             service.getId());
358         return Triple.of(null, null, null);
359       }
360 
361       if (!authorizationService.mayUserRead(service, serviceLayer)) {
362         return Triple.of(null, null, null);
363       }
364 
365       serviceLayerServiceIds.put(serviceLayer, service.getId());
366 
367       if (mr.getServices().stream()
368           .filter(s -> s.getId().equals(service.getId()))
369           .findAny()
370           .isEmpty()) {
371         mr.addServicesItem(service.toJsonPojo(geoServiceHelper));
372       }
373 
374       GeoServiceLayerSettings layerSettings = service.getLayerSettings(layerRef.getLayerName());
375       return Triple.of(service, serviceLayer, layerSettings);
376     }
