45  /**
46   * Service for managing drawings.
47   *
48   * <p>This service provides methods for creating, updating, reading, and deleting drawings and persisting these
49   * operations in the data schema of the tailormap database. Any call can throw a {@link ResponseStatusException} if the
50   * user is not allowed to perform the operation.
51   */
52  @Service
53  public class DrawingService {
54    private static final Logger logger =
55        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
56  
57    private final JdbcClient jdbcClient;
58    private final RowMapper<Drawing> drawingRowMapper;
59    private final ObjectMapper objectMapper;
60  
61    public DrawingService(JdbcClient jdbcClient, ObjectMapper objectMapper) {
62      this.jdbcClient = jdbcClient;
63      this.objectMapper = objectMapper;
64  
65      final GenericConversionService conversionService = new GenericConversionService();
66      DefaultConversionService.addDefaultConverters(conversionService);
67  
68      conversionService.addConverter(new Converter<String, Drawing.AccessEnum>() {
69        @Override
70        public Drawing.AccessEnum convert(@NonNull String source) {
71          return Drawing.AccessEnum.fromValue(source);
72        }
73      });
74  
75      conversionService.addConverter(new Converter<PGobject, Map<String, Object>>() {
76        @Override
77        @SuppressWarnings("unchecked")
78        public Map<String, Object> convert(@NonNull PGobject source) {
79          try {
80            return objectMapper.readValue(source.getValue(), Map.class);
81          } catch (JsonProcessingException e) {
82            throw new IllegalArgumentException("Failed to convert PGobject to Map", e);
83          }
84        }
85      });
86  
87      drawingRowMapper = new SimplePropertyRowMapper<>(Drawing.class, conversionService) {
88        @Override
89        @NonNull public Drawing mapRow(@NonNull ResultSet rs, int rowNum) throws SQLException {
90          return super.mapRow(rs, rowNum);
91        }
92      };
93    }

95    /**
96     * Create a new drawing.
97     *
98     * @param drawing the drawing to create
99     * @param authentication the current user
100    * @return the created drawing
101    */
102   @Transactional
103   public Drawing createDrawing(@NonNull Drawing drawing, @NonNull Authentication authentication)
104       throws JsonProcessingException {
105 
106     canCreateDrawing(authentication);
107 
108     logger.trace(
109         "creating new drawing: {}, domainData {}, createdAt {}",
110         drawing,
111         objectMapper.writeValueAsString(drawing.getDomainData()),
112         OffsetDateTime.now(ZoneId.systemDefault()));
113 
114     Drawing storedDrawing = jdbcClient
115         .sql(
116             """
117 INSERT INTO data.drawing (name, description, domain_data, access, created_at, created_by,srid)
118 VALUES (?, ?, ?::jsonb, ?, ?, ?, ?) RETURNING *
119 """)
120         .param(drawing.getName())
121         .param(drawing.getDescription())
122         .param(objectMapper.writeValueAsString(drawing.getDomainData()))
123         .param(drawing.getAccess().getValue())
124         .param(OffsetDateTime.now(ZoneId.systemDefault()))
125         .param(authentication.getName())
126         .param(drawing.getSrid())
127         .query(drawingRowMapper)
128         .single();
129 
130     if (drawing.getFeatureCollection() != null) {
131       ObjectNode featureCollection = insertGeoJsonFeatureCollection(
132           storedDrawing.getId(),
133           drawing.getSrid(),
134           objectMapper.writeValueAsString(drawing.getFeatureCollection()));
135       storedDrawing.setFeatureCollection(featureCollection);
136     }
137 
138     logger.trace("stored new drawing: {}", storedDrawing);
139     return storedDrawing;
140   }

142   private ObjectNode insertGeoJsonFeatureCollection(UUID drawingId, int srid, String featureCollectionToStore)
143       throws JsonProcessingException {
144     List<JsonNode> storedFeatures = jdbcClient
145         .sql(
146             """
147 WITH jsonData AS (SELECT :featureCollectionToStore::json AS featureCollection)
148 INSERT INTO data.drawing_feature (drawing_id, geometry, properties)
149 SELECT :drawingId::uuid AS drawing_id,
150 ST_SetSRID(ST_GeomFromGeoJSON(feature ->> 'geometry'), :srid) AS geometry,
151 feature -> 'properties' AS properties
152 FROM (SELECT json_array_elements(featureCollection -> 'features') AS feature
153 FROM jsonData)
154 AS f
155 RETURNING
156 -- since we cannot use aggregate functions in a returning clause, we will return a list of geojson
157 -- features and aggregate them into a featureCollection in the next step
158 ST_AsGeoJSON(data.drawing_feature.*, geom_column =>'geometry', id_column => 'id')::json;
159 """)
160         .param("featureCollectionToStore", featureCollectionToStore)
161         .param("drawingId", drawingId)
162         .param("srid", srid)
163         .query(new RowMapper<JsonNode>() {
164           @Override
165           public JsonNode mapRow(@NonNull ResultSet rs, int rowNum) throws SQLException {
166             try {
167               JsonNode jsonNode = objectMapper.readTree(rs.getString(1));
168               // merge/un-nest properties with nested properties, because we have a jsonb column
169               // called "properties" and we are using the `ST_AsGeoJSON(::record,...)` function
170               final ObjectNode properties = (ObjectNode) jsonNode.get("properties");
171               JsonNode nestedProperties = properties.get("properties");
172               if (nestedProperties != null) {
173                 nestedProperties.properties().stream()
174                     .iterator()
175                     .forEachRemaining(
176                         entry -> properties.putIfAbsent(entry.getKey(), entry.getValue()));
177               }
178               properties.remove("properties");
179               return jsonNode;
180             } catch (JsonProcessingException e) {
181               throw new RuntimeException(e);
182             }
183           }
184         })
185         .list();
186 
187     return objectMapper
188         .createObjectNode()
189         .put("type", "FeatureCollection")
190         .set("features", objectMapper.createArrayNode().addAll(storedFeatures));
191   }

193   /**
194    * Update an existing drawing.
195    *
196    * @param drawing the drawing to create
197    * @param authentication the current user
198    * @return the created drawing
199    */
200   @Transactional
201   public Drawing updateDrawing(@NonNull Drawing drawing, @NonNull Authentication authentication)
202       throws JsonProcessingException {
203 
204     canSaveOrDeleteDrawing(drawing, authentication);
205 
206     logger.trace(
207         "updating drawing: {}, domainData {}, updatedAt {}",
208         drawing,
209         objectMapper.writeValueAsString(drawing.getDomainData()),
210         OffsetDateTime.now(ZoneId.systemDefault()));
211 
212     final Drawing oldDrawing = getDrawing(drawing.getId(), authentication)
213         .orElseThrow(() ->
214             new ResponseStatusException(HttpStatus.NOT_FOUND, "Drawing has been deleted by another user"));
215 
216     if (drawing.getVersion() < oldDrawing.getVersion()) {
217       throw new ResponseStatusException(HttpStatus.CONFLICT, "Drawing has been updated by another user");
218     }
219     drawing.setVersion(drawing.getVersion() + 1);
220 
221     Drawing updatedDrawing = jdbcClient
222         .sql(
223             """
224 UPDATE data.drawing SET
225 id=:id,
226 name=:name,
227 description=:description,
228 domain_data=:domainData::jsonb,
229 access=:access,
230 created_by=:createdBy,
231 created_at=:createdAt,
232 updated_by=:updatedBy,
233 updated_at=:updatedAt,
234 srid=:srid,
235 version=:version
236 WHERE id = :id RETURNING *""")
237         .param("id", drawing.getId())
238         .param("name", drawing.getName())
239         .param("description", drawing.getDescription())
240         .param("domainData", objectMapper.writeValueAsString(drawing.getDomainData()))
241         .param("access", drawing.getAccess().getValue())
242         .param("createdBy", drawing.getCreatedBy())
243         .param("createdAt", drawing.getCreatedAt())
244         .param("updatedBy", authentication.getName())
245         .param("updatedAt", OffsetDateTime.now(ZoneId.systemDefault()))
246         .param("srid", drawing.getSrid())
247         .param("version", drawing.getVersion())
248         .query(drawingRowMapper)
249         .single();
250 
251     // delete even if drawing.getFeatureCollection()==null, because all features could have been removed,
252     // afterwards (re)insert the featureCollection
253     jdbcClient
254         .sql("DELETE FROM data.drawing_feature WHERE drawing_id = ?")
255         .param(drawing.getId())
256         .update();
257 
258     if (drawing.getFeatureCollection() != null) {
259       ObjectNode featureCollection = insertGeoJsonFeatureCollection(
260           drawing.getId(),
261           drawing.getSrid(),
262           objectMapper.writeValueAsString(drawing.getFeatureCollection()));
263       updatedDrawing.setFeatureCollection(featureCollection);
264     }
265 
266     logger.trace("stored updated drawing: {}", updatedDrawing);
267     return updatedDrawing;
268   }

270   /**
271    * Get all drawings for the current user.
272    *
273    * @param authentication the current user
274    * @return the drawings, a possibly empty set
275    */
276   public Set<Drawing> getDrawingsForUser(Authentication authentication) throws ResponseStatusException {
277     if (authentication == null || authentication instanceof AnonymousAuthenticationToken) {
278       return Set.of();
279     }
280     return jdbcClient
281         .sql("SELECT * FROM data.drawing WHERE created_by = :userName OR updated_by = :userName")
282         .param("userName", authentication.getName())
283         .query(drawingRowMapper)
284         .set()
285         .stream()
286         .filter(d -> {
287           try {
288             canReadDrawing(d, authentication);
289             return true;
290           } catch (ResponseStatusException e) {
291             return false;
292           }
293         })
294         .sorted(Comparator.comparing(Drawing::getCreatedAt))
295         .collect(Collectors.toCollection(LinkedHashSet::new));
296   }

309   /**
310    * Get a complete drawing by its ID with GeoJSON geometries in the requested srid.
311    *
312    * @param drawingId the ID of the drawing
313    * @param authentication the current user
314    * @param withGeometries whether to fetch the geometries for the drawing
315    * @param requestedSrid the SRID to return the geometries in
316    * @return the complete drawing
317    */
318   @Transactional
319   public Optional<Drawing> getDrawing(
320       @NonNull UUID drawingId,
321       @NonNull Authentication authentication,
322       boolean withGeometries,
323       int requestedSrid) {
324     Optional<Drawing> drawing =
325         jdbcClient
326             .sql("SELECT * FROM data.drawing WHERE id = ?")
327             .param(1, drawingId)
328             .query(drawingRowMapper)
329             .stream()
330             .findFirst();
331 
332     drawing.ifPresent(d -> {
333       // check if the user is allowed to read the drawing
334       canReadDrawing(d, authentication);
335 
336       d.setSrid(requestedSrid);
337       if (withGeometries) {
338         d.setFeatureCollection(getFeatureCollection(drawingId, requestedSrid));
339       }
340     });
341 
342     return drawing;
343   }

345   /**
346    * Retrieve the feature collection as GeoJSON for a drawing.
347    *
348    * @param drawingId the ID of the drawing
349    * @param srid the SRID to return the geometries in
350    * @return the feature collection as GeoJSON
351    */
352   private JsonNode getFeatureCollection(UUID drawingId, int srid) {
353     return jdbcClient
354         .sql(
355             """
356 SELECT row_to_json(featureCollection) from (
357 SELECT
358 'FeatureCollection' AS type,
359 array_to_json(array_agg(feature)) AS features FROM (
360 SELECT 'Feature' AS type,
361 ST_ASGeoJSON(ST_Transform(geomTable.geometry, :srid))::json AS geometry,
362 row_to_json((SELECT l from (SELECT drawing_id, properties) AS l)) AS properties
363 FROM data.drawing_feature AS geomTable WHERE drawing_id = :drawingId::uuid) AS feature) AS featureCollection
364 """)
365         .param("drawingId", drawingId)
366         .param("srid", srid)
367         .query(new RowMapper<JsonNode>() {
368           @Override
369           public JsonNode mapRow(@NonNull ResultSet rs, int rowNum) throws SQLException {
370             try {
371               JsonNode jsonNode = objectMapper.readTree(rs.getString(1));
372               // merge/un-nest properties with nested properties, because we have a jsonb column
373               // called "properties" and we are using the `ST_AsGeoJSON(::record,...)` function
374               ArrayNode features = (ArrayNode) jsonNode.get("features");
375               features.elements().forEachRemaining(feature -> {
376                 ObjectNode properties = (ObjectNode) feature.get("properties");
377                 JsonNode nestedProperties = properties.get("properties");
378                 if (nestedProperties != null) {
379                   nestedProperties
380                       .fields()
381                       .forEachRemaining(
382                           entry -> properties.putIfAbsent(entry.getKey(), entry.getValue()));
383                 }
384                 properties.remove("properties");
385               });
386               return jsonNode;
387             } catch (JsonProcessingException e) {
388               throw new RuntimeException(e);
389             }
390           }
391         })
392         .single();
393   }

402     canSaveOrDeleteDrawing(

425   /**
426    * Check if the current user can read the drawing. If not, throw an unauthorized exception.
427    *
428    * @param drawing the drawing to check
429    * @param authentication the current user
430    * @throws ResponseStatusException if the user is not allowed to read the drawing
431    */
432   private void canReadDrawing(@NonNull Drawing drawing, @NonNull Authentication authentication)
433       throws ResponseStatusException {
434     // TODO validate this method
435     boolean isAuthenticated = !(authentication instanceof AnonymousAuthenticationToken);
436 
437     boolean canRead =
438         switch (drawing.getAccess()) {
439           case PRIVATE -> isAuthenticated && Objects.equals(authentication.getName(), drawing.getCreatedBy());
440           case SHARED -> isAuthenticated;
441           case PUBLIC -> true;
442         };
443 
444     if (!canRead) {
445       throw new ResponseStatusException(HttpStatus.UNAUTHORIZED, "Insufficient permissions to access drawing");
446     }
447   }

449   /**
450    * Check if the current user can save the drawing. If not, throw an unauthorized exception.
451    *
452    * @param drawing the drawing to check
453    * @param authentication the current user
454    * @throws ResponseStatusException if the user is not allowed to save/delete the drawing
455    */
456   private void canSaveOrDeleteDrawing(@NonNull Drawing drawing, @NonNull Authentication authentication)
457       throws ResponseStatusException {
458     // TODO validate this method
459     if (authentication instanceof AnonymousAuthenticationToken) {
460       // Only authenticated users can save drawings, irrelevant of drawing access level
461       throw new ResponseStatusException(HttpStatus.UNAUTHORIZED, "Insufficient permissions to save drawing");
462     }
463 
464     boolean canSave =
465         switch (drawing.getAccess()) {
466           case PRIVATE -> drawing.getCreatedBy().equals(authentication.getName());
467           case SHARED, PUBLIC -> true;
468         };
469 
470     if (!canSave) {
471       throw new ResponseStatusException(HttpStatus.UNAUTHORIZED, "Insufficient permissions to save drawing");
472     }
473   }
