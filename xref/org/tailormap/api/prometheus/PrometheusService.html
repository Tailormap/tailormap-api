27  /**
28   * Service for managing Prometheus-related operations. This service can be used to manage Prometheus-related operations
29   * such as querying and managing metrics stored in the Prometheus instance.
30   */
31  @Component
32  public class PrometheusService {
33    private static final Logger logger =
34        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
35  
36    @Value("${tailormap-api.prometheus-api-url}")
37    private String prometheusUrl;
38  
39    private final RestTemplate restTemplate;
40  
41    public PrometheusService(RestTemplate template) {
42      this.restTemplate = template;
43    }
44  
45    /** Check if the Prometheus server is available by sending a simple query. */
46    public boolean isPrometheusAvailable() {
47      try {
48        ResponseEntity<String> response =
49            restTemplate.getForEntity(prometheusUrl + "/query?query=up", String.class);
50        if (response.getStatusCode() == HttpStatus.OK) {
51          return true;
52        } else {
53          logger.warn("Prometheus server is not available: {}", response.getStatusCode());
54          return false;
55        }
56      } catch (Exception e) {
57        logger.debug("Error checking Prometheus availability", e);
58        return false;
59      }
60    }
61    /**
62     * Executes a Prometheus query and returns the result.
63     *
64     * @param promQuery the Prometheus query to execute
65     * @return the result of the query as a JSON node
66     * @throws JsonProcessingException if there is an error processing the JSON response
67     * @throws IOException if there is an error executing the query
68     * @see PrometheusResultProcessor
69     */
70    public JsonNode executeQuery(String promQuery) throws JsonProcessingException, IOException {
71      ResponseEntity<String> response;
72      URI promUrl = UriComponentsBuilder.fromUriString(prometheusUrl)
73          .path("/query")
74          .queryParam("query", promQuery)
75          .build()
76          .toUri();
77      logger.trace("Executing Prometheus query (GET): {}", promUrl);
78      try {
79        response = restTemplate.getForEntity(promUrl, String.class);
80        if (response.getStatusCode() != HttpStatus.OK) {
81          logger.error("Failed to execute Prometheus query: {}", response.getStatusCode());
82          throw new IOException("Failed to execute Prometheus query: " + response.getStatusCode());
83        }
84      } catch (RestClientException e) {
85        logger.error("Error executing Prometheus query: {}", e.getMessage());
86        throw new IOException("Error executing Prometheus query: " + e.getMessage(), e);
87      }
88  
89      final JsonNode jsonResponse = new ObjectMapper().readTree(response.getBody());
90      logger.trace("Prometheus query response: {}", jsonResponse.toPrettyString());
91  
92      if (!"success".equals(jsonResponse.path("status").asText())) {
93        logger.error(
94            "Prometheus query failed: {}", jsonResponse.path("error").asText());
95        throw new IOException(
96            "Prometheus query failed: " + jsonResponse.path("error").asText());
97      }
98      return jsonResponse;
99    }
100 
101   /**
102    * Executes a Prometheus delete query.
103    *
104    * @param metricMatches the Prometheus metric matches to be deleted
105    * @throws IOException if there is an error executing the delete query
106    */
107   public void deleteMetric(String... metricMatches) throws IOException, URISyntaxException {
108     final URL url = UriComponentsBuilder.fromUriString(prometheusUrl)
109         .path("/admin/tsdb/delete_series")
110         .queryParam("match[]", (Object[]) metricMatches)
111         .build()
112         .toUri()
113         .toURL();
114     logger.trace("Deleting metrics using (PUT): {}", url);
115     HttpURLConnection httpURLConnection = (HttpURLConnection) url.openConnection();
116     httpURLConnection.setRequestMethod("PUT");
117 
118     if (httpURLConnection.getResponseCode() != HttpStatus.NO_CONTENT.value()) {
119       throw new IOException("Failed to delete Prometheus metric: " + httpURLConnection.getResponseCode());
120     }
121   }
122 
123   /**
124    * Executes a Prometheus tombstone query.
125    *
126    * @throws IOException if there is an error executing the delete query
127    */
128   public void cleanTombstones() throws IOException, URISyntaxException {
129     final URL url = new URI(prometheusUrl + "/admin/tsdb/clean_tombstones").toURL();
130     HttpURLConnection httpURLConnection = (HttpURLConnection) url.openConnection();
131     httpURLConnection.setRequestMethod("PUT");
132 
133     if (httpURLConnection.getResponseCode() != HttpStatus.NO_CONTENT.value()) {
134       throw new IOException("Failed to cleanup Prometheus tombstones: " + httpURLConnection.getResponseCode());
135     }
136   }
