6   package org.tailormap.api.prometheus;
7   
8   import com.fasterxml.jackson.databind.JsonNode;
9   import com.fasterxml.jackson.databind.ObjectMapper;
10  import com.fasterxml.jackson.databind.node.ArrayNode;
11  import java.io.IOException;
12  import java.util.Collection;
13  import java.util.HashMap;
14  import java.util.Map;
15  import org.springframework.stereotype.Component;
16  
17  @Component
18  public class PrometheusResultProcessor implements TagNames {
19  
20    /**
21     * Processes the JSON response from a Prometheus query and groups the results by
22     * {@link TagNames#METRICS_APP_ID_TAG}. Each appId will have a map of metrics, where the key is the metric type
23     * (e.g., "total", "lastUpdated") and the value is the metric value. Additionally, the appName is included in the
24     * map for each appId.
25     *
26     * @param root the root JSON node containing the Prometheus query results (expected structure: root.data.result)
27     *     where each result has a "metric" object with "appId", "type", and "value" fields, and a "value" array where
28     *     the second element is the metric value. The "metric" object may also contain "appName".
29     * @return a map where the key is the appId and the value is another map containing metric types as keys and their
30     *     corresponding values. The inner map also includes the appName under the key "appName". If an appId has no
31     *     metrics, it will not be included in the results.
32     * @see TagNames#METRICS_APP_ID_TAG
33     */
34    public Collection<Map<String, String>> processPrometheusResultsForApplications(JsonNode root) {
35      final Map<String, Map<String, String>> groupedResults = new HashMap<>();
36      for (JsonNode result : root.path("data").path("result")) {
37        String appId = result.path("metric").path(METRICS_APP_ID_TAG).asText();
38        String appName = result.path("metric").path(METRICS_APP_NAME_TAG).asText();
39        String type = result.path("metric").path("type").asText();
40        String value = result.path("value").get(1).asText();
41        // combine measurements
42        groupedResults.computeIfAbsent(appId, k -> new HashMap<>()).put(type, value);
43  
44        // Add appName and appId to the map for this appId
45        groupedResults.get(appId).put(METRICS_APP_NAME_TAG, appName);
46        groupedResults.get(appId).put(METRICS_APP_ID_TAG, appId);
47      }
48  
49      return groupedResults.values();
50    }
51  
52    /**
53     * Processes the JSON response from a Prometheus query and groups the results by
54     * {@link TagNames#METRICS_APP_LAYER_ID_TAG}. Each appLayerId will have a map of metrics, where the key is the
55     * metric type (e.g., "total", "lastUpdated") and the value is the metric value. Additionally, the appName is
56     * included in the map for each appId.
57     *
58     * @param root the root JSON node containing the Prometheus query results (expected structure: root.data.result)
59     *     where each result has a "metric" object with "appId", "type", and "value" fields, and a "value" array where
60     *     the second element is the metric value. The "metric" object may also contain "appName".
61     * @return a map where the key is the appLayerId and the value is another map containing metric types as keys and
62     *     their corresponding values. The inner map also includes the appName under the key "appName". If an appId has
63     *     no metrics, it will not be included in the results.
64     * @see TagNames#METRICS_APP_LAYER_ID_TAG
65     */
66    public Collection<Map<String, String>> processPrometheusResultsForApplicationLayers(JsonNode root) {
67      final Map<String, Map<String, String>> groupedResults = new HashMap<>();
68      for (JsonNode result : root.path("data").path("result")) {
69        String appLayerId =
70            result.path("metric").path(METRICS_APP_LAYER_ID_TAG).asText();
71        String appId = result.path("metric").path(METRICS_APP_ID_TAG).asText();
72        String appName = result.path("metric").path(METRICS_APP_NAME_TAG).asText();
73        String type = result.path("metric").path("type").asText();
74        String value = result.path("value").get(1).asText();
75        // combine all measurements
76        groupedResults.computeIfAbsent(appLayerId, k -> new HashMap<>()).put(type, value);
77  
78        // Add appName and appId and appLayerId to the map for this appLayerId
79        groupedResults.get(appLayerId).put(METRICS_APP_NAME_TAG, appName);
80        groupedResults.get(appLayerId).put(METRICS_APP_ID_TAG, appId);
81        groupedResults.get(appLayerId).put(METRICS_APP_LAYER_ID_TAG, appLayerId);
82      }
83  
84      return groupedResults.values();
85    }
86  
87    /**
88     * Merges two Prometheus JSON responses and processes the results into an ArrayNode. Each appId will have a map of
89     * metrics, where the key is the metric type (e.g., "total", "lastUpdated") and the value is the metric value.
90     *
91     * @param jsonResponse1 the first JSON response from a Prometheus query
92     * @param jsonResponse2 the second JSON response from a Prometheus query
93     * @return an ArrayNode containing maps of metrics for each appId
94     * @throws IOException if there is an error processing the JSON responses
95     */
96    public ArrayNode processPrometheusResultsToJsonArray(JsonNode jsonResponse1, JsonNode jsonResponse2)
97        throws IOException {
98      JsonNode mergedResults =
99          new ObjectMapper().readerForUpdating(jsonResponse1).readValue(jsonResponse2);
100     Collection<Map<String, String>> mergedResultsList = this.processPrometheusResultsForApplications(mergedResults);
101     return new ObjectMapper().valueToTree(mergedResultsList);
102   }
103 }
