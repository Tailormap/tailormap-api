7   package org.tailormap.api.util;
8   
9   import java.util.stream.Collectors;
10  import org.apache.commons.lang3.exception.ExceptionUtils;
11  
12  public class TMExceptionUtils {
13    public static String joinAllThrowableMessages(Throwable t) {
14      return joinAllThrowableMessages(t, ", cause: ");
15    }
16  
17    public static String joinAllThrowableMessages(Throwable t, String delimiter) {
18      return ExceptionUtils.getThrowableList(t).stream()
19          .map(ExceptionUtils::getMessage)
20          .collect(Collectors.joining(delimiter));
21    }
22  }
