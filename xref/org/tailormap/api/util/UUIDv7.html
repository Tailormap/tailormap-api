6   package org.tailormap.api.util;
7   
8   import java.nio.ByteBuffer;
9   import java.security.SecureRandom;
10  import java.time.Instant;
11  import java.util.UUID;
12  
13  /**
14   * A utility class to create UUID version 7, which is time-sortable.
15   *
16   * @see <a href="https://antonz.org/uuidv7/#java">UUIDv7 in 33 languages</a> for more details.
17   * @see <a href="https://bugs.openjdk.org/browse/JDK-8357251">for a potential future built-in implementation in Java
18   *     26.</a>
19   */
20  public class UUIDv7 {
21    private static final SecureRandom random = new SecureRandom();
22  
23    private UUIDv7() {
24      // private constructor for utility class
25    }
26  
27    /**
28     * Create a random version 7 UUID.
29     *
30     * @return a random version 7 UUID
31     */
32    public static UUID randomV7() {
33      byte[] value = randomBytes();
34      ByteBuffer buf = ByteBuffer.wrap(value);
35      long high = buf.getLong();
36      long low = buf.getLong();
37      return new UUID(high, low);
38    }
39  
40    private static byte[] randomBytes() {
41      byte[] value = new byte[16];
42      random.nextBytes(value);
43      ByteBuffer timestamp = ByteBuffer.allocate(Long.BYTES).putLong(System.currentTimeMillis());
44      System.arraycopy(timestamp.array(), 2, value, 0, 6);
45      // set version 7
46      value[6] = (byte) ((value[6] & 0x0F) | 0x70);
47      // set RFC 4122 variant
48      value[8] = (byte) ((value[8] & 0x3F) | 0x80);
49      return value;
50    }
51  
52    /**
53     * Extract the Unix epoch timestamp (milliseconds) from a UUIDv7.
54     *
55     * @param uuidv7 the UUIDv7 value
56     * @return the embedded Unix epoch timestamp in milliseconds
57     * @throws IllegalArgumentException if the given UUID is not version 7
58     */
59    public static long timestamp(UUID uuidv7) throws IllegalArgumentException {
60      if (uuidv7 == null) {
61        throw new IllegalArgumentException("UUID cannot be null");
62      }
63      if (uuidv7.version() != 7) {
64        throw new IllegalArgumentException("UUID is not version 7");
65      }
66      // UUIDv7 stores a 48-bit Unix epoch millisecond timestamp in the first 6 bytes.
67      return (uuidv7.getMostSignificantBits() >>> 16) & 0x0000FFFFFFFFFFFFL;
68    }
69  
70    /**
71     * Extract the Unix epoch timestamp as an Instant from a UUIDv7.
72     *
73     * @param uuidv7 the UUIDv7 value
74     * @return the embedded Unix epoch timestamp in milliseconds
75     * @throws IllegalArgumentException if the given UUID is not version 7
76     * @see #timestamp(UUID)
77     */
78    public static Instant timestampAsInstant(UUID uuidv7) throws IllegalArgumentException {
79      return Instant.ofEpochMilli(timestamp(uuidv7));
80    }
81  
82    /**
83     * Parse a string representation of a UUIDv7.
84     *
85     * <p>Accepts the standard {@code xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx} format.
86     *
87     * @param value the string to parse
88     * @return the parsed UUIDv7
89     * @throws IllegalArgumentException if the string is not a valid UUID or not version 7
90     */
91    public static UUID fromString(String value) throws IllegalArgumentException {
92      if (value == null || value.isBlank()) {
93        throw new IllegalArgumentException("UUID string cannot be null or blank");
94      }
95      // Remove hyphens and parse the 32 hex characters directly
96      String hex = value.replace("-", "");
97      if (hex.length() != 32) {
98        throw new IllegalArgumentException("Invalid UUID string: " + value);
99      }
100     try {
101       long high = Long.parseUnsignedLong(hex, 0, 16, 16);
102       long low = Long.parseUnsignedLong(hex, 16, 32, 16);
103       UUID uuid = new UUID(high, low);
104       if (uuid.version() != 7) {
105         throw new IllegalArgumentException("UUID is not version 7: " + value);
106       }
107       return uuid;
108     } catch (NumberFormatException e) {
109       throw new IllegalArgumentException("Invalid UUID string: " + value, e);
110     }
111   }
112 }
