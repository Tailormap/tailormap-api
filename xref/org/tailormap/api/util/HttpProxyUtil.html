26  public class HttpProxyUtil {
27    public static void addForwardedForRequestHeaders(HttpRequest.Builder requestBuilder, HttpServletRequest request) {
28      try {
29        String ip = request.getRemoteAddr();
30        InetAddress inetAddress = InetAddress.getByName(ip);
31  
32        if (inetAddress instanceof Inet6Address inet6Address) {
33          // https://stackoverflow.com/questions/33168783/
34          int scopeId = inet6Address.getScopeId();
35  
36          if (scopeId > 0) {
37            ip = inet6Address.getHostName().replaceAll("%" + scopeId + "$", "");
38          }
39  
40          // IPv6 address must be bracketed and quoted
41          ip = "\"[" + ip + "]\"";
42        }
43        requestBuilder.header("X-Forwarded-For", ip);
44        requestBuilder.header("Forwarded", "for=" + ip);
45      } catch (UnknownHostException ignored) {
46        // Don't care
47      }
48    }
49  
50    public static void passthroughRequestHeaders(
51        HttpRequest.Builder requestBuilder, HttpServletRequest request, Set<String> headers) {
52      for (String header : headers) {
53        String value = request.getHeader(header);
54        if (value != null) {
55          requestBuilder.header(header, value);
56        }
57      }
58    }
59  
60    public static HttpHeaders passthroughResponseHeaders(
61        java.net.http.HttpHeaders upstreamHeaders, Set<String> allowedResponseHeaders) {
62      HttpHeaders headers = new HttpHeaders();
63      for (String header : allowedResponseHeaders) {
64        headers.addAll(header, upstreamHeaders.allValues(header));
65      }
66      return headers;
67    }
68  
69    public static void setHttpBasicAuthenticationHeader(
70        HttpRequest.Builder requestBuilder, String username, String password) {
71      if (username != null && password != null) {
72        String toEncode = username + ":" + password;
73        String encoded = Base64.getEncoder().encodeToString(toEncode.getBytes(StandardCharsets.UTF_8));
74        requestBuilder.header(AUTHORIZATION, "Basic " + encoded);
75      }
76    }

78    /**
79     * If the original request was a POST with x-www-urlencoded content type, configure the requestBuilder for a proxy
80     * request to do a POST request with all parameters in the body to handle large POST parameters.
81     *
82     * @param requestBuilder builder for proxy request
83     * @param uri URI of the proxy target, including query parameters
84     * @param request the original request to be proxied
85     */
86    public static void configureProxyRequestBuilderForUri(
87        HttpRequest.Builder requestBuilder, URI uri, HttpServletRequest request) {
88      // When the original request is a POST with x-www-form-urlencoded content type, do the same (for long parameters
89      // like CQL_FILTER which may trigger a URI Too Long or Bad Request response)
90      if (HttpMethod.POST.matches(request.getMethod())
91          && request.getContentType() != null
92          && request.getContentType()
93              .toLowerCase(Locale.ROOT)
94              .contains(MediaType.APPLICATION_FORM_URLENCODED_VALUE)) {
95  
96        // The original request could have had some parameters in the URL and some in the body. However, in the
97        // proxied request we put all parameters in the body
98        // Make sure to not decode the query so '+' stays encoded as "%2B" and does not become a space
99        String query = uri.getRawQuery();
100       URI uriWithoutQuery = URI.create(uri.toString().split("\\?", 2)[0]);
101       requestBuilder.uri(uriWithoutQuery);
102 
103       if (isNotEmpty(query)) {
104         requestBuilder
105             .POST(HttpRequest.BodyPublishers.ofString(query))
106             .header(CONTENT_TYPE, MediaType.APPLICATION_FORM_URLENCODED_VALUE);
107       } else {
108         requestBuilder.uri(uri);
109       }
110     } else {
111       requestBuilder.uri(uri);
112     }
113   }
