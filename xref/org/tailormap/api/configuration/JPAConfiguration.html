6   package org.tailormap.api.configuration;
7   
8   import jakarta.persistence.EntityManagerFactory;
9   import org.springframework.boot.autoconfigure.domain.EntityScan;
10  import org.springframework.context.annotation.Bean;
11  import org.springframework.context.annotation.Configuration;
12  import org.springframework.context.annotation.Profile;
13  import org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor;
14  import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
15  import org.springframework.orm.jpa.JpaTransactionManager;
16  import org.springframework.transaction.annotation.EnableTransactionManagement;
17  
18  /**
19   * JPA configuration beans.
20   *
21   * @since 0.1
22   */
23  @Configuration
24  @EnableJpaRepositories(basePackages = {"org.tailormap.api.repository"})
25  @EntityScan(basePackages = {"org.tailormap.api.persistence"})
26  @EnableTransactionManagement
27  @Profile("!test")
28  public class JPAConfiguration {
29  
30    @Bean
31    JpaTransactionManager transactionManager(final EntityManagerFactory entityManagerFactory) {
32      final JpaTransactionManager transactionManager = new JpaTransactionManager();
33      transactionManager.setEntityManagerFactory(entityManagerFactory);
34      return transactionManager;
35    }
36  
37    @Bean
38    public PersistenceExceptionTranslationPostProcessor exceptionTranslation() {
39      return new PersistenceExceptionTranslationPostProcessor();
40    }
41  }
