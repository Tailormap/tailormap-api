6   package org.tailormap.api.configuration;
7   
8   import java.lang.reflect.Type;
9   import org.hibernate.type.descriptor.WrapperOptions;
10  import org.hibernate.type.descriptor.java.JavaType;
11  import org.hibernate.type.format.AbstractJsonFormatMapper;
12  import tools.jackson.core.JacksonException;
13  import tools.jackson.core.JsonGenerator;
14  import tools.jackson.core.JsonParser;
15  import tools.jackson.databind.json.JsonMapper;
16  
17  /**
18   * Custom JSON format mapper for Hibernate that uses the Jackson 3 library for JSON serialization and deserialization.
19   * This format mapper allows Hibernate to handle JSON types using Jackson, which is the primary JSON processing library
20   * used in the application. By implementing this custom format mapper, we ensure that Hibernate can correctly serialize
21   * and deserialize JSON properties in a way that is compatible with the application's JSON processing needs, especially
22   * since Hibernate 7.2.x does not natively support Jackson 3. See {@link HibernateConfiguration} for how this format
23   * mapper is registered with Hibernate and:
24   *
25   * <ul>
26   *   <li><a href="https://github.com/spring-projects/spring-boot/issues/47789">Spring Boot issue #47789</a>
27   *   <li><a href="https://github.com/tGrefsrud/demo-spring-boot-4-serialization-issue/pull/1">Demo project illustrating
28   *       the issue with JSON serialization in Spring Boot 4 and Hibernate 7.2.x</a>
29   *   <li><a href="https://hibernate.atlassian.net/browse/HHH-19890">Hibernate issue HHH-19890 regarding JSON format
30   *       mappers and Jackson 3 support</a>
31   *   <li><a href="https://github.com/hibernate/hibernate-orm/pull/11357">Pull request in Hibernate ORM to add support
32   *       for Jackson 3 in JSON format mappers</a>
33   * </ul>
34   *
35   * for more context on the need for a custom JSON format mapper with Hibernate 7.2.x and Jackson 3. This class should be
36   * removed after Hibernate 7.3.x or higher is released/implemented in TM.
37   */
38  public class Jackson3JsonFormatMapper extends AbstractJsonFormatMapper {
39  
40    public static final String SHORT_NAME = "jackson3";
41  
42    private final JsonMapper jsonMapper;
43  
44    public Jackson3JsonFormatMapper() {
45      this(JsonMapper.shared());
46    }
47  
48    public Jackson3JsonFormatMapper(JsonMapper jsonMapper) {
49      this.jsonMapper = jsonMapper;
50    }
51  
52    @Override
53    public <T> void writeToTarget(T value, JavaType<T> javaType, Object target, WrapperOptions options)
54        throws JacksonException {
55      this.jsonMapper
56          .writerFor(this.jsonMapper.constructType(javaType.getJavaType()))
57          .writeValue((JsonGenerator) target, value);
58    }
59  
60    @Override
61    public <T> T readFromSource(JavaType<T> javaType, Object source, WrapperOptions options) throws JacksonException {
62      return jsonMapper.readValue((JsonParser) source, jsonMapper.constructType(javaType.getJavaType()));
63    }
64  
65    @Override
66    public boolean supportsSourceType(Class<?> sourceType) {
67      return JsonParser.class.isAssignableFrom(sourceType);
68    }
69  
70    @Override
71    public boolean supportsTargetType(Class<?> targetType) {
72      return JsonGenerator.class.isAssignableFrom(targetType);
73    }
74  
75    @Override
76    @SuppressWarnings("TypeParameterUnusedInFormals")
77    public <T> T fromString(CharSequence charSequence, Type type) {
78      try {
79        return jsonMapper.readValue(charSequence.toString(), jsonMapper.constructType(type));
80      } catch (JacksonException e) {
81        throw new IllegalArgumentException("Could not deserialize string to java type: " + type, e);
82      }
83    }
84  
85    @Override
86    public <T> String toString(T value, Type type) {
87      try {
88        return jsonMapper.writerFor(jsonMapper.constructType(type)).writeValueAsString(value);
89      } catch (JacksonException e) {
90        throw new IllegalArgumentException("Could not serialize object of java type: " + type, e);
91      }
92    }
93  }
