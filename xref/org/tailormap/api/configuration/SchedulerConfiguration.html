6   package org.tailormap.api.configuration;
7   
8   import io.sentry.quartz.SentryJobListener;
9   import jakarta.annotation.PostConstruct;
10  import java.lang.invoke.MethodHandles;
11  import org.quartz.SchedulerException;
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  import org.springframework.beans.factory.annotation.Autowired;
15  import org.springframework.context.annotation.Configuration;
16  import org.springframework.scheduling.quartz.SchedulerFactoryBean;
17  import org.tailormap.api.scheduling.DebugLoggingJobListener;
18  import org.tailormap.api.scheduling.DebugLoggingTriggerListener;
19  
20  @Configuration
21  public class SchedulerConfiguration {
22    private static final Logger logger =
23        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
24  
25    private final SchedulerFactoryBean schedulerFactoryBean;
26  
27    public SchedulerConfiguration(@Autowired SchedulerFactoryBean schedulerFactoryBean) {
28      this.schedulerFactoryBean = schedulerFactoryBean;
29    }
30  
31    @PostConstruct
32    public void addListeners() throws SchedulerException {
33      schedulerFactoryBean.getScheduler().getListenerManager().addJobListener(new SentryJobListener());
34      if (logger.isDebugEnabled()) {
35        // Add debug logging listeners to the scheduler
36        schedulerFactoryBean.getScheduler().getListenerManager().addJobListener(new DebugLoggingJobListener());
37  
38        schedulerFactoryBean
39            .getScheduler()
40            .getListenerManager()
41            .addTriggerListener(new DebugLoggingTriggerListener());
42      }
43    }
44  }
