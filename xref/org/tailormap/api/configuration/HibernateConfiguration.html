6   package org.tailormap.api.configuration;
7   
8   import org.hibernate.cfg.AvailableSettings;
9   import org.springframework.boot.hibernate.autoconfigure.HibernatePropertiesCustomizer;
10  import org.springframework.context.annotation.Bean;
11  import org.springframework.context.annotation.Configuration;
12  
13  /**
14   * Configuration class to customize Hibernate properties. This class registers a custom JSON format mapper for Hibernate
15   * to use the Jackson 3 library for JSON serialization and deserialization. By defining a bean of type
16   * {@link HibernatePropertiesCustomizer}, we can customize the Hibernate properties to specify our custom JSON format
17   * mapper, which allows Hibernate to handle JSON types in a way that is compatible with the application's JSON
18   * processing needs, especially since Hibernate 7.2.x does not natively support Jackson 3. See
19   * {@link Jackson3JsonFormatMapper} for the implementation of the custom JSON format mapper and:
20   */
21  @Configuration
22  public class HibernateConfiguration {
23    @Bean
24    HibernatePropertiesCustomizer hibernatePropertiesCustomizer() {
25      return hibernateProperties -> {
26        // Use the custom JSON format mapper for Hibernate to handle JSON types with Jackson.
27        // This allows Hibernate to serialize and deserialize JSON properties using the Jackson 3 library, which is
28        // used throughout the application.
29        // The custom format mapper is necessary because Hibernate 7.2.x does not natively support Jackson 3,
30        // and the default JSON handling may not be compatible with the application's JSON processing needs.
31        hibernateProperties.put(AvailableSettings.JSON_FORMAT_MAPPER, Jackson3JsonFormatMapper.class.getName());
32      };
33    }
34  }
