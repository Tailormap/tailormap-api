7   package org.tailormap.api.configuration.base;
8   
9   import static jakarta.servlet.http.HttpServletResponse.SC_FORBIDDEN;
10  import static jakarta.servlet.http.HttpServletResponse.SC_UNAUTHORIZED;
11  
12  import jakarta.servlet.Filter;
13  import jakarta.servlet.FilterChain;
14  import jakarta.servlet.ServletException;
15  import jakarta.servlet.ServletRequest;
16  import jakarta.servlet.ServletResponse;
17  import jakarta.servlet.http.HttpServletRequest;
18  import jakarta.servlet.http.HttpServletResponse;
19  import java.io.IOException;
20  import java.nio.charset.StandardCharsets;
21  import java.util.Base64;
22  import org.springframework.beans.factory.annotation.Value;
23  import org.springframework.stereotype.Component;
24  
25  @Component
26  public class SourceMapSecurityFilter implements Filter {
27  
28    @Value("${tailormap-api.source-map.auth:#null}")
29    private String sourceMapAuth;
30  
31    @Override
32    public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain chain)
33        throws ServletException, IOException {
34  
35      if (!"public".equals(sourceMapAuth)) {
36        HttpServletRequest request = (HttpServletRequest) servletRequest;
37        HttpServletResponse response = (HttpServletResponse) servletResponse;
38        String path =
39            request.getRequestURI().substring(request.getContextPath().length());
40        if (!path.startsWith("/api/") && path.endsWith(".map")) {
41          if (sourceMapAuth == null) {
42            response.sendError(SC_FORBIDDEN);
43            return;
44          }
45          String sourceMapAuthorization =
46              "Basic " + Base64.getEncoder().encodeToString(sourceMapAuth.getBytes(StandardCharsets.UTF_8));
47          if (!sourceMapAuthorization.equals(request.getHeader("Authorization"))) {
48            response.addHeader("WWW-Authenticate", "Basic realm=\"Source maps\"");
49            response.sendError(SC_UNAUTHORIZED);
50            return;
51          }
52        }
53      }
54  
55      chain.doFilter(servletRequest, servletResponse);
56    }
57  }
