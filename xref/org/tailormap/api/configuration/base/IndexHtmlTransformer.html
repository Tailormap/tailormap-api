7   package org.tailormap.api.configuration.base;
8   
9   import static java.nio.charset.StandardCharsets.UTF_8;
10  
11  import jakarta.servlet.http.HttpServletRequest;
12  import java.io.IOException;
13  import org.springframework.beans.factory.annotation.Value;
14  import org.springframework.core.io.Resource;
15  import org.springframework.lang.NonNull;
16  import org.springframework.stereotype.Component;
17  import org.springframework.web.servlet.resource.ResourceTransformer;
18  import org.springframework.web.servlet.resource.ResourceTransformerChain;
19  import org.springframework.web.servlet.resource.TransformedResource;
20  
21  @Component
22  public class IndexHtmlTransformer implements ResourceTransformer {
23    @Value("${tailormap-api.sentry.dsn:#{null}}")
24    private String sentryDsn;
25  
26    @Override
27    @NonNull public Resource transform(
28        @NonNull HttpServletRequest request,
29        @NonNull Resource resource,
30        @NonNull ResourceTransformerChain transformerChain)
31        throws IOException {
32      resource = transformerChain.transform(request, resource);
33  
34      if (sentryDsn == null || !"index.html".equals(resource.getFilename())) {
35        return resource;
36      }
37  
38      String html = resource.getContentAsString(UTF_8);
39      html = html.replace("@SENTRY_DSN@", sentryDsn);
40      return new TransformedResource(resource, html.getBytes(UTF_8));
41    }
42  }
