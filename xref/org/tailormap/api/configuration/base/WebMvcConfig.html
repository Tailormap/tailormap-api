7   package org.tailormap.api.configuration.base;
8   
9   import java.util.List;
10  import org.springframework.beans.factory.annotation.Value;
11  import org.springframework.context.annotation.Configuration;
12  import org.springframework.context.annotation.Lazy;
13  import org.springframework.format.FormatterRegistry;
14  import org.springframework.http.CacheControl;
15  import org.springframework.lang.NonNull;
16  import org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry;
17  import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;
18  import org.springframework.web.servlet.resource.EncodedResourceResolver;
19  import org.tailormap.api.configuration.CaseInsensitiveEnumConverter;
20  import org.tailormap.api.persistence.json.GeoServiceProtocol;
21  import org.tailormap.api.repository.ApplicationRepository;
22  import org.tailormap.api.repository.ConfigurationRepository;
23  import org.tailormap.api.scheduling.TaskType;
24  
25  @Configuration
26  public class WebMvcConfig implements WebMvcConfigurer {
27    private final IndexHtmlTransformer indexHtmlTransformer;
28  
29    @Value("${spring.web.resources.static-locations:file:/home/spring/static/}")
30    private String resourceLocations;
31  
32    @Value("#{'${tailormap-api.supported-languages:en}'.split(',')}")
33    private List<String> supportedLanguages;
34  
35    @Value("${spring.profiles.active:}")
36    private String activeProfile;
37  
38    private final ConfigurationRepository configurationRepository;
39    private final ApplicationRepository applicationRepository;
40  
41    public WebMvcConfig(
42        IndexHtmlTransformer indexHtmlTransformer,
43        // Inject these repositories lazily because in the static-only profile these are not needed
44        // but also not configured
45        @Lazy ConfigurationRepository configurationRepository,
46        @Lazy ApplicationRepository applicationRepository) {
47      this.indexHtmlTransformer = indexHtmlTransformer;
48      this.configurationRepository = configurationRepository;
49      this.applicationRepository = applicationRepository;
50    }
51  
52    @Override
53    public void addResourceHandlers(ResourceHandlerRegistry registry) {
54      registry
55          .addResourceHandler("/version.json")
56          .addResourceLocations(resourceLocations.split(",", -1)[0])
57          .setCacheControl(CacheControl.noStore());
58      registry
59          .addResourceHandler("/**")
60          .addResourceLocations(resourceLocations.split(",", -1)[0])
61          // no-cache means the browser must revalidate index.html with a conditional HTTP request
62          // using If-Modified-Since. This is needed to always have the latest frontend loaded in the
63          // browser after deployment of a new release.
64          .setCacheControl(CacheControl.noCache())
65          .resourceChain(true)
66          .addResolver(
67              new FrontControllerResolver(
68                  configurationRepository,
69                  applicationRepository,
70                  supportedLanguages,
71                  activeProfile.contains("static-only")))
72          .addResolver(new EncodedResourceResolver())
73          .addTransformer(indexHtmlTransformer);
74    }
75  
76    @Override
77    public void addFormatters(@NonNull FormatterRegistry registry) {
78      registry.addConverter(
79          String.class,
80          GeoServiceProtocol.class,
81          new CaseInsensitiveEnumConverter<>(GeoServiceProtocol.class));
82  
83      registry.addConverter(
84          String.class, TaskType.class, new CaseInsensitiveEnumConverter<>(TaskType.class));
85    }
86  }
