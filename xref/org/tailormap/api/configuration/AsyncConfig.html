7   package org.tailormap.api.configuration;
8   
9   import java.util.concurrent.Executor;
10  import org.springframework.context.annotation.Bean;
11  import org.springframework.context.annotation.Configuration;
12  import org.springframework.scheduling.annotation.EnableAsync;
13  import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;
14  
15  @Configuration
16  @EnableAsync
17  public class AsyncConfig {
18  
19    @Bean(name = "passwordResetTaskExecutor")
20    public Executor passwordResetTaskExecutor() {
21      ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
22      executor.setCorePoolSize(1);
23      executor.setMaxPoolSize(1);
24      executor.setQueueCapacity(10);
25      executor.setThreadNamePrefix("pwd-reset-");
26      executor.setWaitForTasksToCompleteOnShutdown(true);
27      executor.setAwaitTerminationSeconds(30);
28      executor.initialize();
29      return executor;
30    }
31  }
