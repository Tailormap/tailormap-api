7   package org.tailormap.api.configuration;
8   
9   import java.io.InputStream;
10  import java.io.ObjectInputStream;
11  import org.springframework.context.annotation.Bean;
12  import org.springframework.context.annotation.Configuration;
13  import org.springframework.core.convert.ConversionService;
14  import org.springframework.core.convert.support.GenericConversionService;
15  import org.springframework.core.serializer.Deserializer;
16  import org.springframework.core.serializer.support.DeserializingConverter;
17  import org.springframework.core.serializer.support.SerializingConverter;
18  
19  /**
20   * Ignore deserialization exceptions, see <a
21   * href="https://github.com/spring-projects/spring-session/issues/529#issuecomment-2761671945">here</a>.
22   */
23  @Configuration
24  public class JdbcSessionConfiguration {
25    @Bean("springSessionConversionService")
26    public ConversionService springSessionConversionService() {
27      GenericConversionService converter = new GenericConversionService();
28      converter.addConverter(Object.class, byte[].class, new SerializingConverter());
29      converter.addConverter(byte[].class, Object.class, new DeserializingConverter(new CustomDeserializer()));
30      return converter;
31    }
32  
33    static class CustomDeserializer implements Deserializer<Object> {
34      @Override
35      public Object deserialize(InputStream inputStream) {
36        try (ObjectInputStream ois = new ObjectInputStream(inputStream)) {
37          return ois.readObject();
38        } catch (Exception ignored) {
39          return null;
40        }
41      }
42    }
43  }
