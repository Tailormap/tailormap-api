197   @EventListener(ApplicationReadyEvent.class)
198   @Transactional
199   public void populate() throws Exception {
200     InternalAdminAuthentication.setInSecurityContext();
201     try {
202       // Used in conjunction with tailormap-api.database.clean=true so the database has been cleaned
203       // and the latest schema re-created
204       createTestUsersAndGroups();
205       createConfigurationTestData();
206       if (categories.contains("catalog")) {
207         createCatalogTestData();
208       }
209       if (categories.contains("apps")) {
210         createAppTestData();
211       }
212       if (categories.contains("search-index")) {
213         try {
214           createSolrIndex();
215         } catch (Exception e) {
216           logger.error("Exception creating Solr Index for testdata (continuing)", e);
217         }
218       }
219       if (categories.contains("tasks")) {
220         createScheduledTasks();
221       }
222       if (categories.contains("pages")) {
223         createPages();
224       }
225       logger.info("Test entities created");
226       if (categories.contains("drawing")) {
227         insertTestDrawing();
228         logger.info("Test drawing created");
229       }
230     } finally {
231       InternalAdminAuthentication.clearSecurityContextAuthentication();
232     }
233     if (exit) {
234       // Exit after transaction is completed - for 'mvn verify' to populate testdata before
235       // integration tests
236       new Thread(() -> {
237             try {
238               logger.info("Exiting in 10 seconds");
239               Thread.sleep(10000);
240             } catch (InterruptedException ignored) {
241               // Ignore
242             }
243             SpringApplication.exit(appContext, () -> 0);
244             System.exit(0);
245           })
246           .start();
247     }
248   }
