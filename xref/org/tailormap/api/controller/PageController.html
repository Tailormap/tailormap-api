37  @AppRestController
38  public class PageController {
39  
40    private final ConfigurationRepository configurationRepository;
41    private final ApplicationRepository applicationRepository;
42    private final AuthorizationService authorizationService;
43    private final PageRepository pageRepository;
44    private final UploadHelper uploadHelper;
45  
46    public PageController(
47        ConfigurationRepository configurationRepository,
48        ApplicationRepository applicationRepository,
49        AuthorizationService authorizationService,
50        PageRepository pageRepository,
51        UploadHelper uploadHelper) {
52      this.configurationRepository = configurationRepository;
53      this.applicationRepository = applicationRepository;
54      this.pageRepository = pageRepository;
55      this.authorizationService = authorizationService;
56      this.uploadHelper = uploadHelper;
57    }
58  
59    private ResponseStatusException notFound() {
60      return new ResponseStatusException(HttpStatus.NOT_FOUND);
61    }
62  
63    @GetMapping(path = "${tailormap-api.base-path}/page")
64    public PageResponse homePage() {
65      Page page =
66          configurationRepository
67              .findByKey(Configuration.HOME_PAGE)
68              .map(Configuration::getValue)
69              .map(Long::parseLong)
70              .flatMap(pageRepository::findById)
71              .orElseThrow(this::notFound);
72      return getPageResponse(page);
73    }
74  
75    @GetMapping(path = {"${tailormap-api.base-path}/page/{name}"})
76    public PageResponse page(@PathVariable(required = false) String name) {
77      return pageRepository.findByName(name).map(this::getPageResponse).orElseThrow(this::notFound);
78    }
79  
80    private static class ViewerPageTileResult {
81      ViewerPageTile viewerPageTile;
82      boolean shouldBeFiltered;
83    }
84  
85    private PageResponse getPageResponse(Page page) {
86      PageResponse pageResponse = new PageResponse();
87      copyProperties(page, pageResponse);
88      pageResponse.tiles(
89          page.getTiles().stream()
90              .map(this::convert)
91              .filter(viewerPageTileResult -> !viewerPageTileResult.shouldBeFiltered)
92              .map(viewerPageTileResult -> viewerPageTileResult.viewerPageTile)
93              .toList());
94  
95      List<MenuItem> menuItems =
96          configurationRepository
97              .findByKey(Configuration.PORTAL_MENU)
98              .map(Configuration::getJsonValue)
99              .filter(JsonNode::isArray)
100             .map(
101                 jsonNode -> {
102                   try {
103                     return Arrays.asList(
104                         new ObjectMapper().treeToValue(jsonNode, MenuItem[].class));
105                   } catch (IOException e) {
106                     throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, null, e);
107                   }
108                 })
109             .orElse(Collections.emptyList());
110 
111     List<ViewerMenuItem> viewerMenuItems =
112         menuItems.stream()
113             .filter(
114                 menuItem ->
115                     menuItem.getExclusiveOnPageId() == null
116                         || menuItem.getExclusiveOnPageId().equals(page.getId()))
117             .map(
118                 menuItem -> {
119                   ViewerMenuItem viewerMenuItem = new ViewerMenuItem();
120                   copyProperties(menuItem, viewerMenuItem);
121                   Optional.ofNullable(menuItem.getPageId())
122                       .flatMap(pageRepository::findById)
123                       .ifPresent(
124                           linkedPage -> viewerMenuItem.pageUrl("/page/" + linkedPage.getName()));
125                   return viewerMenuItem;
126                 })
127             .toList();
128     pageResponse.setMenu(viewerMenuItems);
129 
130     return pageResponse;
131   }
132 
133   /**
134    * @param tile The page tile configuration
135    * @return A page tile for the viewer with a boolean set to whether the tile should not be shown
136    *     (filtered).
137    */
138   private ViewerPageTileResult convert(PageTile tile) {
139     ViewerPageTile viewerPageTile = new ViewerPageTile();
140     ViewerPageTileResult result = new ViewerPageTileResult();
141     result.viewerPageTile = viewerPageTile;
142     result.shouldBeFiltered = false;
143     copyProperties(tile, viewerPageTile);
144 
145     if (tile.getApplicationId() != null) {
146       Optional.ofNullable(tile.getApplicationId())
147           .flatMap(applicationRepository::findById)
148           .filter(
149               application ->
150                   !Boolean.TRUE.equals(tile.getFilterRequireAuthorization())
151                       || authorizationService.mayUserRead(application))
152           .ifPresentOrElse(
153               application -> {
154                 viewerPageTile.applicationUrl("/app/" + application.getName());
155                 viewerPageTile.setApplicationRequiresLogin(
156                     !authorizationService.mayUserRead(application));
157               },
158               () -> result.shouldBeFiltered = true);
159     }
160 
161     Optional.ofNullable(tile.getPageId())
162         .flatMap(pageRepository::findById)
163         .ifPresent(linkedPage -> viewerPageTile.pageUrl("/page/" + linkedPage.getName()));
164 
165     viewerPageTile.image(
166         uploadHelper.getUrlForImage(tile.getImage(), Upload.CATEGORY_PORTAL_IMAGE));
167 
168     return result;
169   }
