37  @AppRestController
38  public class PageController {
39  
40    private final ConfigurationRepository configurationRepository;
41    private final ApplicationRepository applicationRepository;
42    private final AuthorisationService authorisationService;
43    private final PageRepository pageRepository;
44    private final UploadHelper uploadHelper;
45  
46    public PageController(
47        ConfigurationRepository configurationRepository,
48        ApplicationRepository applicationRepository,
49        AuthorisationService authorisationService,
50        PageRepository pageRepository,
51        UploadHelper uploadHelper) {
52      this.configurationRepository = configurationRepository;
53      this.applicationRepository = applicationRepository;
54      this.pageRepository = pageRepository;
55      this.authorisationService = authorisationService;
56      this.uploadHelper = uploadHelper;
57    }
58  
59    private ResponseStatusException notFound() {
60      return new ResponseStatusException(HttpStatus.NOT_FOUND);
61    }
62  
63    @GetMapping(path = "${tailormap-api.base-path}/page")
64    public PageResponse homePage() {
65      Page page = configurationRepository
66          .findByKey(Configuration.HOME_PAGE)
67          .map(Configuration::getValue)
68          .map(Long::parseLong)
69          .flatMap(pageRepository::findById)
70          .orElseThrow(this::notFound);
71      return getPageResponse(page);
72    }
73  
74    @GetMapping(path = {"${tailormap-api.base-path}/page/{name}"})
75    public PageResponse page(@PathVariable(required = false) String name) {
76      return pageRepository.findByName(name).map(this::getPageResponse).orElseThrow(this::notFound);
77    }
78  
79    private static class ViewerPageTileResult {
80      ViewerPageTile viewerPageTile;
81      boolean shouldBeFiltered;
82    }
83  
84    private PageResponse getPageResponse(Page page) {
85      PageResponse pageResponse = new PageResponse();
86      copyProperties(page, pageResponse);
87      pageResponse.tiles(page.getTiles().stream()
88          .map(this::convert)
89          .filter(viewerPageTileResult -> !viewerPageTileResult.shouldBeFiltered)
90          .map(viewerPageTileResult -> viewerPageTileResult.viewerPageTile)
91          .toList());
92  
93      List<MenuItem> menuItems = configurationRepository
94          .findByKey(Configuration.PORTAL_MENU)
95          .map(Configuration::getJsonValue)
96          .filter(JsonNode::isArray)
97          .map(jsonNode -> {
98            try {
99              return Arrays.asList(new ObjectMapper().treeToValue(jsonNode, MenuItem[].class));
100           } catch (IOException e) {
101             throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, null, e);
102           }
103         })
104         .orElse(Collections.emptyList());
105 
106     List<ViewerMenuItem> viewerMenuItems = menuItems.stream()
107         .filter(menuItem -> menuItem.getExclusiveOnPageId() == null
108             || menuItem.getExclusiveOnPageId().equals(page.getId()))
109         .map(menuItem -> {
110           ViewerMenuItem viewerMenuItem = new ViewerMenuItem();
111           copyProperties(menuItem, viewerMenuItem);
112           Optional.ofNullable(menuItem.getPageId())
113               .flatMap(pageRepository::findById)
114               .ifPresent(linkedPage -> viewerMenuItem.pageUrl("/page/" + linkedPage.getName()));
115           return viewerMenuItem;
116         })
117         .toList();
118     pageResponse.setMenu(viewerMenuItems);
119 
120     return pageResponse;
121   }
122 
123   /**
124    * @param tile The page tile configuration
125    * @return A page tile for the viewer with a boolean set to whether the tile should not be shown (filtered).
126    */
127   private ViewerPageTileResult convert(PageTile tile) {
128     ViewerPageTile viewerPageTile = new ViewerPageTile();
129     ViewerPageTileResult result = new ViewerPageTileResult();
130     result.viewerPageTile = viewerPageTile;
131     result.shouldBeFiltered = false;
132     copyProperties(tile, viewerPageTile);
133 
134     if (tile.getApplicationId() != null) {
135       Optional.ofNullable(tile.getApplicationId())
136           .flatMap(applicationRepository::findById)
137           .filter(application -> !Boolean.TRUE.equals(tile.getFilterRequireAuthorization())
138               || authorisationService.userAllowedToViewApplication(application))
139           .ifPresentOrElse(
140               application -> {
141                 viewerPageTile.applicationUrl("/app/" + application.getName());
142                 viewerPageTile.setApplicationRequiresLogin(
143                     !authorisationService.userAllowedToViewApplication(application));
144               },
145               () -> result.shouldBeFiltered = true);
146     }
147 
148     Optional.ofNullable(tile.getPageId())
149         .flatMap(pageRepository::findById)
150         .ifPresent(linkedPage -> viewerPageTile.pageUrl("/page/" + linkedPage.getName()));
151 
152     viewerPageTile.image(uploadHelper.getUrlForImage(tile.getImage(), Upload.CATEGORY_PORTAL_IMAGE));
153 
154     return result;
155   }
