58  /**
59   * Proxy controller for OGC WMS, WMTS, and 3D Tiles services. Does not attempt to hide the original service URL. Mostly
60   * useful for access to HTTP Basic secured services without sending the credentials to the client. The access control is
61   * handled by Spring Security and the authorizations configured on the service.
62   *
63   * <p>Only supports GET requests. Does not support CORS, only meant for tailormap-viewer from the same origin.
64   *
65   * <p>Implementation note: uses the Java 11 HttpClient. Spring cloud gateway can proxy with many more features but can
66   * not be used in a non-reactive application.
67   */
68  @AppRestController
69  @Validated
70  // Can't use ${tailormap-api.base-path} because linkTo() won't work
71  @RequestMapping(path = "/api/{viewerKind}/{viewerName}/layer/{appLayerId}/proxy")
72  public class GeoServiceProxyController {
73    private static final Logger logger =
74        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
75    private final AuthorizationService authorizationService;
76    public static final String TILES3D_DESCRIPTION_PATH = "tiles3dDescription";
77  
78    public GeoServiceProxyController(AuthorizationService authorizationService) {
79      this.authorizationService = authorizationService;
80    }

82    @RequestMapping(
83        method = {GET, POST},
84        path = "/tiles3d/**")
85    public ResponseEntity<?> proxy3dtiles(
86        @ModelAttribute Application application,
87        @ModelAttribute GeoService service,
88        @ModelAttribute GeoServiceLayer layer,
89        HttpServletRequest request) {
90  
91      checkRequestValidity(application, service, layer, GeoServiceProtocol.TILES3D);
92  
93      return doProxy(build3DTilesUrl(service, request), service, request);
94    }

96    @RequestMapping(
97        method = {GET, POST},
98        path = "/{protocol}")
99    @Timed(value = "proxy", description = "Proxy OGC service calls")
100   public ResponseEntity<?> proxy(
101       @ModelAttribute Application application,
102       @ModelAttribute GeoService service,
103       @ModelAttribute GeoServiceLayer layer,
104       @PathVariable("protocol") GeoServiceProtocol protocol,
105       HttpServletRequest request) {
106 
107     checkRequestValidity(application, service, layer, protocol);
108 
109     switch (protocol) {
110       case WMS, WMTS -> {
111         return doProxy(buildWMSUrl(service, request), service, request);
112       }
113       case LEGEND -> {
114         URI legendURI = buildLegendURI(service, layer, request);
115         if (legendURI == null) {
116           logger.warn("No legend URL found for layer {}", layer.getName());
117           return null;
118         }
119         return doProxy(legendURI, service, request);
120       }
121       case TILES3D -> throw new ResponseStatusException(
122           HttpStatus.BAD_REQUEST, "Incorrect 3D Tiles proxy request: No path to capabilities or content");
123       default -> throw new ResponseStatusException(
124           HttpStatus.BAD_REQUEST, "Unsupported proxy protocol: " + protocol);
125     }
126   }

128   private void checkRequestValidity(
129       Application application, GeoService service, GeoServiceLayer layer, GeoServiceProtocol protocol) {
130     if (service == null || layer == null) {
131       throw new ResponseStatusException(HttpStatus.NOT_FOUND);
132     }
133 
134     if (GeoServiceProtocol.XYZ.equals(protocol)) {
135       throw new ResponseStatusException(HttpStatus.NOT_IMPLEMENTED, "XYZ proxying not implemented");
136     }
137 
138     if (!(service.getProtocol().equals(protocol) || GeoServiceProtocol.LEGEND.equals(protocol))) {
139       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Invalid proxy protocol: " + protocol);
140     }
141 
142     if (!service.getSettings().getUseProxy()) {
143       throw new ResponseStatusException(HttpStatus.FORBIDDEN, "Proxy not enabled for requested service");
144     }
145 
146     if (authorizationService.mustDenyAccessForSecuredProxy(application, service)) {
147       logger.warn(
148           "App {} (\"{}\") is using layer \"{}\" from proxied secured service URL {} (username \"{}\"), but app is publicly accessible. Denying proxy, even if user is authenticated.",
149           application.getId(),
150           application.getName(),
151           layer.getName(),
152           service.getUrl(),
153           service.getAuthentication().getUsername());
154       throw new ResponseStatusException(HttpStatus.FORBIDDEN);
155     }
156   }

158   private @Nullable URI buildLegendURI(GeoService service, GeoServiceLayer layer, HttpServletRequest request) {
159     URI legendURI = GeoServiceHelper.getLayerLegendUrlFromStyles(service, layer);
160     if (legendURI == null) {
161       return null;
162     }
163 
164     // If the original service legend URL is not a GetLegendGraphic request, do not add any parameters from the
165     // incoming request and proxy the request as is
166     String wmsRequest = getWmsRequest(legendURI);
167     if (!"getlegendgraphic".equalsIgnoreCase(wmsRequest)) {
168       return legendURI;
169     }
170 
171     // Add all parameters from the incoming request to allow for vendor-specific parameters to enhance the legend
172     // image, such as font antialiasing, hi-DPI, label margins, and so on
173     UriComponentsBuilder uriComponentsBuilder = UriComponentsBuilder.fromUri(legendURI);
174     if (request.getParameterMap() != null) {
175       request.getParameterMap().forEach((key, values) -> {
176         for (String value : values) {
177           uriComponentsBuilder.replaceQueryParam(key, UriUtils.encode(value, StandardCharsets.UTF_8));
178         }
179       });
180     }
181     // Make sure the REQUEST parameter is set to GetLegendGraphic. No SSRF risk as WMSes should not do anything with
182     // other parameters other than returning a legend image
183     uriComponentsBuilder.replaceQueryParam("REQUEST", "GetLegendGraphic");
184 
185     return uriComponentsBuilder.build(true).toUri();
186   }

188   private URI buildWMSUrl(GeoService service, HttpServletRequest request) {
189     final UriComponentsBuilder originalServiceUrl = UriComponentsBuilder.fromUriString(service.getUrl());
190     // request.getParameterMap() includes parameters from an application/x-www-form-urlencoded POST
191     // body
192     final MultiValueMap<String, String> requestParams = request.getParameterMap().entrySet().stream()
193         .map(entry -> new AbstractMap.SimpleEntry<>(
194             entry.getKey(),
195             Arrays.stream(entry.getValue())
196                 .map(value -> UriUtils.encode(value, StandardCharsets.UTF_8))
197                 .collect(Collectors.toList())))
198         .collect(Collectors.toMap(
199             Map.Entry::getKey, Map.Entry::getValue, (x, y) -> y, LinkedMultiValueMap::new));
200     final MultiValueMap<String, String> params =
201         buildOgcProxyRequestParams(originalServiceUrl.build(true).getQueryParams(), requestParams);
202     originalServiceUrl.replaceQueryParams(params);
203     return originalServiceUrl.build(true).toUri();
204   }

206   public static MultiValueMap<String, String> buildOgcProxyRequestParams(
207       MultiValueMap<String, String> originalServiceParams, MultiValueMap<String, String> requestParams) {
208     // Start with all the parameters from the request
209     final MultiValueMap<String, String> params = new LinkedMultiValueMap<>(requestParams);
210 
211     // Add original service URL parameters if they are not required OGC params (case-insensitive)
212     // and not already set by request
213     final List<String> ogcParams = List.of(new String[] {"SERVICE", "REQUEST", "VERSION"});
214     for (Map.Entry<String, List<String>> serviceParam : originalServiceParams.entrySet()) {
215       if (!params.containsKey(serviceParam.getKey())
216           && !ogcParams.contains(serviceParam.getKey().toUpperCase(Locale.ROOT))) {
217         params.put(serviceParam.getKey(), serviceParam.getValue());
218       }
219     }
220     return params;
221   }

223   private URI build3DTilesUrl(GeoService service, HttpServletRequest request) {
224     // The URL in the GeoService refers to the location of the JSON file describing the tileset,
225     // e.g. example.com/buildings/3dtiles. The paths to the subtrees and tiles of the tilesets do
226     // not include the '/3dtiles' (or '/tileset.json') part of the path. Their paths are e.g.
227     // example.com/buildings/subtrees/... or example.com/buildings/t/...
228     final UriComponentsBuilder originalServiceUrl = UriComponentsBuilder.fromUriString(service.getUrl());
229     String baseUrl = originalServiceUrl.build(true).toUriString();
230     String pathToContent = request.getRequestURI().split("/proxy/tiles3d/", 2)[1];
231 
232     // Return service URL when the request is for the JSON file describing the tileset
233     if (Objects.equals(pathToContent, TILES3D_DESCRIPTION_PATH)) {
234       return UriComponentsBuilder.fromUriString(baseUrl).build(true).toUri();
235     }
236 
237     // Remove the part of the service URL referring to the JSON file describing the tileset
238     int lastSlashIndex = baseUrl.lastIndexOf('/');
239     if (lastSlashIndex != -1) {
240       baseUrl = baseUrl.substring(0, lastSlashIndex + 1);
241     }
242 
243     // Return final URL with specific path to the tile or subtree
244     String finalUrl = baseUrl + pathToContent;
245     return UriComponentsBuilder.fromUriString(finalUrl).build(true).toUri();
246   }

248   private static ResponseEntity<?> doProxy(URI uri, GeoService service, HttpServletRequest request) {
249     final HttpClient.Builder builder = HttpClient.newBuilder().followRedirects(HttpClient.Redirect.NORMAL);
250     final HttpClient httpClient = builder.build();
251 
252     HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(uri);
253 
254     addForwardedForRequestHeaders(requestBuilder, request);
255 
256     passthroughRequestHeaders(
257         requestBuilder,
258         request,
259         Set.of(
260             "Accept",
261             "If-Modified-Since",
262             "If-Unmodified-Since",
263             "If-Match",
264             "If-None-Match",
265             "If-Range",
266             "Range",
267             "Referer",
268             "User-Agent"));
269 
270     if (service.getAuthentication() != null
271         && service.getAuthentication().getMethod() == ServiceAuthentication.MethodEnum.PASSWORD) {
272       setHttpBasicAuthenticationHeader(
273           requestBuilder,
274           service.getAuthentication().getUsername(),
275           service.getAuthentication().getPassword());
276     }
277 
278     try {
279       // TODO: close JPA connection before proxying
280       HttpResponse<InputStream> response =
281           httpClient.send(requestBuilder.build(), HttpResponse.BodyHandlers.ofInputStream());
282 
283       // If the server does not accept our credentials it might 'hide' the layer or even send a 401
284       // Unauthorized status. We do not send the WWW-Authenticate header back, so the client will
285       // get the error but not an authorization popup.
286       // It would be nice if proxy (auth) errors were logged and available in the admin interface.
287       // Currently, a layer will just stop working if the geo service credentials are changed
288       // without updating them in the geo service registry.
289       InputStreamResource body = new InputStreamResource(response.body());
290       HttpHeaders headers = passthroughResponseHeaders(
291           response.headers(),
292           Set.of(
293               "Content-Type",
294               "Content-Length",
295               "Content-Range",
296               "Content-Disposition",
297               "Cache-Control",
298               "Expires",
299               "Last-Modified",
300               "ETag",
301               "Pragma"));
302       return ResponseEntity.status(response.statusCode()).headers(headers).body(body);
303     } catch (Exception e) {
304       return ResponseEntity.status(HttpStatus.BAD_GATEWAY).body("Bad Gateway");
305     }
306   }
