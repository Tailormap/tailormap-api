54  @AppRestController
55  @Validated
56  @RequestMapping(
57      path = "${tailormap-api.base-path}/{viewerKind}/{viewerName}/layer/{appLayerId}/unique/{attributeName}",
58      produces = MediaType.APPLICATION_JSON_VALUE)
59  public class UniqueValuesController {
60    private static final Logger logger =
61        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
62  
63    private final FeatureSourceFactoryHelper featureSourceFactoryHelper;
64  
65    private final FeatureSourceRepository featureSourceRepository;
66  
67    @Value("${tailormap-api.unique.use_geotools_unique_function:true}")
68    private boolean useGeotoolsUniqueFunction;
69  
70    private final FilterFactory ff = CommonFactoryFinder.getFilterFactory(GeoTools.getDefaultHints());
71  
72    public UniqueValuesController(
73        FeatureSourceFactoryHelper featureSourceFactoryHelper, FeatureSourceRepository featureSourceRepository) {
74      this.featureSourceFactoryHelper = featureSourceFactoryHelper;
75      this.featureSourceRepository = featureSourceRepository;
76    }
77  
78    @Transactional
79    @RequestMapping(method = {GET, POST})
80    @Timed(value = "get_unique_attributes", description = "time spent to process get unique attributes call")
81    @Counted(value = "get_unique_attributes", description = "number of unique attributes calls")
82    public ResponseEntity<Serializable> getUniqueAttributes(
83        @ModelAttribute GeoService service,
84        @ModelAttribute GeoServiceLayer layer,
85        @ModelAttribute Application app,
86        @ModelAttribute AppTreeLayerNode appTreeLayerNode,
87        @PathVariable("attributeName") String attributeName,
88        @RequestParam(required = false) String filter) {
89      if (StringUtils.isBlank(attributeName)) {
90        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Attribute name is required");
91      }
92  
93      TMFeatureType tmft = service.findFeatureTypeForLayer(layer, featureSourceRepository);
94      AppLayerSettings appLayerSettings = app.getAppLayerSettings(appTreeLayerNode);
95      if (tmft == null) {
96        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "Layer does not have feature type");
97      }
98      if (!getNonHiddenAttributeNames(tmft, appLayerSettings).contains(attributeName)) {
99        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Attribute does not exist");
100     }
101     UniqueValuesResponse uniqueValuesResponse = getUniqueValues(tmft, attributeName, filter);
102     return ResponseEntity.status(HttpStatus.OK).body(uniqueValuesResponse);
103   }
104 
105   private UniqueValuesResponse getUniqueValues(TMFeatureType tmft, String attributeName, String filter) {
106     final UniqueValuesResponse uniqueValuesResponse = new UniqueValuesResponse().filterApplied(false);
107     SimpleFeatureSource fs = null;
108     try {
109       Filter existingFilter = null;
110       if (null != filter) {
111         existingFilter = ECQL.toFilter(filter);
112       }
113       logger.trace("existingFilter: {}", existingFilter);
114 
115       Filter notNull = ff.not(ff.isNull(ff.property(attributeName)));
116       Filter f = notNull;
117       if (null != existingFilter) {
118         f = ff.and(notNull, existingFilter);
119         uniqueValuesResponse.filterApplied(true);
120       }
121 
122       Query q = new Query(tmft.getName(), f);
123       q.setPropertyNames(attributeName);
124       q.setSortBy(ff.sort(attributeName, SortOrder.ASCENDING));
125       logger.trace("Unique values query: {}", q);
126 
127       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmft);
128       // and then there are 2 scenarios:
129       // there might be a performance benefit for one or the other
130       if (!useGeotoolsUniqueFunction) {
131         // #1 use a feature visitor to get the unique values
132         // not recommended, as it may not be performant
133         logger.trace("Using feature visitor to get unique values");
134         fs.getFeatures(q)
135             .accepts(
136                 feature -> uniqueValuesResponse.addValuesItem(
137                     feature.getProperty(attributeName).getValue()),
138                 null);
139       } else {
140         // #2 or use a Function to get the unique values
141         // this is the recommended way, uses SQL "distinct"
142         logger.trace("Using geotools unique collection function to get unique values");
143         Function unique = ff.function("Collection_Unique", ff.property(attributeName));
144         Object o = unique.evaluate(fs.getFeatures(q));
145         if (o instanceof Set) {
146           @SuppressWarnings("unchecked")
147           Set<Object> uniqueValues = (Set<Object>) o;
148           uniqueValuesResponse.setValues(new TreeSet<>(uniqueValues));
149         }
150       }
151     } catch (CQLException e) {
152       logger.error("Could not parse requested filter", e);
153       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Could not parse requested filter");
154     } catch (IOException e) {
155       logger.error("Could not retrieve attribute data", e);
156     } finally {
157       if (fs != null) {
158         fs.getDataStore().dispose();
159       }
160     }
161 
162     return uniqueValuesResponse;
163   }
