42  @AppRestController
43  @Validated
44  @RequestMapping(
45      path = "${tailormap-api.base-path}/{viewerKind}/{viewerName}/layer/{appLayerId}/unique/{attributeName}",
46      produces = MediaType.APPLICATION_JSON_VALUE)
47  public class UniqueValuesController {
48    private final FeatureSourceFactoryHelper featureSourceFactoryHelper;
49  
50    private final FeatureSourceRepository featureSourceRepository;
51  
52    @Value("${tailormap-api.unique.use_geotools_unique_function:true}")
53    private boolean useGeotoolsUniqueFunction;
54  
55    private final FilterFactory ff = CommonFactoryFinder.getFilterFactory(GeoTools.getDefaultHints());
56  
57    public UniqueValuesController(
58        FeatureSourceFactoryHelper featureSourceFactoryHelper, FeatureSourceRepository featureSourceRepository) {
59      this.featureSourceFactoryHelper = featureSourceFactoryHelper;
60      this.featureSourceRepository = featureSourceRepository;
61    }
62  
63    @Transactional
64    @RequestMapping(method = {GET, POST})
65    @Timed(value = "get_unique_attributes", description = "time spent to process get unique attributes call")
66    @Counted(value = "get_unique_attributes", description = "number of unique attributes calls")
67    public ResponseEntity<Serializable> getUniqueAttributes(
68        @ModelAttribute GeoService service,
69        @ModelAttribute GeoServiceLayer layer,
70        @ModelAttribute Application app,
71        @ModelAttribute AppTreeLayerNode appTreeLayerNode,
72        @PathVariable("attributeName") String attributeName,
73        @RequestParam(required = false) String filter) {
74      if (StringUtils.isBlank(attributeName)) {
75        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Attribute name is required");
76      }
77  
78      TMFeatureType tmft = service.findFeatureTypeForLayer(layer, featureSourceRepository);
79      AppLayerSettings appLayerSettings = app.getAppLayerSettings(appTreeLayerNode);
80      if (tmft == null) {
81        throw new ResponseStatusException(HttpStatus.NOT_FOUND, "Layer does not have feature type");
82      }
83      if (!getNonHiddenAttributeNames(tmft, appLayerSettings).contains(attributeName)) {
84        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Attribute does not exist");
85      }
86      UniqueValuesResponse uniqueValuesResponse = UniqueValuesHelper.getUniqueValues(
87          tmft, attributeName, filter, ff, featureSourceFactoryHelper, useGeotoolsUniqueFunction);
88      return ResponseEntity.status(HttpStatus.OK).body(uniqueValuesResponse);
89    }
90  }
