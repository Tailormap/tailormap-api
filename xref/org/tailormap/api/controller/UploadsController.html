7   package org.tailormap.api.controller;
8   
9   import static org.springframework.http.HttpStatus.NOT_FOUND;
10  import static org.springframework.http.HttpStatus.NOT_MODIFIED;
11  import static org.tailormap.api.util.Constants.UUID_REGEX;
12  
13  import jakarta.servlet.http.HttpServletRequest;
14  import java.time.OffsetDateTime;
15  import java.time.temporal.ChronoField;
16  import java.util.UUID;
17  import org.springframework.http.CacheControl;
18  import org.springframework.http.ResponseEntity;
19  import org.springframework.web.bind.annotation.GetMapping;
20  import org.springframework.web.bind.annotation.PathVariable;
21  import org.springframework.web.bind.annotation.RequestMapping;
22  import org.springframework.web.bind.annotation.RestController;
23  import org.springframework.web.server.ResponseStatusException;
24  import org.tailormap.api.persistence.Upload;
25  import org.tailormap.api.repository.UploadRepository;
26  
27  @RestController
28  // Can't use ${tailormap-api.base-path} because linkTo() won't work
29  @RequestMapping(path = "/api/uploads/{category}/{id}/{filename}")
30  public class UploadsController {
31    private final UploadRepository uploadRepository;
32  
33    public UploadsController(UploadRepository uploadRepository) {
34      this.uploadRepository = uploadRepository;
35    }
36  
37    @GetMapping
38    public ResponseEntity<byte[]> getUpload(
39        HttpServletRequest request,
40        @PathVariable String category,
41        @PathVariable(name = "id") String idString,
42        @PathVariable(required = false) String filename) {
43  
44      if (!idString.matches(UUID_REGEX)) {
45        return ResponseEntity.badRequest().build();
46      }
47  
48      UUID id = UUID.fromString(idString);
49      long ifModifiedSince = request.getDateHeader("If-Modified-Since");
50      if (ifModifiedSince != -1) {
51        OffsetDateTime uploadLastModified =
52            uploadRepository.findLastModifiedById(id).orElseThrow(() -> new ResponseStatusException(NOT_FOUND));
53        if (ifModifiedSince
54            >= uploadLastModified
55                .with(ChronoField.MILLI_OF_SECOND, 0)
56                .toInstant()
57                .toEpochMilli()) {
58          return ResponseEntity.status(NOT_MODIFIED).build();
59        }
60      }
61  
62      Upload upload = uploadRepository
63          .findWithContentByIdAndCategory(id, category)
64          .orElseThrow(() -> new ResponseStatusException(NOT_FOUND));
65  
66      return ResponseEntity.ok()
67          .header("Content-Type", upload.getMimeType())
68          .lastModified(upload.getLastModified().toInstant())
69          .contentLength(upload.getContentLength())
70          .cacheControl(CacheControl.noCache().cachePublic())
71          .body(upload.getContent());
72    }
73  }
