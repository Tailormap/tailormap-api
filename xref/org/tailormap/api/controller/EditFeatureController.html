89    private static void checkFeatureHasOnlyValidAttributes(
90        Feature feature, TMFeatureType tmFeatureType, AppLayerSettings appLayerSettings) {
91  
92      if (!getNonHiddenAttributeNames(tmFeatureType, appLayerSettings)
93          .containsAll(feature.getAttributes().keySet())) {
94        throw new ResponseStatusException(
95            HttpStatus.BAD_REQUEST,
96            "Feature cannot be edited, one or more requested attributes are not available on the feature type");
97      }
98      if (!Collections.disjoint(
99          getReadOnlyAttributes(tmFeatureType, appLayerSettings),
100         feature.getAttributes().keySet())) {
101       throw new ResponseStatusException(
102           HttpStatus.BAD_REQUEST,
103           "Feature cannot be edited, one or more requested attributes are not editable on the feature type");
104     }
105   }

107   @Transactional
108   @PostMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
109   @Timed(value = "create_feature", description = "time spent to process create feature call")
110   @Counted(value = "create_feature", description = "number of create feature calls")
111   public ResponseEntity<Serializable> createFeature(
112       @ModelAttribute AppTreeLayerNode appTreeLayerNode,
113       @ModelAttribute GeoService service,
114       @ModelAttribute GeoServiceLayer layer,
115       @ModelAttribute Application application,
116       @RequestBody Feature completeFeature) {
117 
118     checkAuthentication();
119 
120     TMFeatureType tmFeatureType = getEditableFeatureType(application, appTreeLayerNode, service, layer);
121     Map<String, Object> attributesMap = completeFeature.getAttributes();
122 
123     AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
124 
125     checkFeatureHasOnlyValidAttributes(completeFeature, tmFeatureType, appLayerSettings);
126 
127     Feature newFeature;
128     SimpleFeatureSource fs = null;
129 
130     try (Transaction transaction = new DefaultTransaction("create")) {
131       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
132 
133       SimpleFeature simpleFeature;
134       SimpleFeatureBuilder simpleFeatureBuilder = new SimpleFeatureBuilder(fs.getSchema());
135       if (null != completeFeature.getFid() && !completeFeature.getFid().isEmpty()) {
136         simpleFeature = simpleFeatureBuilder.buildFeature(completeFeature.getFid());
137         simpleFeature.getUserData().put(Hints.USE_PROVIDED_FID, Boolean.TRUE);
138       } else {
139         simpleFeature = simpleFeatureBuilder.buildFeature(null);
140       }
141 
142       handleGeometryAttributesInput(
143           tmFeatureType, appLayerSettings, completeFeature, attributesMap, application, fs);
144       for (Map.Entry<String, Object> entry : attributesMap.entrySet()) {
145         simpleFeature.setAttribute(entry.getKey(), entry.getValue());
146       }
147 
148       if (fs instanceof SimpleFeatureStore simpleFeatureStore) {
149         simpleFeatureStore.setTransaction(transaction);
150         List<FeatureId> newFids = simpleFeatureStore.addFeatures(DataUtilities.collection(simpleFeature));
151 
152         transaction.commit();
153         // find the created feature to return
154         newFeature = getFeature(fs, ff.id(newFids.get(0)), application);
155       } else {
156         throw new ResponseStatusException(
157             HttpStatus.BAD_REQUEST, "Feature cannot be added, datasource is not editable");
158       }
159     } catch (RuntimeException | IOException | FactoryException e) {
160       // either opening datastore, modify or transaction failed
161       logger.error("Error creating new feature {}", completeFeature, e);
162       String message = e.getMessage();
163       if (null != e.getCause() && null != e.getCause().getMessage()) {
164         message = e.getCause().getMessage();
165       }
166       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, message, e);
167     } finally {
168       if (fs != null) {
169         fs.getDataStore().dispose();
170       }
171     }
172     return new ResponseEntity<>(newFeature, HttpStatus.OK);
173   }

175   @Transactional
176   @PatchMapping(
177       consumes = MediaType.APPLICATION_JSON_VALUE,
178       produces = MediaType.APPLICATION_JSON_VALUE,
179       path = "/{fid}")
180   @Timed(value = "update_feature", description = "time spent to process patch feature call")
181   @Counted(value = "update_feature", description = "number of patch feature calls")
182   public ResponseEntity<Serializable> patchFeature(
183       @ModelAttribute AppTreeLayerNode appTreeLayerNode,
184       @ModelAttribute GeoService service,
185       @ModelAttribute GeoServiceLayer layer,
186       @ModelAttribute Application application,
187       @PathVariable String fid,
188       @RequestBody Feature partialFeature) {
189 
190     checkAuthentication();
191 
192     TMFeatureType tmFeatureType = getEditableFeatureType(application, appTreeLayerNode, service, layer);
193     AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
194 
195     Map<String, Object> attributesMap = partialFeature.getAttributes();
196 
197     checkFeatureHasOnlyValidAttributes(partialFeature, tmFeatureType, appLayerSettings);
198 
199     Feature patchedFeature;
200     SimpleFeatureSource fs = null;
201     try (Transaction transaction = new DefaultTransaction("edit")) {
202       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
203       // find feature to update
204       final Filter filter = ff.id(ff.featureId(fid));
205       if (!fs.getFeatures(filter).isEmpty() && fs instanceof SimpleFeatureStore simpleFeatureStore) {
206         handleGeometryAttributesInput(
207             tmFeatureType, appLayerSettings, partialFeature, attributesMap, application, fs);
208         // NOTE geotools does not report back that the feature was updated, no error === success
209         simpleFeatureStore.setTransaction(transaction);
210         simpleFeatureStore.modifyFeatures(
211             attributesMap.keySet().toArray(new String[] {}),
212             attributesMap.values().toArray(),
213             filter);
214         transaction.commit();
215         // find the updated feature to return
216         patchedFeature = getFeature(fs, filter, application);
217       } else {
218         throw new ResponseStatusException(
219             HttpStatus.BAD_REQUEST, "Feature cannot be edited, it does not exist or is not editable");
220       }
221     } catch (RuntimeException | IOException | FactoryException e) {
222       // either opening datastore, modify or transaction failed
223       logger.error("Error patching feature {}", partialFeature, e);
224       String message = e.getMessage();
225       if (null != e.getCause() && null != e.getCause().getMessage()) {
226         message = e.getCause().getMessage();
227       }
228       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, message, e);
229     } finally {
230       if (fs != null) {
231         fs.getDataStore().dispose();
232       }
233     }
234     return new ResponseEntity<>(patchedFeature, HttpStatus.OK);
235   }

237   @Transactional
238   @DeleteMapping(path = "/{fid}")
239   @Timed(value = "delete_feature", description = "time spent to process delete feature call")
240   @Counted(value = "delete_feature", description = "number of delete feature calls")
241   public ResponseEntity<Void> deleteFeature(
242       @ModelAttribute AppTreeLayerNode appTreeLayerNode,
243       @ModelAttribute GeoService service,
244       @ModelAttribute GeoServiceLayer layer,
245       @ModelAttribute Application application,
246       @PathVariable String fid) {
247 
248     checkAuthentication();
249 
250     TMFeatureType tmFeatureType = getEditableFeatureType(application, appTreeLayerNode, service, layer);
251 
252     SimpleFeatureSource fs = null;
253     try (Transaction transaction = new DefaultTransaction("delete")) {
254       fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
255       Filter filter = ff.id(ff.featureId(fid));
256       if (fs instanceof FeatureStore<?, ?> featureStore) {
257         // NOTE geotools does not report back that the feature does not exist, nor the number of
258         // deleted features, no error === success
259         featureStore.setTransaction(transaction);
260         featureStore.removeFeatures(filter);
261         transaction.commit();
262       } else {
263         throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Layer cannot be edited");
264       }
265     } catch (IOException e) {
266       // either opening datastore or commit failed
267       logger.error("Error deleting feature {}", fid, e);
268       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, e.getMessage(), e);
269     } finally {
270       if (fs != null) {
271         fs.getDataStore().dispose();
272       }
273     }
274 
275     return new ResponseEntity<>(HttpStatus.NO_CONTENT);
276   }

278   private void checkAuthentication() {
279     Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
280     boolean isAuthenticated = authentication != null && !(authentication instanceof AnonymousAuthenticationToken);
281     if (!isAuthenticated) {
282       throw new ResponseStatusException(HttpStatus.UNAUTHORIZED, "Not authenticated");
283     }
284   }

286   /**
287    * Get editable feature type, throws exception if not found or not editable. Will throw a
288    * {@link ResponseStatusException} if the layer does not have an editable featuretype.
289    *
290    * @param application the application that has the editable layer
291    * @param appTreeLayerNode the layer to edit
292    * @param service the service that has the layer
293    * @param layer the layer to edit
294    * @return the editable feature type
295    */
296   private TMFeatureType getEditableFeatureType(
297       Application application, AppTreeLayerNode appTreeLayerNode, GeoService service, GeoServiceLayer layer) {
298 
299     if (null == layer) {
300       throw new ResponseStatusException(HttpStatus.NOT_FOUND, "Cannot find layer " + appTreeLayerNode);
301     }
302 
303     AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
304     if (!Boolean.TRUE.equals(appLayerSettings.getEditable())) {
305       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Layer is not editable");
306     }
307 
308     TMFeatureType tmft = service.findFeatureTypeForLayer(layer, featureSourceRepository);
309     if (null == tmft) {
310       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Layer does not have feature type");
311     }
312     if (!tmft.isWriteable()) {
313       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Feature type is not writeable");
314     }
315 
316     return tmft;
317   }

319   private static Feature getFeature(SimpleFeatureSource fs, Filter filter, Application application)
320       throws IOException, FactoryException {
321     Feature modelFeature = null;
322     try (SimpleFeatureIterator feats = fs.getFeatures(filter).features()) {
323       if (feats.hasNext()) {
324         SimpleFeature simpleFeature = feats.next();
325         modelFeature = new Feature()
326             .geometry(GeometryProcessor.processGeometry(
327                 simpleFeature.getDefaultGeometry(),
328                 false,
329                 true,
330                 TransformationUtil.getTransformationToApplication(application, fs)))
331             .fid(simpleFeature.getID());
332         for (AttributeDescriptor att : simpleFeature.getFeatureType().getAttributeDescriptors()) {
333           Object value = simpleFeature.getAttribute(att.getName());
334           if (value instanceof Geometry geometry) {
335             geometry = GeometryProcessor.transformGeometry(
336                 geometry, TransformationUtil.getTransformationToApplication(application, fs));
337             value = GeometryProcessor.geometryToWKT(geometry);
338           }
339           modelFeature.putAttributesItem(att.getLocalName(), value);
340         }
341       }
342     }
343     return modelFeature;
344   }

346   /** Handle geometry attributes, essentially this is a WKT to Geometry conversion. */
347   private static void handleGeometryAttributesInput(
348       TMFeatureType tmFeatureType,
349       AppLayerSettings appLayerSettings,
350       Feature modelFeature,
351       Map<String, Object> attributesMap,
352       Application application,
353       SimpleFeatureSource fs)
354       throws FactoryException {
355 
356     final MathTransform transform = TransformationUtil.getTransformationToDataSource(application, fs);
357     getNonHiddenAttributes(tmFeatureType, appLayerSettings).stream()
358         .filter(attr -> TMAttributeTypeHelper.isGeometry(attr.getType()))
359         .filter(attr -> modelFeature.getAttributes().containsKey(attr.getName()))
360         .forEach(attr -> {
361           Geometry geometry = GeometryProcessor.wktToGeometry(
362               (String) modelFeature.getAttributes().get(attr.getName()));
363           if (transform != null && geometry != null) {
364             geometry.setSRID(Integer.parseInt(application.getCrs().substring("EPSG:".length())));
365             if (logger.isTraceEnabled()) {
366               logger.trace(
367                   "Transforming geometry {} from {} to {}",
368                   geometry.toText(),
369                   geometry.getSRID(),
370                   fs.getSchema()
371                       .getCoordinateReferenceSystem()
372                       .getIdentifiers());
373             }
374             geometry = GeometryProcessor.transformGeometry(geometry, transform);
375           }
376           attributesMap.put(attr.getName(), geometry);
377         });
378   }
