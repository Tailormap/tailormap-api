7   package org.tailormap.api.controller;
8   
9   import static org.tailormap.api.util.Constants.UUID_REGEX;
10  
11  import io.micrometer.core.annotation.Counted;
12  import io.micrometer.core.annotation.Timed;
13  import org.springframework.http.HttpStatus;
14  import org.springframework.web.bind.annotation.GetMapping;
15  import org.springframework.web.bind.annotation.ModelAttribute;
16  import org.springframework.web.server.ResponseStatusException;
17  import org.tailormap.api.annotation.AppRestController;
18  import org.tailormap.api.persistence.Application;
19  import org.tailormap.api.persistence.Configuration;
20  import org.tailormap.api.persistence.Upload;
21  import org.tailormap.api.persistence.helper.ApplicationHelper;
22  import org.tailormap.api.persistence.helper.UploadHelper;
23  import org.tailormap.api.repository.ApplicationRepository;
24  import org.tailormap.api.repository.ConfigurationRepository;
25  import org.tailormap.api.security.AuthorizationService;
26  import org.tailormap.api.viewer.model.AppStyling;
27  import org.tailormap.api.viewer.model.MapResponse;
28  import org.tailormap.api.viewer.model.ViewerResponse;
29  
30  @AppRestController
31  public class ViewerController {
32  
33    private final ConfigurationRepository configurationRepository;
34    private final ApplicationRepository applicationRepository;
35    private final ApplicationHelper applicationHelper;
36    private final AuthorizationService authorizationService;
37    private final UploadHelper uploadHelper;
38  
39    public ViewerController(
40        ConfigurationRepository configurationRepository,
41        ApplicationRepository applicationRepository,
42        ApplicationHelper applicationHelper,
43        AuthorizationService authorizationService,
44        UploadHelper uploadHelper) {
45      this.configurationRepository = configurationRepository;
46      this.applicationRepository = applicationRepository;
47      this.applicationHelper = applicationHelper;
48      this.authorizationService = authorizationService;
49      this.uploadHelper = uploadHelper;
50    }
51  
52    @GetMapping(path = "${tailormap-api.base-path}/app")
53    @Timed(value = "get_default_app", description = "Get default app")
54    @Counted(value = "get_default_app", description = "Count of get default app")
55    public ViewerResponse defaultApp() {
56      String defaultAppName = configurationRepository.get(Configuration.DEFAULT_APP);
57      Application app = applicationRepository.findByName(defaultAppName);
58      if (app == null) {
59        throw new ResponseStatusException(HttpStatus.NOT_FOUND);
60      }
61      if (!this.authorizationService.userMayView(app)) {
62        throw new ResponseStatusException(HttpStatus.UNAUTHORIZED);
63      }
64      return viewer(app, ViewerResponse.KindEnum.APP);
65    }
66  
67    // We can't parametrize viewerKind (/app/ and /service/) because they'd clash with /api/admin/...
68    // so name the paths separately
69    @GetMapping(
70        path = {"${tailormap-api.base-path}/app/{viewerName}", "${tailormap-api.base-path}/service/{viewerName}"})
71    @Timed(value = "get_named_app", description = "Get named app")
72    @Counted(value = "get_named_app", description = "Count of get named app")
73    public ViewerResponse viewer(@ModelAttribute Application app, @ModelAttribute ViewerResponse.KindEnum viewerKind) {
74      ViewerResponse viewerResponse = app.getViewerResponse().kind(viewerKind);
75  
76      AppStyling styling = viewerResponse.getStyling();
77      if (styling != null) {
78        styling.setLogo(uploadHelper.getUrlForImage(styling.getLogo(), Upload.CATEGORY_APP_LOGO));
79      }
80      return viewerResponse;
81    }
82  
83    @GetMapping(
84        path = {
85          "${tailormap-api.base-path}/app/{viewerName}/map",
86          "${tailormap-api.base-path}/service/{viewerName}/map"
87        })
88    public MapResponse map(@ModelAttribute Application app) {
89      MapResponse mapResponse = applicationHelper.toMapResponse(app);
90      mapResponse.getAppLayers().stream()
91          .filter(l ->
92              l.getLegendImageUrl() != null && l.getLegendImageUrl().matches(UUID_REGEX))
93          .forEach(l -> l.setLegendImageUrl(
94              uploadHelper.getUrlForImage(l.getLegendImageUrl(), Upload.CATEGORY_LEGEND)));
95      return mapResponse;
96    }
97  }
