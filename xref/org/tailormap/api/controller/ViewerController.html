7   package org.tailormap.api.controller;
8   
9   import static org.tailormap.api.util.Constants.UUID_REGEX;
10  
11  import io.micrometer.core.annotation.Counted;
12  import io.micrometer.core.annotation.Timed;
13  import io.micrometer.core.instrument.Metrics;
14  import java.util.Locale;
15  import org.springframework.http.HttpStatus;
16  import org.springframework.web.bind.annotation.GetMapping;
17  import org.springframework.web.bind.annotation.ModelAttribute;
18  import org.springframework.web.server.ResponseStatusException;
19  import org.tailormap.api.annotation.AppRestController;
20  import org.tailormap.api.persistence.Application;
21  import org.tailormap.api.persistence.Configuration;
22  import org.tailormap.api.persistence.Upload;
23  import org.tailormap.api.persistence.helper.ApplicationHelper;
24  import org.tailormap.api.persistence.helper.UploadHelper;
25  import org.tailormap.api.persistence.helper.ViewerHelper;
26  import org.tailormap.api.prometheus.TagNames;
27  import org.tailormap.api.repository.ApplicationRepository;
28  import org.tailormap.api.repository.ConfigurationRepository;
29  import org.tailormap.api.security.AuthorisationService;
30  import org.tailormap.api.viewer.model.AppStyling;
31  import org.tailormap.api.viewer.model.MapResponse;
32  import org.tailormap.api.viewer.model.ViewerResponse;
33  
34  @AppRestController
35  public class ViewerController implements TagNames {
36  
37    private final ConfigurationRepository configurationRepository;
38    private final ApplicationRepository applicationRepository;
39    private final ApplicationHelper applicationHelper;
40    private final AuthorisationService authorisationService;
41    private final UploadHelper uploadHelper;
42    private final ViewerHelper viewerHelper;
43  
44    public ViewerController(
45        ConfigurationRepository configurationRepository,
46        ApplicationRepository applicationRepository,
47        ApplicationHelper applicationHelper,
48        AuthorisationService authorisationService,
49        UploadHelper uploadHelper,
50        ViewerHelper viewerHelper) {
51      this.configurationRepository = configurationRepository;
52      this.applicationRepository = applicationRepository;
53      this.applicationHelper = applicationHelper;
54      this.authorisationService = authorisationService;
55      this.uploadHelper = uploadHelper;
56      this.viewerHelper = viewerHelper;
57    }
58  
59    @GetMapping(path = "${tailormap-api.base-path}/app")
60    @Timed(value = "get_default_app", description = "Get default app")
61    @Counted(value = "get_default_app", description = "Count of get default app")
62    public ViewerResponse defaultApp() {
63      String defaultAppName = configurationRepository.get(Configuration.DEFAULT_APP);
64      Application app = applicationRepository.findByName(defaultAppName);
65      if (app == null) {
66        throw new ResponseStatusException(HttpStatus.NOT_FOUND);
67      }
68      if (!this.authorisationService.userAllowedToViewApplication(app)) {
69        throw new ResponseStatusException(HttpStatus.UNAUTHORIZED);
70      }
71      return viewer(app, ViewerResponse.KindEnum.APP);
72    }
73  
74    // We can't parametrize viewerKind (/app/ and /service/) because they'd clash with /api/admin/...
75    // so name the paths separately
76    @GetMapping(
77        path = {"${tailormap-api.base-path}/app/{viewerName}", "${tailormap-api.base-path}/service/{viewerName}"})
78    @Timed(value = "get_named_app", description = "Get named app")
79    @Counted(value = "get_named_app", description = "Count of get named app")
80    public ViewerResponse viewer(@ModelAttribute Application app, @ModelAttribute ViewerResponse.KindEnum viewerKind) {
81      ViewerResponse viewerResponse = viewerHelper.getViewerResponse(app).kind(viewerKind);
82  
83      AppStyling styling = viewerResponse.getStyling();
84      if (styling != null) {
85        styling.setLogo(uploadHelper.getUrlForImage(styling.getLogo(), Upload.CATEGORY_APP_LOGO));
86      }
87  
88      // count/increment the number of times this viewer has been requested
89      Metrics.counter(
90              METRICS_APP_REQUEST_COUNTER_NAME,
91              METRICS_APP_TYPE_TAG,
92              viewerKind.name().toLowerCase(Locale.getDefault()),
93              METRICS_APP_NAME_TAG,
94              app.getName(),
95              METRICS_APP_ID_TAG,
96              app.getId().toString())
97          .increment();
98  
99      return viewerResponse;
100   }
101 
102   @GetMapping(
103       path = {
104         "${tailormap-api.base-path}/app/{viewerName}/map",
105         "${tailormap-api.base-path}/service/{viewerName}/map"
106       })
107   public MapResponse map(@ModelAttribute Application app) {
108     MapResponse mapResponse = applicationHelper.toMapResponse(app);
109     mapResponse.getAppLayers().stream()
110         .filter(l ->
111             l.getLegendImageUrl() != null && l.getLegendImageUrl().matches(UUID_REGEX))
112         .forEach(l -> l.setLegendImageUrl(
113             uploadHelper.getUrlForImage(l.getLegendImageUrl(), Upload.CATEGORY_LEGEND)));
114     return mapResponse;
115   }
116 }
