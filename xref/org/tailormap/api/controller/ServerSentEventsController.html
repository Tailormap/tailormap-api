6   package org.tailormap.api.controller;
7   
8   import static ch.rasc.sse.eventbus.SseEvent.DEFAULT_EVENT;
9   
10  import ch.rasc.sse.eventbus.SseEvent;
11  import ch.rasc.sse.eventbus.SseEventBus;
12  import java.lang.invoke.MethodHandles;
13  import java.util.Collections;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.beans.factory.annotation.Qualifier;
17  import org.springframework.http.HttpStatus;
18  import org.springframework.scheduling.annotation.Scheduled;
19  import org.springframework.web.bind.annotation.GetMapping;
20  import org.springframework.web.bind.annotation.PathVariable;
21  import org.springframework.web.bind.annotation.RestController;
22  import org.springframework.web.server.ResponseStatusException;
23  import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;
24  import org.tailormap.api.util.UUIDv7;
25  import org.tailormap.api.viewer.model.ServerSentEventResponse;
26  import tools.jackson.core.JacksonException;
27  import tools.jackson.databind.SerializationFeature;
28  import tools.jackson.databind.json.JsonMapper;
29  
30  @RestController
31  public class ServerSentEventsController {
32    private static final Logger logger =
33        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
34  
35    private final SseEventBus eventBus;
36  
37    private final JsonMapper jsonMapper;
38  
39    public ServerSentEventsController(@Qualifier("viewerSseEventBus") SseEventBus eventBus, JsonMapper jsonMapper) {
40      this.eventBus = eventBus;
41      // force unindented/single line output for SSE messages, because we may have set
42      // spring.jackson.serialization.indent_output=true for debugging/development/test
43      if (jsonMapper.isEnabled(SerializationFeature.INDENT_OUTPUT)) {
44        this.jsonMapper = jsonMapper
45            .rebuild()
46            .configure(SerializationFeature.INDENT_OUTPUT, false)
47            .build();
48      } else {
49        this.jsonMapper = jsonMapper;
50      }
51    }
52  
53    @GetMapping(path = "${tailormap-api.base-path}/events/{clientId}")
54    public SseEmitter sse(@PathVariable String clientId) {
55      // tests input against the set allowed by Nano ID
56      if (!clientId.matches("[A-Za-z0-9_-]+")) {
57        logger.warn("Invalid clientId for SSE connection: {}", clientId);
58        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Invalid clientId");
59      }
60      logger.debug("Adding new SSE client with id: {}", clientId);
61      return this.eventBus.createSseEmitter(clientId, 3600_000L, DEFAULT_EVENT);
62    }
63  
64    @Scheduled(fixedRate = 60_000)
65    public void keepAlive() throws JacksonException {
66      this.eventBus.handleEvent(SseEvent.ofData(jsonMapper.writeValueAsString(new ServerSentEventResponse()
67          .eventType(ServerSentEventResponse.EventTypeEnum.KEEP_ALIVE)
68          .id(UUIDv7.randomV7())
69          .details(Collections.emptyMap()))));
70    }
71  }
