7   package org.tailormap.api.controller.admin;
8   
9   import static ch.rasc.sse.eventbus.SseEvent.DEFAULT_EVENT;
10  import static org.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.KEEP_ALIVE;
11  
12  import ch.rasc.sse.eventbus.SseEvent;
13  import ch.rasc.sse.eventbus.SseEventBus;
14  import com.fasterxml.jackson.core.JsonProcessingException;
15  import com.fasterxml.jackson.databind.ObjectMapper;
16  import java.lang.invoke.MethodHandles;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  import org.springframework.scheduling.annotation.Scheduled;
20  import org.springframework.web.bind.annotation.GetMapping;
21  import org.springframework.web.bind.annotation.PathVariable;
22  import org.springframework.web.bind.annotation.RestController;
23  import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;
24  import org.tailormap.api.admin.model.ServerSentEvent;
25  
26  @RestController
27  public class ServerSentEventsAdminController {
28    private static final Logger logger =
29        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
30  
31    private final SseEventBus eventBus;
32  
33    private final ObjectMapper objectMapper;
34  
35    public ServerSentEventsAdminController(SseEventBus eventBus, ObjectMapper objectMapper) {
36      this.eventBus = eventBus;
37      this.objectMapper = objectMapper;
38    }
39  
40    /**
41     * Endpoint for the single-page admin frontend to receive updates to entities from other windows. The frontend does
42     * not load all data on each navigation as a traditional frontend and loads and stores entities in memory. This
43     * requires updating stale entities when these get updated in other windows (from other sessions or the same session
44     * in a different browser tab), otherwise data might get lost or overwritten.
45     *
46     * <p>This endpoint uses <a href="https://html.spec.whatwg.org/multipage/server-sent-events.html">Server-sent
47     * events</a>, which technically is a long-poll HTTP request which allows unidirectional server-initiated
48     * communication. Not bidirectional like websockets, but simpler because of the long-poll HTTP request. The
49     * webserver should ideally use HTTP/2 or higher because HTTP/1.1 limits the amount of simultaneous connections to a
50     * server per browser to 6.
51     *
52     * @return the server-sent events emitter
53     */
54    @GetMapping(path = "${tailormap-api.admin.base-path}/events/{clientId}")
55    public SseEmitter entityEvents(
56        /*@RequestParam(required = false) String[] events,*/ @PathVariable("clientId") String clientId) {
57      logger.debug("New SSE client: {}, all clients: {}", clientId, this.eventBus.getAllClientIds());
58      return this.eventBus.createSseEmitter(clientId, 3600_000L, DEFAULT_EVENT);
59    }
60  
61    @Scheduled(fixedRate = 60_000)
62    public void keepAlive() throws JsonProcessingException {
63      this.eventBus.handleEvent(
64          SseEvent.ofData(objectMapper.writeValueAsString(new ServerSentEvent().eventType(KEEP_ALIVE))));
65    }
66  }
