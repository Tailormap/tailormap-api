7   package org.tailormap.api.controller.admin;
8   
9   import jakarta.servlet.http.HttpServletResponse;
10  import org.springframework.context.ApplicationContext;
11  import org.springframework.data.rest.core.event.BeforeSaveEvent;
12  import org.springframework.data.rest.webmvc.support.RepositoryEntityLinks;
13  import org.springframework.http.HttpStatus;
14  import org.springframework.http.ResponseEntity;
15  import org.springframework.web.bind.annotation.PathVariable;
16  import org.springframework.web.bind.annotation.PostMapping;
17  import org.springframework.web.bind.annotation.RestController;
18  import org.springframework.web.server.ResponseStatusException;
19  import org.tailormap.api.persistence.GeoService;
20  import org.tailormap.api.repository.GeoServiceRepository;
21  
22  @RestController
23  public class GeoServiceAdminController {
24    private final GeoServiceRepository geoServiceRepository;
25    private final ApplicationContext applicationContext;
26    private final RepositoryEntityLinks repositoryEntityLinks;
27  
28    public GeoServiceAdminController(
29        GeoServiceRepository geoServiceRepository,
30        ApplicationContext applicationContext,
31        RepositoryEntityLinks repositoryEntityLinks) {
32      this.geoServiceRepository = geoServiceRepository;
33      this.applicationContext = applicationContext;
34      this.repositoryEntityLinks = repositoryEntityLinks;
35    }
36  
37    @PostMapping(path = "${tailormap-api.admin.base-path}/geo-services/{id}/refresh-capabilities")
38    public ResponseEntity<GeoService> refreshCapabilities(
39        @PathVariable String id, HttpServletResponse httpServletResponse) throws Exception {
40  
41      GeoService geoService =
42          geoServiceRepository.findById(id).orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND));
43  
44      // Authorization check not needed: only admins are allowed on the admin base path, and admins
45      // have all access
46  
47      geoService.setRefreshCapabilities(true);
48      applicationContext.publishEvent(new BeforeSaveEvent(geoService));
49  
50      geoServiceRepository.saveAndFlush(geoService);
51  
52      httpServletResponse.sendRedirect(String.valueOf(
53          repositoryEntityLinks.linkToItemResource(GeoService.class, id).toUri()));
54      return null;
55    }
56  }
