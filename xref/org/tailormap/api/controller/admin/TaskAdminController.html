76    @Operation(
77        summary = "List all tasks, optionally filtered by type",
78        description =
79            """
80  This will return a list of all tasks, optionally filtered by task type.
81  The state of the task is one of the Quartz Trigger states.
82  The state can be one of: NONE, NORMAL, PAUSED, COMPLETE, ERROR, BLOCKED or null in error conditions.
83  """)
84    @GetMapping(path = "${tailormap-api.admin.base-path}/tasks", produces = MediaType.APPLICATION_JSON_VALUE)
85    @ApiResponse(
86        responseCode = "200",
87        description = "List of all tasks, this list may be empty",
88        content =
89            @Content(
90                mediaType = MediaType.APPLICATION_JSON_VALUE,
91                schema =
92                    @Schema(
93                        example =
94                            """
95  {"tasks":[
96  {"uuid":"6308d26e-fe1e-4268-bb28-20db2cd06914","type":"index", "state":"NORMAL", "interruptable": false},
97  {"uuid":"d5ce9152-e90e-4b5a-b129-3b2366cabca8","type":"poc", "state": "BLOCKED", "interruptable": false},
98  {"uuid":"d5ce9152-e90e-4b5a-b129-3b2366cabca9","type":"poc", "state": "PAUSED", "interruptable": false},
99  {"uuid":"d5ce9152-e90e-4b5a-b129-3b2366cabca2","type":"poc", "state": "COMPLETE", "interruptable": false},
100 {"uuid":"d5ce9152-e90e-4b5a-b129-3b2366cabca3","type":"interuptablepoc", "state": "ERROR", "interruptable": true}
101 ]}
102 """)))
103   public ResponseEntity<Object> list(@RequestParam(required = false) String type) throws ResponseStatusException {
104     logger.debug("Listing all tasks (optional type filter: {})", (null == type ? "all" : type));
105     final List<ObjectNode> tasks = new ArrayList<>();
106 
107     final GroupMatcher<JobKey> groupMatcher =
108         (null == type ? GroupMatcher.anyGroup() : GroupMatcher.groupEquals(type));
109     try {
110       scheduler.getJobKeys(groupMatcher).stream()
111           .map(jobKey -> {
112             try {
113               return scheduler.getJobDetail(jobKey);
114             } catch (SchedulerException e) {
115               logger.error("Error getting task detail", e);
116               return null;
117             }
118           })
119           .filter(Objects::nonNull)
120           .forEach(jobDetail -> {
121             Trigger.TriggerState state;
122             try {
123               state = scheduler.getTriggerState(TriggerKey.triggerKey(
124                   jobDetail.getKey().getName(),
125                   jobDetail.getKey().getGroup()));
126             } catch (SchedulerException e) {
127               logger.error("Error getting task state", e);
128               state = null;
129             }
130             tasks.add(new ObjectMapper()
131                 .createObjectNode()
132                 .put(Task.TYPE_KEY, jobDetail.getKey().getGroup())
133                 .put(Task.UUID_KEY, jobDetail.getKey().getName())
134                 .put(
135                     Task.INTERRUPTABLE_KEY,
136                     InterruptableJob.class.isAssignableFrom(jobDetail.getJobClass()))
137                 .put(
138                     Task.DESCRIPTION_KEY,
139                     jobDetail.getJobDataMap().getString(Task.DESCRIPTION_KEY))
140                 .put(
141                     Task.LAST_RESULT_KEY,
142                     jobDetail.getJobDataMap().getString(Task.LAST_RESULT_KEY))
143                 .putPOJO(Task.STATE_KEY, state));
144           });
145     } catch (SchedulerException e) {
146       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, "Error getting tasks", e);
147     }
148 
149     return ResponseEntity.ok(new ObjectMapper()
150         .createObjectNode()
151         .set("tasks", new ObjectMapper().createArrayNode().addAll(tasks)));
152   }

154   @Operation(
155       summary = "List all details for a given task",
156       description =
157           """
158 This will return the details of the task, including the status, progress,
159 result and any other information.
160 """)
161   @GetMapping(
162       path = "${tailormap-api.admin.base-path}/tasks/{type}/{uuid}",
163       produces = MediaType.APPLICATION_JSON_VALUE)
164   @ApiResponse(
165       responseCode = "404",
166       description = "Task not found",
167       content =
168           @Content(
169               mediaType = MediaType.APPLICATION_JSON_VALUE,
170               schema = @Schema(example = "{\"message\":\"Task not found\",\"code\":404}")))
171   @ApiResponse(
172       responseCode = "200",
173       description =
174           "Details of the task. The response content will vay according to the type of task, the most common fields are listed in the Task interface",
175       content =
176           @Content(
177               mediaType = MediaType.APPLICATION_JSON_VALUE,
178               schema =
179                   @Schema(
180                       example =
181                           """
182 {
183 "type":"poc",
184 "uuid":"6308d26e-fe1e-4268-bb28-20db2cd06914",
185 "interruptable":false,
186 "description":"This is a poc task",
187 "startTime":"2024-06-06T12:00:00Z",
188 "nextTime":"2024-06-06T12:00:00Z",
189 "state":"NORMAL",
190 "progress":"...",
191 "result":"...",
192 "message":"something is happening"
193 "jobData":{ "type":"poc", "description":"This is a poc task" }
194 }
195 """)))
196   public ResponseEntity<Object> details(@PathVariable TaskType type, @PathVariable UUID uuid)
197       throws ResponseStatusException {
198     logger.debug("Getting task details for {}:{}", type, uuid);
199 
200     try {
201       JobKey jobKey = taskManagerService.getJobKey(type, uuid);
202       if (null == jobKey) {
203         return handleTaskNotFound();
204       }
205 
206       JobDetail jobDetail = scheduler.getJobDetail(jobKey);
207       JobDataMap jobDataMap = jobDetail.getJobDataMap();
208 
209       /* there should be only one */
210       Trigger trigger = scheduler.getTriggersOfJob(jobDetail.getKey()).get(0);
211       CronTrigger cron = ((CronTrigger) trigger);
212 
213       final Object[] result = new Object[1];
214       scheduler.getCurrentlyExecutingJobs().stream()
215           .filter(Objects::nonNull)
216           .filter(jobExecutionContext ->
217               jobExecutionContext.getJobDetail().getKey().equals(jobKey))
218           .forEach(jobExecutionContext -> {
219             logger.debug(
220                 "currently executing job {} with trigger {}.",
221                 jobExecutionContext.getJobDetail().getKey(),
222                 jobExecutionContext.getTrigger().getKey());
223 
224             result[0] = jobExecutionContext.getResult();
225           });
226 
227       return ResponseEntity.ok(new ObjectMapper()
228           .createObjectNode()
229           // immutable uuid, type and description
230           .put(Task.TYPE_KEY, jobDetail.getKey().getGroup())
231           .put(Task.UUID_KEY, jobDetail.getKey().getName())
232           .put(Task.INTERRUPTABLE_KEY, InterruptableJob.class.isAssignableFrom(jobDetail.getJobClass()))
233           .put(Task.DESCRIPTION_KEY, jobDataMap.getString(Task.DESCRIPTION_KEY))
234           .put(Task.CRON_EXPRESSION_KEY, cron.getCronExpression())
235           // TODO / XXX we could add a human-readable description of the cron expression using
236           // eg.
237           //   com.cronutils:cron-utils like:
238           //     CronParser cronParser = new
239           //         CronParser(CronDefinitionBuilder.instanceDefinitionFor(QUARTZ));
240           //     CronDescriptor.instance(locale).describe(cronParser.parse(cronExpression));
241           //   this could also be done front-end using eg.
242           // https://www.npmjs.com/package/cronstrue
243           //   which has the advantage of knowing the required locale for the human
244           // .put("cronDescription", cron.getCronExpression())
245           .put("timezone", cron.getTimeZone().getID())
246           .putPOJO("startTime", trigger.getStartTime())
247           .putPOJO("lastTime", trigger.getPreviousFireTime())
248           .putPOJO("nextFireTimes", TriggerUtils.computeFireTimes((OperableTrigger) cron, null, 5))
249           .putPOJO(Task.STATE_KEY, scheduler.getTriggerState(trigger.getKey()))
250           .putPOJO("progress", result[0])
251           .put(Task.LAST_RESULT_KEY, jobDataMap.getString(Task.LAST_RESULT_KEY))
252           .putPOJO("jobData", jobDataMap));
253     } catch (SchedulerException e) {
254       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, "Error getting task", e);
255     }
256   }

258   @Operation(summary = "Start a task", description = "This will start the task if it is not already running")
259   @PutMapping(
260       path = "${tailormap-api.admin.base-path}/tasks/{type}/{uuid}/start",
261       produces = MediaType.APPLICATION_JSON_VALUE)
262   @ApiResponse(
263       responseCode = "404",
264       description = "Task not found",
265       content =
266           @Content(
267               mediaType = MediaType.APPLICATION_JSON_VALUE,
268               schema = @Schema(example = "{\"message\":\"Task not found\",\"code\":404}")))
269   @ApiResponse(
270       responseCode = "202",
271       description = "Task is started",
272       content =
273           @Content(
274               mediaType = MediaType.APPLICATION_JSON_VALUE,
275               schema = @Schema(example = "{\"message\":\"Task starting accepted\",\"code\":202}")))
276   public ResponseEntity<Object> startTask(@PathVariable TaskType type, @PathVariable UUID uuid)
277       throws ResponseStatusException {
278     logger.debug("Starting task {}:{}", type, uuid);
279 
280     try {
281       JobKey jobKey = taskManagerService.getJobKey(type, uuid);
282       if (null == jobKey) {
283         return handleTaskNotFound();
284       }
285       scheduler.triggerJob(jobKey);
286       return ResponseEntity.status(HttpStatusCode.valueOf(HTTP_ACCEPTED))
287           .body(new ObjectMapper().createObjectNode().put("message", "Task starting accepted"));
288 
289     } catch (SchedulerException e) {
290       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, "Error getting task", e);
291     }
292   }

294   @Operation(
295       summary = "Stop a task irrevocably",
296       description =
297           """
298 This will stop a running task, if the task is not running, nothing will happen.
299 This can leave the application in an inconsistent state.
300 A task that is not interruptable cannot be stopped.
301 A stopped task cannot be restarted, it fire again depending on the schedule.
302 """)
303   @PutMapping(
304       path = "${tailormap-api.admin.base-path}/tasks/{type}/{uuid}/stop",
305       produces = MediaType.APPLICATION_JSON_VALUE)
306   @ApiResponse(
307       responseCode = "404",
308       description = "Task not found",
309       content =
310           @Content(
311               mediaType = MediaType.APPLICATION_JSON_VALUE,
312               schema = @Schema(example = "{\"message\":\"Task not found\", \"code\":404}")))
313   @ApiResponse(
314       responseCode = "202",
315       description = "Task is stopping",
316       content =
317           @Content(
318               mediaType = MediaType.APPLICATION_JSON_VALUE,
319               schema = @Schema(example = """
320 {
321 "message":"Task stopping accepted".
322 "succes":true
323 }
324 """)))
325   @ApiResponse(
326       responseCode = "400",
327       description = "The task cannot be stopped as it does not implement the InterruptableJob interface.",
328       content =
329           @Content(
330               mediaType = MediaType.APPLICATION_JSON_VALUE,
331               schema = @Schema(example = """
332 {
333 "message":"Task cannot be stopped"
334 }
335 """)))
336   public ResponseEntity<Object> stopTask(@PathVariable TaskType type, @PathVariable UUID uuid)
337       throws ResponseStatusException {
338     logger.debug("Stopping task {}:{}", type, uuid);
339 
340     try {
341       JobKey jobKey = taskManagerService.getJobKey(type, uuid);
342       if (null == jobKey) {
343         return handleTaskNotFound();
344       }
345 
346       if (InterruptableJob.class.isAssignableFrom(
347           scheduler.getJobDetail(jobKey).getJobClass())) {
348         boolean interrupted = scheduler.interrupt(jobKey);
349         return ResponseEntity.status(HttpStatusCode.valueOf(HTTP_ACCEPTED))
350             .body(new ObjectMapper()
351                 .createObjectNode()
352                 .put("message", "Task stopping accepted")
353                 .put("succes", interrupted));
354       } else {
355         return ResponseEntity.status(HttpStatusCode.valueOf(HTTP_BAD_REQUEST))
356             .body(new ObjectMapper()
357                 .createObjectNode()
358                 .put("message", "Task cannot be stopped")
359                 .put("succes", false));
360       }
361 
362     } catch (SchedulerException e) {
363       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, "Error getting task", e);
364     }
365   }

367   @Operation(
368       summary = "Delete a task",
369       description = "This will remove the task from the scheduler and delete all information about the task")
370   @DeleteMapping(
371       path = "${tailormap-api.admin.base-path}/tasks/{type}/{uuid}",
372       produces = MediaType.APPLICATION_JSON_VALUE)
373   @ApiResponse(
374       responseCode = "404",
375       description = "Task not found",
376       content =
377           @Content(
378               mediaType = MediaType.APPLICATION_JSON_VALUE,
379               schema = @Schema(example = "{\"message\":\"Task not found\"}")))
380   @ApiResponse(responseCode = "204", description = "Task is deleted")
381   public ResponseEntity<Object> delete(@PathVariable TaskType type, @PathVariable UUID uuid)
382       throws ResponseStatusException {
383 
384     try {
385       JobKey jobKey = taskManagerService.getJobKey(type, uuid);
386       if (null == jobKey) {
387         return handleTaskNotFound();
388       }
389 
390       boolean succes = scheduler.deleteJob(jobKey);
391       logger.info("Task {}:{} deletion {}", type, uuid, (succes ? "succeeded" : "failed"));
392       // cleanup the schedule from the business objects
393       switch (type) {
394         case INDEX:
395           deleteScheduleFromSearchIndex(uuid);
396           break;
397         case FAILINGPOC:
398         case POC:
399         case INTERRUPTABLEPOC:
400           // no action required, as these are not managed in Tailormap
401         default:
402           break;
403       }
404       return ResponseEntity.noContent().build();
405     } catch (SchedulerException e) {
406       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, "Error getting task", e);
407     }
408   }
409 
410   private void deleteScheduleFromSearchIndex(UUID uuid) {
411     searchIndexRepository.findByTaskScheduleUuid(uuid).forEach(searchIndex -> {
412       searchIndex.setSchedule(null);
413       searchIndexRepository.save(searchIndex);
414     });
415   }
416 
417   private ResponseEntity<Object> handleTaskNotFound() {
418     return ResponseEntity.status(HttpStatusCode.valueOf(HTTP_NOT_FOUND))
419         .contentType(MediaType.APPLICATION_JSON)
420         .body(new ObjectMapper()
421             .createObjectNode()
422             .put("message", "Task not found")
423             .put("code", HTTP_NOT_FOUND));
424   }
