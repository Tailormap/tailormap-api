7   package org.tailormap.api.controller.admin;
8   
9   import jakarta.servlet.http.HttpServletResponse;
10  import java.lang.invoke.MethodHandles;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  import org.springframework.context.ApplicationContext;
14  import org.springframework.data.rest.core.event.BeforeSaveEvent;
15  import org.springframework.data.rest.webmvc.support.RepositoryEntityLinks;
16  import org.springframework.http.HttpStatus;
17  import org.springframework.http.ResponseEntity;
18  import org.springframework.transaction.annotation.Transactional;
19  import org.springframework.web.bind.annotation.PathVariable;
20  import org.springframework.web.bind.annotation.PostMapping;
21  import org.springframework.web.bind.annotation.RestController;
22  import org.springframework.web.server.ResponseStatusException;
23  import org.tailormap.api.persistence.TMFeatureSource;
24  import org.tailormap.api.repository.FeatureSourceRepository;
25  
26  @RestController
27  public class FeatureSourceAdminController {
28    private static final Logger logger =
29        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
30    private final FeatureSourceRepository featureSourceRepository;
31    private final ApplicationContext applicationContext;
32    private final RepositoryEntityLinks repositoryEntityLinks;
33  
34    public FeatureSourceAdminController(
35        FeatureSourceRepository featureSourceRepository,
36        ApplicationContext applicationContext,
37        RepositoryEntityLinks repositoryEntityLinks) {
38      this.featureSourceRepository = featureSourceRepository;
39      this.applicationContext = applicationContext;
40      this.repositoryEntityLinks = repositoryEntityLinks;
41    }
42  
43    @PostMapping(path = "${tailormap-api.admin.base-path}/feature-sources/{id}/refresh-capabilities")
44    @Transactional
45    public ResponseEntity<?> refreshCapabilities(@PathVariable Long id, HttpServletResponse httpServletResponse)
46        throws Exception {
47  
48      TMFeatureSource featureSource = featureSourceRepository
49          .findById(id)
50          .orElseThrow(() -> new ResponseStatusException(HttpStatus.NOT_FOUND));
51  
52      // Authorization check not needed: only admins are allowed on the admin base path, and admins
53      // have all access
54  
55      logger.info("Loading capabilities for feature source {}", featureSource);
56      featureSource.setRefreshCapabilities(true);
57      applicationContext.publishEvent(new BeforeSaveEvent(featureSource));
58  
59      httpServletResponse.sendRedirect(String.valueOf(repositoryEntityLinks
60          .linkToItemResource(TMFeatureSource.class, id)
61          .toUri()));
62      return null;
63    }
64  }
