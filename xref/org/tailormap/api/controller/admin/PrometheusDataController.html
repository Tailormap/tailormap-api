78    @Override
79    public void afterPropertiesSet() throws Exception {
80      if (prometheusService.isPrometheusAvailable()) {
81        logger.info("Prometheus is available, initializing PrometheusDataController.");
82        try {
83          // there should be only one Prometheus ping task, so we delete any existing ones
84          taskManagerService.deleteTasksByGroupName(TaskType.PROMETHEUS_PING.getValue());
85          final UUID taskUuid = taskManagerService.createTask(
86              PrometheusPingTask.class,
87              new TMJobDataMap(Map.of(
88                  Task.TYPE_KEY,
89                  TaskType.PROMETHEUS_PING.getValue(),
90                  Task.DESCRIPTION_KEY,
91                  "Ping Prometheus service for availability.",
92                  Task.PRIORITY_KEY,
93                  57)),
94              prometheusPingCron);
95          logger.debug("Added Prometheus ping task with UUID: {}", taskUuid);
96        } catch (Exception e) {
97          logger.error("Error initializing Prometheus ping task", e);
98        }
99      } else {
100       logger.info("Prometheus is not available, /graph/ endpoint will not be functional.");
101     }
102   }

116   @Operation(
117       summary = "retrieve application graph data",
118       description = "Fetches the totals and last updated time for application metrics from Prometheus.")
119   @GetMapping(
120       path = "${tailormap-api.admin.base-path}/graph/applications",
121       produces = MediaType.APPLICATION_JSON_VALUE)
122   @ApiResponse(
123       responseCode = "200",
124       description = "Array of application metrics with totals and last updated times.",
125       content =
126           @Content(
127               mediaType = MediaType.APPLICATION_JSON_VALUE,
128               schema =
129                   @Schema(
130                       example =
131                           """
132 {"applications":[
133 {"lastUpdateSecondsAgo":"85746","appName":"default","appId":"1","totalCount":"4003"},
134 {"lastUpdateSecondsAgo":"1345","appName":"test","appId":"5","totalCount":"5"}
135 ]}
136 """)))
137   public ResponseEntity<?> getApplicationGraphicData(@RequestParam(defaultValue = "30") int numberOfDays) {
138     if (numberOfDays < 1) {
139       throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Invalid number of days provided.");
140     }
141     if (!prometheusService.isPrometheusAvailable()) {
142       throw new ResponseStatusException(HttpStatus.SERVICE_UNAVAILABLE, "Prometheus is not available.");
143     }
144     // we need to (at least) relabel the "__name__" query result type to avoid conflicts and omissions in the
145     // flattened results
146     final String completeQuery =
147         "label_replace(" + totalsQuery.replace(NUMBER_OF_DAYS_REPLACE_TOKEN, String.valueOf(numberOfDays))
148             + ", \"type\", \"totalCount\", \"__name__\", \".*\")"
149             + " or " + "label_replace("
150             + counterLastUpdatedQuery.replace(NUMBER_OF_DAYS_REPLACE_TOKEN, String.valueOf(numberOfDays))
151             + ", \"type\", \"lastUpdateSecondsAgo\", \"__name__\", \".*\")";
152     logger.trace("Fetching application graph data using query {}.", completeQuery);
153     try {
154       JsonNode totals = prometheusService.executeQuery(completeQuery);
155       logger.trace("Application graph data fetched successfully. {}", totals.toPrettyString());
156       Collection<Map<String, String>> applications =
157           prometheusResultProcessor.processPrometheusResultsForApplications(totals);
158       return ResponseEntity.ok(new ObjectMapper()
159           .createObjectNode()
160           .set("applications", new ObjectMapper().valueToTree(applications)));
161     } catch (IOException e) {
162       logger.error("Error fetching application graph data", e);
163       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, e.getMessage());
164     }
165   }

190   public ResponseEntity<?> getApplicationLayersGraphicData(
191       @PathVariable Long applicationId, @RequestParam(defaultValue = "30") int numberOfDays) {
192     if (applicationId == null || applicationId <= 0 || numberOfDays < 1) {
193       throw new ResponseStatusException(
194           HttpStatus.BAD_REQUEST, "Invalid application id or number of days provided.");
195       // not a problem to have an applicationId that does not exist, as the Prometheus query will
196       // return an empty result set
197     }
198     if (!prometheusService.isPrometheusAvailable()) {
199       throw new ResponseStatusException(HttpStatus.SERVICE_UNAVAILABLE, "Prometheus is not available.");
200     }
201     // relabel the "__name__" query result type to avoid conflicts and omissions in the flattened results and
202     // replace the APP_ID_REPLACE_TOKEN with the actual applicationId and NUMBER_OF_DAYS_REPLACE_TOKEN
203     // with the actual numberOfDays
204     String completeQuery = "label_replace("
205         + appLayersSwitchedOnTotalsQuery
206             .replace(APP_ID_REPLACE_TOKEN, applicationId.toString())
207             .replace(NUMBER_OF_DAYS_REPLACE_TOKEN, String.valueOf(numberOfDays))
208         + ", \"type\", \"totalCount\", \"__name__\", \".*\")"
209         + " or "
210         + "label_replace("
211         + appLayersSwitchedOnLastUpdatedQuery
212             .replace(APP_ID_REPLACE_TOKEN, applicationId.toString())
213             .replace(NUMBER_OF_DAYS_REPLACE_TOKEN, String.valueOf(numberOfDays))
214         + ", \"type\", \"lastUpdateSecondsAgo\", \"__name__\", \".*\")";
215     logger.trace(
216         "Fetching application layers graph data for applicationId: {} with query: {}",
217         applicationId,
218         completeQuery);
219     try {
220       JsonNode results = prometheusService.executeQuery(completeQuery);
221       logger.trace("Application layers graph data fetched successfully. {}", results.toPrettyString());
222       Collection<Map<String, String>> data =
223           prometheusResultProcessor.processPrometheusResultsForApplicationLayers(results);
224       if (!data.isEmpty()) {
225         // enrich data with additional information (appLayerName and/or appLayerTitle)
226         applicationRepository
227             .findById(applicationId)
228             .ifPresent(application -> data.forEach(metric -> {
229               String appLayerId = metric.get(METRICS_APP_LAYER_ID_TAG);
230               if (appLayerId != null && !appLayerId.isEmpty()) {
231                 application
232                     .getAllAppTreeLayerNode()
233                     .filter(node -> appLayerId.equals(node.getId()))
234                     .findFirst()
235                     .ifPresent(node -> {
236                       metric.put(METRICS_APP_LAYER_NAME_TAG, node.getLayerName());
237                       metric.put(
238                           METRICS_APP_LAYER_TITLE_TAG,
239                           application
240                               .getAppLayerSettings(node)
241                               .getTitle());
242                     });
243               }
244             }));
245       }
246 
247       return ResponseEntity.ok(new ObjectMapper()
248           .createObjectNode()
249           .set("applicationLayers", new ObjectMapper().valueToTree(data)));
250     } catch (IOException e) {
251       logger.error("Error fetching application layers graph data", e);
252       throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, e.getMessage());
253     }
254   }
