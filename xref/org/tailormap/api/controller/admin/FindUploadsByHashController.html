7   package org.tailormap.api.controller.admin;
8   
9   import java.util.List;
10  import org.springframework.web.bind.annotation.PathVariable;
11  import org.springframework.web.bind.annotation.PostMapping;
12  import org.springframework.web.bind.annotation.RequestBody;
13  import org.springframework.web.bind.annotation.RestController;
14  import org.tailormap.api.repository.UploadMatch;
15  import org.tailormap.api.repository.UploadRepository;
16  
17  @RestController
18  public class FindUploadsByHashController {
19    private final UploadRepository uploadRepository;
20  
21    public FindUploadsByHashController(UploadRepository uploadRepository) {
22      this.uploadRepository = uploadRepository;
23    }
24  
25    @PostMapping(
26        path = "${tailormap-api.admin.base-path}/uploads/find-by-hash/{category}",
27        consumes = "application/json")
28    public List<UploadMatch> findUploadsByHash(@PathVariable String category, @RequestBody List<String> hashes) {
29      return uploadRepository.findByContentMd5In(category, hashes);
30    }
31  }
