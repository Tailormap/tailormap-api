6   package org.tailormap.api.controller.admin;
7   
8   import java.lang.invoke.MethodHandles;
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  import org.springframework.http.HttpStatus;
12  import org.springframework.http.MediaType;
13  import org.springframework.http.ResponseEntity;
14  import org.springframework.http.converter.HttpMessageNotReadableException;
15  import org.springframework.lang.Nullable;
16  import org.springframework.web.bind.annotation.ExceptionHandler;
17  import org.springframework.web.bind.annotation.RestControllerAdvice;
18  import org.springframework.web.context.request.WebRequest;
19  import org.tailormap.api.security.InvalidPasswordException;
20  import org.tailormap.api.viewer.model.ErrorResponse;
21  
22  @RestControllerAdvice
23  public class RestExceptionHandler {
24    private static final Logger logger =
25        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
26  
27    @ExceptionHandler({HttpMessageNotReadableException.class})
28    @Nullable public ResponseEntity<Object> handleException(final HttpMessageNotReadableException ex, final WebRequest request) {
29      logger.debug("Invalid message exception", ex);
30  
31      // Invalid password was given
32      // ideally we would use @ExceptionHandler({InvalidPasswordException.class}) but that does not
33      // trigger the
34      // exception handler because it is wrapped in HttpMessageNotReadableException
35      if (ex.getCause() != null && ex.getMostSpecificCause() instanceof InvalidPasswordException ipe) {
36        return ResponseEntity.status(HttpStatus.BAD_REQUEST)
37            .contentType(MediaType.APPLICATION_JSON)
38            .body(new ErrorResponse()
39                .code(HttpStatus.BAD_REQUEST.value())
40                .message(ipe.getOriginalMessage()));
41      }
42  
43      return ResponseEntity.status(HttpStatus.BAD_REQUEST)
44          .contentType(MediaType.APPLICATION_JSON)
45          .body(new ErrorResponse()
46              .code(HttpStatus.BAD_REQUEST.value())
47              .message(ex.getMostSpecificCause().getLocalizedMessage()));
48    }
49  }
