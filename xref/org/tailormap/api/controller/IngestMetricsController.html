6   package org.tailormap.api.controller;
7   
8   import io.micrometer.core.instrument.Metrics;
9   import java.io.Serializable;
10  import java.lang.invoke.MethodHandles;
11  import java.util.Locale;
12  import java.util.Set;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  import org.springframework.beans.factory.annotation.Value;
16  import org.springframework.http.HttpStatus;
17  import org.springframework.http.ResponseEntity;
18  import org.springframework.validation.annotation.Validated;
19  import org.springframework.web.bind.annotation.ModelAttribute;
20  import org.springframework.web.bind.annotation.PathVariable;
21  import org.springframework.web.bind.annotation.PutMapping;
22  import org.tailormap.api.annotation.AppRestController;
23  import org.tailormap.api.persistence.Application;
24  import org.tailormap.api.persistence.json.AppTreeLayerNode;
25  import org.tailormap.api.prometheus.TagNames;
26  import org.tailormap.api.viewer.model.ViewerResponse;
27  
28  @AppRestController
29  @Validated
30  public class IngestMetricsController implements TagNames {
31    private static final Logger logger =
32        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
33  
34    @Value("${tailormap-api.allowed-metrics}")
35    private Set<String> allowedMetrics;
36  
37    @PutMapping(
38        path = "${tailormap-api.base-path}/{viewerKind}/{viewerName}/metrics/ingest/{appLayerId}/{allowedMetric}")
39    public ResponseEntity<Serializable> ingestMetric(
40        @ModelAttribute Application app,
41        @ModelAttribute ViewerResponse.KindEnum viewerKind,
42        @ModelAttribute AppTreeLayerNode appTreeLayerNode,
43        @PathVariable String allowedMetric) {
44  
45      if (!allowedMetrics.contains(allowedMetric)) {
46        logger.warn("Invalid metric: {}, not in allowed metrics: {}", allowedMetric, allowedMetrics);
47        return new ResponseEntity<>(HttpStatus.BAD_REQUEST);
48      }
49  
50      logger.debug(
51          "Ingesting metric: {}, appId={}, viewerKind={}, appLayerId={} to be published through the actuator endpoint",
52          allowedMetric,
53          app.getId(),
54          viewerKind,
55          appTreeLayerNode.getId());
56  
57      // count/increment the number of times this layer has been switched on for this viewer
58      Metrics.counter(
59              allowedMetric,
60              METRICS_APP_ID_TAG,
61              app.getId().toString(),
62              METRICS_APP_TYPE_TAG,
63              viewerKind.name().toLowerCase(Locale.getDefault()),
64              METRICS_APP_NAME_TAG,
65              app.getName(),
66              METRICS_APP_LAYER_ID_TAG,
67              appTreeLayerNode.getId())
68          .increment();
69  
70      // Return a 204 No Content response, to indicate that the request was successful but there is no content to
71      // return.
72      return ResponseEntity.status(HttpStatus.NO_CONTENT).build();
73    }
74  }
