6   package org.tailormap.api.controller;
7   
8   import io.micrometer.core.instrument.Metrics;
9   import io.swagger.v3.oas.annotations.Operation;
10  import java.io.Serializable;
11  import java.lang.invoke.MethodHandles;
12  import java.util.Locale;
13  import java.util.Set;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.beans.factory.annotation.Value;
17  import org.springframework.http.HttpStatus;
18  import org.springframework.http.ResponseEntity;
19  import org.springframework.validation.annotation.Validated;
20  import org.springframework.web.bind.annotation.ModelAttribute;
21  import org.springframework.web.bind.annotation.PathVariable;
22  import org.springframework.web.bind.annotation.PutMapping;
23  import org.tailormap.api.annotation.AppRestController;
24  import org.tailormap.api.persistence.Application;
25  import org.tailormap.api.prometheus.TagNames;
26  import org.tailormap.api.viewer.model.ViewerResponse;
27  
28  @AppRestController
29  @Validated
30  public class IngestMetricsController implements TagNames {
31    private static final Logger logger =
32        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
33  
34    @Value("${tailormap-api.allowed-metrics}")
35    private Set<String> allowedMetrics;
36  
37    /**
38     * Ingests a metric for the given application layers and viewer kind.
39     *
40     * @param app the application
41     * @param viewerKind the kind of viewer
42     * @param appLayerIds the application layer IDs to ingest the metric for
43     * @param allowedMetric the metric to ingest
44     * @return a ResponseEntity with status 204 No Content if successful, or 400 Bad Request if the metric is not
45     *     allowed
46     */
47    @Operation(
48        summary = "Ingest a metric for the given application layer(s)",
49        description =
50            """
51  Ingests a metric for the given application layer(s), the metric will be published through the actuator endpoint as
52  a Micrometer counter. All application layers in the appLayerIds that are part of the application will be processed.
53  The metric must be one of the allowed metrics defined in the application configuration.""")
54    @PutMapping(
55        path = "${tailormap-api.base-path}/{viewerKind}/{viewerName}/metrics/ingest/{appLayerIds}/{allowedMetric}")
56    public ResponseEntity<Serializable> ingestMetric(
57        @ModelAttribute Application app,
58        @ModelAttribute ViewerResponse.KindEnum viewerKind,
59        @PathVariable Set<String> appLayerIds,
60        @PathVariable String allowedMetric) {
61  
62      if (!allowedMetrics.contains(allowedMetric)) {
63        logger.warn("Invalid metric: {}, not in allowed metrics: {}", allowedMetric, allowedMetrics);
64        return new ResponseEntity<>(HttpStatus.BAD_REQUEST);
65      }
66      if (appLayerIds.isEmpty()) {
67        return new ResponseEntity<>(HttpStatus.BAD_REQUEST);
68      }
69      logger.trace("Ingesting metric for {} layer(s)", appLayerIds);
70  
71      // Validate that the given application layers in appLayerIds are actually part of this application
72      app.getAllAppTreeLayerNode()
73          .filter(appTreeLayerNode -> appLayerIds.contains(appTreeLayerNode.getId()))
74          // for each valid appLayer publish metrics
75          .forEach(appTreeLayerNode -> {
76            logger.debug(
77                "Ingesting metric: {}, appId={}, viewerKind={}, appLayerId={} to be published through the actuator endpoint",
78                allowedMetric,
79                app.getId(),
80                viewerKind,
81                appTreeLayerNode.getId());
82  
83            // count/increment the number of times this layer has been switched on for this viewer
84            Metrics.counter(
85                    allowedMetric,
86                    METRICS_APP_ID_TAG,
87                    app.getId().toString(),
88                    METRICS_APP_TYPE_TAG,
89                    viewerKind.name().toLowerCase(Locale.getDefault()),
90                    METRICS_APP_NAME_TAG,
91                    app.getName(),
92                    METRICS_APP_LAYER_ID_TAG,
93                    appTreeLayerNode.getId())
94                // we don't add the appLayerName here, as in many cases you want the allLayerTitle, but that
95                // requires an extra lookup in the appLayerSettings when publishing the metric and does not
96                // take
97                // renaming into account, that can better be done when rendering the metrics
98                .increment();
99          });
100     // Return a 204 No Content response, to indicate that the request was successful but there is no content to
101     // return.
102     return ResponseEntity.status(HttpStatus.NO_CONTENT).build();
103   }
104 }
