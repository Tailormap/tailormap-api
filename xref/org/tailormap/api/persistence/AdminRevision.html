6   package org.tailormap.api.persistence;
7   
8   import jakarta.persistence.Column;
9   import jakarta.persistence.ElementCollection;
10  import jakarta.persistence.Entity;
11  import jakarta.persistence.EntityListeners;
12  import jakarta.persistence.JoinColumn;
13  import jakarta.persistence.JoinTable;
14  import jakarta.persistence.Table;
15  import java.util.HashSet;
16  import java.util.Objects;
17  import java.util.Set;
18  import org.hibernate.envers.DateTimeFormatter;
19  import org.hibernate.envers.DefaultRevisionEntity;
20  import org.hibernate.envers.ModifiedEntityNames;
21  import org.hibernate.envers.RevisionEntity;
22  import org.tailormap.api.persistence.listener.AdminRevisionListener;
23  
24  /** Custom revision entity to store additional information about revisions. */
25  @Entity
26  @RevisionEntity
27  @EntityListeners(AdminRevisionListener.class)
28  @Table(name = "revisions", schema = "history")
29  public class AdminRevision extends DefaultRevisionEntity {
30    private String modifiedBy;
31  
32    @ElementCollection
33    @JoinTable(name = "modified_entities", schema = "history", joinColumns = @JoinColumn(name = "revision_number"))
34    @Column(name = "modified_entity_name")
35    @ModifiedEntityNames
36    private Set<String> modifiedEntityNames = new HashSet<>();
37  
38    public String getModifiedBy() {
39      return modifiedBy;
40    }
41  
42    public void setModifiedBy(String modifiedBy) {
43      this.modifiedBy = modifiedBy;
44    }
45  
46    public Set<String> getModifiedEntityNames() {
47      return modifiedEntityNames;
48    }
49  
50    public void setModifiedEntityNames(Set<String> modifiedEntityNames) {
51      this.modifiedEntityNames = modifiedEntityNames;
52    }
53  
54    @Override
55    public final boolean equals(Object o) {
56      if (this == o) return true;
57      if (!(o instanceof AdminRevision that)) return false;
58      if (!super.equals(o)) return false;
59  
60      return Objects.equals(modifiedBy, that.modifiedBy);
61    }
62  
63    @Override
64    public int hashCode() {
65      int result = super.hashCode();
66      result = 31 * result + Objects.hashCode(modifiedBy);
67      return result;
68    }
69  
70    @Override
71    public String toString() {
72      return "AdminRevision{id = " + getId() + ", revisionDate = "
73          + DateTimeFormatter.INSTANCE.format(getRevisionDate()) + ", modifiedBy='"
74          + modifiedBy + '\'' + ", modifiedEntityNames="
75          + String.join(", ", modifiedEntityNames) + '}';
76    }
77  }
