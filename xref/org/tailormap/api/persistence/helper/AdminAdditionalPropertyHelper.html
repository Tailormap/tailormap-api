26    public static void addOrUpdateAdminProperty(
27        List<AdminAdditionalProperty> list, String key, Object value, boolean isPublic) {
28      if (key == null) {
29        return;
30      }
31      list.removeIf(p -> p.getKey().equals(key));
32      list.add(new AdminAdditionalProperty().key(key).value(value).isPublic(isPublic));
33    }
34  
35    public static void mapAdminPropertyValue(
36        List<AdminAdditionalProperty> additionalProperties,
37        String key,
38        boolean isPublic,
39        Function<Object, Object> valueMapper) {
40      if (key == null) {
41        return;
42      }
43      AdminAdditionalProperty property = additionalProperties.stream()
44          .filter(p -> p.getKey().equals(key))
45          .findFirst()
46          .orElseGet(() -> {
47            AdminAdditionalProperty newProperty =
48                new AdminAdditionalProperty().key(key).isPublic(isPublic);
49            additionalProperties.add(newProperty);
50            return newProperty;
51          });
52      property.setValue(valueMapper.apply(property.getValue()));
53    }
54  }
