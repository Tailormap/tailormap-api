7   package org.tailormap.api.persistence.helper;
8   
9   import java.util.List;
10  import java.util.function.Function;
11  import org.tailormap.api.persistence.json.AdminAdditionalProperty;
12  
13  public class AdminAdditionalPropertyHelper {
14    public static void addOrUpdateAdminProperty(
15        List<AdminAdditionalProperty> list, String key, Object value, boolean isPublic) {
16      if (key == null) {
17        return;
18      }
19      list.removeIf(p -> p.getKey().equals(key));
20      list.add(new AdminAdditionalProperty().key(key).value(value).isPublic(isPublic));
21    }
22  
23    public static void mapAdminPropertyValue(
24        List<AdminAdditionalProperty> additionalProperties,
25        String key,
26        boolean isPublic,
27        Function<Object, Object> valueMapper) {
28      if (key == null) {
29        return;
30      }
31      AdminAdditionalProperty property = additionalProperties.stream()
32          .filter(p -> p.getKey().equals(key))
33          .findFirst()
34          .orElseGet(() -> {
35            AdminAdditionalProperty newProperty =
36                new AdminAdditionalProperty().key(key).isPublic(isPublic);
37            additionalProperties.add(newProperty);
38            return newProperty;
39          });
40      property.setValue(valueMapper.apply(property.getValue()));
41    }
42  }
