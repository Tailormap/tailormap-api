7   package org.tailormap.api.persistence.helper;
8   
9   import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.linkTo;
10  
11  import java.util.Optional;
12  import java.util.UUID;
13  import org.springframework.stereotype.Service;
14  import org.tailormap.api.controller.UploadsController;
15  import org.tailormap.api.repository.UploadRepository;
16  
17  @Service
18  public class UploadHelper {
19  
20    private final UploadRepository uploadRepository;
21  
22    public UploadHelper(UploadRepository uploadRepository) {
23      this.uploadRepository = uploadRepository;
24    }
25  
26    public String getUrlForImage(String imageId, String category) {
27      return Optional.ofNullable(imageId)
28          .map(UUID::fromString)
29          .flatMap(imageUuid -> uploadRepository.findByIdAndCategory(imageUuid, category))
30          .map(upload -> {
31            return linkTo(UploadsController.class)
32                .slash("api")
33                .slash("uploads")
34                .slash(category)
35                .slash(imageId)
36                .slash(upload.getFilename())
37                .toString();
38          })
39          .orElse(null);
40    }
41  }
