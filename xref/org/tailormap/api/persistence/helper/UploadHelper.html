7   package org.tailormap.api.persistence.helper;
8   
9   import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.linkTo;
10  
11  import java.util.Map;
12  import java.util.Optional;
13  import java.util.UUID;
14  import org.springframework.stereotype.Service;
15  import org.tailormap.api.controller.UploadsController;
16  import org.tailormap.api.repository.UploadRepository;
17  
18  @Service
19  public class UploadHelper {
20  
21    private final UploadRepository uploadRepository;
22  
23    public UploadHelper(UploadRepository uploadRepository) {
24      this.uploadRepository = uploadRepository;
25    }
26  
27    public String getUrlForImage(String imageId, String category) {
28      return Optional.ofNullable(imageId)
29          .map(UUID::fromString)
30          .flatMap(imageUuid -> uploadRepository.findByIdAndCategory(imageUuid, category))
31          .map(upload -> linkTo(
32                  UploadsController.class,
33                  Map.of("id", imageId, "category", category, "filename", upload.getFilename()))
34              .toString())
35          .orElse(null);
36    }
37  }
