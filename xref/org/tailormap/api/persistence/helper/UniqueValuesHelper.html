6   package org.tailormap.api.persistence.helper;
7   
8   import java.io.IOException;
9   import java.lang.invoke.MethodHandles;
10  import java.util.Set;
11  import java.util.TreeSet;
12  import org.geotools.api.data.Query;
13  import org.geotools.api.data.SimpleFeatureSource;
14  import org.geotools.api.filter.Filter;
15  import org.geotools.api.filter.FilterFactory;
16  import org.geotools.api.filter.expression.Function;
17  import org.geotools.api.filter.sort.SortOrder;
18  import org.geotools.filter.text.cql2.CQLException;
19  import org.geotools.filter.text.ecql.ECQL;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.springframework.http.HttpStatus;
23  import org.springframework.web.server.ResponseStatusException;
24  import org.tailormap.api.geotools.featuresources.FeatureSourceFactoryHelper;
25  import org.tailormap.api.persistence.TMFeatureType;
26  import org.tailormap.api.viewer.model.UniqueValuesResponse;
27  
28  public class UniqueValuesHelper {
29    private static final Logger logger =
30        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
31  
32    public static UniqueValuesResponse getUniqueValues(
33        TMFeatureType tmft,
34        String attributeName,
35        String filter,
36        FilterFactory ff,
37        FeatureSourceFactoryHelper featureSourceFactoryHelper,
38        Boolean useGeotoolsUniqueFunction) {
39      final UniqueValuesResponse uniqueValuesResponse = new UniqueValuesResponse().filterApplied(false);
40      SimpleFeatureSource fs = null;
41      try {
42        Filter existingFilter = null;
43        if (null != filter) {
44          existingFilter = ECQL.toFilter(filter);
45        }
46        logger.trace("existingFilter: {}", existingFilter);
47  
48        Filter notNull = ff.not(ff.isNull(ff.property(attributeName)));
49        Filter f = notNull;
50        if (null != existingFilter) {
51          f = ff.and(notNull, existingFilter);
52          uniqueValuesResponse.filterApplied(true);
53        }
54  
55        Query q = new Query(tmft.getName(), f);
56        q.setPropertyNames(attributeName);
57        q.setSortBy(ff.sort(attributeName, SortOrder.ASCENDING));
58        logger.trace("Unique values query: {}", q);
59  
60        fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmft);
61        // and then there are 2 scenarios:
62        // there might be a performance benefit for one or the other
63        if (!useGeotoolsUniqueFunction) {
64          // #1 use a feature visitor to get the unique values
65          // not recommended, as it may not be performant
66          logger.trace("Using feature visitor to get unique values");
67          fs.getFeatures(q)
68              .accepts(
69                  feature -> uniqueValuesResponse.addValuesItem(
70                      feature.getProperty(attributeName).getValue()),
71                  null);
72        } else {
73          // #2 or use a Function to get the unique values
74          // this is the recommended way, uses SQL "distinct"
75          logger.trace("Using geotools unique collection function to get unique values");
76          Function unique = ff.function("Collection_Unique", ff.property(attributeName));
77          Object o = unique.evaluate(fs.getFeatures(q));
78          if (o instanceof Set<?> uniqueValues) {
79            uniqueValuesResponse.setValues(new TreeSet<>(uniqueValues));
80          }
81        }
82      } catch (CQLException e) {
83        logger.error("Could not parse requested filter", e);
84        throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Could not parse requested filter");
85      } catch (IOException e) {
86        logger.error("Could not retrieve attribute data", e);
87      } finally {
88        if (fs != null) {
89          fs.getDataStore().dispose();
90        }
91      }
92  
93      return uniqueValuesResponse;
94    }
95  }
