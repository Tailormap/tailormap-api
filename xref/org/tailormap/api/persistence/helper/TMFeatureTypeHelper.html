25  public class TMFeatureTypeHelper {
26    public static boolean isEditable(
27        Application application, AppTreeLayerNode appTreeLayerNode, TMFeatureType featureType) {
28      if (featureType == null) {
29        return false;
30      }
31  
32      boolean editable = false;
33      // Currently FeatureSourceHelper#loadCapabilities() sets editable to true when the datastore is
34      // a JDBC DataStore (even when the database account can't write to the table, can't detect this
35      // without trying). Other DataStore types we support (only WFS atm) we don't set as writeable
36  
37      // TODO: in the future, check for authorizations on editing. Currently you only need to be
38      // logged in (the viewer frontend already checks this before showing editable layers so we don't
39      // need to check for an authenticated user here).
40      if (featureType.isWriteable()) {
41        AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
42        editable = Boolean.TRUE.equals(appLayerSettings.getEditable());
43      }
44      return editable;
45    }

47    public static Set<String> getHiddenAttributes(
48        @NotNull TMFeatureType featureType, @NotNull AppLayerSettings appLayerSettings) {
49      Set<String> hiddenAttributes = new HashSet<>();
50      Optional.ofNullable(featureType.getSettings().getHideAttributes()).ifPresent(hiddenAttributes::addAll);
51      Optional.ofNullable(appLayerSettings.getHideAttributes()).ifPresent(hiddenAttributes::addAll);
52      return hiddenAttributes;
53    }
54  
55    public static Set<String> getReadOnlyAttributes(
56        @NotNull TMFeatureType featureType, @NotNull AppLayerSettings appLayerSettings) {
57      Set<String> readOnlyAttributes = new HashSet<>();
58      Optional.ofNullable(featureType.getSettings().getReadOnlyAttributes()).ifPresent(readOnlyAttributes::addAll);
59      Optional.ofNullable(appLayerSettings.getReadOnlyAttributes()).ifPresent(readOnlyAttributes::addAll);
60      return readOnlyAttributes;
61    }

63    /**
64     * Return a map of attribute names (in order, using a LinkedHashMap implementation) to an attribute descriptor and
65     * configured settings pair, taking into account the configured attribute order and hidden attributes.
66     *
67     * @param featureType The feature type
68     * @param appLayerSettings The app layer settings
69     * @return A sorted map as described
70     */
71    public static Map<String, Pair<TMAttributeDescriptor, AttributeSettings>> getConfiguredAttributes(
72        @NotNull TMFeatureType featureType, @NotNull AppLayerSettings appLayerSettings) {
73      LinkedHashMap<String, TMAttributeDescriptor> originalAttributesOrder = new LinkedHashMap<>();
74      for (TMAttributeDescriptor attributeDescriptor : featureType.getAttributes()) {
75        originalAttributesOrder.put(attributeDescriptor.getName(), attributeDescriptor);
76      }
77  
78      // Order of attributes taking into account hidden attributes and configured attribute order
79      LinkedHashSet<String> finalAttributeOrder;
80  
81      if (featureType.getSettings().getAttributeOrder().isEmpty()) {
82        // Use original feature type order
83        finalAttributeOrder = new LinkedHashSet<>(originalAttributesOrder.keySet());
84      } else {
85        finalAttributeOrder = new LinkedHashSet<>(featureType.getSettings().getAttributeOrder());
86        // Remove once ordered attributes which no longer exist in the feature type as saved in the
87        // configuration database
88        finalAttributeOrder.retainAll(originalAttributesOrder.keySet());
89        // Add attributes not named in attributeOrder in feature type order (added to the feature type
90        // after an admin saved/changed the ordering of attributes -- these attributes should not be
91        // hidden).
92        if (finalAttributeOrder.size() != originalAttributesOrder.size()) {
93          finalAttributeOrder.addAll(originalAttributesOrder.keySet());
94        }
95      }
96  
97      getHiddenAttributes(featureType, appLayerSettings).forEach(finalAttributeOrder::remove);
98  
99      Map<String, AttributeSettings> attributeSettings =
100         featureType.getSettings().getAttributeSettings();
101     LinkedHashMap<String, Pair<TMAttributeDescriptor, AttributeSettings>> result = new LinkedHashMap<>();
102     for (String attribute : finalAttributeOrder) {
103       AttributeSettings settings =
104           Optional.ofNullable(attributeSettings.get(attribute)).orElseGet(AttributeSettings::new);
105       TMAttributeDescriptor attributeDescriptor = originalAttributesOrder.get(attribute);
106       result.put(attribute, Pair.of(attributeDescriptor, settings));
107     }
108     return result;
109   }

111   /**
112    * Get the non-hidden attribute descriptors for a feature type.
113    *
114    * @param featureType The feature type
115    * @param appLayerSettings The app layer settings
116    * @return Unordered set of attribute descriptors
117    */
118   public static Set<TMAttributeDescriptor> getNonHiddenAttributes(
119       @NotNull TMFeatureType featureType, @NotNull AppLayerSettings appLayerSettings) {
120     Set<String> hiddenAttributes = getHiddenAttributes(featureType, appLayerSettings);
121     return featureType.getAttributes().stream()
122         .filter(attributeDescriptor -> !hiddenAttributes.contains(attributeDescriptor.getName()))
123         .collect(Collectors.toSet());
124   }
125 
126   /**
127    * Get the non-hidden attribute names for a feature type.
128    *
129    * @param featureType The feature type
130    * @param appLayerSettings The app layer settings
131    * @return Unordered set of attribute names
132    */
133   public static Set<String> getNonHiddenAttributeNames(
134       @NotNull TMFeatureType featureType, @NotNull AppLayerSettings appLayerSettings) {
135     return getNonHiddenAttributes(featureType, appLayerSettings).stream()
136         .map(TMAttributeDescriptor::getName)
137         .collect(Collectors.toSet());
138   }
