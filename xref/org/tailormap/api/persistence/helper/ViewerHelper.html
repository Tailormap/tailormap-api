6   package org.tailormap.api.persistence.helper;
7   
8   import static java.util.Objects.requireNonNullElse;
9   
10  import java.lang.invoke.MethodHandles;
11  import java.util.ArrayList;
12  import java.util.List;
13  import java.util.Objects;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.stereotype.Service;
17  import org.tailormap.api.persistence.Application;
18  import org.tailormap.api.persistence.GeoService;
19  import org.tailormap.api.persistence.json.AppI18nSettings;
20  import org.tailormap.api.persistence.json.AppTreeLayerNode;
21  import org.tailormap.api.persistence.json.AppUiSettings;
22  import org.tailormap.api.persistence.json.FilterGroup;
23  import org.tailormap.api.persistence.json.GeoServiceLayer;
24  import org.tailormap.api.repository.GeoServiceRepository;
25  import org.tailormap.api.security.AuthorisationService;
26  import org.tailormap.api.viewer.model.ViewerResponse;
27  
28  @Service
29  public class ViewerHelper {
30    private static final Logger logger =
31        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
32    private final GeoServiceRepository geoServiceRepository;
33    private final AuthorisationService authorisationService;
34  
35    public ViewerHelper(GeoServiceRepository geoServiceRepository, AuthorisationService authorisationService) {
36      this.geoServiceRepository = geoServiceRepository;
37      this.authorisationService = authorisationService;
38    }
39  
40    public ViewerResponse getViewerResponse(Application application) {
41      //  filter the filterGroups to only have filterGroups with layerIds
42      //  that are allowed for this user.
43      List<FilterGroup> allowedFilterGroups = application.getSettings().getFilterGroups().stream()
44          // for each filterGroup, filter if the layerIds are allowed
45          .filter(fg -> {
46            List<String> allowedLayerIds = new ArrayList<>();
47            // for each layerId, get the service and layer, then check if viewing is allowed
48            for (String layerId : fg.getLayerIds()) {
49              AppTreeLayerNode lyrNode = application
50                  .getAllAppTreeLayerNode()
51                  .filter(node -> node.getId().equals(layerId))
52                  .findFirst()
53                  .orElse(null);
54              if (lyrNode == null) {
55                continue;
56              }
57  
58              GeoService service = geoServiceRepository
59                  .findById(lyrNode.getServiceId())
60                  .orElse(null);
61              if (service == null) {
62                continue;
63              }
64  
65              GeoServiceLayer layer = service.getLayers().stream()
66                  .filter(l -> Objects.equals(l.getName(), lyrNode.getLayerName()))
67                  .findFirst()
68                  .orElse(null);
69              if (layer != null && authorisationService.userAllowedToViewGeoServiceLayer(service, layer)) {
70                allowedLayerIds.add(layerId);
71              }
72            }
73  
74            if (allowedLayerIds.isEmpty()) {
75              // no allowed layers in this filterGroup, skip the entire filterGroup
76              logger.trace(
77                  "Skipping filterGroup {} because there are no allowed layers for this user",
78                  fg.getId());
79              return false;
80            } else {
81              fg.setLayerIds(allowedLayerIds);
82              return true;
83            }
84          })
85          .toList();
86  
87      return new ViewerResponse()
88          .kind(ViewerResponse.KindEnum.APP)
89          .name(application.getName())
90          .title(application.getTitle())
91          .styling(application.getStyling())
92          .components(application.getComponents())
93          .i18nSettings(requireNonNullElse(
94              application.getSettings().getI18nSettings(), new AppI18nSettings().hideLanguageSwitcher(false)))
95          .uiSettings(requireNonNullElse(
96              application.getSettings().getUiSettings(), new AppUiSettings().hideLoginButton(false)))
97          .projections(List.of(application.getCrs()))
98          .filterGroups(allowedFilterGroups);
99    }
100 }
