40  @Service
41  public class ViewerHelper {
42    private static final Logger logger =
43        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
44    private final GeoServiceRepository geoServiceRepository;
45    private final AuthorisationService authorisationService;
46    private final FeatureSourceRepository featureSourceRepository;
47  
48    public ViewerHelper(
49        GeoServiceRepository geoServiceRepository,
50        AuthorisationService authorisationService,
51        FeatureSourceRepository featureSourceRepository) {
52      this.geoServiceRepository = geoServiceRepository;
53      this.authorisationService = authorisationService;
54      this.featureSourceRepository = featureSourceRepository;
55    }

57    @Transactional
58    public ViewerResponse getViewerResponse(Application application) {
59      Map<String, FeatureTypeRef> appLayerIdToFeatureTypeRef = new HashMap<>();
60      //  filter the filterGroups to only have filterGroups with layerIds
61      //  that are allowed for this user.
62      List<FilterGroup> allowedFilterGroups = application.getSettings().getFilterGroups().stream()
63          // for each filterGroup, filter if the layerIds are allowed
64          .filter(fg -> {
65            List<String> allowedLayerIds = new ArrayList<>();
66            // for each layerId, get the service and layer, then check if viewing is allowed
67            for (String layerId : fg.getLayerIds()) {
68              AppTreeLayerNode lyrNode = application
69                  .getAllAppTreeLayerNode()
70                  .filter(node -> node.getId().equals(layerId))
71                  .findFirst()
72                  .orElse(null);
73              if (lyrNode == null) {
74                continue;
75              }
76  
77              GeoService service = geoServiceRepository
78                  .findById(lyrNode.getServiceId())
79                  .orElse(null);
80              if (service == null) {
81                continue;
82              }
83  
84              GeoServiceLayer layer = service.getLayers().stream()
85                  .filter(l -> Objects.equals(l.getName(), lyrNode.getLayerName()))
86                  .findFirst()
87                  .orElse(null);
88              if (layer != null && authorisationService.userAllowedToViewGeoServiceLayer(service, layer)) {
89                allowedLayerIds.add(layerId);
90                if (service.getLayerSettings(layer.getName()).getFeatureType() != null) {
91                  appLayerIdToFeatureTypeRef.put(
92                      layerId,
93                      service.getLayerSettings(layer.getName())
94                          .getFeatureType());
95                }
96              }
97            }
98  
99            if (allowedLayerIds.isEmpty()) {
100             // no allowed layers in this filterGroup, skip the entire filterGroup
101             logger.trace(
102                 "Skipping filterGroup {} because there are no allowed layers for this user",
103                 fg.getId());
104             return false;
105           } else {
106             fg.setLayerIds(allowedLayerIds);
107             return true;
108           }
109         })
110         .toList();
111 
112     List<TMFeatureSource> featureSources =
113         featureSourceRepository.findByIds(appLayerIdToFeatureTypeRef.values().stream()
114             .map(FeatureTypeRef::getFeatureSourceId)
115             .distinct()
116             .toList());
117 
118     List<FilterGroup> verifiedFilterGroups = allowedFilterGroups.stream()
119         .peek(fg -> {
120           List<TMFeatureType> tmfts = new ArrayList<>();
121           for (String layerId : fg.getLayerIds()) {
122             FeatureTypeRef ftr = appLayerIdToFeatureTypeRef.get(layerId);
123             TMFeatureSource tmfs = featureSources.stream()
124                 .filter(fs -> Objects.equals(fs.getId(), ftr.getFeatureSourceId()))
125                 .findFirst()
126                 .orElse(null);
127             if (tmfs == null) {
128               continue;
129             }
130             TMFeatureType tmft = tmfs.findFeatureTypeByName(ftr.getFeatureTypeName());
131             if (tmft != null) {
132               tmfts.add(tmft);
133             }
134           }
135           List<Filter> verifiedFilters = this.verifyFilters(tmfts, fg.getFilters());
136           fg.setFilters(verifiedFilters);
137         })
138         .filter(fg -> !fg.getFilters().isEmpty())
139         .toList();
140 
141     return new ViewerResponse()
142         .kind(ViewerResponse.KindEnum.APP)
143         .name(application.getName())
144         .title(application.getTitle())
145         .styling(application.getStyling())
146         .components(application.getComponents())
147         .i18nSettings(requireNonNullElse(
148             application.getSettings().getI18nSettings(), new AppI18nSettings().hideLanguageSwitcher(false)))
149         .uiSettings(requireNonNullElse(
150             application.getSettings().getUiSettings(), new AppUiSettings().hideLoginButton(false)))
151         .projections(List.of(application.getCrs()))
152         .filterGroups(verifiedFilterGroups);
153   }
154 
155   /**
156    * Verify if the filters are valid for the given feature types. A filter is valid if its attribute exists in all
157    * feature types and is not hidden. If valid, the attribute alias is set from the attribute settings of the first
158    * feature type.
159    *
160    * @param tmfts the feature types to verify against
161    * @param filters the filters to verify
162    * @return the verified filters
163    */
164   public List<Filter> verifyFilters(List<TMFeatureType> tmfts, List<Filter> filters) {
165     if (filters == null || filters.isEmpty() || tmfts == null || tmfts.isEmpty()) {
166       return List.of();
167     }
168 
169     Set<String> sharedAttributeNames = tmfts.stream()
170         .map(tmft -> {
171           List<String> hiddenAttributes = tmft.getSettings().getHideAttributes();
172           return tmft.getAttributes().stream()
173               .map(TMAttributeDescriptor::getName)
174               .filter(attr -> !hiddenAttributes.contains(attr))
175               .collect(Collectors.toSet());
176         })
177         .reduce((a, b) -> {
178           Set<String> intersection = new java.util.HashSet<>(a);
179           intersection.retainAll(b);
180           return intersection;
181         })
182         .orElse(Set.of());
183 
184     Map<String, AttributeSettings> attributeSettings =
185         tmfts.get(0).getSettings().getAttributeSettings();
186     return filters.stream()
187         .filter(filter -> sharedAttributeNames.contains(filter.getAttribute()))
188         .peek(filter -> {
189           AttributeSettings settings = attributeSettings.get(filter.getAttribute());
190           if (settings != null && settings.getTitle() != null) {
191             filter.setAttributeAlias(settings.getTitle());
192           }
193         })
194         .toList();
195   }
