7   package org.tailormap.api.persistence.listener;
8   
9   import static ch.rasc.sse.eventbus.SseEvent.DEFAULT_EVENT;
10  import static org.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.ENTITY_CREATED;
11  import static org.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.ENTITY_DELETED;
12  import static org.tailormap.api.admin.model.ServerSentEvent.EventTypeEnum.ENTITY_UPDATED;
13  
14  import ch.rasc.sse.eventbus.SseEvent;
15  import ch.rasc.sse.eventbus.SseEventBus;
16  import com.fasterxml.jackson.databind.ObjectMapper;
17  import jakarta.persistence.EntityManagerFactory;
18  import jakarta.persistence.PostPersist;
19  import jakarta.persistence.PostRemove;
20  import jakarta.persistence.PostUpdate;
21  import java.lang.invoke.MethodHandles;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.context.annotation.Lazy;
26  import org.springframework.data.rest.webmvc.config.RepositoryRestMvcConfiguration;
27  import org.springframework.stereotype.Component;
28  import org.springframework.web.context.request.RequestContextHolder;
29  import org.tailormap.api.admin.model.EntityEvent;
30  import org.tailormap.api.admin.model.ServerSentEvent;
31  import org.tailormap.api.persistence.TMFeatureType;
32  
33  @Component
34  public class EntityEventPublisher {
35    private static final Logger logger =
36        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
37  
38    @Autowired
39    @Lazy
40    private EntityManagerFactory entityManagerFactory;
41  
42    @Autowired
43    @Lazy
44    private ObjectMapper objectMapper;
45  
46    @Autowired
47    @Lazy
48    private SseEventBus eventBus;
49  
50    @Autowired
51    @Lazy
52    private RepositoryRestMvcConfiguration repositoryRestMvcConfiguration;
53  
54    public EntityEventPublisher() {}
55  
56    private void sendEvent(ServerSentEvent.EventTypeEnum eventTypeEnum, Object entity, boolean serializeEntity) {
57      Object id = null;
58  
59      if (RequestContextHolder.getRequestAttributes() == null) {
60        // No current request -- do not send events / serialize entities during app startup
61        return;
62      }
63  
64      try {
65        id = entityManagerFactory.getPersistenceUnitUtil().getIdentifier(entity);
66        EntityEvent entityEvent = new EntityEvent()
67            .entityName(entity.getClass().getSimpleName())
68            .id(String.valueOf(id));
69        if (serializeEntity) {
70          entityEvent.setObject(
71              repositoryRestMvcConfiguration.objectMapper().valueToTree(entity));
72        }
73        ServerSentEvent event =
74            new ServerSentEvent().eventType(eventTypeEnum).details(entityEvent);
75        this.eventBus.handleEvent(SseEvent.of(DEFAULT_EVENT, objectMapper.writeValueAsString(event)));
76      } catch (Exception e) {
77        logger.error(
78            "Error sending SSE for event type {}, entity {}, id {}",
79            eventTypeEnum,
80            entity != null ? entity.getClass().getSimpleName() : null,
81            id,
82            e);
83      }
84    }
85  
86    @PostPersist
87    public void postPersist(Object entity) {
88      // Feature types are created when TMFeatureSource is created, so do not send event
89      // Note that when TMFeatureSource created event is sent, the oneToMany allFeatureTypes
90      // have no ID's yet. So for a created feature source the frontend must retrieve the
91      // TMFeatureSource separately or use the return value of the POST request.
92      if (!(entity instanceof TMFeatureType)) {
93        sendEvent(ENTITY_CREATED, entity, true);
94      }
95    }
96  
97    @PostRemove
98    public void postRemove(Object entity) {
99      // Feature types are only deleted when refreshing the feature source, so the updated/deleted
100     // event for TMFeatureSource suffices.
101     if (!(entity instanceof TMFeatureType)) {
102       sendEvent(ENTITY_DELETED, entity, false);
103     }
104   }
105 
106   @PostUpdate
107   public void postUpdate(Object entity) {
108     // Note that for an updated TMFeatureSource, new TMFeatureTypes do appear to have ID's set.
109     sendEvent(ENTITY_UPDATED, entity, true);
110   }
111 }
