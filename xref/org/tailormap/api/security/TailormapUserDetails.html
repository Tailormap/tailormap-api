6   package org.tailormap.api.security;
7   
8   import java.time.ZoneId;
9   import java.time.ZonedDateTime;
10  import java.util.Collection;
11  import java.util.HashSet;
12  import org.apache.commons.lang3.StringUtils;
13  import org.springframework.security.core.GrantedAuthority;
14  import org.springframework.security.core.authority.SimpleGrantedAuthority;
15  import org.springframework.security.core.userdetails.UserDetails;
16  import org.tailormap.api.persistence.Group;
17  import org.tailormap.api.persistence.User;
18  
19  public class TailormapUserDetails implements UserDetails {
20  
21    private final User user;
22  
23    public TailormapUserDetails(User user) {
24      this.user = user;
25    }
26  
27    @Override
28    public Collection<? extends GrantedAuthority> getAuthorities() {
29      Collection<GrantedAuthority> authorities = new HashSet<>();
30      user.getGroups().stream()
31          .map(Group::getName)
32          .map(SimpleGrantedAuthority::new)
33          .forEach(authorities::add);
34  
35      user.getGroups().stream()
36          .map(Group::getAliasForGroup)
37          .filter(StringUtils::isNotBlank)
38          .map(SimpleGrantedAuthority::new)
39          .forEach(authorities::add);
40      return authorities;
41    }
42  
43    @Override
44    public String getPassword() {
45      return user.getPassword();
46    }
47  
48    @Override
49    public String getUsername() {
50      return user.getUsername();
51    }
52  
53    @Override
54    public boolean isAccountNonExpired() {
55      return user.getValidUntil() == null || user.getValidUntil().isAfter(ZonedDateTime.now(ZoneId.systemDefault()));
56    }
57  
58    @Override
59    public boolean isEnabled() {
60      return user.isEnabled();
61    }
62  }
