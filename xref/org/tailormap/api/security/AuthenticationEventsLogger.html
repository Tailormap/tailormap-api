7   package org.tailormap.api.security;
8   
9   import static org.springframework.core.NestedExceptionUtils.getRootCause;
10  
11  import io.micrometer.core.instrument.Metrics;
12  import java.lang.invoke.MethodHandles;
13  import org.jspecify.annotations.NonNull;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.context.event.EventListener;
17  import org.springframework.security.authentication.ProviderNotFoundException;
18  import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
19  import org.springframework.security.authentication.event.AbstractAuthenticationEvent;
20  import org.springframework.security.authentication.event.AbstractAuthenticationFailureEvent;
21  import org.springframework.security.authentication.event.AuthenticationSuccessEvent;
22  import org.springframework.security.oauth2.client.authentication.OAuth2LoginAuthenticationToken;
23  import org.springframework.security.oauth2.core.OAuth2AuthenticationException;
24  import org.springframework.security.oauth2.core.oidc.user.DefaultOidcUser;
25  import org.springframework.security.web.authentication.WebAuthenticationDetails;
26  import org.springframework.stereotype.Component;
27  import org.tailormap.api.security.events.DefaultAuthenticationFailureEvent;
28  
29  @Component
30  public class AuthenticationEventsLogger {
31    private static final Logger logger =
32        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
33  
34    private static @NonNull String getUserInfo(AbstractAuthenticationEvent event, boolean includePersonalInfo) {
35      String userInfo = "";
36      if (event.getAuthentication().getPrincipal() != null) {
37        // prevent leaking personal data in logs unless trace logging is enabled or failures are logged
38        userInfo = " for user <username hidden>";
39        if (logger.isTraceEnabled() || includePersonalInfo) {
40          String ipInfo = "";
41          if (event.getAuthentication().getDetails() instanceof WebAuthenticationDetails details) {
42            ipInfo = " (IP: %s)".formatted(details.getRemoteAddress());
43          }
44          userInfo =
45              " for user \"%s\"%s".formatted(event.getAuthentication().getPrincipal(), ipInfo);
46        }
47      }
48      return userInfo;
49    }
50  
51    private record LoginInfo(String authType, String authInfo, String clientId, String clientName) {}
52  
53    private static LoginInfo getLoginInfo(AbstractAuthenticationEvent event) {
54      String authType = "";
55      String clientId = "";
56      String clientName = "";
57      String authInfo = "";
58      if (event.getAuthentication() instanceof OAuth2LoginAuthenticationToken token) {
59        authType = "oauth2";
60        // prevent leaking personal data in logs unless trace logging is enabled
61        String userClaims = "";
62        if (logger.isTraceEnabled() && token.getPrincipal() instanceof DefaultOidcUser oidcUser) {
63          userClaims = ", user claims: " + oidcUser.getUserInfo().getClaims();
64        }
65        clientId = token.getClientRegistration().getClientId();
66        clientName = token.getClientRegistration().getClientName();
67        authInfo = "via OIDC registration \"%s\" with client ID %s%s".formatted(clientName, clientId, userClaims);
68      } else if (event.getAuthentication() instanceof UsernamePasswordAuthenticationToken) {
69        authType = "username_password";
70        authInfo = "using username/password";
71      }
72      return new LoginInfo(authType, authInfo, clientId, clientName);
73    }
74  
75    private static void incrementMetricsCounter(String name, LoginInfo loginInfo) {
76      Metrics.counter(
77              name,
78              "type",
79              loginInfo.authType(),
80              "clientId",
81              loginInfo.clientId(),
82              "clientName",
83              loginInfo.clientName())
84          .increment();
85    }
86  
87    @EventListener
88    public void onSuccess(AuthenticationSuccessEvent success) {
89      LoginInfo loginInfo = getLoginInfo(success);
90      logger.info(
91          "Authentication successful{}, granted authorities: {}, {}",
92          getUserInfo(success, false),
93          success.getAuthentication().getAuthorities(),
94          loginInfo.authInfo());
95      incrementMetricsCounter("tailormap_authentication_success", loginInfo);
96    }
97  
98    @EventListener
99    public void onFailure(AbstractAuthenticationFailureEvent failure) {
100     Throwable exception = failure.getException();
101     Throwable rootCause = getRootCause(exception);
102 
103     if (exception instanceof ProviderNotFoundException
104         && failure.getAuthentication() instanceof OAuth2LoginAuthenticationToken) {
105       // Ignore, because after this generic exception another event will come with details about the actual OIDC
106       // authentication failure
107       // Prevents logging and incrementing the Micrometer counter for the authentication failure twice
108       logger.trace("Ignoring ProviderNotFoundException for OAuth2LoginAuthenticationToken", exception);
109       return;
110     }
111 
112     String oauth2Message = "";
113     if (exception instanceof OAuth2AuthenticationException oauth2Ex && oauth2Ex.getError() != null) {
114       oauth2Message = ", error code %s: %s"
115           .formatted(
116               oauth2Ex.getError().getErrorCode(),
117               oauth2Ex.getError().getDescription());
118     }
119 
120     LoginInfo loginInfo = getLoginInfo(failure);
121 
122     logger.warn(
123         "Authentication failure{}, {}{}, exception {}: {}{}",
124         // in this case logging the "login" is useful/warranted for analysis
125         getUserInfo(failure, true),
126         loginInfo.authInfo(),
127         oauth2Message,
128         exception.getClass().getName(),
129         exception.getMessage(),
130         rootCause != null
131             ? ", root cause %s: %s".formatted(rootCause.getClass().getName(), rootCause.getMessage())
132             : "");
133 
134     logger.debug("Authentication failure stacktrace", exception);
135 
136     incrementMetricsCounter("tailormap_authentication_failure", loginInfo);
137   }
138 
139   @EventListener
140   public void onDefaultAuthenticationFailureEvent(DefaultAuthenticationFailureEvent event) {
141     logger.info("Default authentication failure", event.getException());
142   }
143 }
