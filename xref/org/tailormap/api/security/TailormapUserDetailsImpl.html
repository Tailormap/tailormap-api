6   package org.tailormap.api.security;
7   
8   import java.io.Serial;
9   import java.time.ZoneId;
10  import java.time.ZonedDateTime;
11  import java.util.ArrayList;
12  import java.util.Collection;
13  import java.util.HashSet;
14  import org.apache.commons.lang3.StringUtils;
15  import org.springframework.security.core.GrantedAuthority;
16  import org.springframework.security.core.authority.SimpleGrantedAuthority;
17  import org.tailormap.api.persistence.Group;
18  import org.tailormap.api.persistence.User;
19  import org.tailormap.api.persistence.json.AdminAdditionalProperty;
20  import org.tailormap.api.repository.GroupRepository;
21  
22  /* Do not make public, use the interface */
23  class TailormapUserDetailsImpl implements TailormapUserDetails {
24  
25    @Serial
26    private static final long serialVersionUID = 1L;
27  
28    private final Collection<GrantedAuthority> authorities;
29    private final String username;
30    private final String password;
31    private final ZonedDateTime validUntil;
32    private final boolean enabled;
33    private final String organisation;
34  
35    private final Collection<TailormapAdditionalProperty> additionalProperties = new ArrayList<>();
36    private final Collection<TailormapAdditionalProperty> additionalGroupProperties = new ArrayList<>();
37  
38    public TailormapUserDetailsImpl(User user, GroupRepository groupRepository) {
39      authorities = new HashSet<>();
40      user.getGroups().stream()
41          .map(Group::getName)
42          .map(SimpleGrantedAuthority::new)
43          .forEach(authorities::add);
44  
45      user.getGroups().stream()
46          .map(Group::getAliasForGroup)
47          .filter(StringUtils::isNotBlank)
48          .map(SimpleGrantedAuthority::new)
49          .forEach(authorities::add);
50  
51      username = user.getUsername();
52      password = user.getPassword();
53      validUntil = user.getValidUntil();
54      enabled = user.isEnabled();
55      organisation = user.getOrganisation();
56  
57      if (user.getAdditionalProperties() != null) {
58        for (AdminAdditionalProperty property : user.getAdditionalProperties()) {
59          additionalProperties.add(new TailormapAdditionalProperty(
60              property.getKey(), property.getIsPublic(), property.getValue()));
61        }
62      }
63  
64      // For group properties, look in the database with a list of authorities instead of user.getGroups(), so
65      // aliasForGroup is taken into account
66      this.additionalGroupProperties.addAll(groupRepository.findAdditionalPropertiesByGroups(
67          authorities.stream().map(GrantedAuthority::getAuthority).toList()));
68    }
69  
70    @Override
71    public Collection<? extends GrantedAuthority> getAuthorities() {
72      return authorities;
73    }
74  
75    @Override
76    public String getPassword() {
77      return password;
78    }
79  
80    @Override
81    public String getUsername() {
82      return username;
83    }
84  
85    @Override
86    public boolean isAccountNonExpired() {
87      return validUntil == null || validUntil.isAfter(ZonedDateTime.now(ZoneId.systemDefault()));
88    }
89  
90    @Override
91    public boolean isEnabled() {
92      return enabled;
93    }
94  
95    @Override
96    public String getOrganisation() {
97      return organisation;
98    }
99  
100   @Override
101   public Collection<TailormapAdditionalProperty> getAdditionalProperties() {
102     return additionalProperties;
103   }
104 
105   @Override
106   public Collection<TailormapAdditionalProperty> getAdditionalGroupProperties() {
107     return additionalGroupProperties;
108   }
109 }
