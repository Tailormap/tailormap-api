6   package org.tailormap.api.security;
7   
8   import jakarta.servlet.http.HttpServletRequest;
9   import jakarta.servlet.http.HttpServletResponse;
10  import jakarta.servlet.http.HttpSession;
11  import java.io.IOException;
12  import java.util.Collections;
13  import org.springframework.beans.factory.annotation.Value;
14  import org.springframework.boot.web.server.Cookie;
15  import org.springframework.context.ApplicationEventPublisher;
16  import org.springframework.context.annotation.Bean;
17  import org.springframework.context.annotation.Configuration;
18  import org.springframework.http.HttpStatus;
19  import org.springframework.security.authentication.AuthenticationEventPublisher;
20  import org.springframework.security.authentication.DefaultAuthenticationEventPublisher;
21  import org.springframework.security.config.annotation.method.configuration.EnableMethodSecurity;
22  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
23  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
24  import org.springframework.security.config.annotation.web.configurers.AbstractHttpConfigurer;
25  import org.springframework.security.oauth2.core.OAuth2AuthenticationException;
26  import org.springframework.security.web.DefaultRedirectStrategy;
27  import org.springframework.security.web.RedirectStrategy;
28  import org.springframework.security.web.SecurityFilterChain;
29  import org.springframework.security.web.authentication.AnonymousAuthenticationFilter;
30  import org.springframework.security.web.authentication.AuthenticationSuccessHandler;
31  import org.springframework.security.web.authentication.www.BasicAuthenticationFilter;
32  import org.springframework.security.web.csrf.CookieCsrfTokenRepository;
33  import org.springframework.security.web.csrf.CsrfTokenRequestAttributeHandler;
34  import org.tailormap.api.persistence.Group;
35  import org.tailormap.api.repository.OIDCConfigurationRepository;
36  import org.tailormap.api.security.events.DefaultAuthenticationFailureEvent;
37  import org.tailormap.api.security.events.OAuth2AuthenticationFailureEvent;
38  
39  @Configuration
40  @EnableWebSecurity
41  @EnableMethodSecurity
42  public class ApiSecurityConfiguration {
43    private final TailormapOidcUserService oidcUserService;
44  
45    @Value("${tailormap-api.base-path}")
46    private String apiBasePath;
47  
48    @Value("${tailormap-api.admin.base-path}")
49    private String adminApiBasePath;
50  
51    @Value("${tailormap-api.security.disable-csrf:false}")
52    private boolean disableCsrf;
53  
54    public ApiSecurityConfiguration(TailormapOidcUserService oidcUserService) {
55      this.oidcUserService = oidcUserService;
56    }
57  
58    @Bean
59    public CookieCsrfTokenRepository csrfTokenRepository() {
60      // Spring CSRF protection requires an X-XSRF-TOKEN header read from the XSRF-TOKEN cookie by
61      // JavaScript so set HttpOnly to false. Angular has automatic XSRF protection support:
62      // https://angular.io/guide/http#security-xsrf-protection
63      CookieCsrfTokenRepository csrfTokenRepository = CookieCsrfTokenRepository.withHttpOnlyFalse();
64      // Allow cross-domain non-GET (unsafe) requests for embedding with an iframe
65      csrfTokenRepository.setCookieCustomizer(cookieCustomizer -> {
66        // Do not set SameSite=None when testing with HTTP instead of HTTPS
67        // Ideally use HttpServletRequest.isSecure(), but look at built cookie for now...
68        if (cookieCustomizer.build().isSecure()) {
69          cookieCustomizer.sameSite(Cookie.SameSite.NONE.attributeValue());
70        }
71      });
72      return csrfTokenRepository;
73    }
74  
75    @Bean
76    public AuthenticationEventPublisher authenticationEventPublisher(
77        ApplicationEventPublisher applicationEventPublisher) {
78      DefaultAuthenticationEventPublisher authenticationEventPublisher =
79          new DefaultAuthenticationEventPublisher(applicationEventPublisher);
80  
81      authenticationEventPublisher.setAdditionalExceptionMappings(
82          Collections.singletonMap(OAuth2AuthenticationException.class, OAuth2AuthenticationFailureEvent.class));
83  
84      authenticationEventPublisher.setDefaultAuthenticationFailureEvent(DefaultAuthenticationFailureEvent.class);
85  
86      return authenticationEventPublisher;
87    }
88  
89    @Bean
90    public SecurityFilterChain apiFilterChain(HttpSecurity http, CookieCsrfTokenRepository csrfTokenRepository)
91        throws Exception {
92  
93      // Disable CSRF protection for development with HAL explorer
94      // https://github.com/spring-projects/spring-data-rest/issues/1347
95      if (disableCsrf) {
96        http.csrf(AbstractHttpConfigurer::disable);
97      } else {
98        // https://docs.spring.io/spring-security/reference/servlet/exploits/csrf.html#csrf-integration-javascript-spa
99        http.csrf(csrf -> csrf.csrfTokenRequestHandler(new CsrfTokenRequestAttributeHandler())
100           .csrfTokenRepository(csrfTokenRepository)
101           .ignoringRequestMatchers(
102               // This uses POST for large filter in body, but is safe (read-only)
103               apiBasePath + "/{viewerKind}/{viewerName}/layer/{appLayerId}/features",
104               // Allow PUT for ingest metrics, but only for allowed metrics
105               apiBasePath + "/{viewerKind}/{viewerName}/metrics/ingest/{appLayerId}/{allowedMetric}"));
106       http.addFilterAfter(new CsrfCookieFilter(), BasicAuthenticationFilter.class);
107     }
108 
109     // Before redirecting the user to the OAuth2 authorization endpoint, store the requested
110     // redirect URL.
111     RedirectStrategy redirectStrategy = new DefaultRedirectStrategy() {
112       @Override
113       public void sendRedirect(HttpServletRequest request, HttpServletResponse response, String url)
114           throws IOException {
115         String redirectUrl = request.getParameter("redirectUrl");
116         if (redirectUrl != null && redirectUrl.startsWith("/")) {
117           request.getSession().setAttribute("redirectUrl", redirectUrl);
118         }
119         super.sendRedirect(request, response, url);
120       }
121     };
122 
123     // When OAuth2 authentication succeeds, use the redirect URL stored in the session to send them
124     // back.
125     AuthenticationSuccessHandler authenticationSuccessHandler = (request, response, authentication) -> {
126       HttpSession session = request.getSession(false);
127       if (session != null) {
128         String redirectUrl = (String) session.getAttribute("redirectUrl");
129         if (redirectUrl != null) {
130           response.sendRedirect(redirectUrl);
131           return;
132         }
133       }
134       response.sendRedirect("/");
135     };
136 
137     http.securityMatchers(matchers -> matchers.requestMatchers(apiBasePath + "/**"))
138         .addFilterAfter(
139             /* (debug) log user making the request */ new AuditInterceptor(),
140             AnonymousAuthenticationFilter.class)
141         .authorizeHttpRequests(authorize -> {
142           authorize.requestMatchers(adminApiBasePath + "/**").hasAuthority(Group.ADMIN);
143           authorize.requestMatchers(apiBasePath + "/**").permitAll();
144         })
145         .formLogin(formLogin ->
146             formLogin.loginPage(apiBasePath + "/unauthorized").loginProcessingUrl(apiBasePath + "/login"))
147         .oauth2Login(login -> login.authorizationEndpoint(
148                 endpoint -> endpoint.baseUri(apiBasePath + "/oauth2/authorization")
149                     .authorizationRedirectStrategy(redirectStrategy))
150             .redirectionEndpoint(endpoint -> endpoint.baseUri(apiBasePath + "/oauth2/callback"))
151             .userInfoEndpoint(endpoint -> endpoint.oidcUserService(oidcUserService))
152             .successHandler(authenticationSuccessHandler))
153         .anonymous(anonymous -> anonymous.authorities(Group.ANONYMOUS))
154         .logout(logout -> logout.logoutUrl(apiBasePath + "/logout")
155             .logoutSuccessHandler((request, response, authentication) ->
156                 response.sendError(HttpStatus.OK.value(), "OK")));
157     return http.build();
158   }
159 
160   @Bean
161   public OIDCRepository clientRegistrationRepository(OIDCConfigurationRepository repository) {
162     return new OIDCRepository(repository);
163   }
164 }
