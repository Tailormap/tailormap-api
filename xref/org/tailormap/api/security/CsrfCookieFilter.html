7   package org.tailormap.api.security;
8   
9   import jakarta.servlet.FilterChain;
10  import jakarta.servlet.ServletException;
11  import jakarta.servlet.http.HttpServletRequest;
12  import jakarta.servlet.http.HttpServletResponse;
13  import java.io.IOException;
14  import org.springframework.lang.NonNull;
15  import org.springframework.security.web.csrf.CsrfToken;
16  import org.springframework.web.filter.OncePerRequestFilter;
17  
18  /**
19   * From <a
20   * href="https://docs.spring.io/spring-security/reference/servlet/exploits/csrf.html#csrf-integration-javascript-spa">Spring
21   * Security docs</a>.
22   */
23  public final class CsrfCookieFilter extends OncePerRequestFilter {
24    @Override
25    protected void doFilterInternal(
26        HttpServletRequest request, @NonNull HttpServletResponse response, FilterChain filterChain)
27        throws ServletException, IOException {
28      CsrfToken csrfToken = (CsrfToken) request.getAttribute("_csrf");
29      // Render the token value to a cookie by causing the deferred token to be loaded
30      csrfToken.getToken();
31  
32      filterChain.doFilter(request, response);
33    }
34  }
