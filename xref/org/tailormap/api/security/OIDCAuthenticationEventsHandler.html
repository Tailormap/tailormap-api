7   package org.tailormap.api.security;
8   
9   import java.lang.invoke.MethodHandles;
10  import java.time.Instant;
11  import java.util.Collections;
12  import java.util.HashMap;
13  import java.util.HashSet;
14  import java.util.List;
15  import java.util.Map;
16  import java.util.Optional;
17  import java.util.Set;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  import org.springframework.context.event.EventListener;
21  import org.springframework.security.authentication.event.AuthenticationSuccessEvent;
22  import org.springframework.security.oauth2.client.authentication.OAuth2LoginAuthenticationToken;
23  import org.springframework.security.oauth2.core.oidc.user.DefaultOidcUser;
24  import org.springframework.stereotype.Component;
25  import org.springframework.transaction.annotation.Transactional;
26  import org.tailormap.api.persistence.Group;
27  import org.tailormap.api.repository.GroupRepository;
28  import org.tailormap.api.security.events.OAuth2AuthenticationFailureEvent;
29  
30  @Component
31  public class OIDCAuthenticationEventsHandler {
32    private static final Logger logger =
33        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
34  
35    private final GroupRepository groupRepository;
36  
37    public OIDCAuthenticationEventsHandler(GroupRepository groupRepository) {
38      this.groupRepository = groupRepository;
39    }
40  
41    @EventListener
42    @Transactional
43    public void onSuccess(AuthenticationSuccessEvent success) {
44      if (success.getSource() instanceof OAuth2LoginAuthenticationToken token
45          && token.getPrincipal() instanceof DefaultOidcUser oidcUser) {
46        String clientId = token.getClientRegistration().getClientId();
47  
48        List<String> roles = Optional.ofNullable(oidcUser.getIdToken().getClaimAsStringList("roles"))
49            .orElseGet(Collections::emptyList);
50  
51        for (String role : roles) {
52          Group group = groupRepository.findById(role).orElseGet(() -> new Group().setName(role));
53          group.mapAdminPropertyValue("oidcClientIds", false, value -> {
54            @SuppressWarnings("unchecked")
55            Set<String> clientIds =
56                new HashSet<>(value instanceof List ? (List<String>) value : Collections.emptyList());
57            clientIds.add(clientId);
58            return clientIds;
59          });
60          group.mapAdminPropertyValue("oidcLastSeen", false, value -> {
61            @SuppressWarnings("unchecked")
62            Map<String, String> lastSeenByClientId =
63                value instanceof Map ? (Map<String, String>) value : new HashMap<>();
64            lastSeenByClientId.put(clientId, Instant.now().toString());
65            return lastSeenByClientId;
66          });
67          groupRepository.save(group);
68        }
69      }
70    }
71  
72    @EventListener
73    public void onOAuth2AuthenticationFailureEvent(OAuth2AuthenticationFailureEvent event) {
74      logger.info(
75          "OAuth2 authentication failure: {}, {}", event.getException().getMessage(), event);
76    }
77  }
