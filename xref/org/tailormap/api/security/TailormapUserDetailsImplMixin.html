6   package org.tailormap.api.security;
7   
8   import com.fasterxml.jackson.annotation.JsonAutoDetect;
9   import com.fasterxml.jackson.annotation.JsonCreator;
10  import com.fasterxml.jackson.annotation.JsonIgnore;
11  import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
12  import com.fasterxml.jackson.annotation.JsonProperty;
13  import com.fasterxml.jackson.annotation.JsonTypeInfo;
14  import java.time.ZonedDateTime;
15  import java.util.Collection;
16  import org.springframework.security.core.GrantedAuthority;
17  
18  @JsonTypeInfo(use = JsonTypeInfo.Id.CLASS)
19  @JsonAutoDetect(
20      fieldVisibility = JsonAutoDetect.Visibility.ANY,
21      getterVisibility = JsonAutoDetect.Visibility.NONE,
22      isGetterVisibility = JsonAutoDetect.Visibility.NONE)
23  @JsonIgnoreProperties(ignoreUnknown = true)
24  public abstract class TailormapUserDetailsImplMixin {
25  
26    @JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
27    public TailormapUserDetailsImplMixin(
28        @SuppressWarnings("unused") @JsonProperty("authorities") Collection<GrantedAuthority> authorities,
29        @SuppressWarnings("unused") @JsonProperty("validUntil") ZonedDateTime validUntil,
30        @SuppressWarnings("unused") @JsonProperty("username") String username,
31        @SuppressWarnings("unused") @JsonProperty("organisation") String organisation,
32        @SuppressWarnings("unused") @JsonProperty("password") String password,
33        @SuppressWarnings("unused") @JsonProperty("enabled") boolean enabled,
34        @SuppressWarnings("unused") @JsonProperty("additionalProperties")
35            Collection<TailormapAdditionalProperty> additionalProperties,
36        @SuppressWarnings("unused") @JsonProperty("additionalGroupProperties")
37            Collection<TailormapAdditionalProperty> additionalGroupProperties) {
38      // mixin constructor only for Jackson; no implementation
39    }
40  
41    /** prevents serializing the password. */
42    @JsonIgnore
43    public abstract String getPassword();
44  }
