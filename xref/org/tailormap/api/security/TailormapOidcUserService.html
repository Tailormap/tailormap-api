7   package org.tailormap.api.security;
8   
9   import java.lang.invoke.MethodHandles;
10  import java.util.ArrayList;
11  import java.util.Collection;
12  import java.util.HashSet;
13  import java.util.List;
14  import java.util.Optional;
15  import java.util.Set;
16  import java.util.regex.Pattern;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  import org.springframework.security.core.GrantedAuthority;
20  import org.springframework.security.core.authority.SimpleGrantedAuthority;
21  import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserRequest;
22  import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserService;
23  import org.springframework.security.oauth2.core.OAuth2AuthenticationException;
24  import org.springframework.security.oauth2.core.oidc.user.OidcUser;
25  import org.springframework.stereotype.Service;
26  import org.tailormap.api.persistence.OIDCConfiguration;
27  import org.tailormap.api.repository.GroupRepository;
28  import org.tailormap.api.repository.OIDCConfigurationRepository;
29  
30  @Service
31  public class TailormapOidcUserService extends OidcUserService {
32    private static final Logger logger =
33        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
34  
35    private final GroupRepository groupRepository;
36    private final OIDCConfigurationRepository oidcConfigurationRepository;
37  
38    public TailormapOidcUserService(
39        GroupRepository groupRepository, OIDCConfigurationRepository oidcConfigurationRepository) {
40      this.groupRepository = groupRepository;
41      this.oidcConfigurationRepository = oidcConfigurationRepository;
42    }
43  
44    @Override
45    public OidcUser loadUser(OidcUserRequest userRequest) throws OAuth2AuthenticationException {
46      OidcUser user = super.loadUser(userRequest);
47  
48      String registrationId = userRequest.getClientRegistration().getRegistrationId();
49      Optional<OIDCConfiguration> oidcConfiguration = "static".equals(registrationId)
50          ? Optional.empty()
51          : oidcConfigurationRepository.findById(Long.valueOf(registrationId));
52  
53      Set<String> groups = new HashSet<>();
54  
55      List<String> roles = userRequest.getIdToken().getClaimAsStringList("roles");
56      if (roles != null) {
57        String oidcRegistrationName = userRequest.getClientRegistration().getClientName();
58        logger.debug("[{}] Roles from ID token: {}", oidcRegistrationName, roles);
59        Pattern rolesClaimFilterPattern = oidcConfiguration
60            .map(OIDCConfiguration::getRolesClaimFilterRegex)
61            .map(Pattern::compile)
62            .orElse(null);
63  
64        if (rolesClaimFilterPattern != null) {
65          List<String> filteredOut = roles.stream()
66              .filter(role -> !rolesClaimFilterPattern.matcher(role).find())
67              .toList();
68          if (logger.isDebugEnabled() && !filteredOut.isEmpty()) {
69            logger.debug(
70                "[{}] Roles filtered out by pattern '{}': {}",
71                oidcRegistrationName,
72                rolesClaimFilterPattern,
73                filteredOut);
74          }
75          roles.removeAll(filteredOut);
76        }
77        groups.addAll(roles);
78      }
79  
80      oidcConfiguration.ifPresent(oc -> {
81        if (oc.getDefaultAuthorities() != null) {
82          groups.addAll(oc.getDefaultAuthorities());
83        }
84      });
85  
86      // Add aliases for groups (with the same name as the role) as authorities, even if the group does not exist
87      Set<String> aliases = groupRepository.findAliasesForGroups(groups);
88      groups.addAll(aliases);
89  
90      // Find all additional properties for the groups, taking aliases into account
91      Collection<TailormapAdditionalProperty> groupProperties =
92          groupRepository.findAdditionalPropertiesByGroups(groups);
93  
94      // Adds OIDC_USER and SCOPE_* authorities
95      Collection<GrantedAuthority> authorities = new ArrayList<>(user.getAuthorities());
96      // Add (aliased) group names as authorities
97      groups.stream().map(SimpleGrantedAuthority::new).forEach(authorities::add);
98  
99      String userNameAttributeName = userRequest
100         .getClientRegistration()
101         .getProviderDetails()
102         .getUserInfoEndpoint()
103         .getUserNameAttributeName();
104 
105     String oidcRegistrationName = userRequest.getClientRegistration().getClientName();
106 
107     return new TailormapOidcUser(
108         authorities,
109         user.getIdToken(),
110         user.getUserInfo(),
111         userNameAttributeName,
112         oidcRegistrationName,
113         groupProperties);
114   }
115 }
