7   package org.tailormap.api.security;
8   
9   import java.lang.invoke.MethodHandles;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  import org.springframework.context.event.EventListener;
13  import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
14  import org.springframework.security.authentication.event.AbstractAuthenticationEvent;
15  import org.springframework.security.authentication.event.AbstractAuthenticationFailureEvent;
16  import org.springframework.security.authentication.event.AuthenticationSuccessEvent;
17  import org.springframework.security.oauth2.client.authentication.OAuth2LoginAuthenticationToken;
18  import org.springframework.security.oauth2.core.oidc.user.DefaultOidcUser;
19  import org.springframework.security.web.authentication.WebAuthenticationDetails;
20  import org.springframework.stereotype.Component;
21  import org.tailormap.api.security.events.DefaultAuthenticationFailureEvent;
22  import org.tailormap.api.security.events.OAuth2AuthenticationFailureEvent;
23  
24  @Component
25  public class AuthenticationEvents {
26    private static final Logger logger =
27        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
28  
29    private static String getExtraInfo(AbstractAuthenticationEvent event) {
30      String extraInfo = "";
31      if (logger.isTraceEnabled()
32          && event.getAuthentication().getDetails() instanceof WebAuthenticationDetails details) {
33        extraInfo = String.format(" (IP: %s)", details.getRemoteAddress());
34      }
35      return extraInfo;
36    }
37  
38    @EventListener
39    public void onSuccess(AuthenticationSuccessEvent success) {
40      String authInfo = "";
41      if (success.getSource() instanceof OAuth2LoginAuthenticationToken token) {
42        String userClaims = "";
43        if (token.getPrincipal() instanceof DefaultOidcUser oidcUser) {
44          userClaims = ", user claims: " + oidcUser.getUserInfo().getClaims();
45        }
46        authInfo = String.format(
47            "via OIDC registration \"%s\" with client ID %s%s",
48            token.getClientRegistration().getClientName(),
49            token.getClientRegistration().getClientId(),
50            userClaims);
51      }
52      if (success.getSource() instanceof UsernamePasswordAuthenticationToken) {
53        authInfo = "using username/password";
54      }
55      logger.info(
56          "Authentication successful for user \"{}\"{}, granted authorities: {}, {}",
57          success.getAuthentication().getName(),
58          getExtraInfo(success),
59          success.getAuthentication().getAuthorities().toString(),
60          authInfo);
61    }
62  
63    @EventListener
64    public void onFailure(AbstractAuthenticationFailureEvent failure) {
65      String userInfo = "";
66      if (failure.getAuthentication().getPrincipal() != null) {
67        userInfo = String.format(
68            " for user \"%s\"", failure.getAuthentication().getPrincipal());
69      }
70      logger.info(
71          "Authentication failure: {} {}{}",
72          failure.getException().getMessage(),
73          userInfo,
74          getExtraInfo(failure));
75    }
76  
77    @EventListener
78    public void onOAuth2AuthenticationFailureEvent(OAuth2AuthenticationFailureEvent event) {
79      logger.info(
80          "OAuth2 authentication failure: {}, {}", event.getException().getMessage(), event);
81    }
82  
83    @EventListener
84    public void onDefaultAuthenticationFailureEvent(DefaultAuthenticationFailureEvent event) {
85      logger.info("Default authentication failure", event.getException());
86    }
87  }
