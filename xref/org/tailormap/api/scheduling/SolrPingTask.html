6   package org.tailormap.api.scheduling;
7   
8   import java.lang.invoke.MethodHandles;
9   import java.time.Instant;
10  import org.quartz.DisallowConcurrentExecution;
11  import org.quartz.JobDataMap;
12  import org.quartz.JobExecutionContext;
13  import org.quartz.JobExecutionException;
14  import org.quartz.PersistJobDataAfterExecution;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  import org.springframework.scheduling.quartz.QuartzJobBean;
18  import org.tailormap.api.solr.SolrService;
19  
20  @DisallowConcurrentExecution
21  @PersistJobDataAfterExecution
22  public class SolrPingTask extends QuartzJobBean implements Task {
23    private static final Logger logger =
24        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
25    private final SolrService solrService;
26  
27    public SolrPingTask(SolrService solrService) {
28      this.solrService = solrService;
29    }
30  
31    @Override
32    protected void executeInternal(JobExecutionContext context) throws JobExecutionException {
33      final JobDataMap persistedJobData = context.getJobDetail().getJobDataMap();
34      if (solrService.isSolrServiceAvailable()) {
35        persistedJobData.put(LAST_RESULT_KEY, "Solr is available. Check succeeded.");
36        context.setResult("Solr is available");
37      } else {
38        logger.warn("Solr service is not available");
39        persistedJobData.put(LAST_RESULT_KEY, "Solr is not available. Check failed.");
40        context.setResult("Solr is not available");
41      }
42      persistedJobData.put(EXECUTION_FINISHED_KEY, Instant.now());
43    }
44  
45    @Override
46    public TaskType getType() {
47      return TaskType.SOLR_PING;
48    }
49  
50    @Override
51    public String getDescription() {
52      return "Ping Solr to ensure it is available.";
53    }
54  }
