6   package org.tailormap.api.scheduling;
7   
8   import jakarta.validation.constraints.NotNull;
9   import java.util.HashMap;
10  import java.util.Map;
11  import org.quartz.Trigger;
12  import org.springframework.util.Assert;
13  
14  /** Define a map with minimally required job data for the TailorMap scheduler. */
15  public class TMJobDataMap extends HashMap<String, Object> {
16  
17    /**
18     * Create a new instance of TMJobDataMap.
19     *
20     * @param map the map with job data, must have values for the required parameters {@code type} and
21     *     {@code description}
22     */
23    public TMJobDataMap(Map<String, Object> map) {
24      this((String) map.get("type"), (String) map.get("description"));
25      this.putAll(map);
26    }
27  
28    /**
29     * Create a new instance of TMJobDataMap with a status of {@code Trigger.TriggerState.NONE}.
30     *
31     * @param type the type of the job
32     * @param description a description of the job
33     */
34    public TMJobDataMap(@NotNull String type, @NotNull String description) {
35      this(type, description, Trigger.TriggerState.NONE);
36    }
37  
38    /**
39     * Create a new instance of TMJobDataMap.
40     *
41     * @param type the type of the job
42     * @param description a description of the job
43     * @param status the status of the job
44     */
45    public TMJobDataMap(
46        @NotNull String type, @NotNull String description, @NotNull Trigger.TriggerState status) {
47      super();
48      // Check if the map contains the required parameters
49      Assert.notNull(type, "type must not be null");
50      Assert.notNull(description, "description must not be null");
51      Assert.notNull(status, "status must not be null");
52      super.put("type", type);
53      super.put("description", description);
54      super.put("status", status);
55    }
56  
57    @NotNull
58    public String getType() {
59      return super.get("type").toString();
60    }
61  
62    @NotNull
63    public String getDescription() {
64      return super.get("description").toString();
65    }
66  
67    @NotNull
68    public Trigger.TriggerState getStatus() {
69      return (Trigger.TriggerState) super.get("status");
70    }
71  
72    public void setStatus(Trigger.TriggerState status) {
73      if (null == status) {
74        status = Trigger.TriggerState.NONE;
75      }
76      super.put("status", status);
77    }
78  }
