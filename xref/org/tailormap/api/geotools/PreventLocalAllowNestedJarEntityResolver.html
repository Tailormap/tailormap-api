7   package org.tailormap.api.geotools;
8   
9   import java.io.IOException;
10  import org.geotools.util.PreventLocalEntityResolver;
11  import org.xml.sax.SAXException;
12  
13  /**
14   * An entity resolver that allows nested JAR files used in Spring Boot using the jar:nested: protocol to be resolved,
15   * specifically for schemas that are packed inside JAR files.
16   */
17  public class PreventLocalAllowNestedJarEntityResolver extends PreventLocalEntityResolver {
18    public static final PreventLocalAllowNestedJarEntityResolver INSTANCE =
19        new PreventLocalAllowNestedJarEntityResolver();
20  
21    @Override
22    public org.xml.sax.InputSource resolveEntity(String publicId, String systemId) throws IOException, SAXException {
23      if (systemId != null && systemId.matches("(?i)jar:nested:[^?#;]*\\.xsd")) {
24        return null;
25      }
26      return super.resolveEntity(publicId, systemId);
27    }
28  }
