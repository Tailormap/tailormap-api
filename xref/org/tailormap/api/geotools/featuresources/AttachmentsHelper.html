27  /** Helper class for managing the {@code <FT>_attachments} sidecar tables in JDBC DataStores. */
28  public final class AttachmentsHelper {
29    private static final Logger logger =
30        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
31    private static final List<String> allowedPKTypesSupportingSize = List.of(
32        // list of database types that support size modifiers
33        // for their foreign key columns
34        // PostgreSQL types: https://www.postgresql.org/docs/current/datatype.html
35        "CHARACTER",
36        "CHARACTER VARYING",
37        "CHAR",
38        "VARCHAR",
39        // numeric/decimal takes size and precision but we don't want to use floating point for FK columns...
40        "NUMERIC",
41        "DECIMAL",
42        // SQL Server types:
43        // https://learn.microsoft.com/en-us/sql/t-sql/data-types/data-types-transact-sql?view=sql-server-ver17
44        "NVARCHAR",
45        "NCHAR",
46        // Oracle types
47        "VARCHAR2",
48        "NVARCHAR2",
49        "NUMBER",
50        "RAW");

52    private AttachmentsHelper() {
53      // private constructor for utility class
54    }

56    private static String getPostGISCreateAttachmentsTableStatement(
57        String tableName, String pkColumnName, String fkColumnType, String typeModifier, String schemaPrefix) {
58      if (!schemaPrefix.isEmpty()) {
59        schemaPrefix += ".";
60      }
61      return MessageFormat.format(
62          """
63  CREATE TABLE IF NOT EXISTS {4}{0}_attachments (
64  {0}_pk          {2}{3}        NOT NULL REFERENCES {4}{0}({1}) ON DELETE CASCADE,
65  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
66  file_name       VARCHAR(255),
67  attribute_name  VARCHAR(255) NOT NULL,
68  description     TEXT,
69  attachment      BYTEA        NOT NULL,
70  attachment_size INTEGER      NOT NULL,
71  mime_type       VARCHAR(100),
72  created_at      TIMESTAMPTZ  NOT NULL DEFAULT CURRENT_TIMESTAMP,
73  created_by      VARCHAR(255) NOT NULL)
74  """,
75          tableName, pkColumnName, fkColumnType, typeModifier, schemaPrefix);
76    }

78    private static String getSQLServerCreateAttachmentsTableStatement(
79        String tableName, String pkColumnName, String fkColumnType, String typeModifier, String schemaPrefix) {
80      if (!schemaPrefix.isEmpty()) {
81        schemaPrefix += ".";
82      }
83      return MessageFormat.format(
84          """
85  IF OBJECT_ID(N''{4}{0}_attachments'', ''U'') IS NULL
86  BEGIN
87  CREATE TABLE {4}{0}_attachments (
88  {0}_pk          {2}{3}         NOT NULL REFERENCES {4}{0}({1}) ON DELETE CASCADE,
89  attachment_id   UNIQUEIDENTIFIER NOT NULL DEFAULT NEWID() PRIMARY KEY,
90  file_name       NVARCHAR(255),
91  attribute_name  VARCHAR(255)    NOT NULL,
92  description     NVARCHAR(MAX),
93  attachment      VARBINARY(MAX)   NOT NULL,
94  mime_type       NVARCHAR(100),
95  attachment_size INT              NOT NULL,
96  created_at      DATETIMEOFFSET   NOT NULL DEFAULT SYSDATETIMEOFFSET(),
97  created_by      NVARCHAR(255)    NOT NULL)
98  END
99  """,
100         tableName, pkColumnName, fkColumnType, typeModifier, schemaPrefix);
101   }

103   private static String getOracleCreateAttachmentsTableStatement(
104       String tableName, String pkColumnName, String fkColumnType, String typeModifier, String schemaPrefix) {
105     if (!schemaPrefix.isEmpty()) {
106       schemaPrefix += ".";
107     }
108     // Oracle supports  IF NOT EXISTS since 19.28
109     return MessageFormat.format(
110         """
111 CREATE TABLE IF NOT EXISTS {4}{0}_ATTACHMENTS (
112 {0}_PK          {2}{3}      NOT NULL REFERENCES {4}{0}({1}) ON DELETE CASCADE,
113 ATTACHMENT_ID   RAW(16)       DEFAULT SYS_GUID() PRIMARY KEY,
114 FILE_NAME       VARCHAR2(255),
115 ATTACHMENT      BLOB          NOT NULL,
116 ATTRIBUTE_NAME  VARCHAR2(255) NOT NULL,
117 DESCRIPTION     CLOB,
118 MIME_TYPE       VARCHAR2(100),
119 ATTACHMENT_SIZE INT           NOT NULL,
120 CREATED_AT      TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
121 CREATED_BY      VARCHAR2(255) NOT NULL)
122 """,
123         tableName, pkColumnName, fkColumnType, typeModifier, schemaPrefix);
124   }

126   /**
127    * Create attachment table and index for the given FeatureType. This will validate that any AttachmentAttribute has
128    * a valid name.
129    *
130    * @param featureType the FeatureType to create the attachment table for
131    * @throws IOException when creating the GeoTools datastore fails
132    * @throws SQLException when executing the SQL statements fails
133    * @throws IllegalArgumentException when the FeatureType is invalid
134    */
135   public static void createAttachmentTableForFeatureType(TMFeatureType featureType)
136       throws IOException, SQLException, IllegalArgumentException {
137     if (featureType == null
138         || featureType.getSettings() == null
139         || featureType.getSettings().getAttachmentAttributes() == null
140         || featureType.getSettings().getAttachmentAttributes().isEmpty()) {
141       throw new IllegalArgumentException("FeatureType "
142           + (featureType != null ? featureType.getName() : "null")
143           + " is invalid or has no attachment attributes defined in its settings");
144     }
145     // check if any attachment attribute names are empty or null
146     featureType.getSettings().getAttachmentAttributes().stream()
147         .filter(attachmentAttributeType -> (attachmentAttributeType.getAttributeName() == null
148             || attachmentAttributeType.getAttributeName().isEmpty()))
149         .findAny()
150         .ifPresent(attachmentAttributeType -> {
151           throw new IllegalArgumentException("FeatureType "
152               + featureType.getName()
153               + " has an attachment attribute with invalid (null or empty) attribute name");
154         });
155 
156     logger.debug(
157         "Creating attachment table for FeatureType: {} and attachment names {}",
158         featureType.getName(),
159         featureType.getSettings().getAttachmentAttributes());
160 
161     JDBCDataStore ds = null;
162     try {
163       ds = (JDBCDataStore) new JDBCFeatureSourceHelper().createDataStore(featureType.getFeatureSource());
164 
165       try (Connection conn = ds.getDataSource().getConnection();
166           Statement stmt = conn.createStatement()) {
167         String sql = getCreateAttachmentsForFeatureTypeStatements(featureType, ds);
168         logger.debug("About to create attachments table using {}", sql);
169         stmt.execute(sql);
170         logger.info("Attachment table created for FeatureType: {}", featureType.getName());
171 
172         sql = getCreateAttachmentsIndexForFeatureTypeStatements(featureType, ds);
173         logger.debug("About to create attachments table FK index using {}", sql);
174         stmt.execute(sql);
175         logger.info("Attachment table FK index created for FeatureType: {}", featureType.getName());
176       }
177     } finally {
178       if (ds != null) {
179         ds.dispose();
180       }
181     }
182   }

184   /**
185    * Get the SQL statement to create the attachments table for the given feature type.
186    *
187    * @param featureType The feature type
188    * @return The SQL statement
189    * @throws IOException If an error connecting to the database occurs
190    * @throws IllegalArgumentException If the database type is not supported
191    */
192   private static String getCreateAttachmentsForFeatureTypeStatements(
193       TMFeatureType featureType, @NotNull JDBCDataStore ds)
194       throws IOException, IllegalArgumentException, SQLException {
195 
196     String fkColumnType = null;
197     int fkColumnSize = 0;
198     AttributeDescriptor pkDescriptor =
199         ds.getSchema(featureType.getName()).getDescriptor(featureType.getPrimaryKeyAttribute());
200 
201     try (Connection conn = ((DelegatingConnection) ds.getDataSource().getConnection()).getInnermostDelegate()) {
202       DatabaseMetaData metaData = conn.getMetaData();
203       try (ResultSet rs = metaData.getColumns(
204           conn.getCatalog(),
205           ds.getDatabaseSchema(),
206           featureType.getName(),
207           featureType.getPrimaryKeyAttribute())) {
208         if (rs.next()) {
209           fkColumnType = rs.getString("TYPE_NAME");
210           fkColumnSize = rs.getInt("COLUMN_SIZE");
211         }
212       }
213 
214       // Fallback to upper-case table/column names (common for some DBs, but something must be wrong in our
215       // configuration because we store uppercase when we get that from the database...)
216       if (fkColumnType == null) {
217         try (ResultSet rs = metaData.getColumns(
218             conn.getCatalog(),
219             ds.getDatabaseSchema(),
220             featureType.getName().toUpperCase(Locale.ROOT),
221             featureType.getPrimaryKeyAttribute().toUpperCase(Locale.ROOT))) {
222           if (rs.next()) {
223             fkColumnType = rs.getString("TYPE_NAME");
224             fkColumnSize = rs.getInt("COLUMN_SIZE");
225           }
226         }
227       }
228 
229       // Final fallback to GeoTools nativeType from the attribute descriptor
230       if (fkColumnType == null) {
231         fkColumnType = (String) pkDescriptor.getUserData().get("org.geotools.jdbc.nativeTypeName");
232       }
233     }
234 
235     String typeModifier = "";
236     if (fkColumnSize > 0) {
237       typeModifier = getValidModifier(fkColumnType, fkColumnSize);
238     }
239     logger.debug(
240         "Creating attachment table for feature type with primary key {} (native type: {}, meta type: {}, size: {} (modifier: {}))",
241         pkDescriptor.getLocalName(),
242         fkColumnType,
243         pkDescriptor.getUserData().get("org.geotools.jdbc.nativeTypeName"),
244         fkColumnSize,
245         typeModifier);
246 
247     JDBCConnectionProperties connProperties = featureType.getFeatureSource().getJdbcConnection();
248     switch (connProperties.getDbtype()) {
249       case POSTGIS -> {
250         return getPostGISCreateAttachmentsTableStatement(
251             featureType.getName(),
252             featureType.getPrimaryKeyAttribute(),
253             fkColumnType,
254             typeModifier,
255             ds.getDatabaseSchema());
256       }
257 
258       case ORACLE -> {
259         return getOracleCreateAttachmentsTableStatement(
260             featureType.getName(),
261             featureType.getPrimaryKeyAttribute(),
262             fkColumnType,
263             typeModifier,
264             ds.getDatabaseSchema());
265       }
266       case SQLSERVER -> {
267         return getSQLServerCreateAttachmentsTableStatement(
268             featureType.getName(),
269             featureType.getPrimaryKeyAttribute(),
270             fkColumnType,
271             typeModifier,
272             ds.getDatabaseSchema());
273       }
274       default ->
275         throw new IllegalArgumentException(
276             "Unsupported database type for attachments: " + connProperties.getDbtype());
277     }
278   }

280   private static String getValidModifier(String columnType, int fkColumnSize) {
281     if (fkColumnSize > 0 && allowedPKTypesSupportingSize.contains(columnType.toUpperCase(Locale.ROOT))) {
282       if (columnType.equalsIgnoreCase("NUMERIC")
283           || columnType.equalsIgnoreCase("DECIMAL")
284           || columnType.equalsIgnoreCase("NUMBER")) {
285         // For NUMERIC/DECIMAL we should ideally also get the precision, but for FK columns
286         // we just use size with default precision 0
287         return "(" + fkColumnSize + ",0)";
288       }
289       return "(" + fkColumnSize + ")";
290     } else {
291       return "";
292     }
293   }

295   /**
296    * Get the SQL statement to create the attachments foreign key index for the given feature type.
297    *
298    * @param featureType The feature type
299    * @return The SQL statement
300    * @throws IllegalArgumentException If the database type is not supported
301    */
302   private static String getCreateAttachmentsIndexForFeatureTypeStatements(TMFeatureType featureType, JDBCDataStore ds)
303       throws IllegalArgumentException {
304 
305     String schemaPrefix = ds.getDatabaseSchema();
306     if (!schemaPrefix.isEmpty()) {
307       schemaPrefix += ".";
308     }
309 
310     JDBCConnectionProperties connProperties = featureType.getFeatureSource().getJdbcConnection();
311     switch (connProperties.getDbtype()) {
312       case POSTGIS -> {
313         return MessageFormat.format(
314             "CREATE INDEX IF NOT EXISTS {0}_attachments_fk ON {1}{0}_attachments({0}_pk)",
315             featureType.getName(), schemaPrefix);
316       }
317       case SQLSERVER -> {
318         return MessageFormat.format(
319             """
320 IF NOT EXISTS(SELECT * FROM sys.indexes WHERE name = ''{0}_attachments_fk'' AND object_id = OBJECT_ID(N''{1}{0}_attachments''))
321 BEGIN
322 CREATE INDEX {0}_attachments_fk ON {1}{0}_attachments({0}_pk)
323 END
324 """,
325             featureType.getName(), schemaPrefix);
326       }
327       case ORACLE -> {
328         return MessageFormat.format(
329                 "CREATE INDEX IF NOT EXISTS {1}{0}_attachments_fk ON {1}{0}_attachments({0}_pk)",
330                 featureType.getName(), schemaPrefix)
331             .toUpperCase(Locale.ROOT);
332       }
333       default ->
334         throw new IllegalArgumentException(
335             "Unsupported database type for attachments: " + connProperties.getDbtype());
336     }
337   }
