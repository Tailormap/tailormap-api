6   package org.tailormap.api.geotools;

8   import jakarta.validation.constraints.NotNull;
9   import org.geotools.api.data.SimpleFeatureSource;
10  import org.geotools.api.referencing.FactoryException;
11  import org.geotools.api.referencing.crs.CoordinateReferenceSystem;
12  import org.geotools.api.referencing.operation.MathTransform;
13  import org.geotools.referencing.CRS;
14  import org.tailormap.api.persistence.Application;
15  
16  public class TransformationUtil {
17    private TransformationUtil() {
18      // utility class
19    }
20  
21    /**
22     * Determine whether we need to transform geometries to the application CRS. Note that this uses the "default
23     * geometry" attribute of the feature source, in cases where a feature source has multiple geometry attributes (with
24     * possibly different CRSs) or heterogenous CRSs across a single geometry attribute this may not be accurate.
25     *
26     * @param application the referenced application
27     * @param simpleFeatureSource the feature source used in the application
28     * @return {@code null} when no transform is required and a valid transform otherwise
29     * @throws FactoryException when the CRS cannot be decoded
30     */
31    public static MathTransform getTransformationToApplication(
32        @NotNull Application application, @NotNull SimpleFeatureSource simpleFeatureSource)
33        throws FactoryException {
34      MathTransform transform = null;
35      // this is the CRS of the "default geometry" attribute
36      final CoordinateReferenceSystem dataSourceCRS =
37          simpleFeatureSource.getSchema().getCoordinateReferenceSystem();
38      final CoordinateReferenceSystem appCRS = CRS.decode(application.getCrs());
39      if (!CRS.equalsIgnoreMetadata(dataSourceCRS, appCRS)) {
40        transform = CRS.findMathTransform(dataSourceCRS, appCRS);
41      }
42      return transform;
43    }
