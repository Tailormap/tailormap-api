75    public static URI getOGCRequestURL(
76        String url, String service, String version, String request, MultiValueMap<String, String> parameters) {
77  
78      final MultiValueMap<String, String> params = new LinkedMultiValueMap<>();
79      params.add("SERVICE", service);
80      params.add("VERSION", version);
81      params.add("REQUEST", request);
82      if (parameters != null) {
83        // We need to encode the parameters manually because UriComponentsBuilder annoyingly does not
84        // encode '+' as used in mime types for output formats, see
85        // https://stackoverflow.com/questions/18138011
86        parameters.replaceAll((key, values) -> values.stream()
87            .map(s -> URLEncoder.encode(s, StandardCharsets.UTF_8))
88            .collect(Collectors.toList()));
89        params.addAll(parameters);
90      }
91      return UriComponentsBuilder.fromUriString(url)
92          .replaceQueryParams(params)
93          .build(true)
94          .toUri();
95    }

98     * Get a list of GetFeature output formats for a WFS feature type.
99     *
100    * <p>If there are no specific output formats for the type, the generally supported output formats are returned.
101    * Requests WFS 1.1.0 but also handles WFS 2.0.0 responses.
102    *
103    * <p>Uses a 'lightweight' WFS implementation parsing only the XML WFS capabilities to extract the output formats
104    * using XPath, instead of using a heavyweight GeoTools WFS DataStore which is much slower.
105    */
106   public static List<String> getOutputFormats(String wfsUrl, String typeName, String username, String password)
107       throws Exception {
108     return getOutputFormats(wfsUrl, typeName, username, password, getDefaultHttpClient());
109   }

111   public static List<String> getOutputFormats(
112       String wfsUrl, String typeName, String username, String password, HttpClient httpClient) throws Exception {
113 
114     URI wfsGetCapabilities = getWFSRequestURL(wfsUrl, "GetCapabilities");
115 
116     HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(wfsGetCapabilities);
117 
118     setHttpBasicAuthenticationHeader(requestBuilder, username, password);
119 
120     HttpResponse<InputStream> response =
121         httpClient.send(requestBuilder.build(), HttpResponse.BodyHandlers.ofInputStream());
122 
123     // Parse capabilities in DOM
124 
125     DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
126     documentBuilderFactory.setNamespaceAware(true);
127     documentBuilderFactory.setExpandEntityReferences(false);
128     documentBuilderFactory.setValidating(false);
129     documentBuilderFactory.setXIncludeAware(false);
130     documentBuilderFactory.setCoalescing(true);
131     documentBuilderFactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
132     DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
133     Document doc = documentBuilder.parse(response.body());
134 
135     // WFS 1.1.0 and WFS 2.0.0 use different namespaces, but the same local element names
136     boolean wfs2 = "2.0.0".equals(doc.getDocumentElement().getAttribute("version"));
137     XPath xPath = XPathFactory.newInstance().newXPath();
138     SimpleNamespaceContext namespaceContext = new SimpleNamespaceContext();
139     namespaceContext.bindNamespaceUri("ows", "http://www.opengis.net/ows" + (wfs2 ? "/1.1" : ""));
140     namespaceContext.bindNamespaceUri("wfs", "http://www.opengis.net/wfs" + (wfs2 ? "/2.0" : ""));
141     xPath.setNamespaceContext(namespaceContext);
142 
143     List<String> outputFormats = null;
144 
145     // Find all feature types and loop through to find the typeName. We can't use XPath with the
146     // typeName parameter in it to find the exact FeatureType because escaping characters like '
147     // in the typeName is not possible.
148     NodeList featureTypes =
149         (NodeList) xPath.compile("/wfs:WFS_Capabilities" + "/wfs:FeatureTypeList" + "/wfs:FeatureType")
150             .evaluate(doc, XPathConstants.NODESET);
151 
152     for (int i = 0; i < featureTypes.getLength(); i++) {
153       Element n = (Element) featureTypes.item(i);
154       Element name = DomUtils.getChildElementByTagName(n, "Name");
155       if (name != null && typeName.equals(DomUtils.getTextValue(name))) {
156         Element formatsNode = DomUtils.getChildElementByTagName(n, "OutputFormats");
157         if (formatsNode != null) {
158           outputFormats = DomUtils.getChildElementsByTagName(formatsNode, "Format").stream()
159               .map(DomUtils::getTextValue)
160               .collect(Collectors.toList());
161         }
162         break;
163       }
164     }
165 
166     // No output formats found (or maybe not even the featureType...), return output formats from
167     // OperationsMetadata
168     if (outputFormats == null) {
169       String xpathExpr = "/wfs:WFS_Capabilities"
170           + "/ows:OperationsMetadata"
171           + "/ows:Operation[@name='GetFeature']"
172           + "/ows:Parameter[@name='outputFormat']"
173           + "//ows:Value";
174       NodeList nodes = (NodeList) xPath.compile(xpathExpr).evaluate(doc, XPathConstants.NODESET);
175 
176       outputFormats = new ArrayList<>();
177       for (int i = 0; i < nodes.getLength(); i++) {
178         outputFormats.add(DomUtils.getTextValue((Element) nodes.item(i)));
179       }
180     }
181 
182     return outputFormats;
183   }

190   public static Map<String, SimpleWFSLayerDescription> describeWMSLayers(
191       String url, String username, String password, List<String> layers) {
192     try {
193       HTTPClient client = new SimpleHttpClient();
194       client.setUser(username);
195       client.setPassword(password);
196       client.setConnectTimeout(TIMEOUT);
197       client.setReadTimeout(TIMEOUT);
198       WebMapServer wms = new WebMapServer(new URL(url), client);
199       // Directly create WMS 1.1.1 request. Creating it from WebMapServer errors with GeoServer
200       // about unsupported request in capabilities unless we override WebMapServer to set up
201       // specifications.
202       DescribeLayerRequest describeLayerRequest = new WMS1_1_1().createDescribeLayerRequest(new URL(url));
203       // XXX Otherwise GeoTools will send VERSION=1.1.0...
204       describeLayerRequest.setProperty("VERSION", "1.1.1");
205       describeLayerRequest.setLayers(String.join(",", layers));
206       // GeoTools will throw a ClassCastException when a WMS ServiceException is returned
207       DescribeLayerResponse describeLayerResponse = wms.issueRequest(describeLayerRequest);
208 
209       Map<String, SimpleWFSLayerDescription> descriptions = new HashMap<>();
210       for (LayerDescription ld : describeLayerResponse.getLayerDescs()) {
211         String wfsUrl = getWfsUrl(ld, wms);
212 
213         if (wfsUrl != null && ld.getQueries() != null && ld.getQueries().length != 0) {
214           descriptions.put(ld.getName(), new SimpleWFSLayerDescription(wfsUrl, List.of(ld.getQueries())));
215         }
216       }
217       return Collections.unmodifiableMap(descriptions);
218     } catch (Exception e) {
219       String msg = String.format(
220           "Error in DescribeLayer request to WMS \"%s\": %s: %s", url, e.getClass(), e.getMessage());
221       if (logger.isTraceEnabled()) {
222         logger.trace(msg, e);
223       } else {
224         logger.debug("{}. Set log level to TRACE for stacktrace.", msg);
225       }
226     }
227 
228     return Collections.emptyMap();
229   }

231   private static String getWfsUrl(LayerDescription ld, WebMapServer wms) {
232     String wfsUrl = (ld.getWfs() != null) ? ld.getWfs().toString() : null;
233     if (wfsUrl == null && "WFS".equalsIgnoreCase(ld.getOwsType())) {
234       wfsUrl = ld.getOwsURL().toString();
235     }
236     // OGC 02-070 Annex B says the wfs/owsURL attributed are not required but implied. Some
237     // Deegree instance encountered has all attributes empty, and apparently the meaning is that
238     // the WFS URL is the same as the WMS URL (not explicitly defined in the spec).
239     if (wfsUrl == null) {
240       wfsUrl = wms.getInfo().getSource().toString();
241     }
242     return wfsUrl;
243   }
