7   package org.tailormap.api.geotools;
8   
9   import java.io.ByteArrayInputStream;
10  import javax.xml.XMLConstants;
11  import javax.xml.parsers.DocumentBuilder;
12  import javax.xml.parsers.DocumentBuilderFactory;
13  import org.apache.commons.lang3.StringUtils;
14  import org.w3c.dom.Document;
15  import org.w3c.dom.Element;
16  import org.w3c.dom.Node;
17  import org.w3c.dom.NodeList;
18  
19  public class WMSServiceExceptionUtil {
20  
21    /**
22     * Tries to parse a WMS service exception XML response of any WMS version and extract an error message.
23     *
24     * @param body The response body from a text/xml WMS response
25     * @return A service exception message or null if it can't be extracted
26     */
27    public static String tryGetServiceExceptionMessage(byte[] body) {
28      try {
29        DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
30        documentBuilderFactory.setNamespaceAware(true);
31        documentBuilderFactory.setExpandEntityReferences(false);
32        documentBuilderFactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
33        documentBuilderFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
34        DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
35        Document doc = builder.parse(new ByteArrayInputStream(body));
36  
37        String code = null;
38        String locator = null;
39        String text = null;
40  
41        if ("ServiceExceptionReport".equals(doc.getDocumentElement().getNodeName())) {
42          // WMS 1.3.0
43  
44          NodeList exceptions = doc.getDocumentElement().getElementsByTagName("ServiceException");
45          if (exceptions.getLength() > 0) {
46            Element exception = (Element) exceptions.item(0);
47            code = exception.getAttribute("code");
48            locator = exception.getAttribute("locator");
49            text = exception.getTextContent().trim();
50          }
51        } else if (doc.getDocumentElement().getNodeName().contains(":ExceptionReport")) {
52          // WMS 1.0.0 and 1.1.0
53  
54          NodeList children = doc.getDocumentElement().getChildNodes();
55          Element exception = null;
56          for (int i = 0; i < children.getLength(); i++) {
57            if (children.item(i).getNodeType() == Node.ELEMENT_NODE) {
58              exception = (Element) children.item(i);
59              break;
60            }
61          }
62          if (exception != null) {
63            code = exception.getAttribute("exceptionCode");
64            locator = exception.getAttribute("locator");
65            children = exception.getChildNodes();
66            for (int i = 0; i < children.getLength(); i++) {
67              if (children.item(i).getNodeType() == Node.ELEMENT_NODE) {
68                text = children.item(i).getTextContent().trim();
69                break;
70              }
71            }
72          }
73        }
74  
75        if (text != null) {
76          if (StringUtils.isNotBlank(locator)) {
77            text = "locator: " + locator + ": " + text;
78          }
79          if (StringUtils.isNotBlank(code)) {
80            text = "code: " + code + ": " + text;
81          }
82          return text;
83        } else {
84          return null;
85        }
86      } catch (Exception e) {
87        return null;
88      }
89    }
90  }
