7   package org.tailormap.api.repository.validation;
8   
9   import org.springframework.lang.NonNull;
10  import org.springframework.stereotype.Component;
11  import org.springframework.validation.Errors;
12  import org.springframework.validation.Validator;
13  import org.tailormap.api.persistence.Application;
14  import org.tailormap.api.repository.ApplicationRepository;
15  
16  @Component
17  public class ApplicationValidator implements Validator {
18    private final ApplicationRepository applicationRepository;
19  
20    public ApplicationValidator(ApplicationRepository applicationRepository) {
21      this.applicationRepository = applicationRepository;
22    }
23  
24    @Override
25    public boolean supports(@NonNull Class<?> clazz) {
26      return Application.class.isAssignableFrom(clazz);
27    }
28  
29    @Override
30    public void validate(@NonNull Object target, @NonNull Errors errors) {
31      Application app = (Application) target;
32      Application existing = applicationRepository.findByName(app.getName());
33      if (existing != null && !existing.getId().equals(app.getId())) {
34        errors.rejectValue(
35            "name", "duplicate", String.format("Application with name \"%s\" already exists.", app.getName()));
36      }
37    }
38  }
