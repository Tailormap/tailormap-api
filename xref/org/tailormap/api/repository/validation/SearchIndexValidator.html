6   package org.tailormap.api.repository.validation;
7   
8   import java.lang.invoke.MethodHandles;
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  import org.springframework.lang.NonNull;
12  import org.springframework.stereotype.Component;
13  import org.springframework.validation.Errors;
14  import org.springframework.validation.Validator;
15  import org.tailormap.api.persistence.SearchIndex;
16  import org.tailormap.api.persistence.TMFeatureSource;
17  import org.tailormap.api.repository.FeatureTypeRepository;
18  
19  @Component
20  public class SearchIndexValidator implements Validator {
21    private static final Logger logger =
22        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
23  
24    private final FeatureTypeRepository featureTypeRepository;
25  
26    public SearchIndexValidator(FeatureTypeRepository featureTypeRepository) {
27      this.featureTypeRepository = featureTypeRepository;
28    }
29  
30    @Override
31    public boolean supports(@NonNull Class<?> clazz) {
32      return SearchIndex.class.isAssignableFrom(clazz);
33    }
34  
35    @Override
36    public void validate(@NonNull Object target, @NonNull Errors errors) {
37      final SearchIndex searchIndex = (SearchIndex) target;
38      if (searchIndex.getFeatureTypeId() != null) {
39        featureTypeRepository.findById(searchIndex.getFeatureTypeId()).ifPresent((ft) -> {
40          if (TMFeatureSource.Protocol.WFS.equals(ft.getFeatureSource().getProtocol())) {
41            logger.warn(
42                "Attempt to index feature type '{}' from unsupported WFS source '{}'.",
43                ft.getName(),
44                ft.getFeatureSource().getTitle());
45            errors.rejectValue("featureTypeId", "invalid", "This feature type is not available for indexing.");
46          }
47        });
48      }
49    }
50  }
