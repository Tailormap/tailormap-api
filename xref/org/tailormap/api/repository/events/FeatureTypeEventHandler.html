6   package org.tailormap.api.repository.events;
7   
8   import java.io.IOException;
9   import java.lang.invoke.MethodHandles;
10  import java.sql.SQLException;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  import org.springframework.data.rest.core.annotation.HandleBeforeSave;
14  import org.springframework.data.rest.core.annotation.RepositoryEventHandler;
15  import org.springframework.http.HttpStatus;
16  import org.springframework.stereotype.Component;
17  import org.springframework.web.server.ResponseStatusException;
18  import org.tailormap.api.geotools.featuresources.AttachmentsHelper;
19  import org.tailormap.api.persistence.TMFeatureType;
20  
21  @Component
22  @RepositoryEventHandler
23  public class FeatureTypeEventHandler {
24    private static final Logger logger =
25        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
26  
27    @HandleBeforeSave
28    void handleBeforeSaveTMFeatureType(TMFeatureType featureType) {
29      try {
30        if (featureType.getSettings() != null
31            && featureType.getSettings().getAttachmentAttributes() != null
32            && !featureType.getSettings().getAttachmentAttributes().isEmpty()) {
33          AttachmentsHelper.createAttachmentTableForFeatureType(featureType);
34        }
35      } catch (IllegalArgumentException | SQLException | IOException e) {
36        logger.error(
37            "Error opening GeoTools datastore or creating attachments table or index for FeatureType: {}",
38            featureType.getName(),
39            e);
40        // TODO check if this is the correct exception to throw,
41        //  we may want to throw eg a BadRequest or illegal state/illegal argument exception instead
42        throw new ResponseStatusException(
43            HttpStatus.INTERNAL_SERVER_ERROR,
44            "Error creating attachments table or index for FeatureType: " + featureType.getName(),
45            e);
46      }
47    }
48  }
