6   package org.tailormap.api.repository.events;
7   
8   import org.springframework.data.rest.core.annotation.HandleAfterCreate;
9   import org.springframework.data.rest.core.annotation.HandleAfterDelete;
10  import org.springframework.data.rest.core.annotation.HandleAfterSave;
11  import org.springframework.data.rest.core.annotation.HandleBeforeCreate;
12  import org.springframework.data.rest.core.annotation.HandleBeforeSave;
13  import org.springframework.data.rest.core.annotation.RepositoryEventHandler;
14  import org.springframework.stereotype.Component;
15  import org.tailormap.api.persistence.OIDCConfiguration;
16  import org.tailormap.api.security.OIDCRepository;
17  
18  @RepositoryEventHandler
19  @Component
20  public class OIDCConfigurationEventHandler {
21    private final OIDCRepository oidcRepository;
22  
23    public OIDCConfigurationEventHandler(OIDCRepository oidcRepository) {
24      this.oidcRepository = oidcRepository;
25    }
26  
27    @HandleBeforeCreate
28    @HandleBeforeSave
29    public void handleBeforeCreateOrSave(OIDCConfiguration configuration) {
30      // If the user provided a "full" OIDC discovery URL, strip it.
31      if (configuration.getIssuerUrl().endsWith("/.well-known/openid-configuration")) {
32        String issuerUrl = configuration.getIssuerUrl();
33        issuerUrl = issuerUrl.substring(0, issuerUrl.lastIndexOf("/.well-known/openid-configuration"));
34        configuration.setIssuerUrl(issuerUrl);
35      }
36    }
37  
38    @HandleAfterCreate
39    @HandleAfterSave
40    @HandleAfterDelete
41    public void handleAfterCreateOrSaveOrDelete(OIDCConfiguration configuration) {
42      oidcRepository.synchronize();
43    }
44  }
