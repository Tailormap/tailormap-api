6   package org.tailormap.api.repository;
7   
8   import java.util.Collection;
9   import java.util.List;
10  import java.util.Objects;
11  import java.util.Set;
12  import org.springframework.data.jpa.repository.JpaRepository;
13  import org.springframework.data.jpa.repository.Query;
14  import org.springframework.data.repository.query.Param;
15  import org.springframework.data.rest.core.annotation.RepositoryRestResource;
16  import org.tailormap.api.persistence.Group;
17  import org.tailormap.api.security.TailormapAdditionalProperty;
18  
19  @RepositoryRestResource
20  public interface GroupRepository extends JpaRepository<Group, String> {
21    default List<TailormapAdditionalProperty> findAdditionalPropertiesByGroups(List<String> groups) {
22      return this.findAllById(groups).stream()
23          .map(Group::getAdditionalProperties)
24          .filter(Objects::nonNull)
25          .flatMap(List::stream)
26          .map(p -> new TailormapAdditionalProperty(p.getKey(), p.getIsPublic(), p.getValue()))
27          .toList();
28    }
29  
30    @Query(
31        "select g.aliasForGroup from Group g where g.name in :groups and g.aliasForGroup is not null and g.aliasForGroup <> ''")
32    Set<String> findAliasesForGroups(@Param("groups") Collection<String> groups);
33  }
