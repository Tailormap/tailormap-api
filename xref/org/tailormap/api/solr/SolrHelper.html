199   /**
200    * Add or update a feature type index for a layer.
201    *
202    * @param searchIndex the search index config
203    * @param tmFeatureType the feature type
204    * @param featureSourceFactoryHelper the feature source factory helper
205    * @param searchIndexRepository the search index repository, so we can save the {@code
206    *     searchIndex}
207    * @param progressListener the progress listener callback
208    * @param taskUuid the task UUID, when {@code null} we will attempt to use the UUID from the
209    *     {@code searchIndex#getSchedule()}
210    * @throws IOException if an I/O error occurs
211    * @throws SolrServerException if a Solr error occurs
212    * @return the possibly updated {@code searchIndex} object
213    */
214   @SuppressWarnings("FromTemporalAccessor")
215   public SearchIndex addFeatureTypeIndex(
216       @NotNull SearchIndex searchIndex,
217       @NotNull TMFeatureType tmFeatureType,
218       @NotNull FeatureSourceFactoryHelper featureSourceFactoryHelper,
219       @NotNull SearchIndexRepository searchIndexRepository,
220       @NotNull Consumer<TaskProgressEvent> progressListener,
221       @Nullable UUID taskUuid)
222       throws IOException, SolrServerException {
223 
224     createSchemaIfNotExists();
225 
226     final Instant startedAt = Instant.now();
227     final OffsetDateTime startedAtOffset =
228         startedAt.atOffset(ZoneId.systemDefault().getRules().getOffset(startedAt));
229 
230     if (null == taskUuid && null != searchIndex.getSchedule()) {
231       // this can be the case when this method is called directly such as when creating
232       // the test data. This in itself is not a big problem; it just means that the uuid
233       // in any progress events will be null (for that call).
234       taskUuid = searchIndex.getSchedule().getUuid();
235     }
236 
237     SearchIndexSummary summary =
238         new SearchIndexSummary().startedAt(startedAtOffset).total(0).duration(0.0);
239 
240     if (null == searchIndex.getSearchFieldsUsed()) {
241       logger.warn(
242           "No search fields configured for search index: {}, bailing out.", searchIndex.getName());
243       return searchIndexRepository.save(
244           searchIndex
245               .setStatus(SearchIndex.Status.ERROR)
246               .setSummary(summary.errorMessage("No search fields configured")));
247     }
248 
249     progressListener.accept(
250         new TaskProgressEvent()
251             .type(TaskType.INDEX.getValue())
252             .uuid(taskUuid)
253             .startedAt(startedAtOffset)
254             .progress(0));
255 
256     // set fields while filtering out hidden fields
257     List<String> searchFields =
258         searchIndex.getSearchFieldsUsed().stream()
259             .filter(s -> !tmFeatureType.getSettings().getHideAttributes().contains(s))
260             .toList();
261     List<String> displayFields =
262         searchIndex.getSearchDisplayFieldsUsed().stream()
263             .filter(s -> !tmFeatureType.getSettings().getHideAttributes().contains(s))
264             .toList();
265 
266     if (searchFields.isEmpty()) {
267       logger.warn(
268           "No valid search fields configured for featuretype: {}, bailing out.",
269           tmFeatureType.getName());
270       return searchIndexRepository.save(
271           searchIndex
272               .setStatus(SearchIndex.Status.ERROR)
273               .setSummary(summary.errorMessage("No search fields configured")));
274     }
275 
276     // add search and display properties to query
277     Set<String> propertyNames = new HashSet<>();
278     // always add primary key and default geometry to geotools query
279     propertyNames.add(tmFeatureType.getPrimaryKeyAttribute());
280     propertyNames.add(tmFeatureType.getDefaultGeometryAttribute());
281     propertyNames.addAll(searchFields);
282 
283     if (!displayFields.isEmpty()) {
284       propertyNames.addAll(displayFields);
285     }
286 
287     clearIndexForLayer(searchIndex.getId());
288 
289     logger.info(
290         "Indexing started for index id: {}, feature type: {}",
291         searchIndex.getId(),
292         tmFeatureType.getName());
293     searchIndex = searchIndexRepository.save(searchIndex.setStatus(SearchIndex.Status.INDEXING));
294 
295     // collect features to index
296     SimpleFeatureSource fs = featureSourceFactoryHelper.openGeoToolsFeatureSource(tmFeatureType);
297     Query q = new Query(fs.getName().toString());
298     // filter out any hidden properties (there should be none though)
299     tmFeatureType.getSettings().getHideAttributes().forEach(propertyNames::remove);
300     q.setPropertyNames(List.copyOf(propertyNames));
301     q.setStartIndex(0);
302     // TODO: make maxFeatures configurable?
303     // q.setMaxFeatures(Integer.MAX_VALUE);
304     logger.trace("Indexing query: {}", q);
305     SimpleFeatureCollection simpleFeatureCollection = fs.getFeatures(q);
306     final int total = simpleFeatureCollection.size();
307     List<FeatureIndexingDocument> docsBatch = new ArrayList<>(solrBatchSize);
308     // TODO this does not currently batch/page the feature source query, this doesn't seem to be an
309     //   issue for now but could be if the feature source is very, very large or slow
310     UpdateResponse updateResponse;
311     int indexCounter = 0;
312     int indexSkippedCounter = 0;
313     try (SimpleFeatureIterator iterator = simpleFeatureCollection.features()) {
314       while (iterator.hasNext()) {
315         indexCounter++;
316         SimpleFeature feature = iterator.next();
317         // note that this will create a unique document
318         FeatureIndexingDocument doc =
319             new FeatureIndexingDocument(feature.getID(), searchIndex.getId());
320         List<String> searchValues = new ArrayList<>();
321         List<String> displayValues = new ArrayList<>();
322         propertyNames.forEach(
323             propertyName -> {
324               Object value = feature.getAttribute(propertyName);
325               if (value != null) {
326                 if (value instanceof Geometry
327                     && propertyName.equals(tmFeatureType.getDefaultGeometryAttribute())) {
328                   // We could use GeoJSON, but WKT is more compact and that would also incur a
329                   // change to the API
330                   doc.setGeometry(GeometryProcessor.processGeometry(value, true, true, null));
331                 } else {
332                   if (searchFields.contains(propertyName)) {
333                     searchValues.add(value.toString());
334                   }
335                   if (displayFields.contains(propertyName)) {
336                     displayValues.add(value.toString());
337                   }
338                 }
339               }
340             });
341         if (searchValues.isEmpty() || displayValues.isEmpty()) {
342           // this is a record/document that can either not be found or not be displayed
343           logger.trace(
344               "No search or display values found for feature: {} in featuretype: {}, skipped for indexing",
345               feature.getID(),
346               tmFeatureType.getName());
347           indexSkippedCounter++;
348         } else {
349           doc.setSearchFields(searchValues.toArray(new String[0]));
350           doc.setDisplayFields(displayValues.toArray(new String[0]));
351           docsBatch.add(doc);
352         }
353 
354         if (indexCounter % solrBatchSize == 0) {
355           updateResponse = solrClient.addBeans(docsBatch, solrQueryTimeout);
356           logger.info(
357               "Added {} documents of {} to index, result status: {}",
358               indexCounter - indexSkippedCounter,
359               total,
360               updateResponse.getStatus());
361           progressListener.accept(
362               new TaskProgressEvent()
363                   .type(TaskType.INDEX.getValue())
364                   .uuid(taskUuid)
365                   .startedAt(startedAtOffset)
366                   .progress((indexCounter - indexSkippedCounter))
367                   .total(total));
368           docsBatch.clear();
369         }
370       }
371     } finally {
372       if (fs.getDataStore() != null) fs.getDataStore().dispose();
373     }
374 
375     if (!docsBatch.isEmpty()) {
376       solrClient.addBeans(docsBatch, solrQueryTimeout);
377       logger.info("Added last {} documents of {} to index", docsBatch.size(), total);
378       progressListener.accept(
379           new TaskProgressEvent()
380               .type(TaskType.INDEX.getValue())
381               .uuid(taskUuid)
382               .startedAt(startedAtOffset)
383               .progress((indexCounter - indexSkippedCounter))
384               .total(total));
385     }
386     final Instant finishedAt = Instant.now();
387     final OffsetDateTime finishedAtOffset =
388         finishedAt.atOffset(ZoneId.systemDefault().getRules().getOffset(finishedAt));
389     Duration processTime = Duration.between(startedAt, finishedAt).abs();
390     logger.info(
391         "Indexing finished for index id: {}, featuretype: {} at {} in {}",
392         searchIndex.getId(),
393         tmFeatureType.getName(),
394         finishedAtOffset,
395         processTime);
396     updateResponse = this.solrClient.commit();
397     logger.debug("Update response commit status: {}", updateResponse.getStatus());
398 
399     if (indexSkippedCounter > 0) {
400       logger.warn(
401           "{} features were skipped because no search or display values were found.",
402           indexSkippedCounter);
403     }
404 
405     return searchIndexRepository.save(
406         searchIndex
407             .setLastIndexed(finishedAtOffset)
408             .setStatus(SearchIndex.Status.INDEXED)
409             .setSummary(
410                 summary
411                     .total(total)
412                     .skippedCounter(indexSkippedCounter)
413                     .duration(
414                         BigDecimal.valueOf(processTime.getSeconds())
415                             .add(BigDecimal.valueOf(processTime.getNano(), 9))
416                             .doubleValue())
417                     .errorMessage(null)));
418   }

420   /**
421    * Clear the index for a layer.
422    *
423    * @param searchLayerId the layer id
424    * @throws IOException if an I/O error occurs
425    * @throws SolrServerException if a Solr error occurs
426    */
427   public void clearIndexForLayer(@NotNull Long searchLayerId)
428       throws IOException, SolrServerException {
429 
430     QueryResponse response =
431         solrClient.query(
432             new SolrQuery("exists(query(" + SEARCH_LAYER + ":" + searchLayerId + "))"));
433     if (response.getResults().getNumFound() > 0) {
434       logger.info("Clearing index for searchLayer {}", searchLayerId);
435       UpdateResponse updateResponse = solrClient.deleteByQuery(SEARCH_LAYER + ":" + searchLayerId);
436       logger.debug("Delete response status: {}", updateResponse.getStatus());
437       updateResponse = solrClient.commit();
438       logger.debug("Commit response status: {}", updateResponse.getStatus());
439     } else {
440       logger.info("No index to clear for layer {}", searchLayerId);
441     }
442   }

444   /**
445    * Search in the index for a layer. The given query is augmented to filter on the {@code
446    * solrLayerId}.
447    *
448    * @param searchIndex the search index
449    * @param solrQuery the query, when {@code null} or empty, the query is set to {@code *} (match
450    *     all)
451    * @param solrPoint the point to search around, in (x y) format
452    * @param solrDistance the distance to search around the point in Solr distance units (kilometers)
453    * @param start the start index, starting at 0
454    * @param numResultsToReturn the number of results to return
455    * @return the documents
456    * @throws IOException if an I/O error occurs
457    * @throws SolrServerException if a Solr error occurs
458    */
459   public SearchResponse findInIndex(
460       @NotNull SearchIndex searchIndex,
461       String solrQuery,
462       String solrFilterQuery,
463       String solrPoint,
464       Double solrDistance,
465       int start,
466       int numResultsToReturn)
467       throws IOException, SolrServerException, SolrException {
468 
469     logger.info("Find in index for {}", searchIndex.getId());
470     if (null == solrQuery || solrQuery.isBlank()) {
471       solrQuery = "*";
472     }
473     // TODO We could escape special/syntax characters, but that also prevents using
474     //      keys like ~ and *
475     // solrQuery = ClientUtils.escapeQueryChars(solrQuery);
476 
477     final SolrQuery query =
478         new SolrQuery(INDEX_SEARCH_FIELD + ":" + solrQuery)
479             .setShowDebugInfo(logger.isDebugEnabled())
480             .setTimeAllowed(solrQueryTimeout)
481             .setIncludeScore(true)
482             .setFields(SEARCH_ID_FIELD, INDEX_DISPLAY_FIELD, INDEX_GEOM_FIELD)
483             .addFilterQuery(SEARCH_LAYER + ":" + searchIndex.getId())
484             .setSort("score", SolrQuery.ORDER.desc)
485             .addSort(SEARCH_ID_FIELD, SolrQuery.ORDER.asc)
486             .setRows(numResultsToReturn)
487             .setStart(start);
488 
489     if (null != solrFilterQuery && !solrFilterQuery.isBlank()) {
490       query.addFilterQuery(solrFilterQuery);
491     }
492     if (null != solrPoint && null != solrDistance) {
493       if (null == solrFilterQuery
494           || !(solrFilterQuery.startsWith("{!geofilt") || solrFilterQuery.startsWith("{!bbox"))) {
495         query.addFilterQuery("{!geofilt sfield=" + INDEX_GEOM_FIELD + "}");
496       }
497       query.add("pt", solrPoint);
498       query.add("d", solrDistance.toString());
499     }
500     query.set("q.op", "AND");
501     logger.debug("Solr query: {}", query);
502 
503     final QueryResponse response = solrClient.query(query);
504     logger.debug("response: {}", response);
505 
506     final SolrDocumentList solrDocumentList = response.getResults();
507     logger.debug("Found {} solr documents", solrDocumentList.getNumFound());
508     final SearchResponse searchResponse =
509         new SearchResponse()
510             .total(solrDocumentList.getNumFound())
511             .start(response.getResults().getStart())
512             .maxScore(solrDocumentList.getMaxScore());
513     response
514         .getResults()
515         .forEach(
516             solrDocument -> {
517               List<String> displayValues =
518                   solrDocument.getFieldValues(INDEX_DISPLAY_FIELD).stream()
519                       .map(Object::toString)
520                       .toList();
521               searchResponse.addDocumentsItem(
522                   new SearchDocument()
523                       .fid(solrDocument.getFieldValue(SEARCH_ID_FIELD).toString())
524                       .geometry(solrDocument.getFieldValue(INDEX_GEOM_FIELD).toString())
525                       .displayValues(displayValues));
526             });
527 
528     return searchResponse;
529   }

541   private boolean checkSchemaIfFieldExists(String fieldName) {
542     SchemaRequest.Field fieldCheck = new SchemaRequest.Field(fieldName);
543     try {
544       SchemaResponse.FieldResponse isField = fieldCheck.process(solrClient);
545       logger.debug("Field {} exists", isField.getField());
546       return true;
547     } catch (SolrServerException | BaseHttpSolrClient.RemoteSolrException e) {
548       logger.debug(
549           "Field {} does not exist or could not be retrieved. Assuming it does not exist.",
550           fieldName);
551     } catch (IOException e) {
552       logger.error("Tried getting field: {}, but failed.", fieldName, e);
553     }
554     return false;
555   }

557   /**
558    * @param fieldName the name of the field to create
559    * @throws SolrServerException if a Solr error occurs
560    * @throws IOException if an I/O error occurs
561    */
562   private void createSchemaFieldIfNotExists(String fieldName)
563       throws SolrServerException, IOException {
564     if (!checkSchemaIfFieldExists(fieldName)) {
565       logger.info("Creating Solr field {}.", fieldName);
566       SchemaRequest.AddField schemaRequest = solrSearchFields.get(fieldName);
567       SolrResponse response = schemaRequest.process(solrClient);
568       logger.debug("Field type {} created", response);
569       solrClient.commit();
570     }
571   }

573   /** Programmatically create the schema if it does not exist. */
574   private void createSchemaIfNotExists() {
575     solrSearchFields.forEach(
576         (key, value) -> {
577           try {
578             if (key.equals(INDEX_GEOM_FIELD)) {
579               createGeometryFieldTypeIfNotExists();
580             }
581             createSchemaFieldIfNotExists(key);
582           } catch (SolrServerException | IOException e) {
583             logger.error(
584                 "Error creating schema field: {} indexing may fail. Details: {}",
585                 key,
586                 e.getLocalizedMessage(),
587                 e);
588           }
589         });
590   }

592   private void createGeometryFieldTypeIfNotExists() throws SolrServerException, IOException {
593     SchemaRequest.FieldType fieldTypeCheck = new SchemaRequest.FieldType(SOLR_SPATIAL_FIELDNAME);
594     try {
595       SchemaResponse.FieldTypeResponse isFieldType = fieldTypeCheck.process(solrClient);
596       logger.debug("Field type {} exists", isFieldType.getFieldType());
597       return;
598     } catch (SolrServerException | BaseHttpSolrClient.RemoteSolrException e) {
599       logger.debug(
600           "Field type {} does not exist or could not be retrieved. Assuming it does not exist.",
601           SOLR_SPATIAL_FIELDNAME);
602     } catch (IOException e) {
603       logger.error("Tried getting field type: {}, but failed.", SOLR_SPATIAL_FIELDNAME, e);
604     }
605 
606     logger.info(
607         "Creating Solr field type for {} with validation rule {}",
608         SOLR_SPATIAL_FIELDNAME,
609         solrGeometryValidationRule);
610     FieldTypeDefinition spatialFieldTypeDef = new FieldTypeDefinition();
611     Map<String, Object> spatialFieldAttributes =
612         new HashMap<>(
613             Map.of(
614                 "name", SOLR_SPATIAL_FIELDNAME,
615                 "class", "solr.SpatialRecursivePrefixTreeFieldType",
616                 "spatialContextFactory", "JTS",
617                 "geo", false,
618                 "distanceUnits", "kilometers",
619                 "distCalculator", "cartesian",
620                 "format", "WKT",
621                 "autoIndex", true,
622                 "distErrPct", "0.025",
623                 "maxDistErr", "0.001"));
624     spatialFieldAttributes.putAll(
625         Map.of(
626             "prefixTree",
627             "packedQuad",
628             // see
629             // https://locationtech.github.io/spatial4j/apidocs/org/locationtech/spatial4j/context/jts/ValidationRule.html
630             "validationRule",
631             this.solrGeometryValidationRule,
632             // NOTE THE ODDITY in coordinate order of "worldBounds",
