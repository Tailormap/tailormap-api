7   package org.tailormap.api.service;
8   
9   import java.util.Locale;
10  import java.util.Optional;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  import org.springframework.beans.factory.annotation.Autowired;
14  import org.springframework.beans.factory.annotation.Value;
15  import org.springframework.context.MessageSource;
16  import org.springframework.lang.Nullable;
17  import org.springframework.mail.SimpleMailMessage;
18  import org.springframework.mail.javamail.JavaMailSender;
19  import org.springframework.scheduling.annotation.Async;
20  import org.springframework.stereotype.Service;
21  import org.springframework.transaction.annotation.Transactional;
22  import org.tailormap.api.persistence.TemporaryToken;
23  import org.tailormap.api.persistence.User;
24  import org.tailormap.api.repository.TemporaryTokenRepository;
25  import org.tailormap.api.repository.UserRepository;
26  
27  @Service
28  public class PasswordResetEmailService {
29  
30    private static final Logger logger = LoggerFactory.getLogger(PasswordResetEmailService.class);
31  
32    private final Optional<JavaMailSender> emailSender;
33    private final UserRepository userRepository;
34    private final TemporaryTokenRepository temporaryTokenRepository;
35    private final MessageSource messageSource;
36  
37    @Value("${tailormap-api.mail.from}")
38    private String mailFrom;
39  
40    @Autowired(required = false)
41    public PasswordResetEmailService(
42        @Nullable JavaMailSender emailSender,
43        UserRepository userRepository,
44        TemporaryTokenRepository temporaryTokenRepository,
45        MessageSource messageSource) {
46      this.emailSender = Optional.ofNullable(emailSender);
47      this.userRepository = userRepository;
48      this.temporaryTokenRepository = temporaryTokenRepository;
49      this.messageSource = messageSource;
50    }
51  
52    @Async("passwordResetTaskExecutor")
53    @Transactional
54    public void sendPasswordResetEmailAsync(
55        String email, String absoluteLinkPrefix, Locale locale, int tokenExpiryMinutes) {
56      try {
57        if (emailSender.isEmpty()) {
58          logger.warn("Cannot send password reset email: JavaMailSender is not configured");
59          return;
60        }
61  
62        User user = userRepository.findByEmail(email).orElse(null);
63        if (user == null || !user.isEnabledAndValidUntil()) {
64          return;
65        }
66  
67        TemporaryToken token =
68            new TemporaryToken(TemporaryToken.TokenType.PASSWORD_RESET, user.getUsername(), tokenExpiryMinutes);
69        token = temporaryTokenRepository.save(token);
70  
71        String absoluteLink =
72            absoluteLinkPrefix + "/user/password-reset/" + token.getCombinedTokenAndExpirationAsBase64();
73  
74        SimpleMailMessage message = new SimpleMailMessage();
75        message.setFrom(mailFrom);
76        message.setTo(user.getEmail());
77        message.setSubject(messageSource.getMessage("reset-password-request.email-subject", null, locale));
78        message.setText(
79            messageSource.getMessage("reset-password-request.email-body", new Object[] {absoluteLink}, locale));
80  
81        logger.info("Sending password reset email for user: {}", user.getUsername());
82        emailSender.ifPresent((sender) -> sender.send(message)); // blocking, but run in async thread
83      } catch (Exception e) {
84        logger.error("Failed to send password reset email", e);
85      }
86    }
87  }
